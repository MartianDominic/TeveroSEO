# Phase 102: Advanced Document Builder — Unified PRD

> **Version:** 1.0 | **Status:** Ready for Implementation | **Last Updated:** 2026-05-15

## Executive Summary

Build a document generation system that turns uploaded PDFs into reusable templates with variable injection. Users design externally (Figma, Canva, Word), upload the PDF, click to place `{{variables}}`, and generate personalized proposals at **$0.001/doc** — 100x cheaper than AI-heavy approaches.

**Core Insight:** Nobody solves "I have a beautiful PDF, make it data-driven." Competitors (PandaDoc, Proposify) require rebuilding in their editors. We let users keep their existing designs.

```typescript
    alignment?: 'left' | 'center' | 'right';

  validation?: {
    required: boolean;
    minLength?: number;
    maxLength?: number;
    pattern?: string;              // Regex for custom validation
  };
  
  defaultValue?: string;
  filters?: string[];              // ['truncate:50', 'uppercase']
}
```

## 6. Technical Implementation

### 6.1 PDF Rendering with pdf-lib

```typescript
import { PDFDocument, StandardFonts, rgb } from 'pdf-lib';

interface RenderOptions {
  template: DocumentTemplate;
  values: Record<string, unknown>;
}

export async function renderDocument({ template, values }: RenderOptions): Promise<Uint8Array> {
  // Load template PDF from R2
  const templateBytes = await env.R2.get(template.source_file_key);
  if (!templateBytes) throw new Error('Template not found');
  
  const pdfDoc = await PDFDocument.load(await templateBytes.arrayBuffer());
  
  // Embed fonts
  const fonts = {
    helvetica: await pdfDoc.embedFont(StandardFonts.Helvetica),
    helveticaBold: await pdfDoc.embedFont(StandardFonts.HelveticaBold),
    times: await pdfDoc.embedFont(StandardFonts.TimesRoman),
    courier: await pdfDoc.embedFont(StandardFonts.Courier),
  };
  
  // Process each variable
  for (const variable of template.variables) {
    const page = pdfDoc.getPages()[variable.position.page - 1];
    if (!page) continue;
    
    // Resolve value with filters
    const rawValue = resolveValue(variable.key, values);
    const value = applyFilters(rawValue, variable.filters ?? []);
    
    // Get styling
    const fontSize = variable.style?.fontSize ?? 12;
    const fontKey = variable.style?.fontWeight === 'bold' 
      ? `${variable.style?.fontFamily ?? 'helvetica'}Bold`
      : variable.style?.fontFamily ?? 'helvetica';
    const font = fonts[fontKey] ?? fonts.helvetica;
    
    // Clear existing content with white rectangle
    page.drawRectangle({
      x: variable.position.x,
      y: variable.position.y,
      width: variable.position.width,
      height: variable.position.height,
      color: rgb(1, 1, 1), // White
    });
    
    // Draw new value
    page.drawText(String(value), {
      x: variable.position.x,
      y: variable.position.y + (variable.position.height / 4), // Vertical centering approximation
      size: fontSize,
      font,
      color: variable.style?.color 
        ? hexToRgb(variable.style.color) 
        : rgb(0, 0, 0),
    });
  }
  
  return pdfDoc.save();
}

function resolveValue(key: string, values: Record<string, unknown>): unknown {
  const parts = key.split('.');
  let current: unknown = values;
  
  for (const part of parts) {
    if (current === null || current === undefined) return '';
    current = (current as Record<string, unknown>)[part];
  }
  
  return current ?? '';
}

function applyFilters(value: unknown, filters: string[]): string {
  let result = String(value);
  
  for (const filter of filters) {
    const [name, ...args] = filter.split(':');
    
    switch (name) {
      case 'truncate':
        const maxLength = parseInt(args[0] ?? '50', 10);
        result = result.length > maxLength 
          ? result.slice(0, maxLength - 3) + '...'
          : result;
        break;
        
      case 'uppercase':
        result = result.toUpperCase();
        break;
        
      case 'lowercase':
        result = result.toLowerCase();
        break;
        
      case 'currency':
        const locale = args[0] === 'EUR' ? 'lt-LT' : 'en-US';
        const currency = args[0] ?? 'USD';
        result = new Intl.NumberFormat(locale, { 
          style: 'currency', 
          currency 
        }).format(Number(value));
        break;
        
      case 'date':
        const format = args[0] ?? 'medium';
        const date = new Date(String(value));
        result = date.toLocaleDateString('lt-LT', {
          dateStyle: format as 'short' | 'medium' | 'long',
        });
        break;
    }
  }
  
  return result;
}
```

### 6.2 URL-to-PDF with Puppeteer

```typescript
import puppeteer from 'puppeteer';

interface UrlToPdfOptions {
  url: string;
  variables: Record<string, string>;
  viewport?: { width: number; height: number };
  waitFor?: string;
}

export async function captureUrlAsPdf(options: UrlToPdfOptions): Promise<Uint8Array> {
  const browser = await puppeteer.launch({
    headless: true,
    args: ['--no-sandbox', '--disable-setuid-sandbox'],
  });
  
  try {
    const page = await browser.newPage();
    
    // Set viewport
    await page.setViewport({
      width: options.viewport?.width ?? 1920,
      height: options.viewport?.height ?? 1080,
    });
    
    // Build URL with query params
    const url = new URL(options.url);
    for (const [key, value] of Object.entries(options.variables)) {
      url.searchParams.set(key, value);
    }
    
    // Navigate
    await page.goto(url.toString(), { waitUntil: 'networkidle0' });
    
    // Wait for specific selector if provided
    if (options.waitFor) {
      await page.waitForSelector(options.waitFor, { timeout: 10000 });
    }
    
    // Generate PDF
    const pdfBuffer = await page.pdf({
      format: 'A4',
      printBackground: true,
      margin: { top: '20px', right: '20px', bottom: '20px', left: '20px' },
    });
    
    return new Uint8Array(pdfBuffer);
  } finally {
    await browser.close();
  }
}
```
