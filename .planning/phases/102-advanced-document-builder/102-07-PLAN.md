import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
import { db } from '@/db';
import { uploadedDocuments } from '@/db/schema/document-builder';
import { createId } from '@paralleldrive/cuid2';
import { documentProcessingQueue } from './processing-queue';

const ALLOWED_TYPES = ['application/pdf', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document', 'image/png', 'image/jpeg'];
const MAX_SIZE = 20 * 1024 * 1024; // 20MB

const r2 = new S3Client({
  region: 'auto',
  endpoint: process.env.R2_ENDPOINT,
  credentials: {
    accessKeyId: process.env.R2_ACCESS_KEY_ID!,
    secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
  },
});

export async function uploadDocument(
  file: File,
  workspaceId: string
): Promise<{ documentId: string; status: string }> {
  // Validate
  if (!ALLOWED_TYPES.includes(file.type)) {
    throw new Error(`Unsupported file type: ${file.type}. Allowed: PDF, DOCX, PNG, JPG`);
  }
  if (file.size > MAX_SIZE) {
    throw new Error(`File too large: ${(file.size / 1024 / 1024).toFixed(1)}MB. Max: 20MB`);
  }
  
  const documentId = createId();
  const r2Key = `${workspaceId}/${documentId}/${file.name}`;
  const fileType = file.type.includes('pdf') ? 'pdf' 
    : file.type.includes('document') ? 'docx' 
    : 'image';
  
  // Upload to R2
  const buffer = Buffer.from(await file.arrayBuffer());
  await r2.send(new PutObjectCommand({
    Bucket: process.env.R2_BUCKET || 'documents',
    Key: r2Key,
    Body: buffer,
    ContentType: file.type,
  }));
  
  // Create database record
  await db.insert(uploadedDocuments).values({
    id: documentId,
    workspaceId,
    fileName: file.name,
    fileType,
    fileSize: file.size,
    mimeType: file.type,
    r2Key,
    status: 'pending',
  });
  
  // Queue for processing
  await documentProcessingQueue.add('process', { documentId }, {
    jobId: documentId,
    attempts: 3,
    backoff: { type: 'exponential', delay: 5000 },
  });
  
  return { documentId, status: 'pending' };
}

export async function getDocumentStatus(documentId: string) {
  const doc = await db.query.uploadedDocuments.findFirst({
    where: eq(uploadedDocuments.id, documentId),
  });
  
  if (!doc) throw new Error('Document not found');
  
  return {
    id: doc.id,
    fileName: doc.fileName,
    status: doc.status,
    progress: doc.processingProgress,
    error: doc.processingError,
    ocrTier: doc.ocrTier,
    confidence: doc.ocrConfidence,
  };
}

import { Queue, Worker, Job } from 'bullmq';
import { redis } from '@/lib/redis';
import { db } from '@/db';
import { uploadedDocuments } from '@/db/schema/document-builder';
import { eq } from 'drizzle-orm';

const QUEUE_NAME = 'document-processing';

export const documentProcessingQueue = new Queue(QUEUE_NAME, { 
  connection: redis,
  defaultJobOptions: {
    removeOnComplete: 100,
    removeOnFail: 50,
  }
});

interface ProcessingJob {
  documentId: string;
}

export const documentProcessingWorker = new Worker<ProcessingJob>(
  QUEUE_NAME,
  async (job: Job<ProcessingJob>) => {
    const { documentId } = job.data;
    
    // Update status to processing
    await db.update(uploadedDocuments)
      .set({ 
        status: 'processing',
        processingStartedAt: new Date(),
      })
      .where(eq(uploadedDocuments.id, documentId));
    
    try {
      // Update progress helper
      const updateProgress = async (progress: number) => {
        await db.update(uploadedDocuments)
          .set({ processingProgress: progress })
          .where(eq(uploadedDocuments.id, documentId));
        await job.updateProgress(progress);
      };
      
      // Actual processing will be implemented in 102-08 (parsers) and 102-09 (OCR)
      // For now, just mark complete
      await updateProgress(10);
      
      // Placeholder: parsing + OCR + structure detection
      // This will call: extractText(), detectStructure(), extractTheme()
      
      await updateProgress(100);
      
      await db.update(uploadedDocuments)
        .set({ 
          status: 'completed',
          processingCompletedAt: new Date(),
        })
        .where(eq(uploadedDocuments.id, documentId));
      
      return { documentId, status: 'completed' };
      
    } catch (error) {
      await db.update(uploadedDocuments)
        .set({ 
          status: 'failed',
          processingError: error instanceof Error ? error.message : 'Unknown error',
        })
        .where(eq(uploadedDocuments.id, documentId));
      
      throw error; // Re-throw for BullMQ retry logic
    }
  },
  { 
    connection: redis,
    concurrency: 3, // Process 3 documents in parallel
  }
);

// Progress event emitter for WebSocket/SSE
documentProcessingWorker.on('progress', async (job, progress) => {
  // Emit to connected clients via WebSocket (implemented in UI layer)
  console.log(`Document ${job.data.documentId}: ${progress}%`);
});

import { NextRequest, NextResponse } from 'next/server';
import { auth } from '@clerk/nextjs/server';
import { rateLimit } from '@/lib/middleware/rate-limit';
import { uploadDocument } from '@/lib/document-processing/upload-service';
import { getWorkspaceId } from '@/lib/workspace';

export async function POST(request: NextRequest) {
  try {
    // Auth
    const { userId } = await auth();
    if (!userId) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    // Rate limit: 10 uploads per minute
    const rateLimitResult = await rateLimit(request, { limit: 10, window: 60 });
    if (!rateLimitResult.success) {
      return NextResponse.json({ error: 'Rate limit exceeded' }, { status: 429 });
    }
    
    // Parse form data
    const formData = await request.formData();
    const file = formData.get('file') as File | null;
    const workspaceId = formData.get('workspaceId') as string || await getWorkspaceId(userId);
    
    if (!file) {
      return NextResponse.json({ error: 'No file provided' }, { status: 400 });
    }
    
    // Upload and queue for processing
    const result = await uploadDocument(file, workspaceId);
    
    return NextResponse.json({
      success: true,
      documentId: result.documentId,
      status: result.status,
      message: 'Document uploaded and queued for processing',
    });
    
  } catch (error) {
    console.error('Upload error:', error);
    return NextResponse.json(
      { error: error instanceof Error ? error.message : 'Upload failed' },
      { status: 400 }
    );
  }
}

// Status endpoint
export async function GET(request: NextRequest) {
  const { userId } = await auth();
  if (!userId) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }
  
  const documentId = request.nextUrl.searchParams.get('documentId');
  if (!documentId) {
    return NextResponse.json({ error: 'documentId required' }, { status: 400 });
  }
  
  const { getDocumentStatus } = await import('@/lib/document-processing/upload-service');
  const status = await getDocumentStatus(documentId);
  
  return NextResponse.json(status);
}

import { useState, useCallback, useEffect } from 'react';

type Status = 'idle' | 'uploading' | 'processing' | 'completed' | 'error';

export function useDocumentProcessing() {
  const [status, setStatus] = useState<Status>('idle');
  const [progress, setProgress] = useState(0);
  const [error, setError] = useState<string | null>(null);
  const [documentId, setDocumentId] = useState<string | null>(null);
  
  const upload = useCallback(async (file: File, workspaceId: string) => {
    setStatus('uploading');
    setError(null);
    setProgress(0);
    
    try {
      const formData = new FormData();
      formData.append('file', file);
      formData.append('workspaceId', workspaceId);
      
      const response = await fetch('/api/documents/upload', {
        method: 'POST',
        body: formData,
      });
      
      const data = await response.json();
      
      if (!response.ok) {
        throw new Error(data.error || 'Upload failed');
      }
      
      setDocumentId(data.documentId);
      setStatus('processing');
      
      return data;
    } catch (e) {
      setError(e instanceof Error ? e.message : 'Upload failed');
      setStatus('error');
      return null;
    }
  }, []);
  
  // Poll for progress when processing
  useEffect(() => {
    if (status !== 'processing' || !documentId) return;
    
    const pollInterval = setInterval(async () => {
      try {
        const response = await fetch(`/api/documents/upload?documentId=${documentId}`);
        const data = await response.json();
        
        setProgress(data.progress || 0);
        
        if (data.status === 'completed') {
          setStatus('completed');
          clearInterval(pollInterval);
        } else if (data.status === 'failed') {
          setError(data.error || 'Processing failed');
          setStatus('error');
          clearInterval(pollInterval);
        }
      } catch (e) {
        // Ignore polling errors, will retry
      }
    }, 1000);
    
    return () => clearInterval(pollInterval);
  }, [status, documentId]);
  
  const reset = useCallback(() => {
    setStatus('idle');
    setProgress(0);
    setError(null);
    setDocumentId(null);
  }, []);
  
  return { upload, status, progress, error, documentId, reset };
}
