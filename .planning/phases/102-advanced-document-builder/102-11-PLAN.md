```typescript
import { generateText } from 'ai';
import { google } from '@ai-sdk/google';
import { db } from '@/db';
import { uploadedDocuments, brandThemes } from '@/db/schema/document-builder';
import { eq } from 'drizzle-orm';

export interface BrandTheme {
  colors: string[];
  primaryColor?: string;
  secondaryColor?: string;
  fonts: Array<{ name: string; usage: 'heading' | 'body' | 'accent' }>;
  headingFont?: string;
  bodyFont?: string;
  voiceAttributes: {
    tone: string[];
    vocabulary: string[];
    patterns: string[];
  };
  confidence: number;
}

export async function extractTheme(documentId: string): Promise<BrandTheme> {
  // Get document with extracted metadata
  const doc = await db.query.uploadedDocuments.findFirst({
    where: eq(uploadedDocuments.id, documentId),
  });
  
  if (!doc) throw new Error('Document not found');
  
  const metadata = doc.extractedMetadata as any;
  const text = (doc.extractedText as any)?.text || '';
  
  // Extract colors from metadata (from PDF parser)
  const colors = metadata?.colors || [];
  const primaryColor = colors[0];
  const secondaryColor = colors[1];
  
  // Extract fonts from metadata
  const rawFonts = metadata?.fonts || [];
  const fonts = classifyFonts(rawFonts);
  const headingFont = fonts.find(f => f.usage === 'heading')?.name;
  const bodyFont = fonts.find(f => f.usage === 'body')?.name;
  
  // AI voice analysis
  const voiceAttributes = await analyzeVoice(text);
  
  const theme: BrandTheme = {
    colors,
    primaryColor,
    secondaryColor,
    fonts,
    headingFont,
    bodyFont,
    voiceAttributes,
    confidence: calculateConfidence(colors, fonts, voiceAttributes),
  };
  
  // Save to database
  await db.insert(brandThemes).values({
    documentId,
    workspaceId: doc.workspaceId,
    colors,
    primaryColor,
    secondaryColor,
    fonts,
    headingFont,
    bodyFont,
    voiceAttributes,
    extractionConfidence: theme.confidence,
  });
  
  return theme;
}

function classifyFonts(rawFonts: Array<{ font: string; size: number; usage: number }>) {
  // Sort by usage (character count)
  const sorted = [...rawFonts].sort((a, b) => b.usage - a.usage);
  
  return sorted.slice(0, 5).map((f, i) => ({
    name: f.font,
    usage: f.size > 16 ? 'heading' : i === 0 ? 'body' : 'accent',
  }));
}

async function analyzeVoice(text: string): Promise<BrandTheme['voiceAttributes']> {
  if (text.length < 100) {
    return { tone: [], vocabulary: [], patterns: [] };
  }
  
  const { text: analysis } = await generateText({
    model: google('gemini-1.5-pro-latest'),
    prompt: `Analyze the voice and tone of this business document:

---
${text.slice(0, 5000)}
---

Return JSON:
{
  "tone": ["formal"|"casual"|"technical"|"friendly"|"authoritative"|"persuasive"],
  "vocabulary": ["industry-specific terms found"],
  "patterns": ["common phrases or sentence structures"]
}

Return valid JSON only.`,
    temperature: 0.3,
  });
  
  try {
    return JSON.parse(analysis);
  } catch {
    return { tone: ['professional'], vocabulary: [], patterns: [] };
  }
}

function calculateConfidence(colors: string[], fonts: any[], voice: any): number {
  let score = 50; // Base score
  
  if (colors.length >= 2) score += 15;
  if (fonts.length >= 2) score += 15;
  if (voice.tone.length > 0) score += 10;
  if (voice.vocabulary.length > 0) score += 10;
  
  return Math.min(100, score);
}
```

Create useUndoRedo.ts:

```typescript
import { useState, useCallback, useEffect } from 'react';

interface UndoRedoState<T> {
  past: T[];
  present: T;
  future: T[];
}

export function useUndoRedo<T>(initialState: T) {
  const [state, setState] = useState<UndoRedoState<T>>({
    past: [],
    present: initialState,
    future: [],
  });
  
  const canUndo = state.past.length > 0;
  const canRedo = state.future.length > 0;
  
  const set = useCallback((newState: T | ((prev: T) => T)) => {
    setState(prev => {
      const nextState = typeof newState === 'function' 
        ? (newState as (prev: T) => T)(prev.present)
        : newState;
      
      return {
        past: [...prev.past, prev.present],
        present: nextState,
        future: [], // Clear redo stack on new action
      };
    });
  }, []);
  
  const undo = useCallback(() => {
    setState(prev => {
      if (prev.past.length === 0) return prev;
      
      const previous = prev.past[prev.past.length - 1];
      const newPast = prev.past.slice(0, -1);
      
      return {
        past: newPast,
        present: previous,
        future: [prev.present, ...prev.future],
      };
    });
  }, []);
  
  const redo = useCallback(() => {
    setState(prev => {
      if (prev.future.length === 0) return prev;
      
      const next = prev.future[0];
      const newFuture = prev.future.slice(1);
      
      return {
        past: [...prev.past, prev.present],
        present: next,
        future: newFuture,
      };
    });
  }, []);
  
  const reset = useCallback((newState: T) => {
    setState({
      past: [],
      present: newState,
      future: [],
    });
  }, []);
  
  // Keyboard shortcuts
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'z') {
        if (e.shiftKey) {
          e.preventDefault();
          redo();
        } else {
          e.preventDefault();
          undo();
        }
      }
      if ((e.metaKey || e.ctrlKey) && e.key === 'y') {
        e.preventDefault();
        redo();
      }
    };
    
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [undo, redo]);
  
  return {
    state: state.present,
    set,
    undo,
    redo,
    reset,
    canUndo,
    canRedo,
    historyLength: state.past.length,
  };
}
```

```typescript
import puppeteer from 'puppeteer';
import { db } from '@/db';
import { persuasionBlocks, brandThemes } from '@/db/schema/document-builder';
import { eq } from 'drizzle-orm';
import { interpolateVariables, VariableContext } from './variable-interpolator';

export interface PdfExportOptions {
  proposalId: string;
  variableContext: VariableContext;
  includeTheme?: boolean;
}

export async function exportToPdf(options: PdfExportOptions): Promise<Buffer> {
  const { proposalId, variableContext, includeTheme = true } = options;
  
  // Get blocks
  const blocks = await db.query.persuasionBlocks.findMany({
    where: eq(persuasionBlocks.proposalId, proposalId),
    orderBy: (blocks, { asc }) => [asc(blocks.position)],
  });
  
  // Get theme if available
  let theme = null;
  if (includeTheme) {
    theme = await db.query.brandThemes.findFirst({
      where: eq(brandThemes.workspaceId, blocks[0]?.workspaceId),
    });
  }
  
  // Generate HTML
  const html = generateProposalHtml(blocks, theme, variableContext);
  
  // Render to PDF with Puppeteer
  const browser = await puppeteer.launch({ headless: true });
  const page = await browser.newPage();
  
  await page.setContent(html, { waitUntil: 'networkidle0' });
  
  const pdf = await page.pdf({
    format: 'A4',
    margin: { top: '1in', right: '1in', bottom: '1in', left: '1in' },
    printBackground: true,
  });
  
  await browser.close();
  
  return Buffer.from(pdf);
}

function generateProposalHtml(
  blocks: any[],
  theme: any,
  variableContext: VariableContext
): string {
  const styles = theme ? `
    :root {
      --primary-color: ${theme.primaryColor || '#1a1a1a'};
      --secondary-color: ${theme.secondaryColor || '#666666'};
      --heading-font: ${theme.headingFont || 'Georgia, serif'};
      --body-font: ${theme.bodyFont || 'Arial, sans-serif'};
    }
    h1, h2, h3 { font-family: var(--heading-font); color: var(--primary-color); }
    body { font-family: var(--body-font); color: var(--secondary-color); line-height: 1.6; }
  ` : '';
  
  const blockHtml = blocks.map(block => {
    const content = block.content?.content || [];
    const text = content.map((p: any) => 
      p.content?.map((t: any) => t.text).join('') || ''
    ).join('\n\n');
    
    const { result } = interpolateVariables(text, variableContext);
    
    return `<div class="block block-${block.type}">${result}</div>`;
  }).join('\n');
  
  return `<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <style>
    ${styles}
    .block { margin-bottom: 1.5em; }
  </style>
</head>
<body>
  ${blockHtml}
</body>
</html>`;
}
```
