```typescript
interface DocumentBuilderState {
  // Blocks on canvas
  blocks: PersuasionBlock[];
  selectedBlockId: string | null;
  
  // Framework (optional)
  frameworkId: string | null;
  
  // Actions
  addBlock: (type: PersuasionBlockType, position?: number) => void;
  removeBlock: (id: string) => void;
  moveBlock: (fromIndex: number, toIndex: number) => void;
  updateBlockContent: (id: string, content: unknown) => void;
  selectBlock: (id: string | null) => void;
  setFramework: (frameworkId: string | null) => void;
  
  // Computed
  getBlockAtPosition: (position: number) => PersuasionBlock | undefined;
}
```

- Use immer middleware for immutable updates (already in codebase pattern)
- moveBlock must update position field of all affected blocks
- addBlock generates nanoid for block ID
- Persist to localStorage for draft recovery (persist middleware)
