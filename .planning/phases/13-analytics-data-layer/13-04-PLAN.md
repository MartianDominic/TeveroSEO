```typescript
/**
 * Drizzle ORM schema for analytics snapshot tables.
 *
 * These tables store GSC and GA4 data synced by the analytics worker.
 * Data is written by open-seo-worker and read by the Next.js dashboard.
 */
import {
  pgTable,
  uuid,
  text,
  date,
  integer,
  real,
  timestamp,
  unique,
  index,
} from "drizzle-orm/pg-core";

/**
 * GSC daily aggregate snapshots.
 * One row per client per date.
 */
export const gscSnapshots = pgTable(
  "gsc_snapshots",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    clientId: uuid("client_id").notNull(),
    date: date("date").notNull(),
    siteUrl: text("site_url").notNull(),
    clicks: integer("clicks").notNull().default(0),
    impressions: integer("impressions").notNull().default(0),
    ctr: real("ctr").notNull().default(0),
    position: real("position").notNull().default(0),
    syncedAt: timestamp("synced_at", { withTimezone: true }).defaultNow().notNull(),
  },
  (table) => ({
    clientDateUnique: unique("uq_gsc_snapshots_client_date").on(
      table.clientId,
      table.date,
    ),
    clientDateIdx: index("ix_gsc_snapshots_client_date").on(
      table.clientId,
      table.date,
    ),
  }),
);

/**
 * GSC top queries per day.
 * Up to 50 queries per client per date.
 */
export const gscQuerySnapshots = pgTable(
  "gsc_query_snapshots",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    clientId: uuid("client_id").notNull(),
    date: date("date").notNull(),
    query: text("query").notNull(),
    clicks: integer("clicks").default(0),
    impressions: integer("impressions").default(0),
    ctr: real("ctr").default(0),
    position: real("position").default(0),
  },
  (table) => ({
    clientDateQueryUnique: unique("uq_gsc_query_snapshots_client_date_query").on(
      table.clientId,
      table.date,
      table.query,
    ),
    clientDateIdx: index("ix_gsc_query_snapshots_client_date").on(
      table.clientId,
      table.date,
    ),
  }),
);

/**
 * GA4 daily aggregate snapshots.
 * One row per client per date.
 */
export const ga4Snapshots = pgTable(
  "ga4_snapshots",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    clientId: uuid("client_id").notNull(),
    date: date("date").notNull(),
    propertyId: text("property_id").notNull(),
    sessions: integer("sessions").default(0),
    users: integer("users").default(0),
    newUsers: integer("new_users").default(0),
    bounceRate: real("bounce_rate").default(0),
    avgSessionDuration: real("avg_session_duration").default(0),
    conversions: integer("conversions").default(0),
    revenue: real("revenue").default(0),
    syncedAt: timestamp("synced_at", { withTimezone: true }).defaultNow().notNull(),
  },
  (table) => ({
    clientDateUnique: unique("uq_ga4_snapshots_client_date").on(
      table.clientId,
      table.date,
    ),
    clientDateIdx: index("ix_ga4_snapshots_client_date").on(
      table.clientId,
      table.date,
    ),
  }),
);

// Type exports for use in queries
export type GSCSnapshot = typeof gscSnapshots.$inferSelect;
export type GSCSnapshotInsert = typeof gscSnapshots.$inferInsert;

export type GSCQuerySnapshot = typeof gscQuerySnapshots.$inferSelect;
export type GSCQuerySnapshotInsert = typeof gscQuerySnapshots.$inferInsert;

export type GA4Snapshot = typeof ga4Snapshots.$inferSelect;
export type GA4SnapshotInsert = typeof ga4Snapshots.$inferInsert;
```

Note: No FK to clients table since that lives in AI-Writer's PostgreSQL. The client_id is a UUID reference validated at the application layer.

  <action>
Add an internal endpoint to trigger backfill that can be called from the OAuth callback.

This avoids coupling the AI-Writer OAuth code directly to the open-seo-worker.
Instead, the callback calls an internal endpoint which then notifies the worker.

Add to internal.py:

```python
@router.post("/analytics/backfill/{client_id}")
def trigger_analytics_backfill(
    client_id: str,
    db: Session = Depends(get_shared_db),
    _: None = Depends(verify_internal_api_key),
):
    """
    Trigger a 90-day analytics backfill for a client.
    
    Called after OAuth callback successfully stores Google credentials.
    This endpoint notifies the open-seo-worker to queue a backfill job.
    
    SECURITY: Protected by X-Internal-Api-Key header.
    """
    import httpx
    
    # Notify open-seo-worker to queue backfill job
    worker_url = os.getenv("OPEN_SEO_WORKER_URL", "http://localhost:3001")
    
    try:
        response = httpx.post(
            f"{worker_url}/internal/analytics/backfill",
            json={"clientId": client_id},
            headers={"X-Internal-Api-Key": INTERNAL_API_KEY},
            timeout=5.0,
        )
        response.raise_for_status()
        logger.info(f"Backfill triggered for client {client_id}")
        return {"status": "triggered", "client_id": client_id}
    except Exception as e:
        logger.error(f"Failed to trigger backfill for client {client_id}: {e}")
        # Don't fail the OAuth flow if backfill trigger fails
        # The nightly sync will eventually catch up
        return {"status": "deferred", "client_id": client_id, "reason": str(e)}
```

Also add the httpx import at the top of the file if not present.

Alternative approach (simpler): Instead of HTTP callback, have the OAuth callback directly queue to Redis. But this couples AI-Writer to BullMQ which we want to avoid.

Simpler alternative for MVP: Just log that backfill should happen, and rely on the nightly sync with mode="backfill" detection based on whether snapshots exist.

```typescript
/**
 * Internal API routes for service-to-service communication.
 *
 * These endpoints are called by AI-Writer backend for cross-service operations.
 * Protected by X-Internal-Api-Key header.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { queueBackfillJob } from "@/server/queues/analyticsQueue";

const INTERNAL_API_KEY = process.env.INTERNAL_API_KEY;

function verifyInternalApiKey(request: Request): boolean {
  const apiKey = request.headers.get("X-Internal-Api-Key");
  if (!INTERNAL_API_KEY) {
    console.error("[internal] INTERNAL_API_KEY not configured");
    return false;
  }
  return apiKey === INTERNAL_API_KEY;
}

export const Route = createAPIFileRoute("/internal/analytics/backfill")({
  POST: async ({ request }) => {
    if (!verifyInternalApiKey(request)) {
      return new Response(JSON.stringify({ error: "Unauthorized" }), {
        status: 401,
        headers: { "Content-Type": "application/json" },
      });
    }

    try {
      const body = await request.json();
      const { clientId } = body;

      if (!clientId) {
        return new Response(
          JSON.stringify({ error: "clientId required" }),
          { status: 400, headers: { "Content-Type": "application/json" } },
        );
      }

      await queueBackfillJob(clientId);

      return new Response(
        JSON.stringify({ status: "queued", clientId }),
        { status: 202, headers: { "Content-Type": "application/json" } },
      );
    } catch (err) {
      console.error("[internal] Failed to queue backfill:", err);
      return new Response(
        JSON.stringify({ error: "Failed to queue backfill" }),
        { status: 500, headers: { "Content-Type": "application/json" } },
      );
    }
  },
});
```

Note: The exact API route registration depends on TanStack Start's file-based routing. May need to create this at:
