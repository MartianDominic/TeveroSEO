    ```bash
    # Start a throwaway PG for runtime config
    docker run -d --rm --name openseo-pg-smoke \
      -e POSTGRES_PASSWORD=postgres \
      -e POSTGRES_DB=open_seo \
      -p 5434:5432 \

      if curl -fsS -o /dev/null "http://localhost:3101/" 2>/dev/null; then break; fi
      sleep 1
    done

    # Verify the port is listening
    curl -sS -o /tmp/server-body.txt -w "%{http_code}" "http://localhost:3101/" > /tmp/server-status.txt
    STATUS=$(cat /tmp/server-status.txt)
    echo "GET / → $STATUS"
    # BUILD-03: any non-5xx is fine (200 OK, 302 redirect to login, 404 if no index route — but NOT 500)
    [ "$STATUS" -lt 500 ] || { echo "FAIL: server returned $STATUS"; kill $SERVER_PID; docker stop openseo-pg-smoke; exit 1; }

    # Verify TCP listener is actually bound
    ss -tlnp 2>/dev/null | grep -q ':3101' || netstat -tlnp 2>/dev/null | grep -q ':3101' || { echo "FAIL: nothing listening on 3101"; kill $SERVER_PID; docker stop openseo-pg-smoke; exit 1; }

    # Cleanup
    kill $SERVER_PID 2>/dev/null || true
    wait $SERVER_PID 2>/dev/null || true
    docker stop openseo-pg-smoke
    ```
