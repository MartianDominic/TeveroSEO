## Tasks

### 1. RevolutCheckoutWidget Component (2h)
- [ ] Install `@revolut/checkout` package
  ```bash
  pnpm add @revolut/checkout
  ```

- [ ] Create `apps/web/src/components/payments/RevolutCheckoutWidget.tsx`
  ```tsx
  'use client';
  
  interface Props {
    publicKey: string;
    orderToken: string;
    amount: number;
    currency: string;
    onSuccess: () => void;
    onError: (error: Error) => void;
    onCancel: () => void;
  }
  
  export function RevolutCheckoutWidget(props: Props) {
    useEffect(() => {
      const initCheckout = async () => {
        const RevolutCheckout = (await import('@revolut/checkout')).default;
        
        const instance = await RevolutCheckout({
          publicToken: props.publicKey,
          locale: 'en',
        });
        
        instance.payWithPopup({
          onSuccess: props.onSuccess,
          onError: (e) => props.onError(new Error(e.message)),
          onCancel: props.onCancel,
        });
      };
      
      initCheckout();
    }, []);
    
    return <div id="revolut-checkout-container" />;
  }
  ```
