```typescript
interface ScoreBreakdown {
  // Component scores (0-1)
  relevance: number;
  funnelConfidence: number;
  funnelStage: 'BOFU' | 'MOFU' | 'TOFU';
  geoScore: number;
  geoMatch: string; // e.g., "Šiauliai"
  volumeNormalized: number;
  volume: number; // raw volume
  
  // Composite scores
  baseScore: number;
  priorityMultiplier: number;
  priorityCategory?: string; // e.g., "detailing"
  quickWinBonus: number;
  position?: number; // current position for quick win
  finalScore: number;
}

interface ScoreExplanationProps {
  breakdown: ScoreBreakdown;
  locale?: 'en' | 'lt';
  children: React.ReactNode; // Trigger element
}

export function ScoreExplanation({ breakdown, locale = 'en', children }: ScoreExplanationProps) {
  // Use shadcn Popover component
  return (
    <Popover>
      <PopoverTrigger asChild>{children}</PopoverTrigger>
      <PopoverContent className="w-80">
        <ScoreBreakdownTable breakdown={breakdown} locale={locale} />
      </PopoverContent>
    </Popover>
  );
}
```
