```typescript
async softDelete(clientId: string, userId: string): Promise<void> {
  await withTransaction(async (tx) => {
    const now = new Date();
    
    // 1. Soft delete client
    await tx.update(clients)
      .set({ deletedAt: now, deletedBy: userId })
      .where(eq(clients.id, clientId));
    
    // 2. Cascade to projects
    await tx.update(projects)
      .set({ deletedAt: now })
      .where(eq(projects.clientId, clientId));
    
    // 3. Cascade to audits
    await tx.update(audits)
      .set({ deletedAt: now })
      .where(eq(audits.clientId, clientId));
    
    // 4. Cascade to briefs
    await tx.update(briefs)
      .set({ deletedAt: now })
      .where(eq(briefs.clientId, clientId));
    
    // 5. Invalidate access cache
    await this.invalidateAccessCache(clientId);
  });
}
```
</action>

- [ ] Access cache invalidated after soft delete
