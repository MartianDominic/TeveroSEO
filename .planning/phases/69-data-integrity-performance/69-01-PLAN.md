import { db } from '@/db';
import { AppError } from './errors';

type TransactionCallback<T> = (tx: typeof db) => Promise<T>;

export async function withTransaction<T>(
  callback: TransactionCallback<T>,
  options?: { isolationLevel?: 'read committed' | 'serializable' }
): Promise<T> {
  try {
    return await db.transaction(async (tx) => {
      return callback(tx);
    });
  } catch (error) {
    if (error instanceof AppError) {
      throw error;
    }
    console.error('[withTransaction] Transaction failed:', error);
    throw new AppError('Transaction failed', 500, 'TRANSACTION_FAILED');
  }
}

export interface PostCommitJob {
  queue: string;
  jobName: string;
  data: Record<string, unknown>;
}

export class TransactionContext {
  private postCommitJobs: PostCommitJob[] = [];
  
  addPostCommitJob(job: PostCommitJob): void {
    this.postCommitJobs.push(job);
  }
  
  getPostCommitJobs(): PostCommitJob[] {
    return this.postCommitJobs;
  }
}
```

import { AppError } from './errors';

interface SagaStep<T> {
  name: string;
  execute: () => Promise<T>;
  compensate: (result: T) => Promise<void>;
}

interface SagaResult<T> {
  success: boolean;
  results: T[];
  failedStep?: string;
  error?: Error;
}

export async function executeSaga<T>(
  steps: SagaStep<T>[],
  onCompensationFailure?: (step: string, error: Error) => Promise<void>
): Promise<SagaResult<T>> {
  const results: T[] = [];
  const completedSteps: { step: SagaStep<T>; result: T }[] = [];
  
  for (const step of steps) {
    try {
      const result = await step.execute();
      results.push(result);
      completedSteps.push({ step, result });
    } catch (error) {
      // Run compensations in reverse
      for (const completed of [...completedSteps].reverse()) {
        try {
          await completed.step.compensate(completed.result);
        } catch (compensateError) {
          console.error(`[saga] Compensation failed for ${completed.step.name}:`, compensateError);
          if (onCompensationFailure) {
            await onCompensationFailure(completed.step.name, compensateError as Error);
          }
        }
      }
      
      return {
        success: false,
        results,
        failedStep: step.name,
        error: error as Error,
      };
    }
  }
  
  return { success: true, results };
}
```
