async bulkQueueAnalysis(prospectIds: string[], workspaceId: string): Promise<void> {
  // Batch fetch all prospects in single query
  const prospects = await db.query.prospects.findMany({
    where: and(
      inArray(prospects.id, prospectIds),
      eq(prospects.workspaceId, workspaceId)
    ),
  });
  
  if (prospects.length === 0) return;
  
  // Prepare all jobs
  const jobs = prospects.map(prospect => ({
    name: 'analyze-prospect',
    data: {
      prospectId: prospect.id,
      domain: prospect.domain,
      workspaceId,
    },
    opts: {
      jobId: `analyze-${prospect.id}`,
      removeOnComplete: 100,
      removeOnFail: 50,
    },
  }));
  
  // Batch insert jobs
  await analysisQueue.addBulk(jobs);
  
  // Batch update prospect status
  await db.update(prospects)
    .set({ status: 'analyzing', analyzingStartedAt: new Date() })
    .where(inArray(prospects.id, prospectIds));
}
```

import { sql, SQL, and, gt, lt, eq } from 'drizzle-orm';

interface CursorConfig {
  primaryKey: string;
  sortColumn?: string;
  sortDirection?: 'asc' | 'desc';
}

interface CursorPaginationResult<T> {
  items: T[];
  nextCursor: string | null;
  hasMore: boolean;
}

export function decodeCursor(cursor: string): { id: string; value?: string } {
  const decoded = Buffer.from(cursor, 'base64url').toString('utf-8');
  const [id, value] = decoded.split(':');
  return { id, value };
}

export function encodeCursor(id: string, value?: string): string {
  const data = value ? `${id}:${value}` : id;
  return Buffer.from(data).toString('base64url');
}

export function buildCursorCondition(
  cursor: string | undefined,
  config: CursorConfig,
  table: any
): SQL | undefined {
  if (!cursor) return undefined;
  
  const { id, value } = decodeCursor(cursor);
  const direction = config.sortDirection ?? 'desc';
  
  if (config.sortColumn && value) {
    // Compound cursor: (sortColumn, id)
    const compare = direction === 'desc' ? lt : gt;
    return sql`(${table[config.sortColumn]}, ${table[config.primaryKey]}) ${
      direction === 'desc' ? sql`<` : sql`>`
    } (${value}, ${id})`;
  }
  
  // Simple cursor: id only
  const compare = direction === 'desc' ? lt : gt;
  return compare(table[config.primaryKey], id);
}
```
