Create email.ts:

```typescript
import { Resend } from "resend";
import { readFile, stat } from "node:fs/promises";
import { createLogger } from "./logger";

const MAX_ATTACHMENT_SIZE = 10 * 1024 * 1024; // 10MB
const log = createLogger({ module: "email" });

export interface EmailConfig {
  from: string;
  replyTo?: string;
}

export interface ReportEmailParams {
  to: string[];
  subject: string;
  html: string;
  pdfPath: string;
  downloadUrl?: string; // Fallback URL for large files
}

export async function sendReportEmail(params: ReportEmailParams): Promise<void> {
  const apiKey = process.env.RESEND_API_KEY;
  if (!apiKey) throw new Error("RESEND_API_KEY not configured");
  
  const resend = new Resend(apiKey);
  const fileSize = (await stat(params.pdfPath)).size;
  
  const emailPayload: Parameters<typeof resend.emails.send>[0] = {
    from: process.env.EMAIL_FROM ?? "reports@tevero.io",
    to: params.to,
    subject: params.subject,
    html: params.html,
  };
  
  if (fileSize < MAX_ATTACHMENT_SIZE) {
    const content = await readFile(params.pdfPath);
    emailPayload.attachments = [{
      filename: params.pdfPath.split("/").pop() ?? "report.pdf",
      content: content.toString("base64"),
    }];
  }
  // Note: If file too large, the HTML template includes download link
  
  const { error } = await resend.emails.send(emailPayload);
  if (error) {
    log.error("Email send failed", new Error(error.message), { to: params.to });
    throw new Error(`Email delivery failed: ${error.message}`);
  }
  
  log.info("Report email sent", { to: params.to, attached: fileSize < MAX_ATTACHMENT_SIZE });
}
```

Create email-templates.ts with HTML email template:

```typescript
export interface ReportEmailData {
  clientName: string;
  reportType: string;
  dateRange: { start: string; end: string };
  downloadUrl?: string; // Present when PDF too large for attachment
  locale: string;
}

export function reportDeliveryTemplate(data: ReportEmailData): {
  subject: string;
  html: string;
} {
  const formatDate = (dateStr: string) => {
    return new Intl.DateTimeFormat(data.locale, {
      year: "numeric",
      month: "long",
      day: "numeric",
    }).format(new Date(dateStr));
  };
  
  const reportTitle = data.reportType === "monthly-seo" 
    ? "Monthly SEO Report" 
    : "Weekly Summary Report";
  
  const subject = `${reportTitle} - ${data.clientName} (${formatDate(data.dateRange.start)} - ${formatDate(data.dateRange.end)})`;
  
  const downloadSection = data.downloadUrl 
    ? `<p style="margin-top: 16px;">
        <a href="${data.downloadUrl}" style="background-color: #3b82f6; color: white; padding: 12px 24px; text-decoration: none; border-radius: 6px; display: inline-block;">
          Download Report (PDF)
        </a>
       </p>
       <p style="color: #6b7280; font-size: 12px; margin-top: 8px;">
         The report file is too large to attach. Please use the download link above.
       </p>`
    : `<p style="color: #6b7280; font-size: 14px;">
        The report PDF is attached to this email.
       </p>`;
  
  const html = `
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
</head>
<body style="font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif; line-height: 1.6; color: #1f2937; max-width: 600px; margin: 0 auto; padding: 20px;">
  <div style="border-bottom: 2px solid #e5e7eb; padding-bottom: 16px; margin-bottom: 24px;">
    <h1 style="margin: 0; font-size: 24px; color: #111827;">${reportTitle}</h1>
    <p style="margin: 8px 0 0; color: #6b7280;">${data.clientName}</p>
  </div>
  
  <p>Your scheduled report for the period <strong>${formatDate(data.dateRange.start)}</strong> to <strong>${formatDate(data.dateRange.end)}</strong> is ready.</p>
  
  ${downloadSection}
  
  <div style="margin-top: 32px; padding-top: 16px; border-top: 1px solid #e5e7eb; color: #9ca3af; font-size: 12px;">
    <p>This is an automated email from Tevero. Please do not reply.</p>
  </div>
</body>
</html>
  `.trim();
  
  return { subject, html };
}
```
