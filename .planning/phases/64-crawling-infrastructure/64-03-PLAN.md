```typescript
import { Queue, type JobsOptions } from "bullmq";
import { getSharedBullMQConnection } from "@/server/lib/redis";

// Queue names
export const FAST_API_QUEUE = "crawl-fast-api" as const;
export const HEAVY_CRAWL_QUEUE = "crawl-heavy" as const;

// Job types per CONTEXT.md
export enum CrawlJobType {
  A_FULL_AUDIT = "A", // Heavy: full site audit with crawling
  B_API_ONLY = "B",   // Fast: pure API calls
  C_DELTA = "C",      // Fast: delta crawl (few pages)
  D_SERP = "D",       // Fast: SERP analysis
  E_CONTENT = "E",    // Fast: content analysis
  F_KEYWORD = "F",    // Fast: keyword enrichment
}

export type CrawlLane = "fast-api" | "heavy-crawl";

export interface CrawlJobData {
  jobType: CrawlJobType;
  tenantId: string;
  projectId: string;
  urls: string[];           // URLs to crawl
  lastCrawlDate?: string;   // ISO date for delta comparison
  sitemapUrl?: string;      // For full audit
  options?: {
    enableDelta?: boolean;
    enableSingleflight?: boolean;
    maxPages?: number;
  };
}

// Route job to appropriate lane
export function routeToLane(jobType: CrawlJobType): CrawlLane {
  return jobType === CrawlJobType.A_FULL_AUDIT ? "heavy-crawl" : "fast-api";
}

// Fast lane: high concurrency, short timeout
const FAST_JOB_OPTIONS: JobsOptions = {
  attempts: 2,
  backoff: { type: "fixed", delay: 5_000 },
  removeOnComplete: { count: 500 },
  removeOnFail: { count: 200 },
};

// Heavy lane: lower concurrency, longer timeout
const HEAVY_JOB_OPTIONS: JobsOptions = {
  attempts: 3,
  backoff: { type: "exponential", delay: 30_000 },
  removeOnComplete: { count: 100 },
  removeOnFail: { count: 500 },
};

export const fastApiQueue = new Queue<CrawlJobData>(FAST_API_QUEUE, {
  connection: getSharedBullMQConnection("queue:crawl-fast"),
  defaultJobOptions: FAST_JOB_OPTIONS,
});

export const heavyCrawlQueue = new Queue<CrawlJobData>(HEAVY_CRAWL_QUEUE, {
  connection: getSharedBullMQConnection("queue:crawl-heavy"),
  defaultJobOptions: HEAVY_JOB_OPTIONS,
});

// Convenience: route and add job automatically
export async function addCrawlJob(
  jobId: string,
  data: CrawlJobData
): Promise<void> {
  const lane = routeToLane(data.jobType);
  const queue = lane === "fast-api" ? fastApiQueue : heavyCrawlQueue;
  await queue.add(jobId, data, { jobId }); // Dedupe by jobId
}
```

```typescript
/**
 * Sandboxed BullMQ processor for crawl jobs.
 *
 * Integrates:
 * - Singleflight: Deduplicate concurrent URL requests
 * - Delta Cascade: Skip unchanged content (L0->L1->L2->L3)
 *
 * Runs in child process (BullMQ sandboxed worker).
 */

import type { Job } from "bullmq";
import { CrawlJobType, type CrawlJobData } from "@/server/queues/crawlQueue";
import { crawlSingleflight, singleflight } from "@/server/lib/crawler/singleflight";
import { createDeltaCascade, DeltaLevel } from "@/server/lib/crawler/delta-cascade";
import { HybridCrawler } from "@/server/lib/crawler/hybrid-crawler";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "crawl-processor" });

interface ProcessorResult {
  processed: number;
  skipped: number;
  failed: number;
  metrics: {
    singleflightCoalesced: number;
    deltaSkipped: Record<DeltaLevel, number>;
  };
}

export default async function processor(
  job: Job<CrawlJobData>
): Promise<ProcessorResult> {
  const { jobType, tenantId, urls, lastCrawlDate, options } = job.data;
  const jobLog = createLogger({ module: "crawl-processor", jobId: job.id });

  const deltaCascade = createDeltaCascade({
    enableL0: options?.enableDelta !== false,
    enableL1: options?.enableDelta !== false,
    enableL2: options?.enableDelta !== false,
  });
  const crawler = new HybridCrawler();

  let processed = 0;
  let skipped = 0;
  let failed = 0;

  for (const url of urls) {
    try {
      // Wrap in singleflight to dedupe concurrent requests
      const result = await singleflight(`crawl:${url}`, async () => {
        // Run delta cascade
        const deltaResult = await deltaCascade.evaluate(
          tenantId,
          url,
          null, // TODO: Get from sitemap if available
          lastCrawlDate ? new Date(lastCrawlDate) : new Date(0)
        );

        if (deltaResult.action === "SKIP") {
          return { skipped: true, deltaLevel: deltaResult.skipLevel };
        }

        // Fetch page
        const crawlResult = await crawler.fetchPage(url, {
          etag: deltaResult.etag,
          lastModified: deltaResult.lastModified,
        });

        return { skipped: false, crawlResult };
      });

      if (result.skipped) {
        skipped++;
        jobLog.debug("URL skipped by delta cascade", { url, level: result.deltaLevel });
      } else {
        processed++;
        jobLog.debug("URL processed", { url, statusCode: result.crawlResult?.statusCode });
      }
    } catch (error) {
      failed++;
      jobLog.warn("URL failed", { url, error: error instanceof Error ? error.message : String(error) });
    }

    // Update job progress
    await job.updateProgress(Math.round(((processed + skipped + failed) / urls.length) * 100));
  }

  const singleflightMetrics = crawlSingleflight.getMetrics();
  const deltaMetrics = deltaCascade.getMetrics();

  jobLog.info("Job complete", {
    processed,
    skipped,
    failed,
    singleflightCoalesced: singleflightMetrics.coalesced,
  });

  return {
    processed,
    skipped,
    failed,
    metrics: {
      singleflightCoalesced: singleflightMetrics.coalesced,
      deltaSkipped: {
        [DeltaLevel.L0_SITEMAP]: deltaMetrics.skippedL0,
        [DeltaLevel.L1_CONDITIONAL]: deltaMetrics.skippedL1,
        [DeltaLevel.L2_HASH]: deltaMetrics.skippedL2,
        [DeltaLevel.L3_FULL]: 0, // L3 never skips
      },
    },
  };
}
```

```typescript
/**
 * BullMQ Workers for crawl queues.
 *
 * Two workers, one per lane:
 * - Fast-API: High concurrency (10), short lock duration (60s)
 * - Heavy-Crawl: Low concurrency (2), long lock duration (15min)
 */

import { Worker, type Job } from "bullmq";
import { fileURLToPath } from "node:url";
import { getSharedBullMQConnection } from "@/server/lib/redis";
import { createLogger } from "@/server/lib/logger";
import {
  FAST_API_QUEUE,
  HEAVY_CRAWL_QUEUE,
  type CrawlJobData,
} from "@/server/queues/crawlQueue";
import { getDLQQueue } from "@/server/queues/dlq";

const log = createLogger({ module: "crawl-worker" });

// Processor path for sandboxed execution
const PROCESSOR_PATH = fileURLToPath(
  new URL("./crawl-processor.js", import.meta.url)
);

// Worker settings per lane
const FAST_SETTINGS = {
  concurrency: 10,       // High throughput
  lockDuration: 60_000,  // 1 min (fast jobs)
  maxStalledCount: 2,
};

const HEAVY_SETTINGS = {
  concurrency: 2,          // Limit heavy jobs
  lockDuration: 900_000,   // 15 min (long crawls)
  maxStalledCount: 3,
};

let fastWorker: Worker<CrawlJobData> | null = null;
let heavyWorker: Worker<CrawlJobData> | null = null;

function createWorker(
  queueName: string,
  label: string,
  settings: { concurrency: number; lockDuration: number; maxStalledCount: number }
): Worker<CrawlJobData> {
  const worker = new Worker<CrawlJobData>(
    queueName,
    PROCESSOR_PATH,
    {
      connection: getSharedBullMQConnection(`worker:${label}`),
      ...settings,
    }
  );

  worker.on("ready", () => log.info("Worker ready", { queue: queueName }));
  worker.on("error", (err) => log.error("Worker error", err, { queue: queueName }));
  worker.on("completed", (job) => log.info("Job completed", { jobId: job.id, queue: queueName }));
  worker.on("failed", async (job, err) => {
    if (!job) return;
    log.error("Job failed", err, { jobId: job.id, queue: queueName });
    
    const maxAttempts = job.opts.attempts ?? 1;
    if (job.attemptsMade >= maxAttempts) {
      // Send to DLQ
      await getDLQQueue().add(`dlq-${job.id}`, {
        originalQueue: queueName,
        originalJobId: job.id,
        data: job.data,
        error: err.message,
        failedAt: Date.now(),
      });
    }
  });
  worker.on("stalled", (jobId) => log.warn("Job stalled", { jobId, queue: queueName }));

  return worker;
}

export function startCrawlWorkers(): void {
  if (!fastWorker) {
    fastWorker = createWorker(FAST_API_QUEUE, "crawl-fast", FAST_SETTINGS);
  }
  if (!heavyWorker) {
    heavyWorker = createWorker(HEAVY_CRAWL_QUEUE, "crawl-heavy", HEAVY_SETTINGS);
  }
  log.info("Crawl workers started", { lanes: ["fast-api", "heavy-crawl"] });
}

export async function stopCrawlWorkers(): Promise<void> {
  const timeout = 30_000;
  const workers = [fastWorker, heavyWorker].filter(Boolean);
  
  await Promise.all(
    workers.map(async (w) => {
      const closed = w!.close().then(() => "closed" as const);
      const timeoutPromise = new Promise<"timeout">((r) => setTimeout(() => r("timeout"), timeout));
      const result = await Promise.race([closed, timeoutPromise]);
      if (result === "timeout") {
        await w!.close(true); // Force
      }
    })
  );
  
  fastWorker = null;
  heavyWorker = null;
  log.info("Crawl workers stopped");
}
```
