import { crawlSingleflight, type SingleflightMetrics } from "./singleflight";
import { type DeltaCascadeMetrics, DeltaLevel } from "./delta-cascade";
import { fastApiQueue, heavyCrawlQueue } from "@/server/queues/crawlQueue";
import { redis } from "@/server/lib/redis";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "crawl-metrics" });

// Cost assumptions for savings calculation
const COST_PER_CRAWL = 0.001; // $0.001 per page crawl (estimate)

export interface QueueMetrics {
  name: string;
  waiting: number;
  active: number;
  completed: number;
  failed: number;
  delayed: number;
  avgProcessingTimeMs: number;
}

export interface CrawlMetrics {
  timestamp: string;
  
  // Singleflight metrics
  singleflight: {
    totalRequests: number;
    coalesced: number;
    coalesceRate: number;           // % of requests that were coalesced
    cacheHits: number;
    cacheHitRate: number;           // % cache hits
    estimatedSavings: number;       // $ saved from deduplication
  };
  
  // Delta cascade metrics
  delta: {
    totalUrls: number;
    skippedL0: number;              // Sitemap lastmod
    skippedL1: number;              // Conditional GET
    skippedL2: number;              // Hash comparison
    proceededL3: number;            // Full reprocess
    overallSkipRate: number;        // % skipped by any level
    estimatedSavings: number;       // $ saved from delta skips
  };
  
  // Queue metrics
  queues: {
    fastApi: QueueMetrics;
    heavyCrawl: QueueMetrics;
  };
  
  // Combined savings
  totalEstimatedSavings: number;
}

export class CrawlMetricsCollector {
  private deltaMetrics: DeltaCascadeMetrics = {
    total: 0,
    skippedL0: 0,
    skippedL1: 0,
    skippedL2: 0,
    proceededL3: 0,
  };
  
  /**
   * Record delta cascade result.
   * Called by crawl-processor after each URL evaluation.
   */
  recordDeltaResult(skipLevel: DeltaLevel | null): void {
    this.deltaMetrics.total++;
    if (skipLevel === DeltaLevel.L0_SITEMAP) this.deltaMetrics.skippedL0++;
    else if (skipLevel === DeltaLevel.L1_CONDITIONAL) this.deltaMetrics.skippedL1++;
    else if (skipLevel === DeltaLevel.L2_HASH) this.deltaMetrics.skippedL2++;
    else this.deltaMetrics.proceededL3++;
  }
  
  /**
   * Get aggregated metrics from all sources.
   */
  async getMetrics(): Promise<CrawlMetrics> {
    // Get singleflight metrics
    const sf = crawlSingleflight.getMetrics();
    const sfCoalesceRate = sf.requests > 0 ? sf.coalesced / sf.requests : 0;
    const sfCacheHitRate = sf.requests > 0 ? sf.cacheHits / sf.requests : 0;
    const sfSavings = sf.coalesced * COST_PER_CRAWL;
    
    // Calculate delta metrics
    const totalSkipped = this.deltaMetrics.skippedL0 + this.deltaMetrics.skippedL1 + this.deltaMetrics.skippedL2;
    const overallSkipRate = this.deltaMetrics.total > 0 ? totalSkipped / this.deltaMetrics.total : 0;
    const deltaSavings = totalSkipped * COST_PER_CRAWL;
    
    // Get queue metrics
    const [fastStats, heavyStats] = await Promise.all([
      this.getQueueMetrics(fastApiQueue, "crawl-fast-api"),
      this.getQueueMetrics(heavyCrawlQueue, "crawl-heavy"),
    ]);
    
    return {
      timestamp: new Date().toISOString(),
      singleflight: {
        totalRequests: sf.requests,
        coalesced: sf.coalesced,
        coalesceRate: sfCoalesceRate,
        cacheHits: sf.cacheHits,
        cacheHitRate: sfCacheHitRate,
        estimatedSavings: sfSavings,
      },
      delta: {
        totalUrls: this.deltaMetrics.total,
        skippedL0: this.deltaMetrics.skippedL0,
        skippedL1: this.deltaMetrics.skippedL1,
        skippedL2: this.deltaMetrics.skippedL2,
        proceededL3: this.deltaMetrics.proceededL3,
        overallSkipRate,
        estimatedSavings: deltaSavings,
      },
      queues: {
        fastApi: fastStats,
        heavyCrawl: heavyStats,
      },
      totalEstimatedSavings: sfSavings + deltaSavings,
    };
  }
  
  private async getQueueMetrics(queue: typeof fastApiQueue, name: string): Promise<QueueMetrics> {
    try {
      const [waiting, active, completed, failed, delayed] = await Promise.all([
        queue.getWaitingCount(),
        queue.getActiveCount(),
        queue.getCompletedCount(),
        queue.getFailedCount(),
        queue.getDelayedCount(),
      ]);
      
      // Get avg processing time from recent completed jobs
      const recentJobs = await queue.getCompleted(0, 100);
      const processingTimes = recentJobs
        .filter(j => j.finishedOn && j.processedOn)
        .map(j => j.finishedOn! - j.processedOn!);
      const avgProcessingTimeMs = processingTimes.length > 0
        ? processingTimes.reduce((a, b) => a + b, 0) / processingTimes.length
        : 0;
      
      return { name, waiting, active, completed, failed, delayed, avgProcessingTimeMs };
    } catch (error) {
      log.warn("Failed to get queue metrics", { queue: name, error });
      return { name, waiting: 0, active: 0, completed: 0, failed: 0, delayed: 0, avgProcessingTimeMs: 0 };
    }
  }
  
  /**
   * Reset all metrics counters.
   */
  reset(): void {
    crawlSingleflight.resetMetrics();
    this.deltaMetrics = {
      total: 0,
      skippedL0: 0,
      skippedL1: 0,
      skippedL2: 0,
      proceededL3: 0,
    };
  }
}

// Singleton instance
export const crawlMetrics = new CrawlMetricsCollector();

/**
 * Crawl Metrics API
 *
 * Returns real-time metrics for crawl infrastructure:
 * - Singleflight coalesce rate and savings
 * - Delta cascade skip rates per level
 * - Queue lane statistics
 *
 * Restricted to admin users.
 */

import { createAPIFileRoute } from "@tanstack/start/api";
import { crawlMetrics, type CrawlMetrics } from "@/server/lib/crawler/metrics";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "api:crawl-metrics" });

interface MetricsResponse {
  success: boolean;
  data?: CrawlMetrics;
  error?: string;
}

export const Route = createAPIFileRoute("/api/admin/crawl-metrics")({
  GET: async ({ request }): Promise<Response> => {
    try {
      // TODO: Add admin auth check
      // const session = await getSession(request);
      // if (!session?.user?.isAdmin) {
      //   return new Response(JSON.stringify({ success: false, error: "Unauthorized" }), {
      //     status: 401,
      //     headers: { "Content-Type": "application/json" },
      //   });
      // }
      
      const metrics = await crawlMetrics.getMetrics();
      
      const response: MetricsResponse = {
        success: true,
        data: metrics,
      };
      
      return new Response(JSON.stringify(response), {
        status: 200,
        headers: {
          "Content-Type": "application/json",
          "Cache-Control": "no-cache, no-store, must-revalidate",
        },
      });
    } catch (error) {
      log.error("Failed to get crawl metrics", error instanceof Error ? error : new Error(String(error)));
      
      const response: MetricsResponse = {
        success: false,
        error: "Failed to retrieve metrics",
      };
      
      return new Response(JSON.stringify(response), {
        status: 500,
        headers: { "Content-Type": "application/json" },
      });
    }
  },
  
  POST: async ({ request }): Promise<Response> => {
    try {
      const body = await request.json();
      
      if (body.action === "reset") {
        crawlMetrics.reset();
        log.info("Crawl metrics reset");
        
        return new Response(JSON.stringify({ success: true, message: "Metrics reset" }), {
          status: 200,
          headers: { "Content-Type": "application/json" },
        });
      }
      
      return new Response(JSON.stringify({ success: false, error: "Unknown action" }), {
        status: 400,
        headers: { "Content-Type": "application/json" },
      });
    } catch (error) {
      return new Response(JSON.stringify({ success: false, error: "Invalid request" }), {
        status: 400,
        headers: { "Content-Type": "application/json" },
      });
    }
  },
});

/**
 * E2E Tests for Crawl Infrastructure
 *
 * Validates:
 * 1. Singleflight achieves 98%+ duplicate coalescing
 * 2. Delta cascade achieves 80%+ cache hit on stable content
 * 3. Fast lane SLA <1m
 * 4. Heavy lane SLA <15m
 * 5. Mixed workloads don't block
 */

import { describe, it, expect, beforeAll, afterAll } from "vitest";
import { CrawlSingleflight } from "@/server/lib/crawler/singleflight";
import { createDeltaCascade, DeltaLevel } from "@/server/lib/crawler/delta-cascade";
import { fastApiQueue, heavyCrawlQueue, addCrawlJob, CrawlJobType } from "@/server/queues/crawlQueue";
import { startCrawlWorkers, stopCrawlWorkers } from "@/server/workers/crawl-worker";
import { redis } from "@/server/lib/redis";
import Redis from "ioredis";

// Test Redis instance (use separate DB for isolation)
const testRedis = new Redis(process.env.REDIS_URL ?? "redis://localhost:6379", {
  db: 15, // Use DB 15 for tests
});

describe("Crawl Infrastructure E2E", () => {
  beforeAll(async () => {
    // Clean test data
    await testRedis.flushdb();
  });

  afterAll(async () => {
    await testRedis.quit();
  });

  describe("Singleflight Deduplication", () => {
    it("should coalesce 98%+ concurrent duplicate requests", async () => {
      const singleflight = new CrawlSingleflight(testRedis, {
        lockTtlMs: 60_000,
        resultTtlMs: 60_000,
        pollIntervalMs: 10,
      });

      const testUrl = "https://example.com/test-page";
      let fetchCount = 0;

      // Simulate 100 concurrent requests for the same URL
      const promises = Array.from({ length: 100 }, () =>
        singleflight.execute(`test:${testUrl}`, async () => {
          fetchCount++;
          await new Promise((r) => setTimeout(r, 50)); // Simulate fetch
          return { html: "<html>content</html>" };
        })
      );

      const results = await Promise.all(promises);

      // All should get the same result
      expect(results.every((r) => r.html === "<html>content</html>")).toBe(true);

      // Only 1-2 actual fetches (allow for race condition at start)
      expect(fetchCount).toBeLessThanOrEqual(2);

      // Metrics should show 98%+ coalesce rate
      const metrics = singleflight.getMetrics();
      const coalesceRate = metrics.coalesced / metrics.requests;
      expect(coalesceRate).toBeGreaterThanOrEqual(0.98);
    });
  });

  describe("Delta Cascade Hit Rate", () => {
    it("should achieve 80%+ skip rate on stable content", async () => {
      const deltaCascade = createDeltaCascade();
      const tenantId = "test-tenant";
      const lastCrawlDate = new Date(Date.now() - 24 * 60 * 60 * 1000); // 1 day ago

      // Simulate 100 URLs, 80% unchanged (L0 skip), 20% changed
      const urls = Array.from({ length: 100 }, (_, i) => ({
        url: `https://example.com/page-${i}`,
        lastmod: i < 80 ? new Date(Date.now() - 48 * 60 * 60 * 1000) : new Date(), // 80 unchanged
      }));

      for (const { url, lastmod } of urls) {
        await deltaCascade.evaluate(tenantId, url, lastmod, lastCrawlDate);
      }

      const metrics = deltaCascade.getMetrics();
      const skipRate = (metrics.skippedL0 + metrics.skippedL1 + metrics.skippedL2) / metrics.total;

      expect(skipRate).toBeGreaterThanOrEqual(0.8);
    });
  });

  describe("Queue Lane SLA", () => {
    beforeAll(async () => {
      // Start workers for SLA tests
      startCrawlWorkers();
    });

    afterAll(async () => {
      await stopCrawlWorkers();
    });

    it("should complete fast-api jobs in <1m", async () => {
      const jobId = `e2e-fast-${Date.now()}`;
      const startTime = Date.now();

      await addCrawlJob(jobId, {
        jobType: CrawlJobType.D_SERP, // Fast lane
        tenantId: "e2e-test",
        projectId: "e2e-project",
        urls: ["https://example.com"],
      });

      // Wait for job completion (poll)
      let completed = false;
      const timeout = 60_000; // 1 minute
      while (Date.now() - startTime < timeout && !completed) {
        const job = await fastApiQueue.getJob(jobId);
        if (job && (await job.isCompleted())) {
          completed = true;
        }
        await new Promise((r) => setTimeout(r, 100));
      }

      expect(completed).toBe(true);
      expect(Date.now() - startTime).toBeLessThan(60_000);
    }, 70_000); // Test timeout 70s

    it("should not block fast lane during heavy job", async () => {
      // Start a heavy job
      const heavyJobId = `e2e-heavy-${Date.now()}`;
      await addCrawlJob(heavyJobId, {
        jobType: CrawlJobType.A_FULL_AUDIT, // Heavy lane
        tenantId: "e2e-test",
        projectId: "e2e-project",
        urls: Array.from({ length: 100 }, (_, i) => `https://example.com/page-${i}`),
      });

      // Immediately submit fast job
      const fastJobId = `e2e-fast-concurrent-${Date.now()}`;
      const fastStartTime = Date.now();

      await addCrawlJob(fastJobId, {
        jobType: CrawlJobType.D_SERP,
        tenantId: "e2e-test",
        projectId: "e2e-project",
        urls: ["https://example.com"],
      });

      // Fast job should complete quickly despite heavy job running
      let fastCompleted = false;
      while (Date.now() - fastStartTime < 60_000 && !fastCompleted) {
        const job = await fastApiQueue.getJob(fastJobId);
        if (job && (await job.isCompleted())) {
          fastCompleted = true;
        }
        await new Promise((r) => setTimeout(r, 100));
      }

      expect(fastCompleted).toBe(true);
      expect(Date.now() - fastStartTime).toBeLessThan(60_000);
    }, 70_000);
  });

  describe("Metrics API", () => {
    it("should return valid metrics structure", async () => {
      const response = await fetch("http://localhost:3001/api/admin/crawl-metrics");
      expect(response.status).toBe(200);

      const data = await response.json();
      expect(data.success).toBe(true);
      expect(data.data).toHaveProperty("singleflight");
      expect(data.data).toHaveProperty("delta");
      expect(data.data).toHaveProperty("queues");
      expect(data.data).toHaveProperty("totalEstimatedSavings");
    });
  });
});
