Implement per RESEARCH.md metrics pattern:

```typescript
const COST_PER_CRAWL_DOLLAR = 0.0001;

const metrics: CrawlMetrics = {
  singleflightHits: 0,
  singleflightMisses: 0,
  deltaL0Skips: 0,
  deltaL1Skips: 0,
  deltaL2Skips: 0,
  fullProcessed: 0,
  fastApiCompleted: 0,
  heavyCrawlCompleted: 0,
  costSavingsDollars: 0,
};

export function recordSingleflight(hit: boolean): void {
  if (hit) {
    metrics.singleflightHits++;
    metrics.costSavingsDollars += COST_PER_CRAWL_DOLLAR;
  } else {
    metrics.singleflightMisses++;
  }
}

export function recordDeltaSkip(layer: "L0" | "L1" | "L2"): void {
  switch (layer) {
    case "L0": metrics.deltaL0Skips++; break;
    case "L1": metrics.deltaL1Skips++; break;
    case "L2": metrics.deltaL2Skips++; break;
  }
  metrics.costSavingsDollars += COST_PER_CRAWL_DOLLAR;
}

export function getMetrics(): CrawlMetrics {
  return { ...metrics }; // Immutable copy
}
```
