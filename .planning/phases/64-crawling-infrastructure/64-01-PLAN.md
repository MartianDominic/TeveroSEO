```typescript
// Core interface
export interface SingleflightOptions {
  lockTtlMs?: number;       // Lock expiry (default: 300_000 = 5 min)
  resultTtlMs?: number;     // Result cache TTL (default: 3_600_000 = 1 hour)
  pollIntervalMs?: number;  // Waiter poll interval (default: 100)
  maxWaitMs?: number;       // Max wait before timeout (default: 300_000 = 5 min)
}

export interface SingleflightMetrics {
  requests: number;
  coalesced: number;
  cacheHits: number;
  cacheMisses: number;
}

export class CrawlSingleflight {
  constructor(redis: Redis, options?: SingleflightOptions);
  async execute<T>(key: string, fn: () => Promise<T>): Promise<T>;
  getMetrics(): SingleflightMetrics;
  resetMetrics(): void;
}
```

Implementation details:
1. Lock acquisition: `SET crawl:lock:{key} {workerId} NX EX {lockTtlSeconds}`
2. If lock acquired: Execute fn(), store result at `crawl:result:{key}` with EX, release lock
3. If lock NOT acquired: Poll `crawl:result:{key}` every pollIntervalMs until result or timeout
4. Use worker UUID to identify lock owner for debugging
5. Track metrics: requests, coalesced (waited for another), cacheHits (result existed), cacheMisses

Edge cases:
- Worker crash: Lock expires via EX, next requester can proceed
- Fn throws: Store error marker, waiters receive the error
- Timeout: Throw SingleflightTimeoutError after maxWaitMs
