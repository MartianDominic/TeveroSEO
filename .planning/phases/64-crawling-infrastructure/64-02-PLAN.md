<objective>
Implement L0->L1->L2->L3 delta crawling cascade to skip unchanged content.

Purpose: Progressive filtering reduces crawl cost. L0 (sitemap lastmod) is free, L1 (conditional GET) is cheap, L2 (hash comparison) requires local compute, L3 (full reprocess) is expensive. Each level gates the next.

Output: `delta-cascade.ts` orchestrating the cascade, plus HybridCrawler integration for L1 conditional GET.
</objective>

```typescript
export enum DeltaLevel {
  L0_SITEMAP = "L0_SITEMAP",     // Sitemap lastmod check (free)
  L1_CONDITIONAL = "L1_CONDITIONAL", // HTTP conditional GET (cheap)
  L2_HASH = "L2_HASH",           // Template-aware hash comparison
  L3_FULL = "L3_FULL",           // Full reprocess (expensive)
}

export interface DeltaResult {
  url: string;
  skipLevel: DeltaLevel | null; // Level that decided SKIP, null if proceeding
  action: "SKIP" | "PROCEED";
  changeType: ChangeType;
  etag?: string;           // From L1 response
  lastModified?: string;   // From L1 response
  html?: string;           // Content if fetched
}

export interface DeltaCascadeOptions {
  enableL0?: boolean; // Default true
  enableL1?: boolean; // Default true
  enableL2?: boolean; // Default true
}

export interface DeltaCascadeMetrics {
  total: number;
  skippedL0: number;
  skippedL1: number;
  skippedL2: number;
  proceededL3: number;
}

export class DeltaCascade {
  constructor(
    private readonly deltaSync: DeltaSyncService,
    private readonly options?: DeltaCascadeOptions
  );

  /**
   * Run delta cascade for a single URL.
   * Returns SKIP as soon as any level determines no change.
   */
  async evaluate(
    tenantId: string,
    url: string,
    sitemapLastmod: Date | null,
    lastCrawlDate: Date
  ): Promise<DeltaResult>;

  /**
   * Batch evaluate multiple URLs.
   * Returns array of DeltaResults.
   */
  async evaluateBatch(
    tenantId: string,
    urls: Array<{ url: string; lastmod: Date | null }>,
    lastCrawlDate: Date
  ): Promise<DeltaResult[]>;

  getMetrics(): DeltaCascadeMetrics;
  resetMetrics(): void;
}
```

Implementation logic for `evaluate`:

```typescript
// L0: Sitemap lastmod check (no network, instant)
if (options.enableL0 && sitemapLastmod && sitemapLastmod < lastCrawlDate) {
  metrics.skippedL0++;
  return { url, skipLevel: DeltaLevel.L0_SITEMAP, action: "SKIP", changeType: ChangeType.UNCHANGED };
}

// L1: Conditional GET
if (options.enableL1) {
  const snapshot = await deltaSync.getSnapshot(tenantId, url);
  if (snapshot?.etag || snapshot?.lastModified) {
    const response = await fetch(url, {
      headers: {
        "If-None-Match": snapshot.etag ?? undefined,
        "If-Modified-Since": snapshot.lastModified?.toUTCString() ?? undefined,
      },
    });
    if (response.status === 304) {
      metrics.skippedL1++;
      return { url, skipLevel: DeltaLevel.L1_CONDITIONAL, action: "SKIP", changeType: ChangeType.UNCHANGED };
    }
    // Capture new etag/lastModified for storage
    etag = response.headers.get("ETag");
    lastModified = response.headers.get("Last-Modified");
    html = await response.text();
  }
}

// L2: Hash comparison (requires content)
if (options.enableL2 && html) {
  const newHashes = computeHashesFromHtml(html);
  const snapshot = await deltaSync.getSnapshot(tenantId, url);
  const changeType = detectChange(snapshot ? { seoContentHash: snapshot.seoContentHash, ... } : null, newHashes);
  if (changeType === ChangeType.UNCHANGED) {
    metrics.skippedL2++;
    return { url, skipLevel: DeltaLevel.L2_HASH, action: "SKIP", changeType };
  }
}

// L3: Full reprocess
metrics.proceededL3++;
return { url, skipLevel: null, action: "PROCEED", changeType: ChangeType.ADD, html, etag, lastModified };
```

async fetchPage(url: string, conditionalHeaders?: ConditionalHeaders): Promise<CrawlResult> {
  const headers: Record<string, string> = {
    "User-Agent": this.options.userAgent,
    "Accept-Language": this.options.acceptLanguage,
    Accept: "text/html,application/xhtml+xml",
  };

  // Add conditional GET headers if provided
  if (conditionalHeaders?.etag) {
    headers["If-None-Match"] = conditionalHeaders.etag;
  }
  if (conditionalHeaders?.lastModified) {
    headers["If-Modified-Since"] = conditionalHeaders.lastModified;
  }

  const response = await fetch(url, {
    headers,
    signal: AbortSignal.timeout(this.options.timeoutMs),

  // Handle 304 Not Modified
  if (response.status === 304) {
