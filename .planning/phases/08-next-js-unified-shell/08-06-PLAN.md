Template:
```tsx
"use client";
import { useAuth } from "@clerk/nextjs";
import { useEffect, useState } from "react";
import { useParams } from "next/navigation";

export default function SeoAuditPage() {
  const { clientId } = useParams<{ clientId: string }>();
  const { getToken, isLoaded } = useAuth();
  const [src, setSrc] = useState<string | null>(null);

  useEffect(() => {
    if (!isLoaded) return;
    (async () => {
      const token = await getToken();
      const base = process.env.NEXT_PUBLIC_OPEN_SEO_URL ?? "https://app.openseo.so";
      const url = new URL("/audits", base);
      url.searchParams.set("client_id", clientId);
      if (token) url.searchParams.set("token", token);
      setSrc(url.toString());
    })();
  }, [clientId, getToken, isLoaded]);

  if (!src) return <div className="p-6">Loading SEO tools...</div>;
  return (
    <iframe
      src={src}
      title="SEO Audit"
      className="h-[calc(100vh-4rem)] w-full border-0"
    />
  );
}
```
