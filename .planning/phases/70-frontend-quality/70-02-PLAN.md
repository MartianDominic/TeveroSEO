Ensure all server actions return ActionResult<T>:

```typescript
type ActionResult<T> = 
  | { success: true; data: T }
  | { success: false; error: string };

export async function createBrief(formData: FormData): Promise<ActionResult<Brief>> {
  try {
    const validated = schema.parse(Object.fromEntries(formData));
    const brief = await briefService.create(validated);
    return { success: true, data: brief };
  } catch (error) {
    return { success: false, error: getErrorMessage(error) };
  }
}
```
