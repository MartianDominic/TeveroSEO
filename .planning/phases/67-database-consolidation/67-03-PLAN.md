```typescript
import { db as primaryDb } from './index';
import { createTeveroDb } from './tevero-connection';
import { sharedClients } from './schema/shared-clients';

const SHADOW_WRITE_ENABLED = process.env.SHADOW_WRITE_ENABLED === 'true';

export async function dualWriteClient(operation: 'insert' | 'update', data: ClientData) {
  // Primary write (blocking)
  const result = await primaryDb.insert(clients).values(data).returning();
  
  // Shadow write (fire-and-forget)
  if (SHADOW_WRITE_ENABLED) {
    const teveroDb = createTeveroDb();
    teveroDb.insert(sharedClients).values(data)
      .catch(err => console.error('[dual-write] Shadow write failed:', err));
  }
  
  return result;
}
```

```python
async def shadow_write_client(client_data: dict):
    """Fire-and-forget shadow write to tevero database."""
    if not SHADOW_WRITE_ENABLED or not TEVERO_URL:
        return
    
    try:
        engine = create_async_engine(TEVERO_URL)
        async with engine.begin() as conn:
            await conn.execute(...)
```

```typescript
const DB_READ_PERCENTAGE_TEVERO = parseInt(process.env.DB_READ_PERCENTAGE_TEVERO || '0', 10);

export function shouldReadFromTevero(table: string): boolean {
  const percentage = DB_READ_PERCENTAGE_TEVERO;
  return Math.random() * 100 < percentage;
}

export function getReadDb(table: string) {
  if (shouldReadFromTevero(table)) {
    return createTeveroDb();
  }
  return primaryDb;
}
```

```markdown
# Database Consolidation Cutover Runbook

## Pre-Cutover Checklist
- [ ] Verify migration scripts passed
- [ ] Verify row counts match
- [ ] Backup all databases
- [ ] Schedule maintenance window (low traffic)
- [ ] Notify team

## Phase 1: Enable Shadow Write (Week 1)
1. Set SHADOW_WRITE_ENABLED=true
2. Deploy all services
3. Monitor shadow write errors for 48h
4. Verify data consistency

## Phase 2: Gradual Read Migration (Week 2)
1. Set DB_READ_PERCENTAGE_TEVERO=10
2. Monitor for 24h
3. Increase to 50%, then 90%, then 100%
4. Verify latency and error rates

## Phase 3: Full Cutover (Week 3)
1. Set DATABASE_URL to tevero
2. Disable shadow write
3. Mark old databases read-only
4. Archive old databases

## Rollback Procedure
1. Set DB_READ_PERCENTAGE_TEVERO=0
2. Set SHADOW_WRITE_ENABLED=false
3. Run rollback scripts
4. Verify service health

## Emergency Contacts
```
