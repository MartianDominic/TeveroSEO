### Task 2: Add Missing Tests (if needed)

If tests are missing, add pytest unit tests:

```python
@pytest.mark.asyncio
async def test_build_article_prompt_includes_h2_suggestions():
    """Verify suggested_h2s from brief_context appear in prompt."""
    scheduled_article = ScheduledArticle(
        brief_context={"suggested_h2s": ["Heading One", "Heading Two"]}
    )
    prompt = _build_article_prompt(scheduled_article)
    assert "Heading One" in prompt
    assert "Heading Two" in prompt

@pytest.mark.asyncio
async def test_build_article_prompt_includes_paa_questions():
    """Verify PAA questions from brief_context appear in prompt."""
    scheduled_article = ScheduledArticle(
        brief_context={"paa_questions": ["What is X?", "How does Y work?"]}
    )
    prompt = _build_article_prompt(scheduled_article)
    assert "What is X?" in prompt

@pytest.mark.asyncio
async def test_build_article_prompt_handles_empty_brief_context():
    """Verify prompt generation succeeds with empty brief_context."""
    scheduled_article = ScheduledArticle(brief_context=None)
    prompt = _build_article_prompt(scheduled_article)
    assert prompt is not None
```
