### Task 3: Add Missing Tests (if needed)

```python
@pytest.mark.asyncio
async def test_fetch_voice_profile_returns_profile():
    """Verify voice profile fetch from open-seo API."""
    with respx.mock:
        respx.get("http://localhost:3001/api/seo/voice/client-123").respond(
            json={"tone": "professional", "vocabulary": ["innovation", "scale"]}
        )
        profile = await fetch_voice_profile("client-123")
        assert profile["tone"] == "professional"

@pytest.mark.asyncio
async def test_build_voice_constraints_formats_correctly():
    """Verify voice constraints generate proper LLM instruction."""
    profile = {"tone": "professional", "avoid_words": ["cheap"]}
    constraints = build_voice_constraints_from_profile(profile)
    assert "professional" in constraints.lower()
    assert "cheap" in constraints.lower()
```
