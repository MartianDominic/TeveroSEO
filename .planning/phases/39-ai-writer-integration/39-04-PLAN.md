### Task 3: Add Missing Security Tests (if needed)

```python
def test_link_insertion_rejects_javascript_urls():
    """XSS prevention: javascript: URLs must be rejected."""
    inserter = InternalLinkInserter()
    result = inserter._insert_link(soup, "click here", "javascript:alert(1)")
    assert result is False

def test_link_insertion_rejects_private_ips():
    """SSRF prevention: private IPs must be rejected."""
    inserter = InternalLinkInserter()
    result = inserter._insert_link(soup, "internal", "http://192.168.1.1/page")
    assert result is False

def test_link_insertion_escapes_anchor_text():
    """XSS prevention: anchor text must be HTML-escaped."""
    inserter = InternalLinkInserter()
    html = "<p>Test <script>alert(1)</script> content</p>"
    result = inserter.insert_links(html, client_id, "keyword")
    assert "<script>" not in result

def test_link_insertion_handles_unicode():
    """Unicode normalization: diacritics should match."""
    inserter = InternalLinkInserter()
    html = "<p>café and naïve content</p>"
    # Should match "cafe" or "naive" anchor text
```
