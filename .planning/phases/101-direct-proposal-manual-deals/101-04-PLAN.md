import { eq, and, desc, isNull, ilike, sql, inArray } from "drizzle-orm";
import { db } from "@/db";
import { contentBlocks, blockUsage, ContentBlockInsert, ContentBlockSelect } from "@/db/content-library-schema";
import { nanoid } from "nanoid";

export const ContentBlockRepository = {
  async create(data: Omit<ContentBlockInsert, "id">): Promise<ContentBlockSelect> {
    const id = nanoid();
    const [block] = await db.insert(contentBlocks).values({ ...data, id }).returning();
    return block;
  },

  async findById(id: string, workspaceId: string): Promise<ContentBlockSelect | null> {
    const [block] = await db.select().from(contentBlocks)
      .where(and(
        eq(contentBlocks.id, id),
        eq(contentBlocks.workspaceId, workspaceId),
        isNull(contentBlocks.softDeletedAt)
      ));
    return block ?? null;
  },

  async findByWorkspace(workspaceId: string, limit = 100): Promise<ContentBlockSelect[]> {
    return db.select().from(contentBlocks)
      .where(and(eq(contentBlocks.workspaceId, workspaceId), isNull(contentBlocks.softDeletedAt)))
      .orderBy(desc(contentBlocks.updatedAt))
      .limit(limit);
  },

  async search(
    workspaceId: string,
    options: { query?: string; category?: string; tags?: string[]; limit?: number }
  ): Promise<ContentBlockSelect[]> {
    let query = db.select().from(contentBlocks)
      .where(and(eq(contentBlocks.workspaceId, workspaceId), isNull(contentBlocks.softDeletedAt)));

    if (options.query) {
      query = query.where(ilike(contentBlocks.name, `%${options.query}%`));
    }

    if (options.category) {
      query = query.where(eq(contentBlocks.category, options.category));
    }

    if (options.tags?.length) {
      // Check if any tag matches (jsonb contains)
      query = query.where(
        sql`${contentBlocks.tags} ?| array[${sql.join(options.tags.map(t => sql`${t}`), sql`, `)}]`
      );
    }

    return query.orderBy(desc(contentBlocks.usageCount)).limit(options.limit ?? 50);
  },

  async update(id: string, workspaceId: string, data: Partial<ContentBlockInsert>): Promise<ContentBlockSelect | null> {
    const [updated] = await db.update(contentBlocks)
      .set({ ...data, updatedAt: new Date() })
      .where(and(eq(contentBlocks.id, id), eq(contentBlocks.workspaceId, workspaceId)))
      .returning();
    return updated ?? null;
  },

  async incrementUsage(id: string, workspaceId: string): Promise<void> {
    await db.update(contentBlocks)
      .set({
        usageCount: sql`${contentBlocks.usageCount} + 1`,
        lastUsedAt: new Date(),
        updatedAt: new Date(),
      })
      .where(and(eq(contentBlocks.id, id), eq(contentBlocks.workspaceId, workspaceId)));
  },

  async softDelete(id: string, workspaceId: string, deletedBy: string): Promise<void> {
    await db.update(contentBlocks)
      .set({ softDeletedAt: new Date(), softDeletedBy: deletedBy, updatedAt: new Date() })
      .where(and(eq(contentBlocks.id, id), eq(contentBlocks.workspaceId, workspaceId)));
  },

  async recordUsage(blockId: string, entityType: string, entityId: string, insertedBy?: string): Promise<void> {
    await db.insert(blockUsage).values({
      id: nanoid(),
      blockId,
      entityType,
      entityId,
      insertedBy,
    });
  },

  async getUsageStats(workspaceId: string, limit = 10): Promise<Array<{ blockId: string; name: string; usageCount: number }>> {
    return db.select({
      blockId: contentBlocks.id,
      name: contentBlocks.name,
      usageCount: contentBlocks.usageCount,
    })
      .from(contentBlocks)
      .where(and(eq(contentBlocks.workspaceId, workspaceId), isNull(contentBlocks.softDeletedAt)))
      .orderBy(desc(contentBlocks.usageCount))
      .limit(limit);
  },
};

import { ContentBlockRepository } from "../repositories/ContentBlockRepository";
import { ContentBlockSelect, CONTENT_BLOCK_CATEGORIES } from "@/db/content-library-schema";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "ContentLibraryService" });

type ContentBlockCategory = (typeof CONTENT_BLOCK_CATEGORIES)[number];

interface CreateBlockInput {
  name: string;
  category: ContentBlockCategory;
  content: string;
  contentEn?: string;
  contentLt?: string;
  tags?: string[];
}

interface UpdateBlockInput {
  name?: string;
  category?: ContentBlockCategory;
  content?: string;
  contentEn?: string;
  contentLt?: string;
  tags?: string[];
}

interface SearchOptions {
  query?: string;
  category?: ContentBlockCategory;
  tags?: string[];
  limit?: number;
}

export const ContentLibraryService = {
  /**
   * Create a new content block
   */
  async create(input: CreateBlockInput, workspaceId: string, userId: string): Promise<ContentBlockSelect> {
    const block = await ContentBlockRepository.create({
      workspaceId,
      name: input.name,
      category: input.category,
      content: input.content,
      contentEn: input.contentEn,
      contentLt: input.contentLt,
      tags: input.tags ?? [],
      createdBy: userId,
    });

    log.info("Created content block", { blockId: block.id, category: input.category, workspaceId });
    return block;
  },

  /**
   * Get a single block by ID
   */
  async getById(id: string, workspaceId: string): Promise<ContentBlockSelect | null> {
    return ContentBlockRepository.findById(id, workspaceId);
  },

  /**
   * Search blocks with filters
   */
  async search(workspaceId: string, options: SearchOptions = {}): Promise<ContentBlockSelect[]> {
    return ContentBlockRepository.search(workspaceId, options);
  },

  /**
   * Update a content block
   */
  async update(id: string, workspaceId: string, input: UpdateBlockInput): Promise<ContentBlockSelect | null> {
    const block = await ContentBlockRepository.update(id, workspaceId, input);
    if (block) {
      log.info("Updated content block", { blockId: id, workspaceId });
    }
    return block;
  },

  /**
   * Delete a content block (soft delete)
   */
  async delete(id: string, workspaceId: string, userId: string): Promise<void> {
    await ContentBlockRepository.softDelete(id, workspaceId, userId);
    log.info("Deleted content block", { blockId: id, workspaceId });
  },

  /**
   * Record usage of a block (when inserted into a document)
   */
  async recordUsage(
    blockId: string,
    workspaceId: string,
    entityType: "proposal" | "contract" | "document",
    entityId: string,
    userId?: string
  ): Promise<void> {
    // Verify block exists and belongs to workspace
    const block = await ContentBlockRepository.findById(blockId, workspaceId);
    if (!block) {
      log.warn("Attempted to record usage for non-existent block", { blockId, workspaceId });
      return;
    }

    // Record usage event
    await ContentBlockRepository.recordUsage(blockId, entityType, entityId, userId);

    // Increment usage counter on block
    await ContentBlockRepository.incrementUsage(blockId, workspaceId);

    log.debug("Recorded content block usage", { blockId, entityType, entityId });
  },

  /**
   * Get most popular blocks
   */
  async getPopularBlocks(workspaceId: string, limit = 10): Promise<Array<{ blockId: string; name: string; usageCount: number }>> {
    return ContentBlockRepository.getUsageStats(workspaceId, limit);
  },

  /**
   * Get all available categories
   */
  getCategories(): readonly string[] {
    return CONTENT_BLOCK_CATEGORIES;
  },
};

import { createAPIFileRoute } from "@tanstack/start/api";
import { z } from "zod";
import { ContentLibraryService } from "@/server/features/content-library/services/ContentLibraryService";
import { CONTENT_BLOCK_CATEGORIES } from "@/db/content-library-schema";
import { withWorkspace } from "@/server/lib/middleware/withWorkspace";

const searchQuerySchema = z.object({
  query: z.string().optional(),
  category: z.enum(CONTENT_BLOCK_CATEGORIES).optional(),
  tags: z.string().optional(), // comma-separated
  limit: z.coerce.number().int().min(1).max(100).default(50),
});

const createBlockSchema = z.object({
  name: z.string().min(1).max(255),
  category: z.enum(CONTENT_BLOCK_CATEGORIES),
  content: z.string().min(1),
  contentEn: z.string().optional(),
  contentLt: z.string().optional(),
  tags: z.array(z.string()).default([]),
});

export const Route = createAPIFileRoute("/api/content-library/blocks")({
  GET: withWorkspace(async ({ request, workspaceId }) => {
    const url = new URL(request.url);
    const query = searchQuerySchema.parse({
      query: url.searchParams.get("query") ?? undefined,
      category: url.searchParams.get("category") ?? undefined,
      tags: url.searchParams.get("tags") ?? undefined,
      limit: url.searchParams.get("limit") ?? 50,
    });

    const blocks = await ContentLibraryService.search(workspaceId, {
      query: query.query,
      category: query.category,
      tags: query.tags ? query.tags.split(",").filter(Boolean) : undefined,
      limit: query.limit,
    });

    return Response.json({ success: true, data: blocks });
  }),

  POST: withWorkspace(async ({ request, workspaceId, userId }) => {
    const body = await request.json();
    const input = createBlockSchema.parse(body);

    const block = await ContentLibraryService.create(input, workspaceId, userId);

    return Response.json({ success: true, data: block }, { status: 201 });
  }),
});

import { z } from "zod";

import { CONTENT_BLOCK_CATEGORIES } from "@/db/content-library-schema";

const updateBlockSchema = z.object({
  name: z.string().min(1).max(255).optional(),
  category: z.enum(CONTENT_BLOCK_CATEGORIES).optional(),
  content: z.string().min(1).optional(),
  contentEn: z.string().optional(),

  GET: withWorkspace(async ({ params, workspaceId }) => {
    const block = await ContentLibraryService.getById(params.blockId, workspaceId);
    if (!block) {
      return Response.json({ success: false, error: "Block not found" }, { status: 404 });

    return Response.json({ success: true, data: block });

    const block = await ContentLibraryService.update(params.blockId, workspaceId, input);
    if (!block) {
      return Response.json({ success: false, error: "Block not found" }, { status: 404 });
    }

    return Response.json({ success: true, data: block });
  }),

  DELETE: withWorkspace(async ({ params, workspaceId, userId }) => {
    await ContentLibraryService.delete(params.blockId, workspaceId, userId);
    return Response.json({ success: true });
  }),
});

import { createAPIFileRoute } from "@tanstack/start/api";
import { z } from "zod";
import { ContentLibraryService } from "@/server/features/content-library/services/ContentLibraryService";
import { withWorkspace } from "@/server/lib/middleware/withWorkspace";

const recordUsageSchema = z.object({
  blockId: z.string(),
  entityType: z.enum(["proposal", "contract", "document"]),
  entityId: z.string(),
});

const popularQuerySchema = z.object({
  limit: z.coerce.number().int().min(1).max(50).default(10),
});

export const Route = createAPIFileRoute("/api/content-library/usage")({
  // GET popular blocks
  GET: withWorkspace(async ({ request, workspaceId }) => {

      limit: url.searchParams.get("limit") ?? 10,

    const input = recordUsageSchema.parse(body);

    await ContentLibraryService.recordUsage(
      input.blockId,
      workspaceId,
      input.entityType,
      input.entityId,
      userId
    );

    return Response.json({ success: true });
  }),

  isLoading: boolean;

  // Actions

  close: () => void;

  setLoading: (loading) => set({ isLoading: loading }),
}));

import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { useContentLibraryStore } from "@/stores/contentLibraryStore";
import { useEffect } from "react";

interface ContentBlock {
  id: string;
  name: string;
  category: string;
  content: string;
  contentEn?: string;
  contentLt?: string;
  tags: string[];
  usageCount: number;
  lastUsedAt: string | null;
}

async function fetchBlocks(query?: string, category?: string): Promise<ContentBlock[]> {
  const params = new URLSearchParams();
  if (query) params.set("query", query);
  if (category) params.set("category", category);

  const res = await fetch(`/api/content-library/blocks?${params}`);
  const data = await res.json();
  return data.data ?? [];
}

async function recordBlockUsage(blockId: string, entityType: string, entityId: string): Promise<void> {
  await fetch("/api/content-library/usage", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ blockId, entityType, entityId }),
  });
}

export function useContentLibrary() {
  const { searchQuery, selectedCategory, setBlocks, setLoading } = useContentLibraryStore();
  const queryClient = useQueryClient();

  const { data: blocks = [], isLoading, refetch } = useQuery({
    queryKey: ["content-library", searchQuery, selectedCategory],
    queryFn: () => fetchBlocks(searchQuery || undefined, selectedCategory || undefined),
    staleTime: 30000,
  });

  useEffect(() => {
    setBlocks(blocks);
    setLoading(isLoading);
  }, [blocks, isLoading, setBlocks, setLoading]);

  const recordUsage = useMutation({
    mutationFn: ({ blockId, entityType, entityId }: { blockId: string; entityType: string; entityId: string }) =>
      recordBlockUsage(blockId, entityType, entityId),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["content-library"] });
    },
  });

  return {
    blocks,
    isLoading,
    refetch,
    recordUsage: recordUsage.mutate,
  };
}

"use client";

import { useCallback } from "react";
import { Sheet, SheetContent, SheetHeader, SheetTitle } from "@tevero/ui/components/sheet";
import { Input } from "@tevero/ui/components/input";
import { Button } from "@tevero/ui/components/button";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@tevero/ui/components/tabs";
import { Search, Library, X } from "lucide-react";
import { useContentLibraryStore } from "@/stores/contentLibraryStore";
import { useContentLibrary } from "./useContentLibrary";
import { ContentBlockCard } from "./ContentBlockCard";
import { useDebounce } from "@/hooks/useDebounce";

const CATEGORIES = [
  { value: "all", label: "All" },
  { value: "case_study", label: "Case Studies" },
  { value: "testimonial", label: "Testimonials" },
  { value: "pricing_table", label: "Pricing" },
  { value: "legal_clause", label: "Legal" },
  { value: "team_bio", label: "Team" },
  { value: "methodology", label: "Methodology" },
  { value: "faq", label: "FAQ" },
];

interface ContentLibraryPanelProps {
  entityType: "proposal" | "contract" | "document";
  entityId: string;
  onInsert: (content: string) => void;
}

export function ContentLibraryPanel({ entityType, entityId, onInsert }: ContentLibraryPanelProps) {
  const {
    isOpen,
    close,
    searchQuery,
    setSearchQuery,
    selectedCategory,
    setSelectedCategory,
    blocks,
    isLoading,
  } = useContentLibraryStore();

  const { recordUsage } = useContentLibrary();
  const debouncedQuery = useDebounce(searchQuery, 300);

  const handleInsert = useCallback(
    (block: { id: string; content: string }) => {
      onInsert(block.content);
      recordUsage({ blockId: block.id, entityType, entityId });
      close();
    },
    [onInsert, recordUsage, entityType, entityId, close]
  );

  const filteredBlocks = selectedCategory && selectedCategory !== "all"
    ? blocks.filter((b) => b.category === selectedCategory)
    : blocks;

  return (
    <Sheet open={isOpen} onOpenChange={(open) => !open && close()}>
      <SheetContent side="right" className="w-[400px] sm:w-[540px]">
        <SheetHeader>
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-2">
              <Library className="h-5 w-5" />
              <SheetTitle>Content Library</SheetTitle>
            </div>
            <Button variant="ghost" size="icon" onClick={close}>
              <X className="h-4 w-4" />
            </Button>
          </div>
        </SheetHeader>

        <div className="mt-6 space-y-4">
          {/* Search */}
          <div className="relative">
            <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
            <Input
              placeholder="Search blocks..."
              className="pl-9"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
            />
          </div>

          {/* Category Tabs */}
          <Tabs
            value={selectedCategory ?? "all"}
            onValueChange={(v) => setSelectedCategory(v === "all" ? null : v)}
          >
            <TabsList className="flex flex-wrap h-auto gap-1">
              {CATEGORIES.map((cat) => (
                <TabsTrigger key={cat.value} value={cat.value} className="text-xs">
                  {cat.label}
                </TabsTrigger>
              ))}
            </TabsList>
          </Tabs>

          {/* Block List */}
          <div className="space-y-3 max-h-[calc(100vh-280px)] overflow-y-auto">
            {isLoading ? (
              <div className="text-center py-8 text-muted-foreground">Loading...</div>
            ) : filteredBlocks.length === 0 ? (
              <div className="text-center py-8 text-muted-foreground">
                <Library className="h-8 w-8 mx-auto mb-2 opacity-50" />
                <p className="text-sm">No blocks found</p>
                <p className="text-xs mt-1">Try adjusting your search or category</p>
              </div>
            ) : (
              filteredBlocks.map((block) => (
                <ContentBlockCard key={block.id} block={block} onInsert={handleInsert} />
              ))
            )}
          </div>
        </div>
      </SheetContent>
    </Sheet>
  );
}
