import { eq, and, gte, lte, or, ilike, desc } from "drizzle-orm";
import { db } from "@/db";
import { payments, PaymentSelect, MATCH_TYPES } from "@/db/payment-schema";
import { invoices, InvoiceSelect } from "@/db/invoice-schema";
import { clients } from "@/db/client-schema";
import { PaymentRepository } from "../repositories/PaymentRepository";
import { createLogger } from "@/server/lib/logger";
import { addDays, subDays } from "date-fns";

const log = createLogger({ module: "AutoMatchEngine" });

export type MatchType = (typeof MATCH_TYPES)[number];

export interface MatchResult {
  invoiceId: string | null;
  confidence: number; // 0-100
  matchType: MatchType;
  suggestedInvoices?: Array<{ id: string; invoiceNumber: string; totalCents: number; confidence: number }>;
}

const AUTO_MATCH_THRESHOLD = 90; // >= 90% confidence auto-matches
const FUZZY_AMOUNT_TOLERANCE_CENTS = 50; // EUR 0.50

export const AutoMatchEngine = {
  /**
   * Attempt to auto-match a payment to an invoice.
   * Follows D-02 priority cascade from CONTEXT.md.
   */
  async autoMatch(payment: PaymentSelect): Promise<MatchResult> {
    const { workspaceId, grossAmountCents, payerEmail, payerName, memo, receivedAt } = payment;

    // Priority 1: Invoice # in memo (100% confidence)
    if (memo) {
      const invoiceByMemo = await this.findInvoiceByMemo(memo, workspaceId);
      if (invoiceByMemo) {
        log.info("Matched by invoice memo", { paymentId: payment.id, invoiceId: invoiceByMemo.id, confidence: 100 });
        return { invoiceId: invoiceByMemo.id, confidence: 100, matchType: "invoice_memo" };
      }
    }

    // Priority 2: Exact amount + client email (95% confidence)
    if (payerEmail) {
      const invoiceByAmountEmail = await this.findInvoiceByAmountAndEmail(grossAmountCents, payerEmail, workspaceId);
      if (invoiceByAmountEmail) {
        log.info("Matched by amount + email", { paymentId: payment.id, invoiceId: invoiceByAmountEmail.id, confidence: 95 });
        return { invoiceId: invoiceByAmountEmail.id, confidence: 95, matchType: "exact_amount_email" };
      }
    }

    // Priority 3: Exact amount + date within 7 days (85% confidence)
    const invoiceByAmountDate = await this.findInvoiceByAmountAndDateRange(grossAmountCents, receivedAt, 7, workspaceId);
    if (invoiceByAmountDate) {
      log.info("Matched by amount + date", { paymentId: payment.id, invoiceId: invoiceByAmountDate.id, confidence: 85 });
      return { invoiceId: invoiceByAmountDate.id, confidence: 85, matchType: "exact_amount_date" };
    }

    // Priority 4: Fuzzy amount + client name (70% confidence)
    if (payerName) {
      const invoiceByFuzzyAmountName = await this.findInvoiceByFuzzyAmountAndName(
        grossAmountCents,
        FUZZY_AMOUNT_TOLERANCE_CENTS,
        payerName,
        workspaceId
      );
      if (invoiceByFuzzyAmountName) {
        log.info("Matched by fuzzy amount + name", { paymentId: payment.id, invoiceId: invoiceByFuzzyAmountName.id, confidence: 70 });
        return { invoiceId: invoiceByFuzzyAmountName.id, confidence: 70, matchType: "fuzzy_amount_name" };
      }
    }

    // Priority 5: No match - return suggestions for review queue
    const suggestions = await this.findSuggestedMatches(payment);
    log.info("No auto-match found", { paymentId: payment.id, suggestionCount: suggestions.length });
    return { invoiceId: null, confidence: 0, matchType: "none", suggestedInvoices: suggestions };
  },

  /**
   * Process a payment through auto-match and update its status.
   * Payments >= 90% confidence are auto-matched; others go to review queue.
   */
  async processPayment(paymentId: string, workspaceId: string): Promise<MatchResult> {
    const payment = await PaymentRepository.findById(paymentId, workspaceId);
    if (!payment) {
      throw new Error(`Payment not found: ${paymentId}`);
    }

    const result = await this.autoMatch(payment);

    // Update payment status based on confidence
    const newStatus = result.confidence >= AUTO_MATCH_THRESHOLD ? "matched" : "review";
    await PaymentRepository.updateStatus(paymentId, workspaceId, newStatus, {
      matchedInvoiceId: result.invoiceId ?? undefined,
      confidence: result.confidence,
      matchType: result.matchType,
    });

    return result;
  },

  // --- Private helper methods ---

  async findInvoiceByMemo(memo: string, workspaceId: string): Promise<InvoiceSelect | null> {
    // Extract invoice number patterns like "INV-042", "#042", "Invoice 042"
    const patterns = [
      /INV[#-]?(\d+)/i,
      /#(\d+)/,
      /invoice\s*#?\s*(\d+)/i,
      /saskaita[#-]?\s*(\d+)/i, // Lithuanian
    ];

    for (const pattern of patterns) {
      const match = memo.match(pattern);
      if (match) {
        const invoiceNumber = match[0]; // Use full match to catch prefix
        const [invoice] = await db.select().from(invoices)
          .where(and(
            eq(invoices.workspaceId, workspaceId),
            ilike(invoices.invoiceNumber, `%${match[1]}%`),
            eq(invoices.status, "sent") // Only match unpaid invoices
          ))
          .limit(1);
        if (invoice) return invoice;
      }
    }
    return null;
  },

  async findInvoiceByAmountAndEmail(amountCents: number, email: string, workspaceId: string): Promise<InvoiceSelect | null> {
    // Join invoices with clients to match by email
    const [result] = await db.select({ invoice: invoices })
      .from(invoices)
      .innerJoin(clients, eq(invoices.clientId, clients.id))
      .where(and(
        eq(invoices.workspaceId, workspaceId),
        eq(invoices.totalCents, amountCents),
        eq(invoices.status, "sent"),
        ilike(clients.email, email)
      ))
      .limit(1);
    return result?.invoice ?? null;
  },

  async findInvoiceByAmountAndDateRange(
    amountCents: number,
    paymentDate: Date,
    dayRange: number,
    workspaceId: string
  ): Promise<InvoiceSelect | null> {
    const startDate = subDays(paymentDate, dayRange);
    const endDate = addDays(paymentDate, dayRange);

    const [invoice] = await db.select().from(invoices)
      .where(and(
        eq(invoices.workspaceId, workspaceId),
        eq(invoices.totalCents, amountCents),
        eq(invoices.status, "sent"),
        gte(invoices.sentAt, startDate),
        lte(invoices.sentAt, endDate)
      ))
      .orderBy(desc(invoices.sentAt)) // Most recent first
      .limit(1);
    return invoice ?? null;
  },

  async findInvoiceByFuzzyAmountAndName(
    amountCents: number,
    toleranceCents: number,
    name: string,
    workspaceId: string
  ): Promise<InvoiceSelect | null> {
    const minAmount = amountCents - toleranceCents;
    const maxAmount = amountCents + toleranceCents;

    const [result] = await db.select({ invoice: invoices })
      .from(invoices)
      .innerJoin(clients, eq(invoices.clientId, clients.id))
      .where(and(
        eq(invoices.workspaceId, workspaceId),
        gte(invoices.totalCents, minAmount),
        lte(invoices.totalCents, maxAmount),
        eq(invoices.status, "sent"),
        ilike(clients.name, `%${name}%`)
      ))
      .limit(1);
    return result?.invoice ?? null;
  },

  async findSuggestedMatches(payment: PaymentSelect): Promise<Array<{ id: string; invoiceNumber: string; totalCents: number; confidence: number }>> {
    const { workspaceId, grossAmountCents } = payment;

    // Find unpaid invoices within 20% of payment amount
    const minAmount = Math.floor(grossAmountCents * 0.8);
    const maxAmount = Math.ceil(grossAmountCents * 1.2);

    const candidates = await db.select().from(invoices)
      .where(and(
        eq(invoices.workspaceId, workspaceId),
        eq(invoices.status, "sent"),
        gte(invoices.totalCents, minAmount),
        lte(invoices.totalCents, maxAmount)
      ))
      .orderBy(desc(invoices.sentAt))
      .limit(5);

    return candidates.map(inv => ({
      id: inv.id,
      invoiceNumber: inv.invoiceNumber,
      totalCents: inv.totalCents,
      // Calculate suggestion confidence based on amount similarity
      confidence: Math.max(0, Math.round(100 - Math.abs(inv.totalCents - grossAmountCents) / grossAmountCents * 100)),
    }));
  },
};

export const ClientCreditRepository = {
  async create(data: Omit<ClientCreditInsert, "id">): Promise<ClientCreditSelect> {
    const id = nanoid();
    const [credit] = await db.insert(clientCredits).values({ ...data, id }).returning();

import { db } from "@/db";

import { PaymentAllocationRepository } from "../repositories/PaymentAllocationRepository";
import { ClientCreditRepository } from "../repositories/ClientCreditRepository";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "PaymentAllocationService" });

interface AllocationInput {
  invoiceId: string;
  amountCents: number;
}

export const PaymentAllocationService = {
  /**
   * Allocate a payment to a single invoice.
   * Per D-02: Partial payment tracking (invoice stays open with balance)
   */
  async allocateToInvoice(
    paymentId: string,
    invoiceId: string,
    amountCents: number,
    workspaceId: string
  ): Promise<{ allocated: boolean; overpaymentCents: number }> {
    // Verify payment exists and belongs to workspace
    const payment = await PaymentRepository.findById(paymentId, workspaceId);
    if (!payment) {
      throw new Error(`Payment not found: ${paymentId}`);
    }

    // Get invoice
    const [invoice] = await db.select().from(invoices)
      .where(and(eq(invoices.id, invoiceId), eq(invoices.workspaceId, workspaceId)));
    if (!invoice) {
      throw new Error(`Invoice not found: ${invoiceId}`);
    }

    // Calculate existing allocations to this invoice
    const existingAllocated = await PaymentAllocationRepository.getTotalAllocatedToInvoice(invoiceId);
    const remainingOnInvoice = invoice.totalCents - existingAllocated;

    // Determine allocation amount and overpayment
    const allocateAmount = Math.min(amountCents, remainingOnInvoice);
    const overpaymentCents = amountCents > remainingOnInvoice ? amountCents - remainingOnInvoice : 0;

    // Create allocation
    await PaymentAllocationRepository.create({
      paymentId,
      invoiceId,
      allocatedCents: allocateAmount,
    });

    // Check if invoice is fully paid
    const totalAllocated = existingAllocated + allocateAmount;
    if (totalAllocated >= invoice.totalCents) {
      await db.update(invoices)
        .set({ status: "paid", paidAt: new Date(), updatedAt: new Date() })
        .where(eq(invoices.id, invoiceId));
      log.info("Invoice fully paid", { invoiceId, totalAllocated });
    }

    // Update payment status to allocated
    await PaymentRepository.updateStatus(paymentId, workspaceId, "allocated", {
      matchedInvoiceId: invoiceId,
    });

    log.info("Payment allocated", { paymentId, invoiceId, allocateAmount, overpaymentCents });

    return { allocated: true, overpaymentCents };
  },

  /**
   * Split a payment across multiple invoices.
   * Per D-02: Split payments across invoices (oldest first)
   */
  async allocateToMultiple(
    paymentId: string,
    allocations: AllocationInput[],
    workspaceId: string
  ): Promise<{ allocatedCount: number; totalAllocatedCents: number; overpaymentCents: number }> {
    const payment = await PaymentRepository.findById(paymentId, workspaceId);
    if (!payment) {
      throw new Error(`Payment not found: ${paymentId}`);
    }

    let totalAllocatedCents = 0;
    let allocatedCount = 0;

    // Sort by oldest invoice first (pay oldest first per common pitfall)
    // Allocations should be pre-sorted by caller, but we process in order given
    for (const allocation of allocations) {
      const result = await this.allocateToInvoice(
        paymentId,
        allocation.invoiceId,
        allocation.amountCents,
        workspaceId
      );
      if (result.allocated) {
        totalAllocatedCents += allocation.amountCents - result.overpaymentCents;
        allocatedCount++;
      }
    }

    const overpaymentCents = payment.grossAmountCents - totalAllocatedCents;

    return { allocatedCount, totalAllocatedCents, overpaymentCents };
  },

  /**
   * Create a client credit from overpayment.
   * Per D-02: Overpayment -> auto-credit to client balance
   */
  async createOverpaymentCredit(
    paymentId: string,
    clientId: string,
    overpaymentCents: number,
    workspaceId: string
  ): Promise<void> {
    if (overpaymentCents <= 0) return;

    const payment = await PaymentRepository.findById(paymentId, workspaceId);
    if (!payment) {
      throw new Error(`Payment not found: ${paymentId}`);
    }

    await ClientCreditRepository.create({
      workspaceId,
      clientId,
      sourcePaymentId: paymentId,
      amountCents: overpaymentCents,
      usedCents: 0,
      currency: payment.currency ?? "EUR",
      reason: "overpayment",
    });

    log.info("Created overpayment credit", { paymentId, clientId, overpaymentCents });
  },

  /**
   * Apply existing client credits to an invoice.
   */
  async applyCreditsToInvoice(
    clientId: string,
    invoiceId: string,
    workspaceId: string
  ): Promise<number> {
    const availableCredits = await ClientCreditRepository.findAvailableByClient(clientId, workspaceId);
    
    const [invoice] = await db.select().from(invoices)
      .where(and(eq(invoices.id, invoiceId), eq(invoices.workspaceId, workspaceId)));
    if (!invoice) return 0;

    const existingAllocated = await PaymentAllocationRepository.getTotalAllocatedToInvoice(invoiceId);
    let remainingOnInvoice = invoice.totalCents - existingAllocated;
    let totalApplied = 0;

    for (const credit of availableCredits) {
      if (remainingOnInvoice <= 0) break;

      const available = credit.amountCents - (credit.usedCents ?? 0);
      const applyAmount = Math.min(available, remainingOnInvoice);

      await ClientCreditRepository.updateUsed(credit.id, (credit.usedCents ?? 0) + applyAmount);
      totalApplied += applyAmount;
      remainingOnInvoice -= applyAmount;
    }

    log.info("Applied credits to invoice", { clientId, invoiceId, totalApplied });
    return totalApplied;
  },
};

  /**
   * Get all payments in review queue for a workspace.
   * Per D-02: Review queue UX shows suggested matches with confidence %
   */
  async getReviewQueue(workspaceId: string): Promise<ReviewQueueItem[]> {
    const reviewPayments = await db.select().from(payments)
      .where(and(
        eq(payments.workspaceId, workspaceId),
        eq(payments.status, "review")
      ))
      .orderBy(desc(payments.receivedAt));

    // Enrich with fresh match suggestions
    const enrichedQueue: ReviewQueueItem[] = [];
    for (const payment of reviewPayments) {
      const matchResult = await AutoMatchEngine.autoMatch(payment);
      enrichedQueue.push({
        ...payment,
        suggestedInvoices: matchResult.suggestedInvoices,
      });
    }

    return enrichedQueue;
  },

  /**
   * Manually confirm a suggested match.
   * Per D-02: One-click confirm
   */
  async confirmMatch(
    paymentId: string,
    invoiceId: string,
    clientId: string,
    workspaceId: string
  ): Promise<{ success: boolean; overpaymentCents: number }> {
    const payment = await PaymentRepository.findById(paymentId, workspaceId);
    if (!payment) {
      throw new Error(`Payment not found: ${paymentId}`);
    }

    // Allocate to invoice
    const { overpaymentCents } = await PaymentAllocationService.allocateToInvoice(
      paymentId,
      invoiceId,
      payment.grossAmountCents,
      workspaceId
    );

    // Handle overpayment if any
    if (overpaymentCents > 0) {
      await PaymentAllocationService.createOverpaymentCredit(
        paymentId,
        clientId,
        overpaymentCents,
        workspaceId
      );
    }

    log.info("Match confirmed", { paymentId, invoiceId, overpaymentCents });
    return { success: true, overpaymentCents };
  },

  /**
   * Retry auto-matching for a payment.
   * Per D-02: Manual "Find matches" trigger
   */
  async retryAutoMatch(paymentId: string, workspaceId: string): Promise<ReviewQueueItem> {
    const payment = await PaymentRepository.findById(paymentId, workspaceId);
    if (!payment) {
      throw new Error(`Payment not found: ${paymentId}`);
    }

    const matchResult = await AutoMatchEngine.processPayment(paymentId, workspaceId);

    // Refetch with updated status
    const updated = await PaymentRepository.findById(paymentId, workspaceId);
    return {
      ...updated!,
      suggestedInvoices: matchResult.suggestedInvoices,
    };
  },

   * Create credit without matching to invoice.

   */

      workspaceId

      .where(and(eq(payments.workspaceId, workspaceId), eq(payments.status, "review")));
    return reviewPayments.length;
  },
