    1. Create `open-seo-main/src/server/lib/client-context.ts`:
    ```typescript
    import { alwrityPool } from "@/server/lib/alwrity-db";
    import { AppError } from "@/server/lib/errors";

    export const CLIENT_ID_HEADER = "x-client-id";

    const UUID_RE =
      /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;

    /**
     * Resolve the client_id for a request.
     *
     * Contract (AUTH-03):
     *  - Header absent     → returns null (not every route is client-scoped).
     *  - Header present + valid UUID + exists in alwrity.clients (not archived)
     *                      → returns the UUID string.
     *  - Header present + malformed OR unknown UUID OR archived client
     *                      → throws AppError("FORBIDDEN").
     */
    export async function resolveClientId(
      headers: Headers,
    ): Promise<string | null> {
      const raw = headers.get(CLIENT_ID_HEADER);
      if (!raw) return null;

      const candidate = raw.trim();
      if (!UUID_RE.test(candidate)) {
        throw new AppError("FORBIDDEN", "Invalid X-Client-ID header");
      }

      const { rows } = await alwrityPool.query<{ id: string }>(
        "SELECT id FROM clients WHERE id = $1 AND is_archived = false LIMIT 1",
        [candidate],
      );

      if (rows.length === 0) {
        throw new AppError("FORBIDDEN", "Unknown or archived client_id");
      }

      return rows[0].id;
    }
    ```

    Notes:
    - `Headers#get` is lowercase-insensitive; `"x-client-id"` matches both
      `X-Client-ID` and `x-client-id` variants that AI-Writer sends.
    - We reuse the existing `FORBIDDEN` error code (already in shared/error-codes.ts).
    - The `LIMIT 1` is defensive; `id` is a primary key in AI-Writer, but this
      module must not rely on that invariant.
    - We do NOT cache across requests (per CONTEXT.md); the pool handles connection reuse.

    describe("resolveClientId", () => {

        mockedQuery.mockReset();

      it("returns null when X-Client-ID header is absent", async () => {

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| AI-Writer shell → open-seo API | Untrusted `X-Client-ID` header crosses here |
| open-seo process → alwrity Postgres | Cross-database read; parameterized queries only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Spoofing | `X-Client-ID` header | mitigate | Validate UUID format AND existence in `alwrity.clients`; reject unknown IDs with FORBIDDEN before any data access |
| T-06-02 | Tampering | Raw header → SQL | mitigate | Parameterized query (`$1`) in `alwrityPool.query` — no string interpolation |
| T-06-03 | Information disclosure | `clients` table row | mitigate | `SELECT id` only (do not echo back name/email); keep pool size small to avoid DB pressure |
| T-06-04 | Denial of service | Spammed bad IDs | mitigate | UUID regex rejects most garbage before DB round-trip; pool `connectionTimeoutMillis: 10_000` caps waits |
| T-06-05 | Elevation of privilege | Missing ALWRITY_DATABASE_URL | mitigate | `REQUIRED_ENV_CORE` makes startup fail-fast; server never runs without the check being wired |
</threat_model>
