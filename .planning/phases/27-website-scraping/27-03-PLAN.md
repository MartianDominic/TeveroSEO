       export async function submitManualBusinessInfo(
         prospectId: string,
         analysisId: string,
         info: ManualBusinessInfo
       ): Promise<void> {
         await patchOpenSeo(`/api/prospects/${prospectId}/analyses/${analysisId}/business-info`, {
           businessInfo: {
             products: info.products.split(",").map(s => s.trim()).filter(Boolean),
             brands: info.brands.split(",").map(s => s.trim()).filter(Boolean),
             services: info.services.split(",").map(s => s.trim()).filter(Boolean),
             location: info.location || null,
             targetMarket: info.targetMarket,
             summary: info.summary,
             confidence: 1.0,  // Manual input = 100% confidence
           },
           scrapeMethod: "manual",
         });
         revalidatePath(`/prospects/${prospectId}`);
       }
