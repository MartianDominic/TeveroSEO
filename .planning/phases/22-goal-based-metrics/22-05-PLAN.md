### Task 1: Create GoalAttainmentBadge component

**File:** `apps/web/src/components/dashboard/GoalAttainmentBadge.tsx`

```tsx
import { cn } from "@/lib/utils";
import { ArrowUp, ArrowDown, Minus, Target } from "lucide-react";
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@tevero/ui";

interface GoalAttainmentBadgeProps {
  attainmentPct: number | null;
  goalsMet: number;
  goalsTotal: number;
  trend?: string | null;
  showDetails?: boolean;
}

function getTrendIcon(trend: string | null) {
  if (trend === "up") return <ArrowUp className="h-3 w-3" />;
  if (trend === "down") return <ArrowDown className="h-3 w-3" />;
  return <Minus className="h-3 w-3" />;
}

function getColorClass(pct: number): string {
  if (pct >= 100) return "bg-green-100 text-green-800 border-green-200";
  if (pct >= 80) return "bg-yellow-100 text-yellow-800 border-yellow-200";
  if (pct >= 60) return "bg-orange-100 text-orange-800 border-orange-200";
  return "bg-red-100 text-red-800 border-red-200";
}

export function GoalAttainmentBadge({
  attainmentPct,
  goalsMet,
  goalsTotal,
  trend,
  showDetails = true,
}: GoalAttainmentBadgeProps) {
  const pct = attainmentPct ?? 0;
  
  if (goalsTotal === 0) {
    return (
      <span className="inline-flex items-center gap-1 px-2 py-1 text-xs rounded border bg-muted text-muted-foreground">
        <Target className="h-3 w-3" />
        No goals
      </span>
    );
  }

  const badge = (
    <span
      className={cn(
        "inline-flex items-center gap-1.5 px-2 py-1 text-xs font-medium rounded border",
        getColorClass(pct)
      )}
    >
      <Target className="h-3 w-3" />
      <span>{pct.toFixed(0)}%</span>
      {trend && (
        <span className={cn(
          trend === "up" ? "text-green-600" :
          trend === "down" ? "text-red-600" : "text-muted-foreground"
        )}>
          {getTrendIcon(trend)}
        </span>
      )}
    </span>
  );

  if (!showDetails) return badge;

  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger asChild>{badge}</TooltipTrigger>
        <TooltipContent>
          <div className="text-sm">
            <p className="font-medium">{goalsMet} of {goalsTotal} goals met</p>
            <p className="text-muted-foreground">Average attainment: {pct.toFixed(1)}%</p>
          </div>
        </TooltipContent>
      </Tooltip>
    </TooltipProvider>
  );
}
```

// Add GoalsPreview section
function GoalsPreview({ clientId }: { clientId: string }) {
  const { data: goals } = useClientGoals(clientId);
  
  if (!goals?.length) {
    return (
      <div className="text-sm text-muted-foreground">
        No goals configured
      </div>
    );
  }
  
  return (
    <div className="space-y-2">
      {goals.slice(0, 3).map(({ goal, template }) => (
        <div key={goal.id} className="flex items-center justify-between text-sm">
          <span className="truncate max-w-[150px]">
            {goal.customName ?? template.name}
          </span>
          <span className={cn(
            "font-medium",
            Number(goal.attainmentPct) >= 100 ? "text-green-600" :
            Number(goal.attainmentPct) >= 80 ? "text-yellow-600" : "text-muted-foreground"
          )}>
            {Number(goal.attainmentPct ?? 0).toFixed(0)}%
          </span>
        </div>
      ))}
      {goals.length > 3 && (
        <p className="text-xs text-muted-foreground">
          +{goals.length - 3} more goals
        </p>
      )}
    </div>
  );
}
