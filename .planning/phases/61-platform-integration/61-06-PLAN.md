import { Queue, type ConnectionOptions } from 'bullmq';
import { getSharedBullMQConnection } from '@/server/lib/redis';

export const TOKEN_REFRESH_QUEUE_NAME = 'token-refresh';

export interface TokenRefreshJobData {
  connectionId: string;
  platform: string;
  workspaceId: string;
}

let queue: Queue<TokenRefreshJobData> | null = null;

export function getTokenRefreshQueue(): Queue<TokenRefreshJobData> {
  if (!queue) {
    queue = new Queue<TokenRefreshJobData>(TOKEN_REFRESH_QUEUE_NAME, {
      connection: getSharedBullMQConnection('queue:token-refresh'),
      defaultJobOptions: {
        attempts: 3,
        backoff: { type: 'exponential', delay: 5000 },
        removeOnComplete: { age: 3600 }, // 1 hour
        removeOnFail: { age: 86400 },    // 24 hours
      },
    });
  }
  return queue;
}

/**
 * Initialize scheduler for token refresh.
 * Runs every 15 minutes per D-11.
 */
export async function initTokenRefreshScheduler(): Promise<void> {
  const q = getTokenRefreshQueue();
  
  // Remove existing repeatable job if any
  const repeatableJobs = await q.getRepeatableJobs();
  for (const job of repeatableJobs) {
    if (job.name === 'check-expiring-tokens') {
      await q.removeRepeatableByKey(job.key);
    }
  }
  
  // Add new repeatable job every 15 minutes
  await q.add('check-expiring-tokens', {}, {
    repeat: { every: 15 * 60 * 1000 }, // 15 minutes
    jobId: 'token-refresh-scheduler',
  });
}

import { Job } from 'bullmq';
import { db } from '@/db';
import { platformConnections } from '@/db/platform-connection-schema';
import { and, eq, lt, isNotNull, inArray } from 'drizzle-orm';
import { platformConnectionService } from '@/server/features/platform-oauth/PlatformConnectionService';
import { GoogleOAuthProvider } from '@/server/features/platform-oauth/providers';
import { WixOAuthProvider } from '@/server/features/platform-oauth/providers';
import { createLogger } from '@/server/lib/logger';

const logger = createLogger({ module: 'token-refresh-processor' });

// Platforms that support token refresh
const REFRESHABLE_PLATFORMS = [
  'google_search_console',
  'google_analytics', 
  'google_business_profile',
  'wix',
];

export default async function process(job: Job): Promise<void> {
  logger.info('Starting token refresh check', { jobId: job.id });
  
  // Find tokens expiring within 30 minutes per D-11
  const expiryThreshold = new Date(Date.now() + 30 * 60 * 1000);
  
  const expiring = await db.query.platformConnections.findMany({
    where: and(
      eq(platformConnections.status, 'active'),
      lt(platformConnections.tokenExpiresAt, expiryThreshold),
      isNotNull(platformConnections.refreshTokenEncrypted),
      inArray(platformConnections.platform, REFRESHABLE_PLATFORMS),
    ),
  });
  
  logger.info('Found expiring connections', { count: expiring.length });
  
  for (const connection of expiring) {
    try {
      // Get current tokens
      const tokens = await platformConnectionService.getOAuthTokens(connection.id);
      if (!tokens?.refreshToken) {
        logger.warn('No refresh token available', { connectionId: connection.id });
        continue;
      }
      
      // Get provider and refresh
      const provider = getProvider(connection.platform, connection.platformSiteUrl);
      const newTokens = await provider.refreshAccessToken(tokens.refreshToken);
      
      // Update stored tokens
      await platformConnectionService.updateTokens(
        connection.id,
        newTokens.accessToken,
        newTokens.refreshToken,
        newTokens.expiresIn,
      );
      
      logger.info('Token refreshed successfully', { connectionId: connection.id });
      
    } catch (error) {
      // Mark as error per D-12
      await platformConnectionService.updateStatus(
        connection.id,
        'error',
        error instanceof Error ? error.message : 'Token refresh failed',
      );
      
      logger.error('Token refresh failed', error as Error, { 
        connectionId: connection.id,
        platform: connection.platform,
      });
    }
  }
}

function getProvider(platform: string, siteUrl: string | null) {
  const redirectUri = `${process.env.NEXT_PUBLIC_APP_URL}/api/oauth/${platform.split('_')[0]}/callback`;
  
  switch (platform) {
    case 'google_search_console':
    case 'google_analytics':
    case 'google_business_profile':
      return new GoogleOAuthProvider(redirectUri);
    case 'wix':
      return new WixOAuthProvider(redirectUri);
    default:
      throw new Error(`No refresh support for platform: ${platform}`);
  }
}

'use client';

import { useCallback, useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { Button } from '@ui/button';
import { ConnectionCard } from './ConnectionCard';
import { PlatformConnectionFlow } from './PlatformConnectionFlow';

interface ConnectionStatusDashboardProps {
  prospectId?: string;
}

export function ConnectionStatusDashboard({ prospectId }: ConnectionStatusDashboardProps) {
  const [showAddDialog, setShowAddDialog] = useState(false);
  const queryClient = useQueryClient();
  
  const { data, isLoading } = useQuery({
    queryKey: ['connections', prospectId],
    queryFn: async () => {
      const params = new URLSearchParams();
      if (prospectId) params.set('prospectId', prospectId);
      const res = await fetch(`/api/connections?${params}`);
      if (!res.ok) throw new Error('Failed to fetch connections');
      return res.json();
    },
  });
  
  const syncMutation = useMutation({
    mutationFn: async (id: string) => {
      const res = await fetch(`/api/connections/${id}/sync`, { method: 'POST' });
      if (!res.ok) throw new Error('Sync failed');
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['connections'] });
    },
  });
  
  const disconnectMutation = useMutation({
    mutationFn: async (id: string) => {
      const res = await fetch(`/api/connections/${id}`, { method: 'DELETE' });
      if (!res.ok) throw new Error('Disconnect failed');
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['connections'] });
    },
  });
  
  const connections = data?.connections ?? [];
  
  return (
    <div className="space-y-4">
      <div className="flex justify-between items-center">
        <h2 className="text-lg font-semibold">Platform Connections</h2>
        <Button onClick={() => setShowAddDialog(true)}>
          + Add Connection
        </Button>
      </div>
      
      {isLoading ? (
        <div>Loading...</div>
      ) : connections.length === 0 ? (
        <div className="text-center py-8 text-muted-foreground">
          No connections yet. Connect a platform to get started.
        </div>
      ) : (
        <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
          {connections.map((conn: any) => (
            <ConnectionCard
              key={conn.id}
              {...conn}
              onSync={syncMutation.mutate}
              onDisconnect={disconnectMutation.mutate}
              isLoading={syncMutation.isPending || disconnectMutation.isPending}
            />
          ))}
        </div>
      )}
      
      {showAddDialog && (
        <PlatformConnectionFlow
          prospectId={prospectId}
          onClose={() => setShowAddDialog(false)}
          onSuccess={() => {
            setShowAddDialog(false);
            queryClient.invalidateQueries({ queryKey: ['connections'] });
          }}
        />
      )}
    </div>
  );
}

              </div>
            </div>
            <Button onClick={handleShopifyConnect} disabled={!shopDomain}>
              Connect Shopify
            </Button>
          </div>
        ) : selectedPlatform === 'wix' ? (
          <div className="space-y-4">
            <p>Connect your Wix account to access site data.</p>

        ) : selectedPlatform === 'wordpress' ? (

            <div>

              <Label>Application Password</Label>
              <Input
                type="password"

                onChange={e => setWpCredentials(c => ({ ...c, appPassword: e.target.value }))}
                placeholder="xxxx xxxx xxxx xxxx"
              />
              <a href="#" className="text-xs text-primary">How to create an Application Password</a>
            </div>
            <Button onClick={handleWordPressConnect} disabled={isLoading}>
              {isLoading ? 'Connecting...' : 'Connect WordPress'}
            </Button>
          </div>
        ) : null}
      </DialogContent>
    </Dialog>
  );
}
