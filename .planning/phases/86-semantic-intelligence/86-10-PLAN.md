/**
 * Portal Data Service
 * Phase 86-10: Final Integration
 * 
 * Aggregates portal data including clusters for "growth areas" view.
 */

import { eq } from 'drizzle-orm';
import type { PostgresJsDatabase } from 'drizzle-orm/postgres-js';
import { portalTokens } from '../../../db/schema/portal-tokens';
import { proposals } from '../../../db/schema/proposals';
import { contractGoals } from '../../../db/schema/contract-goals';
import { contractedKeywords } from '../../../db/schema/contracted-keywords';
import type {
  PortalDataResponse,
  PortalCluster,
  PortalKeyword,
  CalendarEvent,
} from './types';

const TIER_ORDER = { pillar: 0, subtopic: 1, longtail: 2 };

/**
 * Service for aggregating portal data.
 */
export class PortalDataService {
  constructor(private db: PostgresJsDatabase) {}

  /**
   * Get portal data for a token.
   */
  async getPortalData(token: string): Promise<PortalDataResponse> {
    // Validate token and get associated data
    const tokenData = await this.validateToken(token);
    if (!tokenData) {
      throw new Error('Invalid or expired portal token');
    }

    // Get proposal with clusters
    const proposal = await this.getProposalWithClusters(tokenData.clientId);
    
    // Get contract goal
    const goal = await this.getContractGoal(tokenData.clientId);
    
    // Get current keyword positions
    const positions = await this.getCurrentPositions(tokenData.clientId);
    
    // Build clusters with progress metrics
    const clusters = this.buildPortalClusters(proposal?.clusters || [], positions);
    
    // Derive flat keywords list from clusters
    const keywords = this.flattenKeywords(clusters);
    
    // Calculate achievement
    const inTop10 = keywords.filter(k => k.status === 'top10').length;
    const achievement = this.calculateAchievement(inTop10, goal);
    
    // Get calendar events
    const calendar = await this.getCalendarEvents(tokenData.clientId);

    return {
      client: {
        name: tokenData.clientName,
        domain: tokenData.clientDomain,
      },
      agency: {
        name: tokenData.agencyName,
        logoUrl: tokenData.agencyLogoUrl,
      },
      goal: {
        metric: goal?.metric || 'top_10',
        target: goal?.targetUpper || 0,
        deadline: goal?.deadline?.toISOString() || '',
        currentCount: inTop10,
        achievementPct: achievement.percentage,
      },
      achievement,
      clusters,
      keywords,
      calendar,
      lastUpdated: new Date().toISOString(),
    };
  }

  /**
   * Validate portal token.
   */
  private async validateToken(token: string) {
    const result = await this.db
      .select()
      .from(portalTokens)
      .where(eq(portalTokens.token, token))
      .limit(1);
    
    const tokenRow = result[0];
    if (!tokenRow) return null;
    if (tokenRow.isRevoked) return null;
    if (tokenRow.expiresAt && tokenRow.expiresAt < new Date()) return null;
    
    // Update last accessed
    await this.db
      .update(portalTokens)
      .set({ lastAccessedAt: new Date(), accessCount: (tokenRow.accessCount || 0) + 1 })
      .where(eq(portalTokens.id, tokenRow.id));
    
    return tokenRow;
  }

  /**
   * Get proposal with clusters JSONB.
   */
  private async getProposalWithClusters(clientId: string) {
    const result = await this.db
      .select()
      .from(proposals)
      .where(eq(proposals.clientId, clientId))
      .orderBy(proposals.createdAt)
      .limit(1);
    
    return result[0];
  }

  /**
   * Get contract goal.
   */
  private async getContractGoal(clientId: string) {
    const result = await this.db
      .select()
      .from(contractGoals)
      .where(eq(contractGoals.clientId, clientId))
      .limit(1);
    
    return result[0];
  }

  /**
   * Get current positions from contracted_keywords.
   */
  private async getCurrentPositions(clientId: string): Promise<Map<string, number | null>> {
    const result = await this.db
      .select({
        keyword: contractedKeywords.keyword,
        currentPosition: contractedKeywords.currentPosition,
        lockedPosition: contractedKeywords.lockedPosition,
      })
      .from(contractedKeywords)
      .where(eq(contractedKeywords.clientId, clientId));
    
    const positions = new Map<string, number | null>();
    for (const row of result) {
      positions.set(row.keyword, row.currentPosition);
    }
    return positions;
  }

  /**
   * Build portal clusters with progress metrics.
   */
  private buildPortalClusters(
    rawClusters: any[],
    positions: Map<string, number | null>
  ): PortalCluster[] {
    const clusters: PortalCluster[] = rawClusters.map(cluster => {
      const keywords: PortalKeyword[] = (cluster.keywords || []).map((kw: any) => {
        const currentPosition = positions.get(kw.keyword) ?? kw.currentPosition ?? null;
        const status = this.getKeywordStatus(currentPosition);
        
        return {
          id: kw.id || crypto.randomUUID(),
          keyword: kw.keyword,
          volume: kw.volume,
          difficulty: kw.difficulty,
          funnelStage: kw.funnelStage,
          currentPosition,
          lockedPosition: kw.lockedPosition ?? null,
          positionChange: this.calculatePositionChange(kw.lockedPosition, currentPosition),
          status,
        };
      });

      const inTop10 = keywords.filter(k => k.status === 'top10').length;
      const inTop20 = keywords.filter(k => k.status === 'top20').length;

      return {
        id: cluster.id,
        tier: cluster.tier,
        label: cluster.labelLt || cluster.label,
        labelEn: cluster.labelEn || cluster.label,
        totalVolume: cluster.totalVolume,
        averageDifficulty: cluster.averageDifficulty,
        dominantFunnel: cluster.dominantFunnel,
        keywords,
        progress: {
          inTop10,
          inTop20,
          total: keywords.length,
          percentComplete: keywords.length > 0 ? Math.round((inTop10 / keywords.length) * 100) : 0,
        },
        parentId: cluster.parentId || null,
      };
    });

    // Sort by tier (pillar first) then by volume descending
    return clusters.sort((a, b) => {
      const tierDiff = TIER_ORDER[a.tier] - TIER_ORDER[b.tier];
      if (tierDiff !== 0) return tierDiff;
      return b.totalVolume - a.totalVolume;
    });
  }

  /**
   * Get keyword status based on position.
   */
  private getKeywordStatus(position: number | null): PortalKeyword['status'] {
    if (position === null) return 'pending';
    if (position <= 10) return 'top10';
    if (position <= 20) return 'top20';
    return 'progress';
  }

  /**
   * Calculate position change.
   */
  private calculatePositionChange(locked: number | null, current: number | null): number | null {
    if (locked === null || current === null) return null;
    return locked - current; // Positive = improved
  }

  /**
   * Flatten keywords from clusters for backwards compatibility.
   */
  private flattenKeywords(clusters: PortalCluster[]): PortalKeyword[] {
    return clusters.flatMap(c => c.keywords);
  }

  /**
   * Calculate achievement metrics.
   */
  private calculateAchievement(current: number, goal: any) {
    const target = goal?.targetUpper || goal?.target || 0;
    const percentage = target > 0 ? Math.round((current / target) * 100) : 0;
    
    // Calculate days ahead/behind (simplified)
    const daysAhead = 0; // TODO: Calculate based on deadline and trajectory
    
    return {
      current,
      target,
      percentage,
      daysAhead,
    };
  }

  /**
   * Get calendar events.
   */
  private async getCalendarEvents(clientId: string): Promise<CalendarEvent[]> {
    // TODO: Query content calendar
    return [];
  }
}

/**
 * Factory function for getting portal data.
 */
export async function getPortalData(
  db: PostgresJsDatabase,
  token: string
): Promise<PortalDataResponse> {
  const service = new PortalDataService(db);
  return service.getPortalData(token);
}

/**
 * Portal Data API Endpoint
 * Phase 86-10: Final Integration
 * 
 * GET /portal/:token/data
 * 
 * Returns portal data including clusters for "growth areas" view.
 * Per CLIENT-PORTAL-SPEC.md, this is the main data endpoint for the portal.
 */

import { createAPIFileRoute } from '@tanstack/start/api';
import { json } from '@tanstack/start';
import { db } from '../../../../db';
import { getPortalData } from '../../../../server/features/portal/PortalDataService';
import type { PortalDataResponse } from '../../../../server/features/portal/types';

export const Route = createAPIFileRoute('/api/portal/$token/data')({
  GET: async ({ params }) => {
    const { token } = params;

    if (!token) {
      return json({ error: 'Token required' }, { status: 400 });
    }

    try {
      const data: PortalDataResponse = await getPortalData(db, token);

      // Response includes clusters for growth areas view
      // Per CLIENT-PORTAL-SPEC.md:
      // - clusters: Array of PortalCluster with progress metrics
      // - keywords: Flat list for simple view toggle (backwards compatible)
      return json(data, {
        headers: {
          // Cache for 5 minutes (portal data doesn't change frequently)
          'Cache-Control': 'private, max-age=300',
        },
      });
    } catch (error) {
      if (error instanceof Error && error.message.includes('Invalid or expired')) {
        return json({ error: 'Invalid or expired portal token' }, { status: 401 });
      }

      console.error('Portal data error:', error);
      return json({ error: 'Failed to load portal data' }, { status: 500 });
    }
  },
});

/**
 * Cluster Contract Mapper
 * Phase 86-10: Final Integration
 * 
 * Maps proposal clusters to contracted_keywords table when proposal is approved.
 * Per CLIENT-PORTAL-SPEC.md, this creates the contract-to-delivery bridge.
 */

import type { PostgresJsDatabase } from 'drizzle-orm/postgres-js';
import { contractedKeywords } from '../../../../db/schema/contracted-keywords';
import type { LabeledCluster, ClusteringInput } from '../clustering/types';

/**
 * Result of mapping clusters to contract.
 */
export interface ClusterContractResult {
  contractId: string;
  keywordsLocked: number;
  clustersMapped: number;
  lockedAt: Date;
}

/**
 * GSC service interface for position lookup.
 */
export interface GscPositionService {
  getPosition(keyword: string): Promise<number | null>;
}

/**
 * Maps clusters to contracted_keywords table.
 */
export class ClusterContractMapper {
  constructor(
    private db: PostgresJsDatabase,
    private gscService: GscPositionService
  ) {}

  /**
   * Map all keywords from clusters to contracted_keywords.
   * Called when proposal is approved and contract is signed.
   */
  async mapClustersToContract(
    contractId: string,
    clientId: string,
    clusters: LabeledCluster[]
  ): Promise<ClusterContractResult> {
    const lockedAt = new Date();
    let totalKeywords = 0;

    await this.db.transaction(async (tx) => {
      for (const cluster of clusters) {
        // Use selectedKeywords (the ones in the proposal, not backfill)
        const keywords = cluster.selectedKeywords || cluster.keywords || [];

        for (const kw of keywords) {
          // Snapshot current position from GSC
          const currentPosition = await this.gscService.getPosition(kw.keyword);

          await tx.insert(contractedKeywords).values({
            contractId,
            clientId,
            keyword: kw.keyword,
            lockedAt,
            lockedPosition: currentPosition,
            lockedSearchVolume: kw.volume,
            status: 'active',
            currentPosition,
            
            // Cluster context for reporting
            clusterId: cluster.id,
            clusterLabel: cluster.labelLt || cluster.labelEn,
            clusterTier: cluster.tier,
            funnelStage: kw.funnelStage,
          });

          totalKeywords++;
        }
      }
    });

    return {
      contractId,
      keywordsLocked: totalKeywords,
      clustersMapped: clusters.length,
      lockedAt,
    };
  }
}

/**
 * Factory function for mapping clusters to contract.
 */
export async function mapClustersToContract(
  db: PostgresJsDatabase,
  gscService: GscPositionService,
  contractId: string,
  clientId: string,
  clusters: LabeledCluster[]
): Promise<ClusterContractResult> {
  const mapper = new ClusterContractMapper(db, gscService);
  return mapper.mapClustersToContract(contractId, clientId, clusters);
}
