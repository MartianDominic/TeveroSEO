"""
Clustering Service
Phase 86-02: HDBSCAN + UMAP Clustering

Provides semantic clustering for keyword embeddings using HDBSCAN
with UMAP dimensionality reduction for improved performance.

CRITICAL: All embeddings MUST be 768-dim (jina-v5-text-nano).
"""

import numpy as np
from typing import TypedDict, Optional
import fast_hdbscan  # Memory-optimized HDBSCAN
import umap


class ClusterConfig(TypedDict, total=False):
    """Configuration for clustering."""
    min_cluster_size: int
    min_samples: int
    umap_dimensions: int
    umap_vis_dimensions: int


class ClusterResult(TypedDict):
    """Result of clustering operation."""
    labels: list[int]
    centroids: list[list[float]]
    vis_coords: list[dict[str, float]]
    cluster_count: int
    noise_count: int


DEFAULT_CONFIG: ClusterConfig = {
    "min_cluster_size": 3,
    "min_samples": 2,
    "umap_dimensions": 15,
    "umap_vis_dimensions": 2,
}

# Expected embedding dimension from jina-v5-text-nano
EXPECTED_EMBEDDING_DIM = 768


class EmbeddingDimensionError(ValueError):
    """Raised when embedding dimension does not match expected 768-dim."""
    pass


class ClusteringService:
    """
    Semantic clustering service using HDBSCAN.
    
    HDBSCAN (Hierarchical Density-Based Spatial Clustering of Applications
    with Noise) automatically determines the number of clusters based on
    density. Keywords that don't fit any cluster are marked as noise (-1).
    
    UMAP is used for dimensionality reduction:
    - 768D -> 15D for clustering (6-7x speedup)
    - 768D -> 2D for visualization
    
    CRITICAL: low_memory=True is REQUIRED for 24GB VPS with 100 concurrent prospects.
    """
    
    def __init__(self, config: Optional[ClusterConfig] = None):
        self.config = {**DEFAULT_CONFIG, **(config or {})}
    
    def _validate_embeddings(self, embeddings: list[list[float]]) -> None:
        """
        Validate embedding dimensions.
        
        Args:
            embeddings: List of embedding vectors
            
        Raises:
            EmbeddingDimensionError: If any embedding is not 768-dim
        """
        for i, emb in enumerate(embeddings):
            if len(emb) != EXPECTED_EMBEDDING_DIM:
                raise EmbeddingDimensionError(
                    f"Embedding at index {i} has {len(emb)} dimensions, "
                    f"expected {EXPECTED_EMBEDDING_DIM} (jina-v5-text-nano). "
                    f"Check that embeddings are from Phase 78 RelevanceScorer."
                )
    
    def cluster_keywords(
        self,
        embeddings: list[list[float]],
        config: Optional[ClusterConfig] = None
    ) -> ClusterResult:
        """
        Cluster keyword embeddings using HDBSCAN.
        
        Args:
            embeddings: List of 768-dim embedding vectors
            config: Optional config override
            
        Returns:
            ClusterResult with labels, centroids, and visualization coords
            
        Raises:
            EmbeddingDimensionError: If embeddings are not 768-dim
        """
        cfg = {**self.config, **(config or {})}
        
        if len(embeddings) == 0:
            return {
                "labels": [],
                "centroids": [],
                "vis_coords": [],
                "cluster_count": 0,
                "noise_count": 0,
            }
        
        # CRITICAL: Validate embedding dimensions before processing
        self._validate_embeddings(embeddings)
        
        # Convert to numpy array
        X = np.array(embeddings, dtype=np.float32)
        n_samples = X.shape[0]
        
        # Handle edge case: too few samples for UMAP/HDBSCAN
        if n_samples < cfg["min_cluster_size"]:
            return {
                "labels": [-1] * n_samples,  # All noise
                "centroids": [],
                "vis_coords": [{"x": 0.0, "y": 0.0} for _ in range(n_samples)],
                "cluster_count": 0,
                "noise_count": n_samples,
            }
        
        # Step 1: UMAP reduction for clustering (768D -> 15D)
        n_neighbors = min(15, n_samples - 1)
        umap_cluster = umap.UMAP(
            n_components=cfg["umap_dimensions"],
            n_neighbors=n_neighbors,
            min_dist=0.0,  # Tighter clustering
            metric="cosine",
            random_state=42,
        )
        X_reduced = umap_cluster.fit_transform(X)
        
        # Step 2: HDBSCAN clustering (using fast-hdbscan for memory efficiency)
        # CRITICAL: low_memory=True is REQUIRED for VPS with 24GB RAM
        clusterer = fast_hdbscan.HDBSCAN(
            min_cluster_size=cfg["min_cluster_size"],
            min_samples=cfg["min_samples"],
            low_memory=True,  # REQUIRED for large datasets on 24GB VPS
        )
        labels = clusterer.fit_predict(X_reduced)
        
        # Step 3: UMAP reduction for visualization (768D -> 2D)
        umap_vis = umap.UMAP(
            n_components=cfg["umap_vis_dimensions"],
            n_neighbors=n_neighbors,
            min_dist=0.1,  # Slight spread for visualization
            metric="cosine",
            random_state=42,
        )
        X_vis = umap_vis.fit_transform(X)
        
        # Step 4: Calculate centroids (in original 768D space)
        unique_labels = set(labels)
        unique_labels.discard(-1)  # Remove noise label
        centroids = []
        
        for label in sorted(unique_labels):
            mask = labels == label
            centroid = X[mask].mean(axis=0).tolist()
            centroids.append(centroid)
        
        # Build result
        vis_coords = [
            {"x": float(X_vis[i, 0]), "y": float(X_vis[i, 1])}
            for i in range(n_samples)
        ]
        
        noise_count = int(np.sum(labels == -1))
        cluster_count = len(centroids)
        
        return {
            "labels": labels.tolist(),
            "centroids": centroids,
            "vis_coords": vis_coords,
            "cluster_count": cluster_count,
            "noise_count": noise_count,
        }


def cluster_keywords(
    embeddings: list[list[float]],
    config: Optional[ClusterConfig] = None
) -> ClusterResult:
    """
    Factory function for quick clustering.
    
    Args:
        embeddings: List of 768-dim embedding vectors
        config: Optional configuration
        
    Returns:
        ClusterResult
        
    Raises:
        EmbeddingDimensionError: If embeddings are not 768-dim
    """
    service = ClusteringService(config)
    return service.cluster_keywords(embeddings)

"""
Clustering API Endpoints
Phase 86-02: HDBSCAN + UMAP Clustering

FastAPI router for keyword clustering requests from open-seo-main.
"""

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel, Field
from typing import Optional
import logging

from services.clustering_service import (
    ClusteringService,
    ClusterConfig,
    ClusterResult,
    EmbeddingDimensionError,
    EXPECTED_EMBEDDING_DIM,
)


logger = logging.getLogger(__name__)
router = APIRouter(prefix="/clustering", tags=["clustering"])


class ClusterRequest(BaseModel):
    """Request body for clustering endpoint."""
    
    embeddings: list[list[float]] = Field(
        ...,
        description=f"List of {EXPECTED_EMBEDDING_DIM}-dim embedding vectors (jina-v5-text-nano)",
        min_length=0,
        max_length=10000,
    )
    min_cluster_size: int = Field(
        default=3,
        ge=2,
        le=100,
        description="Minimum points to form a cluster",
    )
    min_samples: int = Field(
        default=2,
        ge=1,
        le=50,
        description="Minimum samples for core point",
    )
    umap_dimensions: int = Field(
        default=15,
        ge=2,
        le=50,
        description="UMAP target dimensions for clustering (768D -> 15D)",
    )
    umap_vis_dimensions: int = Field(
        default=2,
        ge=2,
        le=3,
        description="UMAP target dimensions for visualization (768D -> 2D)",
    )
    
    class Config:
        json_schema_extra = {
            "example": {
                "embeddings": [[0.1] * 768, [0.2] * 768, [0.3] * 768],
                "min_cluster_size": 3,
                "min_samples": 2,
            }
        }


class ClusterResponse(BaseModel):
    """Response from clustering endpoint."""
    
    labels: list[int] = Field(
        ...,
        description="Cluster label per keyword (-1 = noise)",
    )
    centroids: list[list[float]] = Field(
        ...,
        description="Centroid embedding per cluster (768-dim)",
    )
    vis_coords: list[dict[str, float]] = Field(
        ...,
        description="2D visualization coordinates per keyword",
    )
    cluster_count: int = Field(
        ...,
        description="Number of clusters found",
    )
    noise_count: int = Field(
        ...,
        description="Number of noise points (cluster=-1)",
    )


# Singleton service instance
_clustering_service: Optional[ClusteringService] = None


def get_clustering_service() -> ClusteringService:
    """Get or create clustering service instance."""
    global _clustering_service
    if _clustering_service is None:
        _clustering_service = ClusteringService()
    return _clustering_service


@router.post(
    "/cluster",
    response_model=ClusterResponse,
    summary="Cluster keyword embeddings",
    description=f"""
    Cluster keyword embeddings using HDBSCAN with UMAP dimensionality reduction.
    
    **CRITICAL:** All embeddings MUST be {EXPECTED_EMBEDDING_DIM}-dim (jina-v5-text-nano from Phase 78).
    Embeddings with wrong dimensions will be rejected with HTTP 400.
    
    HDBSCAN automatically determines the optimal number of clusters based on
    density. Keywords that don't fit any cluster are marked as noise (label=-1).
    
    Returns cluster labels, centroids, and 2D visualization coordinates.
    """,
)
async def cluster_embeddings(request: ClusterRequest) -> ClusterResponse:
    """Cluster keyword embeddings using HDBSCAN."""
    
    try:
        # Build config from request
        config: ClusterConfig = {
            "min_cluster_size": request.min_cluster_size,
            "min_samples": request.min_samples,
            "umap_dimensions": request.umap_dimensions,
            "umap_vis_dimensions": request.umap_vis_dimensions,
        }
        
        # Run clustering (includes dimension validation)
        service = get_clustering_service()
        result: ClusterResult = service.cluster_keywords(
            embeddings=request.embeddings,
            config=config,
        )
        
        logger.info(
            f"Clustered {len(request.embeddings)} embeddings into "
            f"{result['cluster_count']} clusters, {result['noise_count']} noise"
        )
        
        return ClusterResponse(
            labels=result["labels"],
            centroids=result["centroids"],
            vis_coords=result["vis_coords"],
            cluster_count=result["cluster_count"],
            noise_count=result["noise_count"],
        )
        
    except EmbeddingDimensionError as e:
        # Clear error for wrong embedding dimensions
        logger.warning(f"Embedding dimension error: {e}")
        raise HTTPException(
            status_code=400,
            detail=str(e),
        )
    except HTTPException:
        raise
    except Exception as e:
        logger.exception("Clustering failed")
        raise HTTPException(
            status_code=500,
            detail=f"Clustering failed: {str(e)}",
        )

/**
 * HDBSCAN Clusterer
 * Phase 86-02: Semantic Intelligence Pipeline
 * 
 * TypeScript wrapper for Python HDBSCAN clustering service.
 * Calls AI-Writer microservice and maps results to TypeScript types.
 * 
 * CRITICAL: Uses 60s timeout via AbortSignal.timeout() to prevent hanging.
 */

import type {
  ClusteringInput,
  ClusteringConfig,
  ClusteringResult,
  ClusteringStats,
  KeywordCluster,
} from './types';
import { DEFAULT_CLUSTERING_CONFIG } from './types';

/**
 * Python API response shape.
 */
interface PythonClusterResponse {
  labels: number[];
  centroids: number[][];
  vis_coords: Array<{ x: number; y: number }>;
  cluster_count: number;
  noise_count: number;
}

/**
 * HTTP timeout for clustering requests (60 seconds).
 * Clustering 2000 keywords with UMAP + HDBSCAN can take 30-45s.
 */
const CLUSTERING_TIMEOUT_MS = 60000;

/**
 * HDBSCAN Clusterer using Python microservice.
 */
export class HDBSCANClusterer {
  private config: ClusteringConfig;
  private apiUrl: string;

  constructor(
    config: Partial<ClusteringConfig> = {},
    apiUrl?: string
  ) {
    this.config = { ...DEFAULT_CLUSTERING_CONFIG, ...config };
    this.apiUrl = apiUrl || process.env.AI_WRITER_URL || 'http://localhost:8000';
  }

  /**
   * Cluster keywords using HDBSCAN via Python API.
   */
  async clusterKeywords(inputs: ClusteringInput[]): Promise<ClusteringResult> {
    const startTime = Date.now();
    
    // Handle empty input
    if (inputs.length === 0) {
      return {
        clusters: [],
        noise: [],
        stats: {
          inputCount: 0,
          clusterCount: 0,
          noiseCount: 0,
          avgClusterSize: 0,
          processingTimeMs: 0,
        },
      };
    }

    // Extract embeddings for Python API
    const embeddings = inputs.map(i => i.embedding);
    
    // Call Python clustering service
    const response = await this.callPythonApi(embeddings);
    
    // Map results back to TypeScript types
    const result = this.mapResponse(inputs, response);
    
    // Calculate stats
    const processingTimeMs = Date.now() - startTime;
    result.stats = {
      inputCount: inputs.length,
      clusterCount: response.cluster_count,
      noiseCount: response.noise_count,
      avgClusterSize: response.cluster_count > 0
        ? (inputs.length - response.noise_count) / response.cluster_count
        : 0,
      processingTimeMs,
    };

    return result;
  }

  /**
   * Call Python clustering API.
   * 
   * CRITICAL: Uses AbortSignal.timeout(60000) to prevent hanging requests.
   * Clustering can take 30-45s for large datasets.
   */
  private async callPythonApi(
    embeddings: number[][]
  ): Promise<PythonClusterResponse> {
    const url = `${this.apiUrl}/api/clustering/cluster`;
    
    const response = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        embeddings,
        min_cluster_size: this.config.minClusterSize,
        min_samples: this.config.minSamples,
        umap_dimensions: this.config.umapDimensions,
        umap_vis_dimensions: this.config.umapVisDimensions,
      }),
      // CRITICAL: 60s timeout to prevent hanging requests
      signal: AbortSignal.timeout(CLUSTERING_TIMEOUT_MS),
    });

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`Clustering API error: ${response.status} - ${error}`);
    }

    return response.json();
  }

  /**
   * Map Python response to TypeScript ClusteringResult.
   */
  private mapResponse(
    inputs: ClusteringInput[],
    response: PythonClusterResponse
  ): ClusteringResult {
    // Group inputs by cluster label
    const clusterGroups = new Map<number, ClusteringInput[]>();
    const noise: ClusteringInput[] = [];
    
    for (let i = 0; i < inputs.length; i++) {
      const label = response.labels[i];
      const input = {
        ...inputs[i],
        visCoords: response.vis_coords[i],
      } as ClusteringInput & { visCoords: { x: number; y: number } };
      
      if (label === -1) {
        noise.push(input);
      } else {
        if (!clusterGroups.has(label)) {
          clusterGroups.set(label, []);
        }
        clusterGroups.get(label)!.push(input);
      }
    }

    // Build cluster objects
    const clusters: KeywordCluster[] = [];
    let centroidIndex = 0;
    
    for (const [clusterId, keywords] of clusterGroups) {
      const cluster = this.buildCluster(
        clusterId,
        keywords,
        response.centroids[centroidIndex]
      );
      clusters.push(cluster);
      centroidIndex++;
    }

    return {
      clusters,
      noise,
      stats: {} as ClusteringStats, // Filled by caller
    };
  }

  /**
   * Build KeywordCluster from grouped keywords.
   */
  private buildCluster(
    clusterId: number,
    keywords: ClusteringInput[],
    centroid: number[]
  ): KeywordCluster {
    // Calculate aggregates
    const totalVolume = keywords.reduce((sum, k) => sum + k.volume, 0);
    const averageDifficulty = keywords.reduce((sum, k) => sum + k.difficulty, 0) / keywords.length;
    
    // Calculate funnel breakdown
    const funnelBreakdown = { bofu: 0, mofu: 0, tofu: 0 };
    for (const k of keywords) {
      funnelBreakdown[k.funnelStage]++;
    }
    
    // Determine dominant funnel
    const dominantFunnel = Object.entries(funnelBreakdown)
      .sort(([, a], [, b]) => b - a)[0][0] as 'bofu' | 'mofu' | 'tofu';

    return {
      clusterId,
      keywords,
      centroid,
      totalVolume,
      averageDifficulty,
      dominantFunnel,
      funnelBreakdown,
    };
  }
}

/**
 * Factory function for quick clustering.
 */
export async function clusterKeywords(
  inputs: ClusteringInput[],
  config?: Partial<ClusteringConfig>,
  apiUrl?: string
): Promise<ClusteringResult> {
  const clusterer = new HDBSCANClusterer(config, apiUrl);
  return clusterer.clusterKeywords(inputs);
}
