```python
from google_auth_oauthlib.flow import Flow

def get_oauth_url(self, user_id: str) -> str:
    flow = Flow.from_client_config(
        self.client_config,
        scopes=self.scopes,
        redirect_uri=redirect_uri
    )
    random_state = secrets.token_urlsafe(32)
    state = f"{user_id}:{random_state}"
    authorization_url, _ = flow.authorization_url(
        access_type='offline',
        include_granted_scopes='true',
        prompt='consent',
        state=state
    )
    return authorization_url
```

| T-12-05 | Tampering | Invite token replay | mitigate | completed_at set on first use; subsequent callbacks rejected |
| T-12-06 | Information Disclosure | Access token in response | mitigate | ConnectionResponse schema excludes encrypted fields; write-only pattern |
| T-12-07 | Spoofing | Invite token brute force | accept | 256-bit entropy (2^256 guesses); rate limiting at nginx layer |
