```python
#!/usr/bin/env python3
"""
Migrate per-user GSC/Bing credentials from SQLite to per-client PostgreSQL.

Usage:
    python scripts/migrate_credentials.py --dry-run    # Preview only
    python scripts/migrate_credentials.py              # Execute migration
    python scripts/migrate_credentials.py --verbose    # With detailed logging

The script:
1. Scans USER_DATA_DIR for user SQLite databases
2. For each user with gsc_credentials:
   a. Finds their most recently active client (by scheduled_articles.created_at or client.updated_at)
   b. Re-encrypts tokens with Fernet
   c. Inserts into client_oauth_tokens with provider='google'
3. Logs migration status for each user

Idempotency: Skips users whose credentials already exist in client_oauth_tokens
(checked by UNIQUE(client_id, provider) constraint).
"""

import argparse
import json
import os
import sqlite3
import sys
from datetime import datetime
from pathlib import Path

# Add backend to path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from loguru import logger
from sqlalchemy.orm import Session
from sqlalchemy.exc import IntegrityError

from services.database import get_user_db_path, USER_DATA_DIR
from services.shared_db import get_shared_db_session
from services.encryption import encrypt_value
from models.client import Client
from models.client_oauth import ClientOAuthToken

def get_user_gsc_credentials(user_id: str) -> dict | None:
    """Load GSC credentials from user's SQLite database."""

def find_most_recent_client(db: Session, user_id: str) -> Client | None:
    """Find the client most recently active for this user."""

def migrate_user_credentials(
    db: Session,
    user_id: str,
    client: Client,
    credentials: dict,
    dry_run: bool = False
) -> str:
    """Migrate one user's credentials. Returns status: 'success', 'skipped', 'failed'."""

def migrate_all_credentials(dry_run: bool = False, verbose: bool = False) -> dict:
    """Main migration function. Returns {migrated: int, skipped: int, failed: int}."""

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Migrate credentials to per-client storage")
    parser.add_argument("--dry-run", action="store_true", help="Preview without writing")
    parser.add_argument("--verbose", action="store_true", help="Detailed logging")
    args = parser.parse_args()
    
    results = migrate_all_credentials(dry_run=args.dry_run, verbose=args.verbose)
    print(f"Migration complete: {results}")
```

Key logic:
- Scan USER_DATA_DIR for *.db files (each is a user's SQLite)
- For each user: check gsc_credentials table exists and has rows
- Find client by joining scheduled_articles or using client.updated_at
- Skip if ClientOAuthToken already exists for (client_id, 'google') - idempotency
- Re-encrypt: extract token/refresh_token from JSON, encrypt with Fernet
- Insert with connected_by='migration:$USER_ID' for audit trail
