```python
async def get_categories(self, site_id: str) -> List[Dict]:
    """Fetch real categories from Wix Blog API."""
    try:
        access_token = await self._get_access_token(site_id)
        
        async with httpx.AsyncClient(timeout=30.0) as client:
            response = await client.get(
                "https://www.wixapis.com/blog/v3/categories",
                headers={
                    "Authorization": access_token,
                    "wix-site-id": site_id
                }
            )
            response.raise_for_status()
            data = response.json()
            
            return [
                {
                    "id": cat["id"],
                    "name": cat.get("label", cat.get("title", "Untitled")),
                    "slug": cat.get("slug", cat["id"]),
                    "description": cat.get("description", "")
                }
                for cat in data.get("categories", [])
            ]
    except httpx.HTTPStatusError as e:
        logger.error(f"[wix] Failed to fetch categories: {e}")
        raise WixApiError(f"Failed to fetch categories: {e.response.status_code}")
    except Exception as e:
        logger.error(f"[wix] Unexpected error fetching categories: {e}")
        raise WixApiError(f"Failed to fetch categories: {str(e)}")
```

```typescript
'use server';

import { postFastApi } from '@/lib/server-fetch';

interface TestConnectionParams {
  platform: 'wordpress' | 'shopify' | 'wix' | 'webhook';
  credentials: Record<string, string>;
}

interface TestConnectionResult {
  success: boolean;
  message?: string;
  error?: string;
}

export async function testCmsConnection(
  clientId: string,
  params: TestConnectionParams
): Promise<TestConnectionResult> {
  try {
    const result = await postFastApi<TestConnectionResult>(
      `/api/clients/${clientId}/test-connection`,
      params
    );
    return result;
  } catch (error) {
    return {
      success: false,
      error: error instanceof Error ? error.message : 'Connection test failed'
    };
  }
}
```

```python
@router.post("/api/clients/{client_id}/test-connection")
async def test_cms_connection(
    client_id: str,
    params: TestConnectionParams,
    current_user: User = Depends(get_current_user)
):
    """Test CMS connection with provided credentials."""
    platform = params.platform
    credentials = params.credentials
    
    try:
        if platform == "wordpress":
            return await _test_wordpress_connection(credentials)
        elif platform == "shopify":
            return await _test_shopify_connection(credentials)
        elif platform == "wix":
            return await _test_wix_connection(credentials)
        elif platform == "webhook":
            return await _test_webhook_connection(credentials)
        else:
            return {"success": False, "error": f"Unknown platform: {platform}"}
    except Exception as e:
        logger.error(f"Connection test failed: {e}")
        return {"success": False, "error": str(e)}

async def _test_wordpress_connection(credentials: Dict) -> Dict:
    """Test WordPress REST API connection."""
    url = credentials.get("wp_url", "").rstrip("/")
    username = credentials.get("wp_username")
    app_password = credentials.get("wp_app_password")
    
    if not all([url, username, app_password]):
        return {"success": False, "error": "Missing WordPress credentials"}
    
    async with httpx.AsyncClient(timeout=10.0) as client:
        response = await client.get(
            f"{url}/wp-json/wp/v2/users/me",
            auth=(username, app_password)
        )
        
        if response.status_code == 200:
            user_data = response.json()
            return {
                "success": True,
                "message": f"Connected as {user_data.get('name', username)}"
            }
        else:
            return {
                "success": False,
                "error": f"WordPress returned {response.status_code}"
            }

async def _test_shopify_connection(credentials: Dict) -> Dict:
    """Test Shopify Admin API connection."""
    store_url = credentials.get("shopify_store_url", "").rstrip("/")
    access_token = credentials.get("shopify_access_token")
    
    if not all([store_url, access_token]):
        return {"success": False, "error": "Missing Shopify credentials"}
    
    async with httpx.AsyncClient(timeout=10.0) as client:
        response = await client.get(
            f"{store_url}/admin/api/2024-01/shop.json",
            headers={"X-Shopify-Access-Token": access_token}
        )
        
        if response.status_code == 200:
            shop_data = response.json().get("shop", {})
            return {
                "success": True,
                "message": f"Connected to {shop_data.get('name', store_url)}"
            }
        else:
            return {
                "success": False,
                "error": f"Shopify returned {response.status_code}"
            }

async def _test_wix_connection(credentials: Dict) -> Dict:
    """Test Wix API connection."""
    site_id = credentials.get("wix_site_id")
    access_token = credentials.get("wix_access_token")
    
    if not all([site_id, access_token]):
        return {"success": False, "error": "Missing Wix credentials"}
    
    async with httpx.AsyncClient(timeout=10.0) as client:
        response = await client.get(
            "https://www.wixapis.com/site-properties/v4/site",
            headers={
                "Authorization": access_token,
                "wix-site-id": site_id
            }
        )
        
        if response.status_code == 200:
            site_data = response.json()
            return {
                "success": True,
                "message": f"Connected to {site_data.get('displayName', site_id)}"
            }
        else:
            return {
                "success": False,
                "error": f"Wix returned {response.status_code}"
            }

async def _test_webhook_connection(credentials: Dict) -> Dict:
    """Test webhook endpoint with a ping."""
    webhook_url = credentials.get("webhook_url")
    
    if not webhook_url:
        return {"success": False, "error": "Missing webhook URL"}
    
    async with httpx.AsyncClient(timeout=10.0) as client:
        response = await client.post(
            webhook_url,
            json={"type": "ping", "timestamp": datetime.utcnow().isoformat()}
        )
        
        if response.status_code in (200, 201, 202, 204):
            return {"success": True, "message": "Webhook responded successfully"}
        else:
            return {
                "success": False,
                "error": f"Webhook returned {response.status_code}"
            }
```

```typescript
export async function getTopOpportunities(
  workspaceId: string,
  limit = 10
): Promise<Opportunity[]> {
  // Get all clients in workspace
  const clients = await getOpenSeo<{ id: string; name: string }[]>(
    `/api/workspaces/${workspaceId}/clients`
  );
  
  // Aggregate opportunities from all clients
  const allOpportunities: Opportunity[] = [];
  
  await Promise.all(
    clients.map(async (client) => {
      try {
        const clientOpportunities = await getClientOpportunities(client.id, 20);
        allOpportunities.push(
          ...clientOpportunities.map(opp => ({
            ...opp,
            clientId: client.id,
            clientName: client.name
          }))
        );
      } catch (error) {
        console.warn(`Failed to fetch opportunities for client ${client.id}:`, error);
      }
    })
  );
  
  // Sort by potential impact and return top N
  return allOpportunities
    .sort((a, b) => (b.potentialClicks ?? 0) - (a.potentialClicks ?? 0))
    .slice(0, limit);
}
```

### Task 5: Add workspace clients endpoint
**File:** `open-seo-main/src/routes/api/workspaces/[workspaceId]/clients.ts`

```typescript
import { createAPIFileRoute } from '@tanstack/start/api';
import { db } from '~/db';
import { clients } from '~/db/client-schema';
import { eq } from 'drizzle-orm';

export const Route = createAPIFileRoute('/api/workspaces/$workspaceId/clients')({
  GET: async ({ params }) => {
    const { workspaceId } = params;
    
    const workspaceClients = await db.query.clients.findMany({
      where: eq(clients.workspaceId, workspaceId),
      columns: {
        id: true,
        name: true,
        domain: true,
        createdAt: true
      },
      orderBy: (clients, { desc }) => [desc(clients.createdAt)]
    });
    
    return Response.json(workspaceClients);
  }
});
```
