### Task 1: Implement opportunity detection from GSC
**File:** `AI-Writer/backend/services/intelligence/autonomous_pipeline.py`

```python
from services.gsc_service import GSCService
from services.client_context import get_client_context

class AutonomousPipeline:
    def __init__(self):
        self.gsc_service = GSCService()
    
    async def detect_opportunities(self, client_id: str) -> List[Dict[str, Any]]:
        """Identify keywords with high impressions but low clicks (CTR opportunities)."""
        context = await get_client_context(client_id)
        
        # Get last 28 days of GSC data
        analytics = await self.gsc_service.get_search_analytics(
            site_url=context.site_url,
            start_date=(datetime.now() - timedelta(days=28)).isoformat(),
            end_date=datetime.now().isoformat(),
            dimensions=['query'],
            row_limit=100
        )
        
        opportunities = []
        for row in analytics.get('rows', []):
            impressions = row.get('impressions', 0)
            clicks = row.get('clicks', 0)
            position = row.get('position', 100)
            
            # Opportunity: High impressions, low CTR, rankable position
            ctr = clicks / impressions if impressions > 0 else 0
            expected_ctr = self._expected_ctr_for_position(position)
            
            if impressions > 100 and ctr < expected_ctr * 0.5 and position <= 20:
                opportunities.append({
                    'keyword': row['keys'][0],
                    'impressions': impressions,
                    'clicks': clicks,
                    'position': position,
                    'ctr': ctr,
                    'expected_ctr': expected_ctr,
                    'opportunity_type': 'ctr_improvement',
                    'priority': self._calculate_priority(impressions, position, ctr, expected_ctr)
                })
        
        return sorted(opportunities, key=lambda x: x['priority'], reverse=True)[:10]
    
    def _expected_ctr_for_position(self, position: int) -> float:
        """Industry standard CTR by position."""
        ctr_curve = {
            1: 0.28, 2: 0.15, 3: 0.11, 4: 0.08, 5: 0.06,
            6: 0.05, 7: 0.04, 8: 0.03, 9: 0.03, 10: 0.02
        }
        return ctr_curve.get(int(position), 0.01)
    
    def _calculate_priority(self, impressions: int, position: float, ctr: float, expected_ctr: float) -> float:
        """Higher priority = more impressions + closer to page 1 + bigger CTR gap."""
        impression_score = min(impressions / 1000, 10)
        position_score = max(0, 10 - position) / 2
        ctr_gap_score = (expected_ctr - ctr) * 100
        return impression_score + position_score + ctr_gap_score
```

### Task 2: Implement brief creation from opportunity
**File:** `AI-Writer/backend/services/intelligence/autonomous_pipeline.py`

```python
import httpx
import os

OPEN_SEO_API_URL = os.getenv("OPEN_SEO_API_URL", "http://localhost:3001")

async def create_brief_for_opportunity(
    self, 
    client_id: str, 
    opportunity: Dict[str, Any]
) -> Dict[str, Any]:
    """Create content brief via open-seo API."""
    async with httpx.AsyncClient(timeout=60.0) as client:
        response = await client.post(
            f"{OPEN_SEO_API_URL}/api/seo/briefs",
            json={
                "clientId": client_id,
                "keyword": opportunity['keyword'],
                "targetUrl": None,  # New content
                "action": "create",
                "voiceMode": "adaptive",
                "includeSerp": True,
                "includePaa": True
            },
            headers={"X-Internal-Api-Key": os.getenv("INTERNAL_API_KEY")}
        )
        response.raise_for_status()
        return response.json()
```

```python
async def validate_article(self, article_id: str) -> Dict[str, Any]:
    """Check article quality via open-seo quality gate."""
    # Get article content
    article = await self._get_article(article_id)
    
    async with httpx.AsyncClient(timeout=30.0) as client:
        response = await client.post(
            f"{OPEN_SEO_API_URL}/api/seo/content/validate",
            json={
                "clientId": article['client_id'],
                "content": article['content_html'],
                "keyword": article['keyword'],
                "voiceProfileId": article.get('voice_profile_id')
            },
            headers={"X-Internal-Api-Key": os.getenv("INTERNAL_API_KEY")}
        )
        response.raise_for_status()
        result = response.json()
        
        return {
            'passed': result.get('score', 0) >= 80,
            'score': result.get('score', 0),
            'issues': result.get('issues', [])
        }
```

### Task 5: Implement full autonomous cycle
**File:** `AI-Writer/backend/services/intelligence/autonomous_pipeline.py`

```python
async def run_autonomous_cycle(self, client_id: str) -> Dict[str, Any]:
    """Execute one full autonomous optimization cycle."""
    logger.info(f"[autonomous] Starting cycle for client {client_id}")
    
    actions_taken = []
    errors = []
    
    try:
        # Step 1: Detect opportunities
        opportunities = await self.detect_opportunities(client_id)
        logger.info(f"[autonomous] Found {len(opportunities)} opportunities")
        
        if not opportunities:
            return {
                "status": "complete",
                "actions_taken": [],
                "message": "No opportunities detected"
            }
        
        # Step 2: Process top opportunity
        top_opportunity = opportunities[0]
        
        # Step 3: Create brief
        brief = await self.create_brief_for_opportunity(client_id, top_opportunity)
        actions_taken.append({
            "action": "brief_created",
            "keyword": top_opportunity['keyword'],
            "brief_id": brief['id']
        })
        
        # Step 4: Generate article
        article = await self.generate_article_from_brief(client_id, brief)
        actions_taken.append({
            "action": "article_generated",
            "article_id": article['id'],
            "title": article['title']
        })
        
        # Step 5: Validate quality
        validation = await self.validate_article(article['id'])
        actions_taken.append({
            "action": "quality_validated",
            "passed": validation['passed'],
            "score": validation['score']
        })
        
        if not validation['passed']:
            return {
                "status": "pending_review",
                "actions_taken": actions_taken,
                "message": f"Article quality score {validation['score']} < 80, needs review"
            }
        
        # Step 6: Auto-approve for publishing (will be picked up by auto_publish_executor)
        await self._approve_article(article['id'])
        actions_taken.append({
            "action": "article_approved",
            "article_id": article['id']
        })
        
        return {
            "status": "complete",
            "actions_taken": actions_taken,
            "message": f"Successfully processed opportunity: {top_opportunity['keyword']}"
        }
        
    except Exception as e:
        logger.error(f"[autonomous] Cycle failed: {e}")
        return {
            "status": "error",
            "actions_taken": actions_taken,
            "error": str(e)
        }
```

```python
from services.intelligence.autonomous_pipeline import AutonomousPipeline

# Add to scheduler initialization
scheduler.add_job(
    run_autonomous_cycles,
    CronTrigger(hour=3, minute=0),  # Run at 3 AM daily
    id='autonomous_seo_cycle',
    name='Autonomous SEO Optimization Cycle',
    replace_existing=True
)

async def run_autonomous_cycles():
    """Run autonomous cycle for all clients with auto_optimize enabled."""
    pipeline = AutonomousPipeline()
    
    # Get clients with auto-optimization enabled
    clients = await get_auto_optimize_clients()
    
    for client in clients:
        try:
            result = await pipeline.run_autonomous_cycle(client['id'])
            logger.info(f"[autonomous] Client {client['id']}: {result['status']}")
        except Exception as e:
            logger.error(f"[autonomous] Client {client['id']} failed: {e}")
        
        # Rate limit between clients
        await asyncio.sleep(5)
```
