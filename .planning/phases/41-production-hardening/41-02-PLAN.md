### Task 1: Create GSC aggregate endpoint in open-seo
**File:** `open-seo-main/src/routes/api/workspaces/[workspaceId]/traffic-data.ts`

```typescript
import { createAPIFileRoute } from '@tanstack/start/api';
import { db } from '~/db';
import { gscSnapshots } from '~/db/analytics-schema';
import { eq, desc, and, gte } from 'drizzle-orm';

export const Route = createAPIFileRoute('/api/workspaces/$workspaceId/traffic-data')({
  GET: async ({ request, params }) => {
    const { workspaceId } = params;
    
    // Get all clients in workspace
    const clients = await db.query.clients.findMany({
      where: eq(clients.workspaceId, workspaceId),
      columns: { id: true, name: true }
    });
    
    // Get last 4 weeks of GSC data per client
    const fourWeeksAgo = new Date();
    fourWeeksAgo.setDate(fourWeeksAgo.getDate() - 28);
    
    const trafficData = await Promise.all(
      clients.map(async (client) => {
        const snapshots = await db.query.gscSnapshots.findMany({
          where: and(
            eq(gscSnapshots.clientId, client.id),
            gte(gscSnapshots.date, fourWeeksAgo)
          ),
          orderBy: desc(gscSnapshots.date),
          limit: 28
        });
        
        // Aggregate by week
        const weeklyClicks = aggregateByWeek(snapshots);
        const trend = calculateTrend(weeklyClicks);
        
        return {
          clientId: client.id,
          clientName: client.name,
          weeklyClicks,
          trend,
          status: trend < -20 ? 'dropped' : trend > 10 ? 'growing' : 'stable'
        };
      })
    );
    
    return Response.json(trafficData);
  }
});

function aggregateByWeek(snapshots: typeof gscSnapshots.$inferSelect[]): number[] {
  // Group by week and sum clicks
  const weeks: number[] = [0, 0, 0, 0];
  snapshots.forEach((s, i) => {
    const weekIndex = Math.floor(i / 7);
    if (weekIndex < 4) weeks[weekIndex] += s.clicks;
  });
  return weeks;
}

function calculateTrend(weeklyClicks: number[]): number {
  if (weeklyClicks.length < 2) return 0;
  const latest = weeklyClicks[0];
  const previous = weeklyClicks[1];
  if (previous === 0) return 0;
  return Math.round(((latest - previous) / previous) * 100);
}
```

### Task 2: Create ranking data endpoint in open-seo
**File:** `open-seo-main/src/routes/api/workspaces/[workspaceId]/ranking-data.ts`

```typescript
import { createAPIFileRoute } from '@tanstack/start/api';
import { db } from '~/db';
import { keywordRankings, savedKeywords } from '~/db/schema';
import { eq, desc, and, gte } from 'drizzle-orm';

export const Route = createAPIFileRoute('/api/workspaces/$workspaceId/ranking-data')({
  GET: async ({ request, params }) => {
    const { workspaceId } = params;
    
    // Get clients with tracked keywords
    const clients = await db.query.clients.findMany({
      where: eq(clients.workspaceId, workspaceId)
    });
    
    const rankingData = await Promise.all(
      clients.map(async (client) => {
        // Get keywords for this client's projects
        const keywords = await db.query.savedKeywords.findMany({
          where: and(
            eq(savedKeywords.projectId, client.projectId),
            eq(savedKeywords.trackingEnabled, true)
          ),
          limit: 100
        });
        
        // Get latest rankings
        const rankings = await Promise.all(
          keywords.slice(0, 10).map(async (kw) => {
            const latest = await db.query.keywordRankings.findFirst({
              where: eq(keywordRankings.keywordId, kw.id),
              orderBy: desc(keywordRankings.date)
            });
            const previous = await db.query.keywordRankings.findFirst({
              where: eq(keywordRankings.keywordId, kw.id),
              orderBy: desc(keywordRankings.date),
              offset: 7 // Week ago
            });
            
            return {
              keyword: kw.keyword,
              currentPosition: latest?.position ?? null,
              previousPosition: previous?.position ?? null,
              change: latest && previous ? previous.position - latest.position : 0
            };
          })
        );
        
        return {
          clientId: client.id,
          clientName: client.name,
          topKeywords: rankings,
          improvedCount: rankings.filter(r => r.change > 0).length,
          droppedCount: rankings.filter(r => r.change < 0).length
        };
      })
    );
    
    return Response.json(rankingData);
  }
});
```

```typescript
export async function detectPatterns(workspaceId: string): Promise<DetectedPattern[]> {
  // Fetch real data from open-seo
  const [trafficData, rankingData] = await Promise.all([
    getOpenSeo<ClientTrafficData[]>(`/api/workspaces/${workspaceId}/traffic-data`),
    getOpenSeo<ClientRankingData[]>(`/api/workspaces/${workspaceId}/ranking-data`)
  ]);
  
  const patterns: DetectedPattern[] = [];
  
  // Detect traffic drop patterns
  const droppedClients = trafficData.filter(c => c.status === 'dropped');
  if (droppedClients.length >= 3) {
    patterns.push({
      id: `traffic-drop-${Date.now()}`,
      type: 'traffic_drop',
      severity: droppedClients.length >= 5 ? 'critical' : 'warning',
      title: `Traffic drop detected across ${droppedClients.length} clients`,
      description: `Clients affected: ${droppedClients.map(c => c.clientName).join(', ')}`,
      affectedClients: droppedClients.map(c => c.clientId),
      detectedAt: new Date().toISOString()
    });
  }
  
  // Detect ranking improvement patterns
  const improvingClients = rankingData.filter(c => c.improvedCount > c.droppedCount);
  if (improvingClients.length >= 3) {
    patterns.push({
      id: `ranking-improvement-${Date.now()}`,
      type: 'ranking_improvement',
      severity: 'info',
      title: `Ranking improvements across ${improvingClients.length} clients`,
      description: `Positive momentum detected`,
      affectedClients: improvingClients.map(c => c.clientId),
      detectedAt: new Date().toISOString()
    });
  }
  
  return patterns;
}
```

### Task 5: Add pattern API endpoint
**File:** `open-seo-main/src/routes/api/patterns/[patternId].ts`

```typescript
import { createAPIFileRoute } from '@tanstack/start/api';
import { db } from '~/db';
import { detectedPatterns } from '~/db/pattern-schema';
import { eq } from 'drizzle-orm';

export const Route = createAPIFileRoute('/api/patterns/$patternId')({
  PATCH: async ({ request, params }) => {
    const { patternId } = params;
    const body = await request.json();
    
    await db.update(detectedPatterns)
      .set(body)
      .where(eq(detectedPatterns.id, patternId));
    
    return Response.json({ success: true });
  }
});
```
