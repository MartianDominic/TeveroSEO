```typescript
/**
 * Detected cross-client patterns.
 */
export const detectedPatterns = pgTable(
  "detected_patterns",
  {
    id: text("id").primaryKey(),
    workspaceId: text("workspace_id").notNull(),
    
    patternType: text("pattern_type").notNull(),  // 'traffic_drop', 'ranking_shift', 'industry_trend', 'serp_change'
    title: text("title").notNull(),
    description: text("description"),
    
    // Affected clients
    affectedClientIds: jsonb("affected_client_ids").$type<string[]>(),
    affectedCount: integer("affected_count").default(0),
    
    // Pattern details
    magnitude: numeric("magnitude"),  // Avg change %
    direction: text("direction"),     // 'up', 'down', 'volatile'
    confidence: numeric("confidence"), // 0-100 confidence score
    
    // Time range
    startDate: timestamp("start_date", { withTimezone: true }),
    endDate: timestamp("end_date", { withTimezone: true }),
    
    // Status
    status: text("status").default("active"),  // 'active', 'resolved', 'dismissed'
    resolvedAt: timestamp("resolved_at", { withTimezone: true }),
    
    detectedAt: timestamp("detected_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    index("idx_detected_patterns_workspace").on(table.workspaceId),
    index("idx_detected_patterns_type").on(table.patternType),
    index("idx_detected_patterns_status").on(table.status),
  ]
);

export type DetectedPatternSelect = typeof detectedPatterns.$inferSelect;
export type DetectedPatternInsert = typeof detectedPatterns.$inferInsert;
```

```typescript
import { db } from "@/db";
import { gscSnapshots, keywordRankings, clientDashboardMetrics, detectedPatterns } from "@/db/schema";
import { eq, and, gte, lte, sql } from "drizzle-orm";
import { subDays, startOfDay, endOfDay } from "date-fns";
import { nanoid } from "nanoid";

export async function detectPatterns(workspaceId: string): Promise<void> {
  const results = await Promise.all([
    detectTrafficPatterns(workspaceId),
    detectRankingPatterns(workspaceId),
    detectSERPChanges(workspaceId),
  ]);

  // Store detected patterns
  for (const pattern of results.flat()) {
    if (pattern.affectedCount >= 3 && pattern.confidence >= 70) {
      await db.insert(detectedPatterns).values({
        id: nanoid(),
        workspaceId,
        ...pattern,
      });
    }
  }
}

async function detectTrafficPatterns(workspaceId: string) {
  const patterns: Array<Partial<typeof detectedPatterns.$inferInsert>> = [];
  const today = new Date();
  const weekAgo = subDays(today, 7);
  const twoWeeksAgo = subDays(today, 14);

  // Get week-over-week traffic changes for all clients
  const trafficChanges = await db.execute(sql`
    WITH this_week AS (
      SELECT client_id, SUM(clicks) as clicks
      FROM gsc_snapshots
      WHERE workspace_id = ${workspaceId}
        AND date >= ${weekAgo}
        AND date <= ${today}
      GROUP BY client_id
    ),
    last_week AS (
      SELECT client_id, SUM(clicks) as clicks
      FROM gsc_snapshots
      WHERE workspace_id = ${workspaceId}
        AND date >= ${twoWeeksAgo}
        AND date < ${weekAgo}
      GROUP BY client_id
    )
    SELECT 
      tw.client_id,
      tw.clicks as this_week,
      lw.clicks as last_week,
      CASE WHEN lw.clicks > 0 
        THEN ((tw.clicks - lw.clicks) / lw.clicks * 100)
        ELSE 0 
      END as change_pct
    FROM this_week tw
    JOIN last_week lw ON tw.client_id = lw.client_id
    ORDER BY change_pct
  `);

  const rows = trafficChanges.rows as Array<{
    client_id: string;
    this_week: number;
    last_week: number;
    change_pct: number;
  }>;

  // Detect significant drops (> 20%) affecting multiple clients
  const droppedClients = rows.filter((r) => r.change_pct < -20);
  if (droppedClients.length >= 3) {
    const avgDrop = droppedClients.reduce((sum, r) => sum + r.change_pct, 0) / droppedClients.length;
    
    patterns.push({
      patternType: "traffic_drop",
      title: `Traffic Drop Affecting ${droppedClients.length} Clients`,
      description: `Multiple clients experienced significant traffic drops (avg ${avgDrop.toFixed(1)}%) this week. This may indicate an algorithm update or seasonal trend.`,
      affectedClientIds: droppedClients.map((r) => r.client_id),
      affectedCount: droppedClients.length,
      magnitude: String(Math.abs(avgDrop)),
      direction: "down",
      confidence: String(Math.min(95, 70 + droppedClients.length * 5)),
      startDate: weekAgo,
      endDate: today,
    });
  }

  // Detect significant gains (> 20%) affecting multiple clients
  const gainedClients = rows.filter((r) => r.change_pct > 20);
  if (gainedClients.length >= 3) {
    const avgGain = gainedClients.reduce((sum, r) => sum + r.change_pct, 0) / gainedClients.length;
    
    patterns.push({
      patternType: "industry_trend",
      title: `Traffic Surge Across ${gainedClients.length} Clients`,
      description: `Multiple clients experienced significant traffic gains (avg +${avgGain.toFixed(1)}%). This may indicate a favorable algorithm update or seasonal opportunity.`,
      affectedClientIds: gainedClients.map((r) => r.client_id),
      affectedCount: gainedClients.length,
      magnitude: String(avgGain),
      direction: "up",
      confidence: String(Math.min(95, 70 + gainedClients.length * 5)),
      startDate: weekAgo,
      endDate: today,
    });
  }

  return patterns;
}

async function detectRankingPatterns(workspaceId: string) {
  const patterns: Array<Partial<typeof detectedPatterns.$inferInsert>> = [];
  const today = new Date();
  const weekAgo = subDays(today, 7);

  // Detect significant position changes across clients
  const rankingChanges = await db.execute(sql`
    WITH recent AS (
      SELECT client_id, keyword, position
      FROM keyword_rankings
      WHERE workspace_id = ${workspaceId}
        AND checked_at >= ${weekAgo}
      ORDER BY checked_at DESC
    ),
    changes AS (
      SELECT 
        r1.client_id,
        r1.keyword,
        r1.position as current_pos,
        r2.position as prev_pos,
        r2.position - r1.position as improvement
      FROM recent r1
      JOIN recent r2 ON r1.client_id = r2.client_id AND r1.keyword = r2.keyword
      WHERE r1.position != r2.position
    )
    SELECT 
      keyword,
      COUNT(DISTINCT client_id) as client_count,
      AVG(improvement) as avg_change
    FROM changes
    GROUP BY keyword
    HAVING COUNT(DISTINCT client_id) >= 3
    ORDER BY client_count DESC
    LIMIT 10
  `);

  for (const row of rankingChanges.rows as any[]) {
    if (Math.abs(row.avg_change) >= 5) {
      patterns.push({
        patternType: "ranking_shift",
        title: `Ranking Shift for "${row.keyword}" (${row.client_count} clients)`,
        description: `The keyword "${row.keyword}" saw position changes across ${row.client_count} clients (avg ${row.avg_change > 0 ? "+" : ""}${row.avg_change.toFixed(1)} positions). This may indicate SERP volatility for this term.`,
        affectedCount: row.client_count,
        magnitude: String(Math.abs(row.avg_change)),
        direction: row.avg_change > 0 ? "up" : "down",
        confidence: String(75),
        startDate: weekAgo,
        endDate: today,
      });
    }
  }

  return patterns;
}
```

```typescript
import { Worker, Queue } from "bullmq";
import { redis } from "@/lib/redis";
import { detectPatterns } from "@/lib/pattern-detection";
import { db } from "@/db";
import { workspaces } from "@/db/schema";

const QUEUE_NAME = "pattern-detection";

export const patternDetectionQueue = new Queue(QUEUE_NAME, {
  connection: redis,
  defaultJobOptions: {
    removeOnComplete: 50,
    removeOnFail: 20,
  },
});

// Schedule daily pattern detection
export async function schedulePatternDetection() {
  const allWorkspaces = await db.select({ id: workspaces.id }).from(workspaces);
  
  for (const workspace of allWorkspaces) {
    await patternDetectionQueue.add(
      "detect",
      { workspaceId: workspace.id },
      {
        repeat: { pattern: "0 6 * * *" }, // Daily at 6 AM
        jobId: `pattern-detection-${workspace.id}`,
      }
    );
  }
}

export const patternDetectionWorker = new Worker(
  QUEUE_NAME,
  async (job) => {
    const { workspaceId } = job.data;
    await detectPatterns(workspaceId);
    return { success: true, workspaceId };
  },
  { connection: redis, concurrency: 2 }
);
```

### Task 4: Create patterns API

**File:** `apps/web/src/actions/patterns/get-patterns.ts`

```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { db } from "@/db";
import { detectedPatterns, clients } from "@/db/schema";
import { eq, desc, and, inArray } from "drizzle-orm";

export interface PatternWithClients {
  id: string;
  patternType: string;
  title: string;
  description: string | null;
  affectedClients: Array<{ id: string; name: string }>;
  magnitude: number;
  direction: string | null;
  confidence: number;
  startDate: Date | null;
  endDate: Date | null;
  status: string;
  detectedAt: Date;
}

export async function getPatterns(
  workspaceId: string,
  options: { status?: string; limit?: number } = {}
): Promise<PatternWithClients[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const { status = "active", limit = 20 } = options;

  const patterns = await db
    .select()
    .from(detectedPatterns)
    .where(
      and(
        eq(detectedPatterns.workspaceId, workspaceId),
        status ? eq(detectedPatterns.status, status) : undefined
      )
    )
    .orderBy(desc(detectedPatterns.detectedAt))
    .limit(limit);

  // Fetch client names for affected clients
  const results: PatternWithClients[] = [];

  for (const pattern of patterns) {
    const clientIds = pattern.affectedClientIds as string[] | null;
    let affectedClients: Array<{ id: string; name: string }> = [];

    if (clientIds?.length) {
      const clientData = await db
        .select({ id: clients.id, name: clients.name })
        .from(clients)
        .where(inArray(clients.id, clientIds));
      affectedClients = clientData;
    }

    results.push({
      id: pattern.id,
      patternType: pattern.patternType,
      title: pattern.title,
      description: pattern.description,
      affectedClients,
      magnitude: Number(pattern.magnitude ?? 0),
      direction: pattern.direction,
      confidence: Number(pattern.confidence ?? 0),
      startDate: pattern.startDate,
      endDate: pattern.endDate,
      status: pattern.status ?? "active",
      detectedAt: pattern.detectedAt!,
    });
  }

  return results;
}

export async function dismissPattern(patternId: string): Promise<void> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  await db
    .update(detectedPatterns)
    .set({ status: "dismissed" })
    .where(eq(detectedPatterns.id, patternId));
}

export async function resolvePattern(patternId: string): Promise<void> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  await db
    .update(detectedPatterns)
    .set({ status: "resolved", resolvedAt: new Date() })
    .where(eq(detectedPatterns.id, patternId));
}
```

```tsx
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import {
  Card,

  CardHeader,
  CardTitle,
  Badge,
  Button,
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@tevero/ui";
import {
  TrendingDown,
  TrendingUp,
  Activity,
  ChevronDown,
  Check,
  X,
  Users,
} from "lucide-react";
import { getPatterns, dismissPattern, resolvePattern } from "@/actions/patterns/get-patterns";
import { cn } from "@/lib/utils";
import Link from "next/link";

interface PatternsPanelProps {
  workspaceId: string;
}

const patternIcons: Record<string, React.ReactNode> = {
  traffic_drop: <TrendingDown className="h-4 w-4 text-red-500" />,
  industry_trend: <TrendingUp className="h-4 w-4 text-green-500" />,
  ranking_shift: <Activity className="h-4 w-4 text-yellow-500" />,
  serp_change: <Activity className="h-4 w-4 text-blue-500" />,
};

export function PatternsPanel({ workspaceId }: PatternsPanelProps) {
  const queryClient = useQueryClient();

  const { data: patterns, isLoading } = useQuery({
    queryKey: ["patterns", workspaceId],
    queryFn: () => getPatterns(workspaceId),
  });

  const dismissMutation = useMutation({
    mutationFn: dismissPattern,
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["patterns"] }),
  });

  const resolveMutation = useMutation({
    mutationFn: resolvePattern,
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["patterns"] }),
  });

  if (isLoading) {
    return <PatternsPanelSkeleton />;
  }

  if (!patterns?.length) {
    return null; // Don't show if no patterns
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Activity className="h-5 w-5" />
          Detected Patterns
          <Badge variant="secondary">{patterns.length}</Badge>
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-3">
        {patterns.map((pattern) => (
          <Collapsible key={pattern.id}>
            <div className="border rounded-lg p-3">
              <CollapsibleTrigger className="w-full">
                <div className="flex items-center gap-3">
                  {patternIcons[pattern.patternType]}
                  <div className="flex-1 text-left">
                    <p className="font-medium text-sm">{pattern.title}</p>
                    <p className="text-xs text-muted-foreground">
                      {pattern.affectedClients.length} clients affected •{" "}
                      {pattern.confidence.toFixed(0)}% confidence
                    </p>
                  </div>
                  <Badge
                    variant={
                      pattern.direction === "down" ? "destructive" : "default"
                    }
                  >
                    {pattern.direction === "down" ? "-" : "+"}
                    {pattern.magnitude.toFixed(1)}%
                  </Badge>
                  <ChevronDown className="h-4 w-4 text-muted-foreground transition-transform [[data-state=open]_&]:rotate-180" />
                </div>
              </CollapsibleTrigger>

              <CollapsibleContent className="mt-3 pt-3 border-t">
                <p className="text-sm text-muted-foreground mb-3">
                  {pattern.description}
                </p>

                <div className="flex items-center gap-2 mb-3">
                  <Users className="h-4 w-4 text-muted-foreground" />
                  <span className="text-sm">Affected clients:</span>
                </div>
                <div className="flex flex-wrap gap-2 mb-3">
                  {pattern.affectedClients.slice(0, 5).map((client) => (
                    <Link
                      key={client.id}
                      href={`/clients/${client.id}`}
                      className="text-xs bg-muted px-2 py-1 rounded hover:bg-muted/80"
                    >
                      {client.name}
```
