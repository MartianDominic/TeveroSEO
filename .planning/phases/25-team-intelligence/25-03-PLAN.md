```typescript
import { db } from "@/db";
import { gscSnapshots, clientGoals, goalSnapshots, clientDashboardMetrics } from "@/db/schema";
import { eq, and, gte, desc } from "drizzle-orm";
import { subDays, addDays, differenceInDays } from "date-fns";
import type { Prediction, GoalProjection } from "@/types/predictions";

/**
 * Linear regression for trend prediction
 */
function linearRegression(data: { x: number; y: number }[]): { slope: number; intercept: number; r2: number } {
  const n = data.length;
  if (n < 2) return { slope: 0, intercept: 0, r2: 0 };

  const sumX = data.reduce((sum, p) => sum + p.x, 0);
  const sumY = data.reduce((sum, p) => sum + p.y, 0);
  const sumXY = data.reduce((sum, p) => sum + p.x * p.y, 0);
  const sumXX = data.reduce((sum, p) => sum + p.x * p.x, 0);
  const sumYY = data.reduce((sum, p) => sum + p.y * p.y, 0);

  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
  const intercept = (sumY - slope * sumX) / n;

  // R-squared
  const meanY = sumY / n;
  const ssTotal = data.reduce((sum, p) => sum + Math.pow(p.y - meanY, 2), 0);
  const ssResidual = data.reduce((sum, p) => sum + Math.pow(p.y - (slope * p.x + intercept), 2), 0);
  const r2 = 1 - ssResidual / ssTotal;

  return { slope, intercept, r2 };
}

export async function generatePredictions(clientId: string): Promise<Prediction[]> {
  const predictions: Prediction[] = [];

  // Traffic prediction
  const trafficPrediction = await predictTrafficDecline(clientId);
  if (trafficPrediction) predictions.push(trafficPrediction);

  // Goal predictions
  const goalPredictions = await predictGoalOutcomes(clientId);
  predictions.push(...goalPredictions);

  return predictions;
}

async function predictTrafficDecline(clientId: string): Promise<Prediction | null> {
  const thirtyDaysAgo = subDays(new Date(), 30);

  const snapshots = await db
    .select({
      date: gscSnapshots.date,
      clicks: gscSnapshots.clicks,
    })
    .from(gscSnapshots)
    .where(and(eq(gscSnapshots.clientId, clientId), gte(gscSnapshots.date, thirtyDaysAgo)))
    .orderBy(gscSnapshots.date);

  if (snapshots.length < 7) return null;

  // Convert to regression data
  const data = snapshots.map((s, i) => ({
    x: i,
    y: Number(s.clicks),
  }));

  const { slope, r2 } = linearRegression(data);

  // Predict decline if negative slope and good fit
  if (slope < 0 && r2 > 0.5) {
    const currentAvg = data.slice(-7).reduce((sum, p) => sum + p.y, 0) / 7;
    const predictedInTwoWeeks = currentAvg + slope * 14;
    const declinePercent = ((currentAvg - predictedInTwoWeeks) / currentAvg) * 100;

    if (declinePercent > 10) {
      return {
        id: `pred-traffic-${clientId}`,
        clientId,
        predictionType: "traffic_decline",
        title: "Traffic Decline Predicted",
        description: `Based on the last 30 days, traffic is projected to decline ${declinePercent.toFixed(0)}% over the next 2 weeks.`,
        confidence: r2 * 100,
        severity: declinePercent > 25 ? "critical" : declinePercent > 15 ? "warning" : "info",
        predictedDate: addDays(new Date(), 14),
        currentValue: currentAvg,
        predictedValue: predictedInTwoWeeks,
        createdAt: new Date(),
      };
    }
  }

  return null;
}

export async function predictGoalOutcomes(clientId: string): Promise<Prediction[]> {
  const predictions: Prediction[] = [];

  const goals = await db
    .select()
    .from(clientGoals)
    .where(eq(clientGoals.clientId, clientId));

  for (const goal of goals) {
    const projection = await projectGoal(goal);
    
    if (projection.projectedDate) {
      const daysToTarget = differenceInDays(projection.projectedDate, new Date());
      
      if (projection.trend === "declining" || daysToTarget > 90) {
        predictions.push({
          id: `pred-goal-risk-${goal.id}`,
          clientId,
          predictionType: "goal_at_risk",
          title: `Goal "${goal.customName}" at Risk`,
          description: projection.trend === "declining"
            ? `Progress is declining. Current attainment: ${projection.attainmentPct.toFixed(0)}%`
            : `At current pace, goal won't be reached for ${daysToTarget} days.`,
          confidence: projection.confidence,
          severity: projection.trend === "declining" ? "critical" : "warning",
          currentValue: projection.currentValue,
          predictedValue: Number(goal.targetValue),
          createdAt: new Date(),
        });
      } else if (daysToTarget <= 30 && projection.confidence >= 70) {
        predictions.push({
          id: `pred-goal-achievable-${goal.id}`,
          clientId,
          predictionType: "goal_achievable",
          title: `Goal "${goal.customName}" Within Reach`,
          description: `On track to achieve goal in ~${daysToTarget} days with ${projection.confidence.toFixed(0)}% confidence.`,
          confidence: projection.confidence,
          severity: "info",
          predictedDate: projection.projectedDate,
          currentValue: projection.currentValue,
          predictedValue: Number(goal.targetValue),
          createdAt: new Date(),
        });
      }
    }
  }

  return predictions;
}

export async function projectGoal(goal: typeof clientGoals.$inferSelect): Promise<GoalProjection> {
  const thirtyDaysAgo = subDays(new Date(), 30);

  const snapshots = await db
    .select({
      date: goalSnapshots.snapshotDate,
      value: goalSnapshots.currentValue,
    })
    .from(goalSnapshots)
    .where(and(eq(goalSnapshots.goalId, goal.id), gte(goalSnapshots.snapshotDate, thirtyDaysAgo)))
    .orderBy(goalSnapshots.snapshotDate);

  const currentValue = Number(goal.currentValue ?? 0);
  const targetValue = Number(goal.targetValue);
  const attainmentPct = (currentValue / targetValue) * 100;

  if (snapshots.length < 7) {
    return {
      goalId: goal.id,
      goalName: goal.customName ?? "Goal",
      currentValue,
      targetValue,
      attainmentPct,
      trend: "steady",
      projectedDate: null,
      daysToTarget: null,
      confidence: 30, // Low confidence without enough data
      weeklyVelocity: 0,
    };
  }

  // Calculate velocity
  const data = snapshots.map((s, i) => ({
    x: i,
    y: Number(s.value),
  }));

  const { slope, r2 } = linearRegression(data);
  const weeklyVelocity = slope * 7;

  // Determine trend
  let trend: GoalProjection["trend"];
  if (slope < -0.1) trend = "declining";
  else if (slope < 0.5) trend = "steady";
  else if (slope > weeklyVelocity * 1.2) trend = "accelerating";
  else trend = "steady";

  // Project completion date
  let projectedDate: Date | null = null;
  let daysToTarget: number | null = null;

  if (slope > 0 && currentValue < targetValue) {
    const remaining = targetValue - currentValue;
    daysToTarget = Math.ceil(remaining / slope);
    if (daysToTarget > 0 && daysToTarget < 365) {
      projectedDate = addDays(new Date(), daysToTarget);
    }
  }

  return {
    goalId: goal.id,
    goalName: goal.customName ?? "Goal",
    currentValue,
    targetValue,
    attainmentPct,
    trend,
    projectedDate,
    daysToTarget,
    confidence: Math.min(95, r2 * 100),
    weeklyVelocity,
  };
}
```

### Task 3: Create predictions API

**File:** `apps/web/src/actions/predictions/get-predictions.ts`

```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { db } from "@/db";
import { clients, clientGoals } from "@/db/schema";
import { eq } from "drizzle-orm";
import { generatePredictions, projectGoal } from "@/lib/predictions";
import type { Prediction, GoalProjection } from "@/types/predictions";

export async function getClientPredictions(clientId: string): Promise<Prediction[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  return generatePredictions(clientId);
}

export async function getGoalProjections(clientId: string): Promise<GoalProjection[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const goals = await db
    .select()
    .from(clientGoals)
    .where(eq(clientGoals.clientId, clientId));

  const projections = await Promise.all(goals.map(projectGoal));
  return projections;
}

export async function getWorkspacePredictions(
  workspaceId: string
): Promise<Prediction[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const clientList = await db
    .select({ id: clients.id })
    .from(clients)
    .where(eq(clients.workspaceId, workspaceId))
    .limit(100); // Process first 100

  const allPredictions: Prediction[] = [];

  for (const client of clientList) {
    const predictions = await generatePredictions(client.id);
    allPredictions.push(...predictions);
  }

  // Sort by severity and confidence
  return allPredictions.sort((a, b) => {
    const severityOrder = { critical: 0, warning: 1, info: 2 };
    const severityDiff = severityOrder[a.severity] - severityOrder[b.severity];
    if (severityDiff !== 0) return severityDiff;
    return b.confidence - a.confidence;
  });
}
```

```tsx
"use client";

import { useQuery } from "@tanstack/react-query";
import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
  Progress,
  Badge,
} from "@tevero/ui";
import { Target, TrendingUp, TrendingDown, Minus, Calendar } from "lucide-react";
import { getGoalProjections } from "@/actions/predictions/get-predictions";
import { format } from "date-fns";
import { cn } from "@/lib/utils";

interface GoalProjectionCardProps {
  clientId: string;
}

const trendIcons = {
  accelerating: <TrendingUp className="h-4 w-4 text-green-500" />,
  steady: <Minus className="h-4 w-4 text-yellow-500" />,
  decelerating: <TrendingDown className="h-4 w-4 text-orange-500" />,
  declining: <TrendingDown className="h-4 w-4 text-red-500" />,
};

export function GoalProjectionCard({ clientId }: GoalProjectionCardProps) {
  const { data: projections, isLoading } = useQuery({
    queryKey: ["goal-projections", clientId],
    queryFn: () => getGoalProjections(clientId),
  });

  if (isLoading) {
    return <GoalProjectionSkeleton />;
  }

  if (!projections?.length) {
    return null;
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2 text-base">
          <Target className="h-4 w-4" />
          Goal Projections
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        {projections.map((projection) => (
          <div key={projection.goalId} className="space-y-2">
            <div className="flex items-center justify-between">
              <span className="text-sm font-medium">{projection.goalName}</span>
              <div className="flex items-center gap-2">
                {trendIcons[projection.trend]}
                <Badge variant="outline" className="text-xs">
                  {projection.confidence.toFixed(0)}% confidence
                </Badge>
              </div>
            </div>

            <Progress
              value={Math.min(projection.attainmentPct, 100)}
              className={cn(
                "h-2",
                projection.attainmentPct >= 100
                  ? "[&>div]:bg-green-500"
                  : projection.attainmentPct >= 80
                  ? "[&>div]:bg-yellow-500"
                  : "[&>div]:bg-blue-500"
              )}
            />

            <div className="flex items-center justify-between text-xs text-muted-foreground">
              <span>
                {projection.currentValue.toLocaleString()} /{" "}
                {projection.targetValue.toLocaleString()}
              </span>
              {projection.projectedDate ? (
                <span className="flex items-center gap-1">
                  <Calendar className="h-3 w-3" />
                  Est. {format(projection.projectedDate, "MMM d, yyyy")}
                </span>
              ) : projection.trend === "declining" ? (
                <span className="text-red-600">Goal at risk</span>
              ) : (
                <span>Insufficient data</span>
              )}
            </div>

            {projection.weeklyVelocity !== 0 && (
              <p className="text-xs text-muted-foreground">
                {projection.weeklyVelocity > 0 ? "+" : ""}
                {projection.weeklyVelocity.toFixed(1)} per week
              </p>
            )}
          </div>
        ))}
      </CardContent>
    </Card>
  );
}
```

```tsx
"use client";

import { useQuery } from "@tanstack/react-query";
import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
  Badge,
} from "@tevero/ui";
import { AlertTriangle, TrendingDown, Target, Activity } from "lucide-react";
import { getWorkspacePredictions } from "@/actions/predictions/get-predictions";
import { cn } from "@/lib/utils";
import Link from "next/link";

interface PredictiveAlertsPanelProps {
  workspaceId: string;
}

const predictionIcons = {
  traffic_decline: <TrendingDown className="h-4 w-4" />,
  goal_at_risk: <Target className="h-4 w-4" />,
  goal_achievable: <Target className="h-4 w-4" />,
  ranking_drop: <Activity className="h-4 w-4" />,
  ctr_decline: <Activity className="h-4 w-4" />,
};

export function PredictiveAlertsPanel({ workspaceId }: PredictiveAlertsPanelProps) {
  const { data: predictions, isLoading } = useQuery({
    queryKey: ["predictions", workspaceId],
    queryFn: () => getWorkspacePredictions(workspaceId),
    staleTime: 5 * 60 * 1000, // 5 minutes
  });

  if (isLoading) {
    return <PredictiveAlertsSkeleton />;
  }

  const criticalPredictions = predictions?.filter((p) => p.severity === "critical") ?? [];
  const warningPredictions = predictions?.filter((p) => p.severity === "warning") ?? [];

  if (criticalPredictions.length === 0 && warningPredictions.length === 0) {
    return null;
  }

  return (
    <Card className="border-yellow-200 bg-yellow-50/50">
      <CardHeader>
        <CardTitle className="flex items-center gap-2 text-yellow-800">
          <AlertTriangle className="h-5 w-5" />
          Predictive Alerts
          {criticalPredictions.length > 0 && (
            <Badge variant="destructive">{criticalPredictions.length} critical</Badge>
          )}
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-2">
        {[...criticalPredictions, ...warningPredictions].slice(0, 5).map((prediction) => (
          <Link
            key={prediction.id}
            href={`/clients/${prediction.clientId}`}
            className={cn(
              "flex items-center gap-3 p-2 rounded-lg transition-colors",
              prediction.severity === "critical"
                ? "bg-red-100 hover:bg-red-200"
                : "bg-yellow-100 hover:bg-yellow-200"
            )}
          >
            <span
              className={cn(
                prediction.severity === "critical" ? "text-red-600" : "text-yellow-600"
              )}
            >
              {predictionIcons[prediction.predictionType]}
            </span>
            <div className="flex-1 min-w-0">
              <p className="text-sm font-medium truncate">{prediction.title}</p>
              <p className="text-xs text-muted-foreground truncate">
                {prediction.description}
              </p>
            </div>
            <Badge variant="outline" className="text-xs shrink-0">
              {prediction.confidence.toFixed(0)}%
            </Badge>
          </Link>
        ))}
      </CardContent>
    </Card>
  );
}

function PredictiveAlertsSkeleton() {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Predictive Alerts</CardTitle>
      </CardHeader>
      <CardContent className="space-y-2">
        {[1, 2, 3].map((i) => (
          <div key={i} className="flex items-center gap-3 p-2 animate-pulse">
            <div className="w-4 h-4 bg-muted rounded" />
            <div className="flex-1 space-y-1">
              <div className="h-4 w-32 bg-muted rounded" />
              <div className="h-3 w-48 bg-muted rounded" />
            </div>
          </div>
        ))}
      </CardContent>
    </Card>
  );
}
```
