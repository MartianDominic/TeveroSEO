### Task 2: Create team metrics computation

**File:** `apps/worker/src/lib/compute-team-metrics.ts`

```typescript
import { db } from "@/db";
import {
  teamMemberMetrics,
  clientAssignments,
  clientDashboardMetrics,
  clientTouches,
} from "@/db/schema";
import { eq, sql, and, count, avg, max, gte } from "drizzle-orm";
import { subDays } from "date-fns";
import { nanoid } from "nanoid";

export async function computeTeamMetrics(workspaceId: string): Promise<void> {
  // Get all team members with assignments
  const memberStats = await db
    .select({
      userId: clientAssignments.userId,
      clientCount: count(clientAssignments.clientId),
    })
    .from(clientAssignments)
    .where(eq(clientAssignments.workspaceId, workspaceId))
    .groupBy(clientAssignments.userId);

  for (const member of memberStats) {
    // Get client IDs for this member
    const assignments = await db
      .select({ clientId: clientAssignments.clientId })
      .from(clientAssignments)
      .where(
        and(
          eq(clientAssignments.workspaceId, workspaceId),
          eq(clientAssignments.userId, member.userId)
        )
      );

    const clientIds = assignments.map((a) => a.clientId);

    if (clientIds.length === 0) continue;

    // Get metrics for assigned clients
    const clientMetrics = await db
      .select()
      .from(clientDashboardMetrics)
      .where(sql`${clientDashboardMetrics.clientId} = ANY(${clientIds})`);

    // Calculate aggregates
    const clientsOnTrack = clientMetrics.filter(
      (c) => Number(c.goalAttainmentPct) >= 80
    ).length;
    const clientsWatching = clientMetrics.filter((c) => {
      const pct = Number(c.goalAttainmentPct);
      return pct >= 60 && pct < 80;
    }).length;
    const clientsCritical = clientMetrics.filter(
      (c) => Number(c.goalAttainmentPct) < 60
    ).length;

    const avgGoalAttainment =
      clientMetrics.reduce((sum, c) => sum + Number(c.goalAttainmentPct ?? 0), 0) /
      clientMetrics.length;

    const totalAlertsCritical = clientMetrics.reduce(
      (sum, c) => sum + (c.alertsCritical ?? 0),
      0
    );

    // Get last touch for each client
    const sevenDaysAgo = subDays(new Date(), 7);
    const touches = await db
      .select({
        clientId: clientTouches.clientId,
        lastTouch: max(clientTouches.touchedAt),
      })
      .from(clientTouches)
      .where(sql`${clientTouches.clientId} = ANY(${clientIds})`)
      .groupBy(clientTouches.clientId);

    const touchMap = new Map(touches.map((t) => [t.clientId, t.lastTouch]));

    let totalDaysSinceTouch = 0;
    let clientsNeglected = 0;
    for (const clientId of clientIds) {
      const lastTouch = touchMap.get(clientId);
      if (!lastTouch) {
        clientsNeglected++;
        totalDaysSinceTouch += 30; // Assume 30 days if never touched
      } else {
        const daysSince = Math.floor(
          (Date.now() - new Date(lastTouch).getTime()) / (1000 * 60 * 60 * 24)
        );
        totalDaysSinceTouch += daysSince;
        if (daysSince > 7) clientsNeglected++;
      }
    }

    const avgDaysSinceTouch = totalDaysSinceTouch / clientIds.length;

    // Get max capacity (or default)
    const [existing] = await db
      .select({ maxCapacity: teamMemberMetrics.maxCapacity })
      .from(teamMemberMetrics)
      .where(
        and(
          eq(teamMemberMetrics.workspaceId, workspaceId),
          eq(teamMemberMetrics.userId, member.userId)
        )
      );

    const maxCapacity = existing?.maxCapacity ?? 50;
    const capacityUtilization = (member.clientCount / maxCapacity) * 100;

    // Upsert metrics
    await db
      .insert(teamMemberMetrics)
      .values({
        id: `tm-${workspaceId}-${member.userId}`,
        workspaceId,
        userId: member.userId,
        clientCount: member.clientCount,
        clientsOnTrack,
        clientsWatching,
        clientsCritical,
        avgGoalAttainment: String(avgGoalAttainment),
        totalAlertsCritical,
        avgDaysSinceTouch: String(avgDaysSinceTouch),
        clientsNeglected,
        maxCapacity,
        capacityUtilization: String(capacityUtilization),
        computedAt: new Date(),
      })
      .onConflictDoUpdate({
        target: [teamMemberMetrics.workspaceId, teamMemberMetrics.userId],
        set: {
          clientCount: member.clientCount,
          clientsOnTrack,
          clientsWatching,
          clientsCritical,
          avgGoalAttainment: String(avgGoalAttainment),
          totalAlertsCritical,
          avgDaysSinceTouch: String(avgDaysSinceTouch),
          clientsNeglected,
          capacityUtilization: String(capacityUtilization),
          computedAt: new Date(),
        },
      });
  }
}
```

```typescript
"use server";

import { auth, clerkClient } from "@clerk/nextjs/server";
import { db } from "@/db";
import { teamMemberMetrics, clientAssignments } from "@/db/schema";
import { eq, sql } from "drizzle-orm";

export interface TeamMemberDashboard {
  userId: string;
  name: string;
  email: string;
  imageUrl?: string;
  metrics: {
    clientCount: number;
    clientsOnTrack: number;
    clientsWatching: number;
    clientsCritical: number;
    avgGoalAttainment: number;
    totalAlertsCritical: number;
    avgDaysSinceTouch: number;
    clientsNeglected: number;
    maxCapacity: number;
    capacityUtilization: number;
  };
}

export async function getTeamDashboard(
  workspaceId: string
): Promise<TeamMemberDashboard[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  // Get all team metrics
  const metrics = await db
    .select()
    .from(teamMemberMetrics)
    .where(eq(teamMemberMetrics.workspaceId, workspaceId));

  // Get user details from Clerk
  const userIds = metrics.map((m) => m.userId);
  const users = await clerkClient.users.getUserList({
    userId: userIds,
  });

  const userMap = new Map(users.data.map((u) => [u.id, u]));

  return metrics.map((m) => {
    const user = userMap.get(m.userId);
    return {
      userId: m.userId,
      name: user ? `${user.firstName} ${user.lastName}` : "Unknown",
      email: user?.primaryEmailAddress?.emailAddress ?? "",
      imageUrl: user?.imageUrl,
      metrics: {
        clientCount: m.clientCount ?? 0,
        clientsOnTrack: m.clientsOnTrack ?? 0,
        clientsWatching: m.clientsWatching ?? 0,
        clientsCritical: m.clientsCritical ?? 0,
        avgGoalAttainment: Number(m.avgGoalAttainment ?? 0),
        totalAlertsCritical: m.totalAlertsCritical ?? 0,
        avgDaysSinceTouch: Number(m.avgDaysSinceTouch ?? 0),
        clientsNeglected: m.clientsNeglected ?? 0,
        maxCapacity: m.maxCapacity ?? 50,
        capacityUtilization: Number(m.capacityUtilization ?? 0),
      },
    };
  });
}

export async function getUnassignedClients(workspaceId: string) {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  // Find clients without assignments
  const unassigned = await db.execute(sql`
    SELECT c.id, c.name, cdm.goal_attainment_pct, cdm.alerts_critical
    FROM clients c
    LEFT JOIN client_assignments ca ON c.id = ca.client_id
    LEFT JOIN client_dashboard_metrics cdm ON c.id = cdm.client_id
    WHERE c.workspace_id = ${workspaceId}
      AND ca.id IS NULL
    ORDER BY cdm.priority_score DESC
  `);

  return unassigned.rows;
}
```

### Task 4: Create workload balancing suggestions

**File:** `apps/web/src/lib/team/workload-balancer.ts`

```typescript
import type { TeamMemberDashboard } from "@/actions/team/get-team-dashboard";

export interface ReassignmentSuggestion {
  clientId: string;
  clientName: string;
  fromUserId: string;
  fromUserName: string;
  toUserId: string;
  toUserName: string;
  reason: string;
  priority: "high" | "medium" | "low";
}

export function generateReassignmentSuggestions(
  team: TeamMemberDashboard[],
  clients: Array<{
    id: string;
    name: string;
    assignedTo: string;
    priorityScore: number;
    alertsCritical: number;
    daysSinceTouch: number;
  }>
): ReassignmentSuggestion[] {
  const suggestions: ReassignmentSuggestion[] = [];

  // Find overloaded and underloaded members
  const overloaded = team.filter((m) => m.metrics.capacityUtilization > 100);
  const underloaded = team.filter((m) => m.metrics.capacityUtilization < 70);

  if (overloaded.length === 0 || underloaded.length === 0) {
    return suggestions;
  }

  // For each overloaded member, suggest moving clients to underloaded
  for (const from of overloaded) {
    const theirClients = clients
      .filter((c) => c.assignedTo === from.userId)
      .sort((a, b) => a.priorityScore - b.priorityScore); // Move lowest priority first

    const clientsToMove = Math.min(
      Math.ceil((from.metrics.capacityUtilization - 90) / 10), // Target 90%
      theirClients.length
    );

    for (let i = 0; i < clientsToMove; i++) {
      const client = theirClients[i];
      if (!client) continue;

      // Find best underloaded member
      const to = underloaded
        .filter((m) => m.metrics.capacityUtilization < 90)
        .sort((a, b) => a.metrics.capacityUtilization - b.metrics.capacityUtilization)[0];

      if (!to) break;

      suggestions.push({
        clientId: client.id,
        clientName: client.name,
        fromUserId: from.userId,
        fromUserName: from.name,
        toUserId: to.userId,
        toUserName: to.name,
        reason: `${from.name} is at ${from.metrics.capacityUtilization.toFixed(0)}% capacity, ${to.name} is at ${to.metrics.capacityUtilization.toFixed(0)}%`,
        priority: client.alertsCritical > 0 ? "high" : "medium",
      });

      // Update simulated capacity
      to.metrics.clientCount++;
      to.metrics.capacityUtilization =
        (to.metrics.clientCount / to.metrics.maxCapacity) * 100;
    }
  }

  // Also suggest reassigning neglected clients
  const neglected = clients.filter((c) => c.daysSinceTouch > 7);
  for (const client of neglected.slice(0, 5)) {
    const currentOwner = team.find((m) => m.userId === client.assignedTo);
    if (!currentOwner || currentOwner.metrics.clientsNeglected <= 2) continue;

    const available = team
      .filter(
        (m) =>
          m.userId !== client.assignedTo &&
          m.metrics.avgDaysSinceTouch < 5 &&
          m.metrics.capacityUtilization < 80
      )
      .sort((a, b) => a.metrics.avgDaysSinceTouch - b.metrics.avgDaysSinceTouch)[0];

    if (available) {
      suggestions.push({
        clientId: client.id,
        clientName: client.name,
        fromUserId: currentOwner.userId,
        fromUserName: currentOwner.name,
        toUserId: available.userId,
        toUserName: available.name,
        reason: `Client neglected for ${client.daysSinceTouch} days, ${available.name} has better response time`,
        priority: "high",
      });
    }
  }

  return suggestions.slice(0, 10); // Top 10 suggestions
}
```

```tsx
"use client";

import { useState } from "react";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
  Button,
  Badge,
} from "@tevero/ui";
import { Scale, ArrowRight, Check, X } from "lucide-react";
import { getTeamDashboard } from "@/actions/team/get-team-dashboard";
import { reassignClient } from "@/actions/clients/reassign";
import { generateReassignmentSuggestions } from "@/lib/team/workload-balancer";
import { toast } from "sonner";

interface WorkloadBalancerProps {
  workspaceId: string;
  clients: Array<{
    id: string;
    name: string;
    assignedTo: string;
    priorityScore: number;
    alertsCritical: number;
    daysSinceTouch: number;
  }>;
}

export function WorkloadBalancer({ workspaceId, clients }: WorkloadBalancerProps) {
  const queryClient = useQueryClient();
  const [appliedIds, setAppliedIds] = useState<Set<string>>(new Set());

  const { data: team } = useQuery({
    queryKey: ["team", "dashboard", workspaceId],
    queryFn: () => getTeamDashboard(workspaceId),
  });

  const suggestions = team
    ? generateReassignmentSuggestions(team, clients)
    : [];

  const mutation = useMutation({
    mutationFn: reassignClient,
    onSuccess: (_, variables) => {
      setAppliedIds((prev) => new Set([...prev, variables.clientId]));
      toast.success("Client reassigned successfully");
      queryClient.invalidateQueries({ queryKey: ["team"] });
    },
    onError: () => {
      toast.error("Failed to reassign client");
    },
  });

  if (!suggestions.length) {
    return (
      <Card>
        <CardContent className="py-8 text-center text-muted-foreground">
          <Scale className="h-8 w-8 mx-auto mb-2 opacity-50" />
          Workload is balanced. No reassignments suggested.
        </CardContent>
      </Card>
    );
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Scale className="h-5 w-5" />
          Workload Balancing Suggestions
        </CardTitle>
      </CardHeader>
      <CardContent>
        <div className="space-y-3">
          {suggestions.map((suggestion) => (
            <div
              key={`${suggestion.clientId}-${suggestion.toUserId}`}
              className="flex items-center gap-3 p-3 border rounded-lg"
            >
              <Badge
                variant={
                  suggestion.priority === "high"
                    ? "destructive"
                    : suggestion.priority === "medium"
                    ? "default"
                    : "secondary"
                }
              >
                {suggestion.priority}
              </Badge>

              <div className="flex-1">
                <div className="flex items-center gap-2 text-sm">
                  <span className="font-medium">{suggestion.clientName}</span>
                  <span className="text-muted-foreground">
                    {suggestion.fromUserName}
                  </span>
                  <ArrowRight className="h-4 w-4 text-muted-foreground" />
                  <span className="text-muted-foreground">
                    {suggestion.toUserName}
                  </span>
                </div>
                <p className="text-xs text-muted-foreground mt-1">
                  {suggestion.reason}
                </p>
              </div>

              {appliedIds.has(suggestion.clientId) ? (
                <Badge variant="outline" className="gap-1">
                  <Check className="h-3 w-3" />
                  Applied
                </Badge>
              ) : (
                <div className="flex items-center gap-1">
                  <Button
                    variant="outline"
                    size="sm"
                    onClick={() =>
                      mutation.mutate({
                        clientId: suggestion.clientId,
                        toUserId: suggestion.toUserId,
                      })
                    }
                    disabled={mutation.isPending}
                  >
                    Apply
                  </Button>
                  <Button
                    variant="ghost"
                    size="sm"
                    onClick={() =>
                      setAppliedIds((prev) => new Set([...prev, suggestion.clientId]))
                    }
                  >
                    <X className="h-4 w-4" />
                  </Button>
                </div>
              )}
            </div>
          ))}
        </div>

        <Button
          className="w-full mt-4"
          onClick={() => {
            for (const s of suggestions.filter(
              (s) => !appliedIds.has(s.clientId)
            )) {
              mutation.mutate({
                clientId: s.clientId,
                toUserId: s.toUserId,
              });
            }
          }}
          disabled={mutation.isPending || appliedIds.size === suggestions.length}
        >
          Apply All Suggestions
        </Button>
      </CardContent>
    </Card>
  );
}
```
