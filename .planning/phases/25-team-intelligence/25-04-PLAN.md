```typescript
export type OpportunityType =
  | "ctr_improvement"    // High impressions, low CTR
  | "ranking_gap"        // Close to page 1 but not there
  | "content_gap"        // Missing content for high-volume terms
  | "quick_win"          // Easy improvements available
  | "featured_snippet"   // Potential for featured snippet
  | "local_pack";        // Local SEO opportunity

export interface Opportunity {
  id: string;
  clientId: string;
  type: OpportunityType;
  title: string;
  description: string;
  impact: "high" | "medium" | "low";
  effort: "low" | "medium" | "high";
  priority: number;  // Computed: impact / effort
  
  // Supporting data
  keywords?: string[];
  pages?: string[];
  metrics: {
    currentValue?: number;
    potentialValue?: number;
    estimatedGain?: number;
  };
  
  createdAt: Date;
  expiresAt?: Date;
}

export interface OpportunityFilter {
  types?: OpportunityType[];
  minImpact?: "high" | "medium" | "low";
  maxEffort?: "low" | "medium" | "high";
}
```

### Task 2: Create opportunity detection algorithms

**File:** `apps/worker/src/lib/opportunities/index.ts`

```typescript
import { db } from "@/db";
import { gscQuerySnapshots, keywordRankings, savedKeywords } from "@/db/schema";
import { eq, and, gte, lte, desc, sql } from "drizzle-orm";
import { subDays } from "date-fns";
import type { Opportunity, OpportunityType } from "@/types/opportunities";
import { nanoid } from "nanoid";

export async function findOpportunities(clientId: string): Promise<Opportunity[]> {
  const opportunities: Opportunity[] = [];

  const [ctrOpps, rankingGaps, quickWins] = await Promise.all([
    findCTRImprovements(clientId),
    findRankingGaps(clientId),
    findQuickWins(clientId),
  ]);

  opportunities.push(...ctrOpps, ...rankingGaps, ...quickWins);

  // Sort by priority (impact/effort ratio)
  return opportunities.sort((a, b) => b.priority - a.priority);
}

async function findCTRImprovements(clientId: string): Promise<Opportunity[]> {
  const opportunities: Opportunity[] = [];
  const sevenDaysAgo = subDays(new Date(), 7);

  // Find queries with high impressions but low CTR
  const queries = await db
    .select({
      query: gscQuerySnapshots.query,
      impressions: sql<number>`SUM(${gscQuerySnapshots.impressions})`,
      clicks: sql<number>`SUM(${gscQuerySnapshots.clicks})`,
      avgPosition: sql<number>`AVG(${gscQuerySnapshots.position})`,
    })
    .from(gscQuerySnapshots)
    .where(
      and(
        eq(gscQuerySnapshots.clientId, clientId),
        gte(gscQuerySnapshots.date, sevenDaysAgo)
      )
    )
    .groupBy(gscQuerySnapshots.query)
    .having(sql`SUM(${gscQuerySnapshots.impressions}) > 100`)
    .orderBy(desc(sql`SUM(${gscQuerySnapshots.impressions})`))
    .limit(50);

  for (const query of queries) {
    const ctr = query.impressions > 0 ? (query.clicks / query.impressions) * 100 : 0;
    
    // Expected CTR based on position
    const expectedCTR = getExpectedCTR(query.avgPosition);
    const ctrGap = expectedCTR - ctr;

    if (ctrGap > 2 && query.avgPosition <= 10) {
      const potentialClicks = Math.round((expectedCTR / 100) * query.impressions);
      const estimatedGain = potentialClicks - query.clicks;

      opportunities.push({
        id: nanoid(),
        clientId,
        type: "ctr_improvement",
        title: `Improve CTR for "${query.query}"`,
        description: `This query has ${query.impressions.toLocaleString()} impressions but only ${ctr.toFixed(1)}% CTR (expected ${expectedCTR.toFixed(1)}% for position ${query.avgPosition.toFixed(0)}). Improving title/meta description could add ~${estimatedGain} clicks/week.`,
        impact: estimatedGain > 50 ? "high" : estimatedGain > 20 ? "medium" : "low",
        effort: "low",
        priority: estimatedGain > 50 ? 9 : estimatedGain > 20 ? 6 : 3,
        keywords: [query.query],
        metrics: {
          currentValue: ctr,
          potentialValue: expectedCTR,
          estimatedGain,
        },
        createdAt: new Date(),
      });
    }
  }

  return opportunities.slice(0, 10);
}

async function findRankingGaps(clientId: string): Promise<Opportunity[]> {
  const opportunities: Opportunity[] = [];

  // Find keywords ranking 11-20 (almost on page 1)
  const keywords = await db
    .select({
      keyword: keywordRankings.keyword,
      position: keywordRankings.position,
      url: keywordRankings.url,
    })
    .from(keywordRankings)
    .where(
      and(
        eq(keywordRankings.clientId, clientId),
        gte(keywordRankings.position, 11),
        lte(keywordRankings.position, 20)
      )
    )
    .orderBy(keywordRankings.position)
    .limit(20);

  for (const kw of keywords) {
    const positionsToMove = kw.position - 10;

    opportunities.push({
      id: nanoid(),
      clientId,
      type: "ranking_gap",
      title: `Push "${kw.keyword}" to Page 1`,
      description: `Currently at position ${kw.position}, just ${positionsToMove} spots from page 1. Small content improvements could boost visibility significantly.`,
      impact: positionsToMove <= 3 ? "high" : positionsToMove <= 5 ? "medium" : "low",
      effort: positionsToMove <= 3 ? "low" : "medium",
      priority: positionsToMove <= 3 ? 8 : positionsToMove <= 5 ? 5 : 3,
      keywords: [kw.keyword],
      pages: kw.url ? [kw.url] : [],
      metrics: {
        currentValue: kw.position,
        potentialValue: 10,
      },
      createdAt: new Date(),
    });
  }

  return opportunities.slice(0, 5);
}

async function findQuickWins(clientId: string): Promise<Opportunity[]> {
  const opportunities: Opportunity[] = [];

  // Find keywords that recently dropped and could be recovered
  const recentDrops = await db.execute(sql`
    WITH ranked AS (
      SELECT 
        keyword,
        position,
        LAG(position) OVER (PARTITION BY keyword ORDER BY checked_at) as prev_position,
        checked_at
      FROM keyword_rankings
      WHERE client_id = ${clientId}
        AND checked_at >= ${subDays(new Date(), 14)}
    )
    SELECT 
      keyword,
      position as current_position,
      prev_position,
      prev_position - position as drop_amount
    FROM ranked
    WHERE prev_position IS NOT NULL
      AND prev_position - position > 3
      AND prev_position <= 10
      AND position > 10
    ORDER BY drop_amount DESC
    LIMIT 5
  `);

  for (const row of recentDrops.rows as any[]) {
    opportunities.push({
      id: nanoid(),
      clientId,
      type: "quick_win",
      title: `Recover "${row.keyword}" Ranking`,
      description: `This keyword dropped from position ${row.prev_position} to ${row.current_position} recently. Investigating and fixing the cause could quickly recover the ranking.`,
      impact: "high",
      effort: "medium",
      priority: 7,
      keywords: [row.keyword],
      metrics: {
        currentValue: row.current_position,
        potentialValue: row.prev_position,
      },
      createdAt: new Date(),
    });
  }

  return opportunities;
}

function getExpectedCTR(position: number): number {
  // Industry average CTR by position
  const ctrByPosition: Record<number, number> = {
    1: 28.5,
    2: 15.7,
    3: 11.0,
    4: 8.0,
    5: 7.2,
    6: 5.1,
    7: 4.0,
    8: 3.2,
    9: 2.8,
    10: 2.5,
  };
  const pos = Math.round(position);
  return ctrByPosition[pos] ?? 1.0;
}
```

```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { findOpportunities } from "@/lib/opportunities";
import type { Opportunity, OpportunityFilter } from "@/types/opportunities";

export async function getClientOpportunities(
  clientId: string,
  filter?: OpportunityFilter
): Promise<Opportunity[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  let opportunities = await findOpportunities(clientId);

  // Apply filters
  if (filter?.types?.length) {
    opportunities = opportunities.filter((o) => filter.types!.includes(o.type));
  }

  if (filter?.minImpact) {
    const impactOrder = { high: 3, medium: 2, low: 1 };
    const minImpactNum = impactOrder[filter.minImpact];
    opportunities = opportunities.filter(
      (o) => impactOrder[o.impact] >= minImpactNum
    );
  }

  if (filter?.maxEffort) {
    const effortOrder = { low: 1, medium: 2, high: 3 };
    const maxEffortNum = effortOrder[filter.maxEffort];
    opportunities = opportunities.filter(
      (o) => effortOrder[o.effort] <= maxEffortNum
    );
  }

  return opportunities;
}
```

```tsx
import { getClientOpportunities } from "@/actions/opportunities/get-opportunities";
```

```tsx
  Card,

  CardHeader,
  CardTitle,
  Tabs,
  TabsContent,
  TabsList,
  TabsTrigger,
} from "@tevero/ui";

interface OpportunitiesPageProps {
  params: { clientId: string };

          <TabsTrigger value="ranking">Ranking ({rankingOpps.length})</TabsTrigger>

        <TabsContent value="all" className="mt-4">

        </TabsContent>

        <TabsContent value="ctr" className="mt-4">
          <OpportunityList opportunities={ctrOpps} />
        </TabsContent>

        <TabsContent value="ranking" className="mt-4">
          <OpportunityList opportunities={rankingOpps} />
        </TabsContent>

        <TabsContent value="quick" className="mt-4">
          <OpportunityList opportunities={quickWins} />
        </TabsContent>
      </Tabs>

}

function OpportunityList({ opportunities }: { opportunities: any[] }) {
  if (opportunities.length === 0) {
    return (
      <Card>
        <CardContent className="py-8 text-center text-muted-foreground">
          No opportunities in this category.
        </CardContent>
      </Card>
    );
  }

  return (
    <div className="grid gap-4">
      {opportunities.map((opp) => (
        <Card key={opp.id}>
          <CardHeader>
            <CardTitle className="text-lg">{opp.title}</CardTitle>
          </CardHeader>
          <CardContent>
            <p className="text-muted-foreground mb-4">{opp.description}</p>
            {opp.keywords?.length > 0 && (
              <div className="mb-2">
                <span className="text-sm font-medium">Keywords:</span>
                <div className="flex flex-wrap gap-1 mt-1">
                  {opp.keywords.map((kw: string) => (
                    <code key={kw} className="text-xs bg-muted px-2 py-1 rounded">
                      {kw}
                    </code>
                  ))}
                </div>
              </div>
            )}
          </CardContent>
        </Card>
      ))}
    </div>
  );
}
```
