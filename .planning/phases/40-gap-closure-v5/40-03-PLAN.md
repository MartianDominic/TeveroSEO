```typescript
// src/routes/api/voice/$clientId/constraints.ts
import { createFileRoute } from "@tanstack/react-router";
import { VoiceConstraintBuilder } from "@/server/features/voice/services/VoiceConstraintBuilder";
import { VoiceProfileRepository } from "@/server/features/voice/repositories/VoiceProfileRepository";
import type { VoiceMode } from "@/db/voice-schema";

export const Route = createFileRoute("/api/voice/$clientId/constraints")({
  server: {
    handlers: {
      GET: async ({ params, request }) => {
        const url = new URL(request.url);
        const mode = (url.searchParams.get("mode") || "application") as VoiceMode;
        
        // Check if client has a voice profile
        const profile = await VoiceProfileRepository.getByClientId(params.clientId);
        
        if (!profile && mode !== "best_practices") {
          return Response.json({
            hasProfile: false,
            mode: "best_practices",
            constraints: getDefaultConstraints(),
          });
        }
        
        const builder = new VoiceConstraintBuilder();
        const constraints = await builder.build(params.clientId, mode);
        
        return Response.json({
          hasProfile: !!profile,
          mode,
          constraints: {
            systemPrompt: constraints.systemPrompt,
            tone: constraints.tone,
            formality: constraints.formality,
            personality: constraints.personality,
            preferredVocabulary: constraints.preferredVocabulary || [],
            forbiddenPhrases: constraints.forbiddenPhrases || [],
            sentenceLengthGuidance: constraints.sentenceLengthGuidance,
            headingStyle: constraints.headingStyle,
            ctaTemplate: constraints.ctaTemplate,
          },
        });
      },
    },
  },
});

function getDefaultConstraints() {
  return {
    systemPrompt: "Write in a professional, SEO-optimized style.",
    tone: "professional",
    formality: "semi-formal",
    personality: "helpful and authoritative",
    preferredVocabulary: [],
    forbiddenPhrases: [],
    sentenceLengthGuidance: "Vary sentence length. Average 15-20 words.",
    headingStyle: "Action-oriented and benefit-focused",
    ctaTemplate: null,
  };
}
```

### T-40-03-02: Quality Gate Enforcement (P39)

**Problem**: `auto_publish_executor.py` uses boolean `auto_publish`, no quality scoring.

**File**: `AI-Writer/backend/services/auto_publish_executor.py`

**Steps**:
1. Add quality check before publish approval
2. Call content validation endpoint
3. Check voice compliance if profile exists
4. Block auto-publish if score < 80
5. Move to review queue with reason

**Code**:
```python
# AI-Writer/backend/services/auto_publish_executor.py

class QualityGate:
    """Quality gate for content publication."""
    
    SEO_THRESHOLD = 80
    VOICE_THRESHOLD = 75
    
    def __init__(self, open_seo_client: OpenSeoClient):
        self.open_seo_client = open_seo_client
    
    async def check(self, article: Article) -> tuple[bool, str, dict]:
        """
        Check if article passes quality gate.
        Returns: (passed, reason, details)
        """
        details = {}
        
        # 1. SEO Score Check
        try:
            seo_result = await self.open_seo_client.post(
                "/api/seo/content/validate",
                json={
                    "html": article.content,
                    "keyword": article.keyword,
                }
            )
            seo_score = seo_result.get("score", 0)
            details["seo_score"] = seo_score
            details["seo_findings"] = seo_result.get("findings", [])
            
            if seo_score < self.SEO_THRESHOLD:
                return False, f"SEO score {seo_score} < {self.SEO_THRESHOLD}", details
        except Exception as e:
            logger.error(f"SEO validation failed: {e}")
            return False, f"SEO validation error: {e}", details
        
        # 2. Voice Compliance Check (if voice mode set)
        if article.voice_mode and article.voice_mode != "best_practices":
            try:
                voice_result = await self.open_seo_client.post(
                    f"/api/voice/{article.client_id}/compliance",
                    json={"content": article.content}
                )
                voice_score = voice_result.get("overallScore", 0)
                details["voice_score"] = voice_score
                
                if voice_score < self.VOICE_THRESHOLD:
                    return False, f"Voice score {voice_score} < {self.VOICE_THRESHOLD}", details
            except Exception as e:
                logger.warning(f"Voice compliance check failed: {e}")
                # Don't block on voice check failure
        
        return True, "Quality gate passed", details


class AutoPublishExecutor:
    def __init__(self):
        self.quality_gate = QualityGate(OpenSeoClient())
    
    async def _process_article(self, article: Article):
        """Process article for publishing."""
        
        # Check quality gate before publishing
        if article.auto_publish:
            passed, reason, details = await self.quality_gate.check(article)
            
            if not passed:
                logger.info(f"Article {article.id} failed quality gate: {reason}")
                await self._move_to_review(article, reason, details)
                return
        
        # Proceed with publishing
        await self._publish_article(article)
    
    async def _move_to_review(self, article: Article, reason: str, details: dict):
        """Move article to review queue."""
        article.status = "needs_review"
        article.review_reason = reason
        article.quality_details = details
        await self.article_repository.update(article)
        
        # Notify relevant parties
        await self._send_review_notification(article)
```

**Verification**:
- [ ] Articles with SEO score < 80 blocked from auto-publish
- [ ] Articles moved to "needs_review" status with reason
- [ ] Quality details stored for reviewer
- [ ] Articles with score ≥ 80 proceed normally

```python
# AI-Writer/backend/services/gsc_service.py

from google.oauth2 import service_account
from googleapiclient.discovery import build

class GSCService:
    INDEXING_SCOPES = ["https://www.googleapis.com/auth/indexing"]
    
    async def submit_url_for_indexing(
        self, 
        client_id: str, 
        url: str,
        action: str = "URL_UPDATED"
    ) -> dict:
        """
        Submit URL to Google Indexing API.
        
        Args:
            client_id: Client identifier
            url: URL to submit
            action: "URL_UPDATED" or "URL_DELETED"
        
        Returns:
            dict with success status and metadata
        """
        try:
            # Get service account credentials
            credentials = await self._get_indexing_credentials(client_id)
            if not credentials:
                return {
                    "success": False, 
                    "error": "No Indexing API credentials configured"
                }
            
            # Build Indexing API service
            service = build("indexing", "v3", credentials=credentials)
            
            # Submit URL notification
            response = service.urlNotifications().publish(
                body={
                    "url": url,
                    "type": action
                }
            ).execute()
            
            logger.info(f"GSC URL submitted: {url}", extra={
                "client_id": client_id,
                "response": response
            })
            
            return {
                "success": True,
                "url": url,
                "notifyTime": response.get("urlNotificationMetadata", {}).get("latestUpdate", {}).get("notifyTime"),
            }
            
        except Exception as e:
            logger.error(f"GSC URL submission failed: {e}")
            return {"success": False, "error": str(e)}
    
    async def _get_indexing_credentials(self, client_id: str):
        """Get service account credentials for Indexing API."""
        # Check for client-specific service account
        client_creds = await self.credential_repository.get_indexing_credentials(client_id)
        if client_creds:
            return service_account.Credentials.from_service_account_info(
                client_creds,
                scopes=self.INDEXING_SCOPES
            )
        
        # Fall back to platform service account
        if settings.GSC_SERVICE_ACCOUNT_JSON:
            return service_account.Credentials.from_service_account_file(
                settings.GSC_SERVICE_ACCOUNT_JSON,
                scopes=self.INDEXING_SCOPES
            )
        
        return None
```

```typescript
import { createFileRoute } from "@tanstack/react-router";
import { VoiceComplianceService } from "@/server/features/voice/services/VoiceComplianceService";
import { VoiceProfileRepository } from "@/server/features/voice/repositories/VoiceProfileRepository";

export const Route = createFileRoute("/api/voice/$clientId/compliance")({
  server: {
    handlers: {
      POST: async ({ params, request }) => {
        const { content } = await request.json();
        
        if (!content) {
          return Response.json({ error: "content required" }, { status: 400 });
        }
        
        const profile = await VoiceProfileRepository.getByClientId(params.clientId);
        if (!profile) {
          return Response.json({
            hasProfile: false,
            overallScore: 100, // No profile = no voice requirements
            message: "No voice profile configured",
          });
        }
        
        const complianceService = new VoiceComplianceService();
        const result = await complianceService.scoreContent(content, profile);
        
        return Response.json({
          hasProfile: true,
          overallScore: result.overallScore,
          dimensions: result.dimensions,
          violations: result.violations,
          suggestions: result.suggestions,
        });
      },
    },
  },
});
```
