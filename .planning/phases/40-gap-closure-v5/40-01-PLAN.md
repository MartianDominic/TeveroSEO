### T-40-01-02: GSC Integration in CannibalizationService (P35)

**Problem**: `CannibalizationService.getGscKeywordData()` returns empty array (stub).

**File**: `open-seo-main/src/server/features/linking/services/CannibalizationService.ts:243-251`

**Steps**:
1. Import gscSnapshots from analytics schema
2. Query gsc_snapshots table for matching keywords
3. Return structured GscKeywordData array

**Code**:
```typescript
import { db } from "@/db";
import { gscSnapshots } from "@/db/analytics-schema";
import { eq, and, ilike, desc } from "drizzle-orm";

private async getGscKeywordData(
  clientId: string,
  keyword: string
): Promise<GscKeywordData[]> {
  const snapshots = await db
    .select({
      query: gscSnapshots.query,
      page: gscSnapshots.page,
      clicks: gscSnapshots.clicks,
      impressions: gscSnapshots.impressions,
      position: gscSnapshots.position,
    })
    .from(gscSnapshots)
    .where(
      and(
        eq(gscSnapshots.clientId, clientId),
        ilike(gscSnapshots.query, `%${keyword}%`)
      )
    )
    .orderBy(desc(gscSnapshots.impressions))
    .limit(50);

  return snapshots.map((s) => ({
    keyword: s.query,
    url: s.page,
    clicks: s.clicks ?? 0,
    impressions: s.impressions ?? 0,
    position: s.position ?? 0,
  }));
}
```

**Code**:
```typescript
// src/routes/api/voice/$clientId/preview.ts
import { createFileRoute } from "@tanstack/react-router";
import { VoiceProfileRepository } from "@/server/features/voice/repositories/VoiceProfileRepository";
import { VoiceConstraintBuilder } from "@/server/features/voice/services/VoiceConstraintBuilder";
import { VoiceComplianceService } from "@/server/features/voice/services/VoiceComplianceService";
import { generateText } from "@/server/lib/ai/claude";

export const Route = createFileRoute("/api/voice/$clientId/preview")({
  server: {
    handlers: {
      POST: async ({ params, request }) => {
        const { sampleText, voiceMode } = await request.json();
        
        const profile = await VoiceProfileRepository.getByClientId(params.clientId);
        if (!profile) {
          return Response.json({ error: "No voice profile found" }, { status: 404 });
        }
        
        const builder = new VoiceConstraintBuilder();
        const constraints = await builder.build(params.clientId, voiceMode);
        
        const prompt = `Rewrite the following text in the specified voice:\n\n${constraints.systemPrompt}\n\nOriginal text:\n${sampleText}\n\nRewritten text:`;
        
        const preview = await generateText(prompt);
        
        const complianceService = new VoiceComplianceService();
        const compliance = await complianceService.scoreContent(preview, profile);
        
        return Response.json({
          preview,
          compliance: {
            overallScore: compliance.overallScore,
            dimensions: compliance.dimensions,
            violations: compliance.violations,
          },
        });
      },
    },
  },
});
```

**Verification**:
- [ ] `POST /api/voice/{clientId}/preview` generates transformed text
- [ ] Response includes compliance scores
- [ ] VoicePreviewPanel displays results correctly
