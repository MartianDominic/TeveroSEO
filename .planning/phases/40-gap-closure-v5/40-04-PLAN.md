```typescript
// src/routes/api/seo/links/suggestions.ts
import { createFileRoute } from "@tanstack/react-router";
import { LinkSuggestionService } from "@/server/features/linking/services/LinkSuggestionService";
import { VelocityService } from "@/server/features/linking/services/VelocityService";

export const Route = createFileRoute("/api/seo/links/suggestions")({
  server: {
    handlers: {
      POST: async ({ request }) => {
        const { clientId, content, keyword, maxLinks = 7 } = await request.json();
        
        if (!clientId || !content) {
          return Response.json(
            { error: "clientId and content required" },
            { status: 400 }
          );
        }
        
        // Check velocity limits
        const velocityService = new VelocityService();
        const remaining = await velocityService.getRemainingDailyQuota(clientId);
        
        if (remaining <= 0) {
          return Response.json({
            links: [],
            reason: "Daily link insertion quota exhausted",
            remainingQuota: 0,
          });
        }
        
        // Get suggestions
        const suggestionService = new LinkSuggestionService();
        const suggestions = await suggestionService.getSuggestionsForContent(
          clientId,
          content,
          keyword,
          Math.min(maxLinks, remaining)
        );
        
        // Filter to auto-applicable only
        const autoApplicable = suggestions.filter(s => s.isAutoApplicable);
        
        return Response.json({
          links: autoApplicable.map(s => ({
            anchorText: s.anchorText,
            targetUrl: s.targetUrl,
            confidence: s.confidence,
            method: s.method,
            position: s.suggestedPosition,
          })),
          totalSuggestions: suggestions.length,
          autoApplicable: autoApplicable.length,
          remainingQuota: remaining - autoApplicable.length,
        });
      },
    },
  },
});
```

**Steps**:

**Code**:
```python
# AI-Writer/backend/services/internal_link_inserter.py

import re
from bs4 import BeautifulSoup, NavigableString
from typing import List, Dict
import httpx

class InternalLinkInserter:
    """Auto-insert internal links into generated content."""
    
    MIN_LINKS = 3
    MAX_LINKS = 7
    
    def __init__(self, open_seo_url: str):
        self.open_seo_url = open_seo_url
        self.client = httpx.AsyncClient()
    
    async def insert_links(
        self, 
        html: str, 
        client_id: str, 
        keyword: str
    ) -> tuple[str, List[Dict]]:
        """
        Insert internal links into HTML content.
        
        Returns:
            tuple of (modified_html, inserted_links)
        """
        # 1. Get link suggestions
        suggestions = await self._get_suggestions(client_id, html, keyword)
        
        if not suggestions:
            return html, []
        
        # 2. Parse HTML
        soup = BeautifulSoup(html, "html.parser")
        
        # 3. Insert links
        inserted = []
        for suggestion in suggestions[:self.MAX_LINKS]:
            if len(inserted) >= self.MAX_LINKS:
                break
            
            success = self._insert_link(
                soup, 
                suggestion["anchorText"], 
                suggestion["targetUrl"]
            )
            
            if success:
                inserted.append(suggestion)
        
        # 4. Check minimum
        if len(inserted) < self.MIN_LINKS:
            # Log warning but don't fail
            logger.warning(
                f"Only inserted {len(inserted)} links (min: {self.MIN_LINKS})"
            )
        
        return str(soup), inserted
    
    async def _get_suggestions(
        self, 
        client_id: str, 
        content: str, 
        keyword: str
    ) -> List[Dict]:
        """Fetch link suggestions from open-seo-main."""
        try:
            response = await self.client.post(
                f"{self.open_seo_url}/api/seo/links/suggestions",
                json={
                    "clientId": client_id,
                    "content": content,
                    "keyword": keyword,
                    "maxLinks": self.MAX_LINKS,
                },
                timeout=30.0
            )
            response.raise_for_status()
            data = response.json()
            return data.get("links", [])
        except Exception as e:
            logger.error(f"Failed to get link suggestions: {e}")
            return []
    
    def _insert_link(
        self, 
        soup: BeautifulSoup, 
        anchor_text: str, 
        target_url: str
    ) -> bool:
        """
        Insert a single link into the soup.
        Returns True if successful.
        """
        # Find text nodes containing anchor text (case-insensitive)
        pattern = re.compile(re.escape(anchor_text), re.IGNORECASE)
        
        # Search in paragraph, list item, and div elements
        for tag in soup.find_all(["p", "li", "div"]):
            # Skip if already has links
            if tag.find("a"):
                continue
            
            for text_node in tag.find_all(string=True):
                if not isinstance(text_node, NavigableString):
                    continue
                
                # Skip if inside a link already
                if text_node.parent.name == "a":
                    continue
                
                match = pattern.search(str(text_node))
                if match:
                    # Found the anchor text - replace with link
                    original = str(text_node)
                    matched_text = original[match.start():match.end()]
                    
                    # Create link element
                    new_html = original[:match.start()]
                    new_html += f'<a href="{target_url}">{matched_text}</a>'
                    new_html += original[match.end():]
                    
                    # Replace text node with new HTML
                    new_soup = BeautifulSoup(new_html, "html.parser")
                    text_node.replace_with(new_soup)
                    
                    return True
        
        return False
```

### T-40-04-03: Link Graph Update on Publish (P39)

**Problem**: Link graph not updated when new content published.

**Files**:

- MODIFY: `AI-Writer/backend/services/auto_publish_executor.py`

**Steps**:
1. Create endpoint to accept new page URL and HTML
2. Extract all links from HTML
3. Upsert to link_graph table
4. Update page_links metrics
5. Recalculate orphan status
6. Call from AI-Writer after publish

**Code (open-seo-main)**:
```typescript
// src/routes/api/seo/links/graph/update.ts
import { createFileRoute } from "@tanstack/react-router";
import { extractLinksFromHtml } from "@/server/lib/linking/link-extractor";
import { LinkGraphRepository } from "@/server/features/linking/repositories/LinkGraphRepository";
import { ClickDepthService } from "@/server/lib/linking/click-depth";
import { db } from "@/db";
import { linkGraph, pageLinks, orphanPages } from "@/db/link-schema";
import { eq, and } from "drizzle-orm";

export const Route = createFileRoute("/api/seo/links/graph/update")({
  server: {
    handlers: {
      POST: async ({ request }) => {
        const { clientId, url, html } = await request.json();
        
        if (!clientId || !url || !html) {
          return Response.json(
            { error: "clientId, url, and html required" },
            { status: 400 }
          );
        }
        
        try {
          // 1. Extract links from new content
          const links = extractLinksFromHtml(html, url);
          
          // 2. Clear existing outbound links from this page
          await db.delete(linkGraph).where(
            and(
              eq(linkGraph.clientId, clientId),
              eq(linkGraph.sourceUrl, url)
            )
          );
          
          // 3. Insert new links
          if (links.length > 0) {
            await db.insert(linkGraph).values(
              links.map(link => ({
                clientId,
                sourceUrl: url,
                targetUrl: link.href,
                anchorText: link.text,
                position: link.position,
                isInternal: link.isInternal,
              }))
            );
          }
          
          // 4. Update page_links for source page
          await db.insert(pageLinks)
            .values({
              clientId,
              url,
              outboundTotal: links.filter(l => l.isInternal).length,
              outboundExternal: links.filter(l => !l.isInternal).length,
              lastUpdated: new Date(),
            })
            .onConflictDoUpdate({
              target: [pageLinks.clientId, pageLinks.url],
              set: {
                outboundTotal: links.filter(l => l.isInternal).length,
                outboundExternal: links.filter(l => !l.isInternal).length,
                lastUpdated: new Date(),
              },
            });
          
          // 5. Update inbound counts for target pages
          const internalTargets = [...new Set(
            links.filter(l => l.isInternal).map(l => l.href)
          )];
          
          for (const targetUrl of internalTargets) {
            await LinkGraphRepository.incrementInboundCount(clientId, targetUrl);
          }
          
          // 6. Remove this page from orphans (it's being published = has inbound from somewhere)
          await db.delete(orphanPages).where(
            and(
              eq(orphanPages.clientId, clientId),
              eq(orphanPages.url, url)
            )
          );
          
          // 7. Queue click depth recalculation (async)
          // This is expensive, so we queue it rather than doing it inline
          await queueClickDepthRecalculation(clientId);
          
          return Response.json({
            success: true,
            linksExtracted: links.length,
            internalLinks: links.filter(l => l.isInternal).length,
            externalLinks: links.filter(l => !l.isInternal).length,
          });
          
        } catch (error) {
          console.error("Link graph update failed:", error);
          return Response.json(
            { error: "Failed to update link graph" },
            { status: 500 }
          );
        }
      },
    },
  },
});
```

```typescript
  url: string,

  options?: { tiers?: number[] }
): Promise<CheckResult[]> {
  try {
```
