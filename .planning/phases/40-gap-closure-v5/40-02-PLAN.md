**Problem**: `SerpAnalyzer.extractCommonH2s()` returns empty array (stub).

5. Return H2s appearing in 2+ pages

**Code**:
```typescript
import * as cheerio from "cheerio";
import { dataForSeoClient } from "@/server/lib/dataforseo";

interface H2Frequency {
  heading: string;
  frequency: number;
}

export async function extractCommonH2s(
  urls: string[]
): Promise<H2Frequency[]> {
  const h2Counts = new Map<string, number>();

  // Fetch HTML for top 5 competitors
  const htmlResults = await Promise.allSettled(
    urls.slice(0, 5).map(async (url) => {
      const response = await dataForSeoClient.post("/on_page/instant_pages", [{
        url,
        enable_javascript: true,
        load_resources: false,
      }]);
      return response.data?.tasks?.[0]?.result?.[0]?.items?.[0]?.fetch_html || null;
    })
  );

  // Parse H2s from each successful fetch
  for (const result of htmlResults) {
    if (result.status !== "fulfilled" || !result.value) continue;
    
    const $ = cheerio.load(result.value);
    $("h2").each((_, el) => {
      const text = $(el).text().trim().toLowerCase();
      // Filter reasonable H2 lengths
      if (text.length >= 5 && text.length <= 100) {
        h2Counts.set(text, (h2Counts.get(text) || 0) + 1);
      }
    });
  }

  // Return H2s appearing in 2+ pages, sorted by frequency
  return Array.from(h2Counts.entries())
    .filter(([_, count]) => count >= 2)
    .map(([heading, frequency]) => ({ heading, frequency }))
    .sort((a, b) => b.frequency - a.frequency)
    .slice(0, 10);
}
```

**Cost**: ~$0.02/page × 5 = $0.10 per brief

```typescript
interface WordCountStats {
  min: number;
  max: number;
  avg: number;
}

export async function calculateWordCountStats(
  urls: string[]
): Promise<WordCountStats> {
  const wordCounts: number[] = [];

  const htmlResults = await Promise.allSettled(
    urls.slice(0, 5).map(async (url) => {
      const response = await dataForSeoClient.post("/on_page/instant_pages", [{
        url,
        enable_javascript: true,
        load_resources: false,
      }]);
      return response.data?.tasks?.[0]?.result?.[0]?.items?.[0]?.fetch_html || null;
    })
  );

  for (const result of htmlResults) {
    if (result.status !== "fulfilled" || !result.value) continue;
    
    const $ = cheerio.load(result.value);
    
    // Remove non-content elements
    $("script, style, nav, footer, header, aside, .sidebar, .comments").remove();
    
    // Get main content
    const content = $("article, main, .content, .post-content, [role='main']")
      .first()
      .text();
    
    // Fallback to body if no main content found
    const text = content || $("body").text();
    
    // Count words
    const words = text.split(/\s+/).filter((w) => w.length > 0);
    if (words.length > 100) { // Filter out failed extractions
      wordCounts.push(words.length);
    }
  }

  if (wordCounts.length === 0) {
    return { min: 1500, max: 2500, avg: 2000 }; // Sensible defaults
  }

  return {
    min: Math.min(...wordCounts),
    max: Math.max(...wordCounts),
    avg: Math.round(wordCounts.reduce((a, b) => a + b, 0) / wordCounts.length),
  };
}
```

```typescript
import * as cheerio from "cheerio";
import { dataForSeoClient } from "@/server/lib/dataforseo";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "SerpContentAnalyzer" });

export interface SerpContentAnalysis {
  commonH2s: { heading: string; frequency: number }[];
  wordCountStats: { min: number; max: number; avg: number };
  analyzedUrls: number;
}

export async function analyzeSerpContent(
  urls: string[]
): Promise<SerpContentAnalysis> {
  const h2Counts = new Map<string, number>();
  const wordCounts: number[] = [];
  let analyzedUrls = 0;

  // Batch fetch - DataForSEO supports up to 100 URLs per request
  const response = await dataForSeoClient.post("/on_page/instant_pages", 
    urls.slice(0, 5).map(url => ({
      url,
      enable_javascript: true,
      load_resources: false,
    }))
  );

  const results = response.data?.tasks?.[0]?.result || [];

  for (const result of results) {
    const html = result?.items?.[0]?.fetch_html;
    if (!html) continue;

    analyzedUrls++;
    const $ = cheerio.load(html);

    // Extract H2s
    $("h2").each((_, el) => {
      const text = $(el).text().trim().toLowerCase();
      if (text.length >= 5 && text.length <= 100) {
        h2Counts.set(text, (h2Counts.get(text) || 0) + 1);
      }
    });

    // Extract word count
    $("script, style, nav, footer, header, aside, .sidebar").remove();
    const content = $("article, main, .content, [role='main']").first().text() 
                    || $("body").text();
    const words = content.split(/\s+/).filter((w) => w.length > 0);
    if (words.length > 100) {
      wordCounts.push(words.length);
    }
  }

  log.info("SERP content analysis complete", { analyzedUrls, h2Count: h2Counts.size });

  return {
    commonH2s: Array.from(h2Counts.entries())
      .filter(([_, count]) => count >= 2)
      .map(([heading, frequency]) => ({ heading, frequency }))
      .sort((a, b) => b.frequency - a.frequency)
      .slice(0, 10),
    wordCountStats: wordCounts.length > 0
      ? {
          min: Math.min(...wordCounts),
          max: Math.max(...wordCounts),
          avg: Math.round(wordCounts.reduce((a, b) => a + b, 0) / wordCounts.length),
        }
      : { min: 1500, max: 2500, avg: 2000 },
    analyzedUrls,
  };
}
```

- [ ] Cost is $0.10 per brief (not $0.20)

```python
def _build_article_prompt(self, settings: ArticleSettings) -> str:
    prompt_parts = [
        f"Write a comprehensive article about: {settings.keyword}",
        f"Target word count: {settings.target_word_count} words",
    ]
    
    # Add H2 suggestions
    if settings.suggested_h2s:
        h2_list = "\n".join(f"- {h2}" for h2 in settings.suggested_h2s)
        prompt_parts.append(
            f"\nInclude these H2 headings (adapt wording as needed):\n{h2_list}"
        )
    
    # Add PAA questions as FAQ section
    if settings.paa_questions and len(settings.paa_questions) > 0:
        paa_list = "\n".join(f"- {q}" for q in settings.paa_questions[:5])
        prompt_parts.append(
            f"\nInclude a FAQ section that answers these questions from Google's 'People Also Ask':\n{paa_list}"
        )
    
    # Add voice constraints if provided
    if hasattr(settings, 'voice_constraints') and settings.voice_constraints:
        prompt_parts.append(f"\nVoice and tone guidelines:\n{settings.voice_constraints}")
    
    return "\n\n".join(prompt_parts)
```
