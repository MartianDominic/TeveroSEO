```typescript
import { GoogleGenerativeAI } from "@google/generative-ai";
import { z } from "zod";
import { CircuitBreaker, CircuitOpenError } from "../services/CircuitBreaker";
import {
  ClassificationResponseSchema,
  type BusinessContext,
  type ClassificationItem,
} from "./types";
import { GEMINI_CONFIG, CLASSIFICATION_CONFIG } from "./config";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "GeminiClassifier" });

export class GeminiClassifier {
  private client: GoogleGenerativeAI;
  private model: ReturnType<GoogleGenerativeAI["getGenerativeModel"]>;
  private circuit: CircuitBreaker;

  constructor(apiKey: string) {
    this.client = new GoogleGenerativeAI(apiKey);
    this.model = this.client.getGenerativeModel({
      model: GEMINI_CONFIG.model,
      generationConfig: {
        maxOutputTokens: GEMINI_CONFIG.maxTokens,
        temperature: GEMINI_CONFIG.temperature,
        responseMimeType: "application/json",
      },
    });
    this.circuit = new CircuitBreaker({
      name: "gemini-classifier",
      failureThreshold: 3,
      resetTimeout: 60000,
    });
  }

  async classify(
    keywords: string[],
    context: BusinessContext
  ): Promise<ClassificationItem[]> {
    if (!this.circuit.allowsRequest) {
      throw new CircuitOpenError("gemini-classifier");
    }

    if (keywords.length > CLASSIFICATION_CONFIG.BATCH_SIZE) {
      return this.classifyBatched(keywords, context);
    }

    try {
      const prompt = this.buildPrompt(keywords, context);
      const result = await this.model.generateContent(prompt);
      const text = result.response.text();

      const parsed = ClassificationResponseSchema.safeParse(JSON.parse(text));

      if (!parsed.success) {
        log.warn("Invalid Gemini response", { error: parsed.error.message });
        this.circuit.recordFailure();
        throw new Error(`Invalid Gemini response: ${parsed.error.message}`);
      }

      this.circuit.recordSuccess();
      return parsed.data.classifications;
    } catch (error) {
      this.circuit.recordFailure();
      throw error;
    }
  }

  private async classifyBatched(
    keywords: string[],
    context: BusinessContext
  ): Promise<ClassificationItem[]> {
    const results: ClassificationItem[] = [];
    for (let i = 0; i < keywords.length; i += CLASSIFICATION_CONFIG.BATCH_SIZE) {
      const batch = keywords.slice(i, i + CLASSIFICATION_CONFIG.BATCH_SIZE);
      const batchResults = await this.classify(batch, context);
      results.push(...batchResults);
    }
    return results;
  }

  private buildPrompt(keywords: string[], context: BusinessContext): string {
    // Same prompt structure as GrokClassifier for consistency
    const negativeSection = context.negativeAssociations
      ? `
NOT services: ${context.negativeAssociations.notServices.join(", ")}
Adjacent verticals to EXCLUDE: ${context.negativeAssociations.adjacentVerticals.join(", ")}`
      : "";

    return `You are an expert keyword classifier for Lithuanian e-commerce and B2B businesses.

<business-context>
Business: ${context.businessName}
Industry: ${context.industry}
Services: ${context.services.join(", ")}
Target: ${context.targetAudience}
${negativeSection}
</business-context>

<keywords>
${keywords.map((k, i) => `${i + 1}. ${k}`).join("\n")}
</keywords>

Return JSON: {"classifications": [{"keyword": "...", "include": true/false, "confidence": 0.0-1.0, "type": "product"|"long_tail"|"question"|"local"|"comparison"|null, "reasoning": "..."}]}`;
  }

  get isCircuitOpen(): boolean {
    return !this.circuit.allowsRequest;
  }

  resetCircuit(): void {
    this.circuit.reset();
  }
}
```

```typescript
import { GrokClassifier } from "./GrokClassifier";
import { GeminiClassifier } from "./GeminiClassifier";
import { ResilientClassifier } from "../services/ResilientClassifier";
import { CLASSIFICATION_CONFIG } from "./config";
import type { BusinessContext, ClassifiedKeyword, ClassificationItem } from "./types";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "ClassificationPipeline" });

export interface ClassificationStats {
  totalInput: number;
  pass1Resolved: number;
  pass2Resolved: number;
  excluded: number;
  included: number;
  pass1Rate: number;
}

export interface PipelineConfig {
  xaiApiKey?: string;
  geminiApiKey?: string;
  claudeApiKey?: string;
  confidenceThreshold?: number;
}

export class ClassificationPipeline {
  private grok: GrokClassifier | null;
  private gemini: GeminiClassifier | null;
  private claude: ResilientClassifier;
  private confidenceThreshold: number;

  constructor(config: PipelineConfig) {
    this.grok = config.xaiApiKey ? new GrokClassifier(config.xaiApiKey) : null;
    this.gemini = config.geminiApiKey ? new GeminiClassifier(config.geminiApiKey) : null;
    this.claude = new ResilientClassifier({
      claudeApiKey: config.claudeApiKey,
    });
    this.confidenceThreshold = config.confidenceThreshold ?? CLASSIFICATION_CONFIG.CONFIDENCE_THRESHOLD;
  }

  async classify(
    keywords: string[],
    context: BusinessContext
  ): Promise<{ keywords: ClassifiedKeyword[]; stats: ClassificationStats }> {
    const stats: ClassificationStats = {
      totalInput: keywords.length,
      pass1Resolved: 0,
      pass2Resolved: 0,
      excluded: 0,
      included: 0,
      pass1Rate: 0,
    };

    // Pass 1: Grok (or Gemini fallback)
    let pass1Results: ClassificationItem[];

    if (this.grok && !this.grok.isCircuitOpen) {
      try {
        pass1Results = await this.grok.classify(keywords, context);
        log.info("Pass 1 (Grok) complete", { count: pass1Results.length });
      } catch (error) {
        log.warn("Grok failed, falling back to Gemini", { error });
        pass1Results = await this.runGeminiFallback(keywords, context);
      }
    } else {
      pass1Results = await this.runGeminiFallback(keywords, context);
    }

    // Separate resolved vs uncertain
    const resolved: ClassifiedKeyword[] = [];
    const uncertain: string[] = [];

    for (const result of pass1Results) {
      if (result.confidence >= this.confidenceThreshold) {
        resolved.push({
          ...result,
          pass: 1,
        });
        stats.pass1Resolved++;
        if (result.include) {
          stats.included++;
        } else {
          stats.excluded++;
        }
      } else {
        uncertain.push(result.keyword);
      }
    }

    log.info("Pass 1 results", {
      resolved: resolved.length,
      uncertain: uncertain.length,
      threshold: this.confidenceThreshold,
    });

    // Pass 2: Claude Sonnet for uncertain keywords
    if (uncertain.length > 0) {
      const categories = context.services; // Use services as classification categories
      const pass2Results = await this.claude.classifyBatch(uncertain, categories);

      for (const result of pass2Results) {
        resolved.push({
          keyword: result.keyword,
          include: result.confidence >= 0.5, // Lower threshold for Pass 2
          confidence: result.confidence,
          type: null, // Claude doesn't return type in current implementation
          reasoning: result.reasoning || "",
          pass: 2,
        });
        stats.pass2Resolved++;
        if (result.confidence >= 0.5) {
          stats.included++;
        } else {
          stats.excluded++;
        }
      }

      log.info("Pass 2 (Claude) complete", { count: pass2Results.length });
    }

    stats.pass1Rate = (stats.pass1Resolved / stats.totalInput) * 100;

    // Filter to included keywords only for output
    const included = resolved.filter((k) => k.include);

    return { keywords: included, stats };
  }

  private async runGeminiFallback(
    keywords: string[],
    context: BusinessContext
  ): Promise<ClassificationItem[]> {
    if (!this.gemini || this.gemini.isCircuitOpen) {
      log.warn("No fallback available, returning empty classification");
      return keywords.map((keyword) => ({
        keyword,
        include: true,
        confidence: 0.5,
        type: null,
        reasoning: "No classifier available",
      }));
    }

    return this.gemini.classify(keywords, context);
  }

  getCircuitStates(): { grok: boolean; gemini: boolean; claude: { claude: string; openai: string } } {
    return {
      grok: this.grok?.isCircuitOpen ?? true,
      gemini: this.gemini?.isCircuitOpen ?? true,
      claude: this.claude.getCircuitStates(),
    };
  }
}

export function createClassificationPipeline(): ClassificationPipeline {
  return new ClassificationPipeline({
    xaiApiKey: process.env.XAI_API_KEY,
    geminiApiKey: process.env.GEMINI_API_KEY,
    claudeApiKey: process.env.ANTHROPIC_API_KEY,
  });
}
```

```typescript
import pLimit from "p-limit";
import { KeywordDeduplicator } from "../services/KeywordDeduplicator";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "KeywordUniverseBuilder" });

export interface DataForSEOAutocomplete {
  keywords(keyword: string, options?: { location_code?: number }): Promise<{ keywords: string[] }>;
  keywordIdeas(keyword: string, options?: { location_code?: number; limit?: number }): Promise<{ keywords: { keyword: string; search_volume?: number }[] }>;
  relatedKeywords(keyword: string, options?: { location_code?: number; limit?: number }): Promise<{ keywords: { keyword: string; search_volume?: number }[] }>;
}

export interface UniverseBuilderConfig {
  maxKeywordsPerSeed?: number;
  locationCode?: number;
  concurrency?: number;
}

const DEFAULT_CONFIG: Required<UniverseBuilderConfig> = {
  maxKeywordsPerSeed: 30,
  locationCode: 2440, // Lithuania
  concurrency: 3,
};

export class KeywordUniverseBuilder {
  private dataForSEO: DataForSEOAutocomplete;
  private deduplicator: KeywordDeduplicator;
  private config: Required<UniverseBuilderConfig>;

  constructor(dataForSEO: DataForSEOAutocomplete, config?: UniverseBuilderConfig) {
    this.dataForSEO = dataForSEO;
    this.deduplicator = new KeywordDeduplicator();
    this.config = { ...DEFAULT_CONFIG, ...config };
  }

  async expand(seeds: string[], domain?: string): Promise<string[]> {
    if (seeds.length === 0) {
      return [];
    }

    const limit = pLimit(this.config.concurrency);
    const allKeywords: string[] = [...seeds];

    log.info("Starting keyword expansion", { seedCount: seeds.length });

    // Process seeds in parallel with concurrency limit
    const expansions = await Promise.all(
      seeds.map((seed) =>
        limit(async () => {
          try {
            return await this.expandSeed(seed);
          } catch (error) {
            log.warn("Failed to expand seed", { seed, error });
            return [];
          }
        })
      )
    );

    for (const expanded of expansions) {
      allKeywords.push(...expanded);
    }

    // Deduplicate
    const unique = this.deduplicator.deduplicate(allKeywords);

    log.info("Keyword expansion complete", {
      seeds: seeds.length,
      raw: allKeywords.length,
      unique: unique.length,
    });

    return unique;
  }

  private async expandSeed(seed: string): Promise<string[]> {
    const results: string[] = [];

    // Autocomplete suggestions
    try {
      const autocomplete = await this.dataForSEO.keywords(seed, {
        location_code: this.config.locationCode,
      });
      results.push(...autocomplete.keywords.slice(0, 10));
    } catch {
      log.debug("Autocomplete failed for seed", { seed });
    }

    // Keyword ideas
    try {
      const ideas = await this.dataForSEO.keywordIdeas(seed, {
        location_code: this.config.locationCode,
        limit: this.config.maxKeywordsPerSeed,
      });
      results.push(...ideas.keywords.map((k) => k.keyword));
    } catch {
      log.debug("Keyword ideas failed for seed", { seed });
    }

    // Related keywords
    try {
      const related = await this.dataForSEO.relatedKeywords(seed, {
        location_code: this.config.locationCode,
        limit: 10,
      });
      results.push(...related.keywords.map((k) => k.keyword));
    } catch {
      log.debug("Related keywords failed for seed", { seed });
    }

    return results;
  }
}

export function createKeywordUniverseBuilder(
  dataForSEO: DataForSEOAutocomplete,
  config?: UniverseBuilderConfig
): KeywordUniverseBuilder {
  return new KeywordUniverseBuilder(dataForSEO, config);
}
```
