/**
 * Site Connections API Routes
 * Phase 31-04: API Endpoints
 *
 * CRUD operations for site connections.
 * All credential data is encrypted by ConnectionService.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { json } from "@tanstack/start";
import { z } from "zod";
import { connectionService } from "@/server/features/connections/services/ConnectionService";
import { PLATFORM_TYPES } from "@/server/features/connections/types";

const CreateConnectionSchema = z.object({
  clientId: z.string().min(1),
  platform: z.enum(PLATFORM_TYPES),
  siteUrl: z.string().url(),
  displayName: z.string().optional(),
  credentials: z.record(z.string()),
});

export const Route = createAPIFileRoute("/api/connections")({
  // GET /api/connections?clientId=X
  GET: async ({ request }) => {
    const url = new URL(request.url);
    const clientId = url.searchParams.get("clientId");

    if (!clientId) {
      return json({ error: "clientId query parameter required" }, { status: 400 });
    }

    try {
      const connections = await connectionService.getConnectionsForClient(clientId);
      return json(connections);
    } catch (error) {
      return json({ error: (error as Error).message }, { status: 500 });
    }
  },

  // POST /api/connections - create new connection
  POST: async ({ request }) => {
    try {
      const body = await request.json();
      const parsed = CreateConnectionSchema.parse(body);

      const connection = await connectionService.createConnection({
        clientId: parsed.clientId,
        platform: parsed.platform,
        siteUrl: parsed.siteUrl,
        displayName: parsed.displayName,
        credentials: parsed.credentials,
      });

      return json(connection, { status: 201 });
    } catch (error) {
      if (error instanceof z.ZodError) {
        return json({ error: "Invalid input", details: error.errors }, { status: 400 });
      }
      return json({ error: (error as Error).message }, { status: 500 });
    }
  },
});
```

/**
 * Single Connection API Routes
 * GET, DELETE operations for a specific connection.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { json } from "@tanstack/start";
import { connectionService } from "@/server/features/connections/services/ConnectionService";

export const Route = createAPIFileRoute("/api/connections/$id")({
  // GET /api/connections/:id
  GET: async ({ params }) => {
    try {
      const connection = await connectionService.getConnection(params.id);

      if (!connection) {
        return json({ error: "Connection not found" }, { status: 404 });
      }

      return json(connection);
    } catch (error) {
      return json({ error: (error as Error).message }, { status: 500 });
    }
  },

  // DELETE /api/connections/:id
  DELETE: async ({ params }) => {
    try {
      await connectionService.deleteConnection(params.id);
      return new Response(null, { status: 204 });
    } catch (error) {
      return json({ error: (error as Error).message }, { status: 500 });
    }
  },
});

/**
 * Platform Detection API Route
 * Phase 31-04: API Endpoints
 *
 * Detects the CMS platform of a given domain using multi-probe fingerprinting.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { json } from "@tanstack/start";
import { z } from "zod";
import { detectPlatform } from "@/server/features/connections/services/PlatformDetector";

const DetectPlatformSchema = z.object({
  domain: z.string().min(1, "Domain is required"),
});

export const Route = createAPIFileRoute("/api/detect-platform")({
  // POST /api/detect-platform
  // Body: { domain: "example.com" }
  // Returns: { platform, confidence, signals }
  POST: async ({ request }) => {
    try {
      const body = await request.json();
      const parsed = DetectPlatformSchema.parse(body);

      const result = await detectPlatform(parsed.domain);
      return json(result);
    } catch (error) {
      if (error instanceof z.ZodError) {
        return json({ error: "Invalid input", details: error.errors }, { status: 400 });
      }
      // detectPlatform handles its own errors and returns a result
      // This catch is for unexpected errors
      return json({ error: (error as Error).message }, { status: 500 });
    }
  },
});
```
