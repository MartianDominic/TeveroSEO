**File:** `open-seo-main/src/client/hooks/useKeywordAnalysis.ts`

```typescript
const SSE_CONFIG = {
  maxRetries: 5,
  baseDelay: 1000,      // 1 second
  maxDelay: 30000,      // 30 seconds
  backoffMultiplier: 2,
};

function useSSEWithReconnect(url: string, options: SSEOptions) {
  const [retryCount, setRetryCount] = useState(0);
  const eventSourceRef = useRef<EventSource | null>(null);
  
  const connect = useCallback(() => {
    const es = new EventSource(url);
    eventSourceRef.current = es;
    
    es.onopen = () => {
      setRetryCount(0); // Reset on successful connection
      options.onOpen?.();
    };
    
    es.onerror = (error) => {
      es.close();
      
      if (retryCount < SSE_CONFIG.maxRetries) {
        const delay = Math.min(
          SSE_CONFIG.baseDelay * Math.pow(SSE_CONFIG.backoffMultiplier, retryCount),
          SSE_CONFIG.maxDelay
        );
        
        console.log(`SSE reconnecting in ${delay}ms (attempt ${retryCount + 1}/${SSE_CONFIG.maxRetries})`);
        
        setTimeout(() => {
          setRetryCount(prev => prev + 1);
          connect();
        }, delay);
      } else {
        options.onMaxRetriesExceeded?.();
      }
    };
    
    es.onmessage = options.onMessage;
  }, [url, retryCount, options]);
  
  useEffect(() => {
    connect();
    return () => eventSourceRef.current?.close();
  }, [connect]);
  
  return { retryCount, isConnected: eventSourceRef.current?.readyState === 1 };
}
```

**Verification:** Kill network mid-analysis → reconnects and resumes

### Task 2: IndexedDB Checkpoint System (3 hours)

```typescript
import { openDB, IDBPDatabase } from 'idb';

interface AnalysisCheckpoint {
  sessionId: string;
  timestamp: number;
  stage: 'constraints' | 'embedding' | 'clustering' | 'scoring' | 'complete';
  progress: number;          // 0-100
  constraints: Constraints;
  keywords: Keyword[];
  partialResults?: PartialResults;
}

const DB_NAME = 'keyword-analysis';
const STORE_NAME = 'checkpoints';

class CheckpointManager {
  private db: IDBPDatabase | null = null;
  
  async init() {
    this.db = await openDB(DB_NAME, 1, {
      upgrade(db) {
        db.createObjectStore(STORE_NAME, { keyPath: 'sessionId' });
      },
    });
  }
  
  async saveCheckpoint(checkpoint: AnalysisCheckpoint): Promise<void> {
    if (!this.db) await this.init();
    await this.db!.put(STORE_NAME, {
      ...checkpoint,
      timestamp: Date.now(),
    });
  }
  
  async getCheckpoint(sessionId: string): Promise<AnalysisCheckpoint | null> {
    if (!this.db) await this.init();
    return await this.db!.get(STORE_NAME, sessionId);
  }
  
  async getLatestCheckpoint(): Promise<AnalysisCheckpoint | null> {
    if (!this.db) await this.init();
    const all = await this.db!.getAll(STORE_NAME);
    if (all.length === 0) return null;
    
    // Return most recent
    return all.sort((a, b) => b.timestamp - a.timestamp)[0];
  }
  
  async clearCheckpoint(sessionId: string): Promise<void> {
    if (!this.db) await this.init();
    await this.db!.delete(STORE_NAME, sessionId);
  }
  
  async clearOldCheckpoints(maxAgeMs: number = 24 * 60 * 60 * 1000): Promise<void> {
    if (!this.db) await this.init();
    const all = await this.db!.getAll(STORE_NAME);
    const cutoff = Date.now() - maxAgeMs;
    
    for (const checkpoint of all) {
      if (checkpoint.timestamp < cutoff) {
        await this.db!.delete(STORE_NAME, checkpoint.sessionId);
      }
    }
  }
}

export const checkpointManager = new CheckpointManager();
```

### Task 4: Graceful Degradation (2 hours)

**File:** `open-seo-main/src/server/features/keywords/lib/graceful-degradation.ts`

```typescript
interface DegradationConfig {
  stage: string;
  optional: boolean;
  fallback?: () => Promise<any>;
  timeoutMs: number;
}

const PIPELINE_STAGES: DegradationConfig[] = [
  { stage: 'constraint_extraction', optional: false, timeoutMs: 30000 },
  { stage: 'keyword_enrichment', optional: true, fallback: useRawKeywords, timeoutMs: 60000 },
  { stage: 'embedding', optional: false, timeoutMs: 120000 },
  { stage: 'clustering', optional: true, fallback: skipClustering, timeoutMs: 60000 },
  { stage: 'funnel_classification', optional: false, timeoutMs: 60000 },
  { stage: 'scoring', optional: false, timeoutMs: 30000 },
  { stage: 'labeling', optional: true, fallback: useHeuristicLabels, timeoutMs: 30000 },
];

async function runWithDegradation<T>(
  stage: string,
  operation: () => Promise<T>,
): Promise<T | null> {
  const config = PIPELINE_STAGES.find(s => s.stage === stage);
  if (!config) throw new Error(`Unknown stage: ${stage}`);
  
  try {
    return await withTimeout(operation(), config.timeoutMs);
  } catch (error) {
    if (config.optional && config.fallback) {
      console.warn(`Stage ${stage} failed, using fallback`, error);
      return await config.fallback();
    }
    throw error;
  }
}

// Fallback implementations
async function useRawKeywords(keywords: string[]): Promise<EnrichedKeyword[]> {
  // Return keywords without DataForSEO enrichment
  return keywords.map(kw => ({
    keyword: kw,
    volume: null,
    difficulty: null,
    cpc: null,
    source: 'raw',
  }));
}

async function skipClustering(keywords: KeywordWithEmbedding[]): Promise<Cluster[]> {
  // Put all keywords in a single "Unclustered" group
  return [{
    id: 'unclustered',
    label: 'All Keywords',
    keywords,
  }];
}

async function useHeuristicLabels(clusters: Cluster[]): Promise<Cluster[]> {
  // Generate labels from top keyword + intent
  return clusters.map(c => ({
    ...c,
    label: `${c.keywords[0].keyword} (${c.keywords.length} keywords)`,
  }));
}
```

```typescript
export enum ErrorCode {
  NETWORK_ERROR = 'NETWORK_ERROR',
  API_TIMEOUT = 'API_TIMEOUT',
  RATE_LIMITED = 'RATE_LIMITED',

  EMBEDDING_FAILED = 'EMBEDDING_FAILED',
  LLM_ERROR = 'LLM_ERROR',

interface ErrorTemplate {
  code: ErrorCode;
  title: string;
  message: string;
  action: string;
  retryable: boolean;
}

    code: ErrorCode.RATE_LIMITED,
    title: 'Rate Limited',
    message: 'Too many requests. Please wait a moment before trying again.',
    action: 'Wait 60 seconds',

  },
  [ErrorCode.INVALID_INPUT]: {
    code: ErrorCode.INVALID_INPUT,
    title: 'Invalid Input',
    message: 'Some keywords could not be processed. Check for special characters or duplicates.',
    action: 'Fix input',
    retryable: false,
  },
  [ErrorCode.EMBEDDING_FAILED]: {
    code: ErrorCode.EMBEDDING_FAILED,
    title: 'Embedding Error',
    message: 'Failed to generate keyword embeddings. The analysis will continue with reduced accuracy.',
    action: 'Continue anyway',
    retryable: true,
  },
  [ErrorCode.LLM_ERROR]: {
    code: ErrorCode.LLM_ERROR,
    title: 'AI Classification Error',
    message: 'The AI classification service is temporarily unavailable. Using pattern-based fallback.',
    action: 'Continue with patterns',
    retryable: false,
  },
  [ErrorCode.INTERNAL_ERROR]: {
    code: ErrorCode.INTERNAL_ERROR,
    title: 'Something Went Wrong',
    message: 'An unexpected error occurred. Our team has been notified.',
    action: 'Report issue',
    retryable: false,
  },
};

export function getErrorTemplate(code: ErrorCode): ErrorTemplate {
  return ERROR_TEMPLATES[code] ?? ERROR_TEMPLATES[ErrorCode.INTERNAL_ERROR];
}

export function classifyError(error: Error): ErrorCode {
  const message = error.message.toLowerCase();
  
  if (message.includes('network') || message.includes('fetch')) {
    return ErrorCode.NETWORK_ERROR;
  }
  if (message.includes('timeout') || message.includes('timed out')) {
    return ErrorCode.API_TIMEOUT;
  }
  if (message.includes('429') || message.includes('rate limit')) {
    return ErrorCode.RATE_LIMITED;
  }
  if (message.includes('embedding') || message.includes('jina')) {
    return ErrorCode.EMBEDDING_FAILED;
  }
  if (message.includes('groq') || message.includes('llm') || message.includes('grok')) {
    return ErrorCode.LLM_ERROR;
  }
  
  return ErrorCode.INTERNAL_ERROR;
}
```

```typescript
/**
 * Queue database operations when connection is unavailable.
 * Replay when connection is restored.
 */
import { openDB } from 'idb';

interface QueuedOperation {
  id: string;
  table: string;
  operation: 'insert' | 'update' | 'delete';
  data: any;
  timestamp: number;
}

class OfflineQueue {
  private db: IDBPDatabase | null = null;
  private isOnline: boolean = true;
  
  async init() {
    this.db = await openDB('offline-queue', 1, {
      upgrade(db) {
        db.createObjectStore('operations', { keyPath: 'id' });
      },
    });
    
    // Listen for online/offline events
    window.addEventListener('online', () => this.flush());
    window.addEventListener('offline', () => this.isOnline = false);
  }
  
  async enqueue(operation: Omit<QueuedOperation, 'id' | 'timestamp'>) {
    if (!this.db) await this.init();
    
    await this.db!.put('operations', {
      ...operation,
      id: crypto.randomUUID(),
      timestamp: Date.now(),
    });
  }
  
  async flush() {
    if (!this.db) return;
    
    const operations = await this.db.getAll('operations');
    
    for (const op of operations.sort((a, b) => a.timestamp - b.timestamp)) {
      try {
        await this.executeOperation(op);
        await this.db.delete('operations', op.id);
      } catch (error) {
        console.error('Failed to flush operation', op, error);
        break; // Stop on first failure to maintain order
      }
```

### Task 7: Rate Limit Backoff + Batch Splitting (1 hour)

```typescript
async function embedWithBackoff(
  texts: string[],
  maxRetries: number = 3,
): Promise<number[][]> {
  let retries = 0;
  let batchSize = EMBEDDING_CONFIG.batchSize;
  
  while (retries < maxRetries) {
    try {
      return await embedBatched(texts, batchSize);
    } catch (error) {
      if (isRateLimitError(error)) {
        // Wait and retry
        const delay = Math.pow(2, retries) * 1000;
        console.log(`Rate limited, waiting ${delay}ms...`);
        await sleep(delay);
        retries++;
      } else if (isBatchTooLargeError(error)) {
        // Split batch and retry
        batchSize = Math.max(1, Math.floor(batchSize / 2));
        console.log(`Batch too large, reducing to ${batchSize}`);
      } else {
        throw error;
      }
    }
  }
  
  throw new Error('Max retries exceeded for embedding');
}
```
