### Task 1: Progressive Model Selection (2 hours)

**File:** `open-seo-main/src/server/features/keywords/services/model-router.ts`

```typescript
/**
 * Route to cheapest model that can handle the task.
 * Simple tasks → cheap models, complex → expensive.
 */

interface ModelConfig {
  id: string;
  provider: 'groq' | 'grok' | 'openai';
  costPerMToken: { input: number; output: number };
  capabilities: Set<string>;
  maxBatchSize: number;
  rateLimit: number;  // RPM
}

const MODELS: ModelConfig[] = [
  {
    id: 'llama-3.1-8b-instant',
    provider: 'groq',
    costPerMToken: { input: 0.05, output: 0.08 },
    capabilities: new Set(['classification', 'simple_labeling']),
    maxBatchSize: 200,
    rateLimit: 6000,
  },
  {
    id: 'llama-3.3-70b',
    provider: 'groq',
    costPerMToken: { input: 0.59, output: 0.79 },
    capabilities: new Set(['classification', 'labeling', 'reasoning', 'lithuanian_complex']),
    maxBatchSize: 100,
    rateLimit: 1000,
  },
  {
    id: 'grok-2-mini',
    provider: 'grok',
    costPerMToken: { input: 0.30, output: 1.00 },
    capabilities: new Set(['classification', 'labeling', 'reasoning']),
    maxBatchSize: 50,
    rateLimit: 60,
  },
];

type TaskType = 'classification' | 'labeling' | 'reasoning' | 'lithuanian_complex';

class ModelRouter {
  private circuitBreakers = new Map<string, CircuitBreaker>();
  
  selectModel(task: TaskType, batchSize: number): ModelConfig {
    // Filter by capability and batch size
    const candidates = MODELS.filter(m => 
      m.capabilities.has(task) && 
      m.maxBatchSize >= batchSize &&
      !this.circuitBreakers.get(m.id)?.isOpen
    );
    
    if (candidates.length === 0) {
      throw new Error(`No model available for task: ${task}`);
    }
    
    // Sort by cost (cheapest first)
    candidates.sort((a, b) => 
      (a.costPerMToken.input + a.costPerMToken.output) - 
      (b.costPerMToken.input + b.costPerMToken.output)
    );
    
    return candidates[0];
  }
  
  async call(
    task: TaskType,
    messages: Message[],
    batchSize: number = 1,
  ): Promise<{ result: string; model: string; cost: number }> {
    const model = this.selectModel(task, batchSize);
    const breaker = this.getOrCreateBreaker(model.id);
    
    return breaker.execute(async () => {
      const response = await this.callProvider(model, messages);
      const cost = this.calculateCost(model, response.usage);
      
      return {
        result: response.content,
        model: model.id,
        cost,
      };
    });
  }
  
  private calculateCost(model: ModelConfig, usage: TokenUsage): number {
    return (usage.input / 1_000_000 * model.costPerMToken.input) +
           (usage.output / 1_000_000 * model.costPerMToken.output);
  }
}

export const modelRouter = new ModelRouter();
```

### Task 3: Usage API Endpoints (1.5 hours)

**File:** `open-seo-main/src/server/api/usage.ts`

```typescript
import { Router } from 'express';
import { db } from '../db';

const router = Router();

// GET /api/usage/stats
router.get('/stats', async (req, res) => {
  const workspaceId = req.user.workspaceId;
  const days = parseInt(req.query.days as string) || 30;
  
  const since = new Date();
  since.setDate(since.getDate() - days);
  
  // Daily aggregates
  const daily = await db.query(`
    SELECT 
      DATE(timestamp) as date,
      COUNT(DISTINCT session_id) as analyses,
      SUM(keyword_count) as keywords,
      SUM(cost) as cost
    FROM api_costs
    WHERE workspace_id = $1 AND timestamp >= $2
    GROUP BY DATE(timestamp)
    ORDER BY date
  `, [workspaceId, since]);
  
  // Totals
  const totals = await db.queryOne(`
    SELECT 
      COUNT(DISTINCT session_id) as analyses,
      SUM(keyword_count) as keywords,
      SUM(cost) as cost
    FROM api_costs
    WHERE workspace_id = $1 AND timestamp >= $2
  `, [workspaceId, since]);
  
  // Breakdown by service
  const breakdown = await db.query(`
    SELECT 
      service,
      SUM(cost) as cost
    FROM api_costs
    WHERE workspace_id = $1 AND timestamp >= $2
    GROUP BY service
    ORDER BY cost DESC
  `, [workspaceId, since]);
  
  const totalCost = breakdown.reduce((sum, b) => sum + b.cost, 0);
  
  res.json({
    daily: daily.rows,
    totals: {
      analyses: totals.analyses,
      keywords: totals.keywords,
      cost: totals.cost,
      avgCostPerAnalysis: totals.analyses > 0 ? totals.cost / totals.analyses : 0,
      avgCostPerKeyword: totals.keywords > 0 ? totals.cost / totals.keywords : 0,
    },
    breakdown: breakdown.rows.map(b => ({
      service: b.service,
      cost: b.cost,
      percentage: totalCost > 0 ? (b.cost / totalCost) * 100 : 0,
    })),
  });
});

// GET /api/usage/analysis/:sessionId
router.get('/analysis/:sessionId', async (req, res) => {
  const { sessionId } = req.params;
  
  const costs = await db.query(`
    SELECT service, operation, input_tokens, output_tokens, api_calls, cost, timestamp
    FROM api_costs
    WHERE session_id = $1
    ORDER BY timestamp
  `, [sessionId]);
  
  const total = costs.rows.reduce((sum, c) => sum + c.cost, 0);
  
  res.json({
    sessionId,
    totalCost: total,
    entries: costs.rows,
  });
});

export default router;
```

**Note:** Uses existing `api_costs` table from `api-costs-schema.ts`

interface AnalysisCostSummaryProps {
  sessionId: string;
}

export function AnalysisCostSummary({ sessionId }: AnalysisCostSummaryProps) {
  const { data: cost } = useQuery({
    queryKey: ['analysis-cost', sessionId],
    queryFn: () => fetch(`/api/usage/analysis/${sessionId}`).then(r => r.json()),
  });
  
  if (!cost) return null;
  
  return (
    <div className="flex items-center gap-2 text-xs text-muted-foreground">
      <span>Analysis cost:</span>
      <span className="font-mono">${cost.totalCost.toFixed(4)}</span>
      <TooltipProvider>
        <Tooltip>
          <TooltipTrigger>
            <InfoIcon className="h-3 w-3" />
          </TooltipTrigger>
          <TooltipContent>
            <div className="space-y-1">
              {cost.entries.map((entry, i) => (
                <div key={i} className="flex justify-between gap-4">
                  <span>{entry.service}</span>
                  <span className="font-mono">${entry.cost.toFixed(6)}</span>
                </div>
              ))}
            </div>
          </TooltipContent>
        </Tooltip>
      </TooltipProvider>
    </div>
  );
}
