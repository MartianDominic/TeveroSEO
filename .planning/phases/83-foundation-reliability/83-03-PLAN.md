### Task 1: Extend Embedding Cache with Redis L2 (2 hours)

**File:** `open-seo-main/src/server/features/keywords/services/ResilientEmbedding.ts`

Extend existing `InMemoryEmbeddingCache` to add Redis as L2 layer:

```typescript
import { Redis } from 'ioredis';

interface CacheStats {
  l1Hits: number;
  l2Hits: number;
  misses: number;
  hitRate: number;
}

/**
 * 2-tier cache: L1 (memory) → L2 (Redis)
 * Extends existing InMemoryEmbeddingCache with Redis persistence.
 */
export class TieredEmbeddingCache implements EmbeddingCache {
  private readonly l1: InMemoryEmbeddingCache;
  private readonly l2: Redis | null;
  private stats: CacheStats = { l1Hits: 0, l2Hits: 0, misses: 0, hitRate: 0 };
  
  constructor(redis?: Redis, l1MaxSize = 10000, l1TtlMs = 3600000) {
    this.l1 = new InMemoryEmbeddingCache(l1MaxSize, l1TtlMs);
    this.l2 = redis || null;
  }
  
  async get(text: string): Promise<number[] | null> {
    // L1: Memory (existing implementation)
    const l1Result = await this.l1.get(text);
    if (l1Result) {
      this.stats.l1Hits++;
      return l1Result;
    }
    
    // L2: Redis
    if (this.l2) {
      const key = this.getRedisKey(text);
      const l2Result = await this.l2.get(key);
      if (l2Result) {
        this.stats.l2Hits++;
        const vector = JSON.parse(l2Result);
        await this.l1.set(text, vector);  // Promote to L1
        return vector;
      }
    }
    
    this.stats.misses++;
    return null;
  }
  
  async set(text: string, vector: number[]): Promise<void> {
    // Write to L1
    await this.l1.set(text, vector);
    
    // Write to L2 (async, fire-and-forget for speed)
    if (this.l2) {
      const key = this.getRedisKey(text);
      this.l2.setex(key, 30 * 24 * 60 * 60, JSON.stringify(vector)).catch(() => {});
    }
  }
  
  private getRedisKey(text: string): string {
    const normalized = text.toLowerCase().trim();
    // Use SHA256 for consistent key generation
    const hash = require('crypto').createHash('sha256').update(normalized).digest('hex').slice(0, 16);
    return `emb:v5:${hash}`;
  }
  
  async getMany(texts: string[]): Promise<Array<number[] | null>> {
    return Promise.all(texts.map((t) => this.get(t)));
  }
  
  async setMany(texts: string[], vectors: number[][]): Promise<void> {
    await Promise.all(texts.map((t, i) => this.set(t, vectors[i])));
  }
  
  getStats(): CacheStats {
    const total = this.stats.l1Hits + this.stats.l2Hits + this.stats.misses;
    return {
      ...this.stats,
      hitRate: total > 0 ? (this.stats.l1Hits + this.stats.l2Hits) / total : 0,
    };
  }
}
```

**Verification:** Cache hit rate >60% after warm-up

```typescript
function getConversationHash(messages: Message[]): string {
  const content = messages
    .filter(m => m.role === 'user')
    .map(m => m.content)
    .join('|');
  return createHash('sha256').update(content).digest('hex').slice(0, 16);
}

export async function extractConstraints(messages: Message[]): Promise<ExtractedConstraints> {
  const hash = getConversationHash(messages);
  
  const cached = constraintCache.get(hash);
  if (cached) {
    log.debug('Constraint cache hit', { hash });
    return cached;
  }
  
  const constraints = await llmExtractConstraints(messages);
  constraintCache.set(hash, constraints);
  
  return constraints;
}
```

### Task 3: Parallel Pipeline Execution (2.5 hours)

**File:** `open-seo-main/src/server/features/keywords/pipeline/parallel-pipeline.ts`

```typescript
import pLimit from 'p-limit';

interface PipelineStage {
  name: string;
  dependencies: string[];
  execute: (input: unknown) => Promise<unknown>;
}

const PIPELINE_STAGES: PipelineStage[] = [
  { name: 'enrichment', dependencies: [], execute: enrichKeywords },
  { name: 'embedding', dependencies: [], execute: embedKeywords },
  { name: 'patternClassification', dependencies: [], execute: patternClassify },
  { name: 'llmClassification', dependencies: ['patternClassification'], execute: llmClassify },
  { name: 'clustering', dependencies: ['embedding', 'llmClassification'], execute: cluster },
  { name: 'scoring', dependencies: ['clustering'], execute: score },
  { name: 'labeling', dependencies: ['clustering'], execute: label },
];

export class ParallelPipeline {
  private readonly limit = pLimit(4);  // Max 4 concurrent stages
  
  async execute(keywords: string[]): Promise<AnalysisResult> {
    const results = new Map<string, unknown>();
    const completed = new Set<string>();
    const pending = new Map<string, Promise<void>>();
    
    const canRun = (stage: PipelineStage): boolean => {
      return stage.dependencies.every(d => completed.has(d));
    };
    
    const runStage = async (stage: PipelineStage): Promise<void> => {
      // Wait for dependencies
      await Promise.all(
        stage.dependencies
          .filter(d => pending.has(d))
          .map(d => pending.get(d))
      );
      
      const input = this.gatherInputs(stage.dependencies, results);
      const output = await this.limit(() => stage.execute(input));
      
      results.set(stage.name, output);
      completed.add(stage.name);
    };
    
    // Start all stages that can run (respecting dependencies)
    for (const stage of PIPELINE_STAGES) {
      pending.set(stage.name, runStage(stage));
    }
    
    await Promise.all(pending.values());
    return this.assembleResult(results);
  }
  
  private gatherInputs(deps: string[], results: Map<string, unknown>): unknown {
    if (deps.length === 0) return undefined;
    if (deps.length === 1) return results.get(deps[0]);
    return Object.fromEntries(deps.map(d => [d, results.get(d)]));
  }
  
  private assembleResult(results: Map<string, unknown>): AnalysisResult {
    return {
      clusters: results.get('clustering') as Cluster[],
      scores: results.get('scoring') as Score[],
      labels: results.get('labeling') as Label[],
    };
  }
}
```

### Task 4: Worker Thread Pool for CPU Stages (2 hours)

**File:** `open-seo-main/src/server/features/keywords/lib/worker-pool.ts`

```typescript
import { Worker, isMainThread, parentPort } from 'worker_threads';
import os from 'os';

const POOL_SIZE = Math.max(2, os.cpus().length - 2);

interface WorkerTask {
  id: string;
  type: 'hdbscan' | 'scoring' | 'pattern_match';
  data: unknown;
}

interface WorkerResult {
  id: string;
  result: unknown;
  error?: string;
}

class WorkerPool {
  private workers: Worker[] = [];
  private queue: Array<{ task: WorkerTask; resolve: (r: unknown) => void; reject: (e: Error) => void }> = [];
  private busy = new Set<Worker>();
  
  constructor(workerScript: string) {
    for (let i = 0; i < POOL_SIZE; i++) {
      const worker = new Worker(workerScript);
      worker.on('message', (msg: WorkerResult) => this.handleResult(worker, msg));
      worker.on('error', (err) => this.handleError(worker, err));
      this.workers.push(worker);
    }
  }
  
  async execute<T>(type: WorkerTask['type'], data: unknown): Promise<T> {
    const id = crypto.randomUUID();
    
    return new Promise((resolve, reject) => {
      const task: WorkerTask = { id, type, data };
      const idleWorker = this.workers.find(w => !this.busy.has(w));
      
      if (idleWorker) {
        this.busy.add(idleWorker);
        idleWorker.postMessage(task);
        this.queue.push({ task, resolve: resolve as (r: unknown) => void, reject });
      } else {
        this.queue.push({ task, resolve: resolve as (r: unknown) => void, reject });
      }
    });
  }
  
  private handleResult(worker: Worker, msg: WorkerResult): void {
    const idx = this.queue.findIndex(q => q.task.id === msg.id);
    if (idx >= 0) {
      const { resolve, reject } = this.queue.splice(idx, 1)[0];
      if (msg.error) {
        reject(new Error(msg.error));
      } else {
        resolve(msg.result);
      }
    }
    
    this.busy.delete(worker);
    this.processQueue(worker);
  }
  
  private handleError(worker: Worker, error: Error): void {
    this.busy.delete(worker);
    // Find pending task for this worker and reject
  }
  
  private processQueue(worker: Worker): void {
    if (this.queue.length > 0 && !this.busy.has(worker)) {
      const next = this.queue.find(q => !this.busy.has(worker));
      if (next) {
        this.busy.add(worker);
        worker.postMessage(next.task);
      }
    }
  }
  
  async shutdown(): Promise<void> {
    await Promise.all(this.workers.map(w => w.terminate()));
  }
}

export const workerPool = new WorkerPool('./clustering-worker.js');
```

```typescript
import v8 from 'v8';

interface MemoryStatus {
  heapUsed: number;
  heapTotal: number;
  usagePercent: number;
  status: 'ok' | 'warning' | 'critical';
}

const THRESHOLDS = {
  warning: 0.7,   // 70% heap used
  critical: 0.85, // 85% heap used
};

class MemoryMonitor {
  private interval: NodeJS.Timeout | null = null;
  private listeners: ((status: MemoryStatus) => void)[] = [];
  
  start(intervalMs: number = 5000): void {
    this.interval = setInterval(() => {
      const status = this.check();
      if (status.status !== 'ok') {
        this.listeners.forEach(l => l(status));
      }
    }, intervalMs);
  }
  
  stop(): void {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
    }
  }
  
  check(): MemoryStatus {
    const heap = v8.getHeapStatistics();
    const usagePercent = heap.used_heap_size / heap.heap_size_limit;
    
    let status: MemoryStatus['status'] = 'ok';
    if (usagePercent > THRESHOLDS.critical) status = 'critical';
    else if (usagePercent > THRESHOLDS.warning) status = 'warning';
    
    return {
      heapUsed: heap.used_heap_size,
      heapTotal: heap.heap_size_limit,
      usagePercent,
      status,
    };
  }
  
  onPressure(callback: (status: MemoryStatus) => void): () => void {
    this.listeners.push(callback);
    return () => {
      const idx = this.listeners.indexOf(callback);
      if (idx >= 0) this.listeners.splice(idx, 1);
    };
  }
  
  forceGC(): void {
    if (global.gc) {
      global.gc();
    }
  }
}

export const memoryMonitor = new MemoryMonitor();
```

### Task 6: Progressive Result Streaming (1.5 hours)

```typescript
import { useState, useCallback } from 'react';

interface ProgressiveResults {
  keywords: Keyword[];
  clusters: Cluster[];
  stage: 'initializing' | 'embedding' | 'clustering' | 'scoring' | 'complete';
  progress: number;
}

export function useProgressiveResults() {
  const [results, setResults] = useState<ProgressiveResults>({
    keywords: [],
    clusters: [],
    stage: 'initializing',
    progress: 0,
  });
  
  const handleSSEMessage = useCallback((event: MessageEvent) => {
    const data = JSON.parse(event.data);
    
    switch (data.type) {
      case 'keywords_partial':
        setResults(prev => ({
          ...prev,
          keywords: [...prev.keywords, ...data.keywords],
          progress: data.progress,
        }));
        break;
        
      case 'cluster_ready':
        setResults(prev => ({
          ...prev,
          clusters: [...prev.clusters, data.cluster],
        }));
        break;
        
      case 'stage_change':
        setResults(prev => ({
          ...prev,
          stage: data.stage,
          progress: data.progress,
        }));
        break;
        
      case 'complete':
        setResults(prev => ({
          ...prev,
          stage: 'complete',
          progress: 100,
        }));
        break;
    }
  }, []);
  
  return { results, handleSSEMessage };
}
```

### Task 7: Client-Side RAF Buffering (1 hour)

**File:** `open-seo-main/src/client/hooks/useBufferedUpdates.ts`

```typescript
import { useState, useCallback, useRef, useEffect } from 'react';

export function useBufferedUpdates<T>(initialValue: T): [T, (update: (prev: T) => T) => void] {
  const [value, setValue] = useState(initialValue);
  const pendingRef = useRef<T>(initialValue);
  const rafRef = useRef<number | null>(null);
  
  const bufferedUpdate = useCallback((update: (prev: T) => T) => {
    pendingRef.current = update(pendingRef.current);
    
    if (rafRef.current === null) {
      rafRef.current = requestAnimationFrame(() => {
        setValue(pendingRef.current);
        rafRef.current = null;
      });
    }
  }, []);
  
  useEffect(() => {
    return () => {
      if (rafRef.current !== null) {
        cancelAnimationFrame(rafRef.current);
      }
    };
  }, []);
  
  return [value, bufferedUpdate];
}
```
