### Task 3: Create saved views server actions

**File:** `apps/web/src/actions/views/saved-views.ts`

```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { db } from "@/db";
import { savedViews } from "@/db/schema";
import { eq, and, or } from "drizzle-orm";
import { nanoid } from "nanoid";
import type { ViewConfig, SavedView } from "@/types/saved-views";

export async function getSavedViews(workspaceId: string): Promise<SavedView[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const views = await db
    .select()
    .from(savedViews)
    .where(
      and(
        eq(savedViews.workspaceId, workspaceId),
        or(
          eq(savedViews.userId, userId),
          eq(savedViews.isShared, true)
        )
      )
    )
    .orderBy(savedViews.name);

  return views.map((v) => ({
    id: v.id,
    name: v.name,
    description: v.description ?? undefined,
    config: {
      columns: JSON.parse(v.columns),
      filters: JSON.parse(v.filters),
      sortBy: v.sortBy ?? undefined,
      sortDir: v.sortDir as "asc" | "desc" | undefined,
    },
    isShared: v.isShared ?? false,
    isDefault: v.isDefault ?? false,
    createdAt: v.createdAt!,
    userId: v.userId,
  }));
}

export async function createSavedView(
  workspaceId: string,
  name: string,
  config: ViewConfig,
  options?: { description?: string; isShared?: boolean }
): Promise<SavedView> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const id = nanoid();
  
  await db.insert(savedViews).values({
    id,
    workspaceId,
    userId,
    name,
    description: options?.description,
    columns: JSON.stringify(config.columns),
    filters: JSON.stringify(config.filters),
    sortBy: config.sortBy,
    sortDir: config.sortDir,
    isShared: options?.isShared ?? false,
  });

  return {
    id,
    name,
    description: options?.description,
    config,
    isShared: options?.isShared ?? false,
    isDefault: false,
    createdAt: new Date(),
    userId,
  };
}

export async function updateSavedView(
  viewId: string,
  updates: Partial<{ name: string; config: ViewConfig; isShared: boolean }>
): Promise<void> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const updateData: Record<string, any> = { updatedAt: new Date() };
  
  if (updates.name) updateData.name = updates.name;
  if (updates.isShared !== undefined) updateData.isShared = updates.isShared;
  if (updates.config) {
    updateData.columns = JSON.stringify(updates.config.columns);
    updateData.filters = JSON.stringify(updates.config.filters);
    updateData.sortBy = updates.config.sortBy;
    updateData.sortDir = updates.config.sortDir;
  }

  await db
    .update(savedViews)
    .set(updateData)
    .where(and(eq(savedViews.id, viewId), eq(savedViews.userId, userId)));
}

export async function deleteSavedView(viewId: string): Promise<void> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  await db
    .delete(savedViews)
    .where(and(eq(savedViews.id, viewId), eq(savedViews.userId, userId)));
}

export async function setDefaultView(viewId: string, workspaceId: string): Promise<void> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  // Clear existing default
  await db
    .update(savedViews)
    .set({ isDefault: false })
    .where(and(eq(savedViews.workspaceId, workspaceId), eq(savedViews.userId, userId)));

  // Set new default
  await db
    .update(savedViews)
    .set({ isDefault: true })
    .where(eq(savedViews.id, viewId));
}
```

### Task 4: Create SavedViewSelector component

**File:** `apps/web/src/components/dashboard/SavedViewSelector.tsx`

```tsx
"use client";

import { useState } from "react";
import {
  Button,
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuSeparator,
  DropdownMenuTrigger,
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
  Input,
  Label,
  Switch,
} from "@tevero/ui";
import { ChevronDown, Plus, Star, Share2, Trash2 } from "lucide-react";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import {
  getSavedViews,
  createSavedView,
  deleteSavedView,
  setDefaultView,
} from "@/actions/views/saved-views";
import type { ViewConfig, SavedView } from "@/types/saved-views";

interface SavedViewSelectorProps {
  workspaceId: string;
  currentConfig: ViewConfig;
  onViewSelect: (config: ViewConfig) => void;
}

export function SavedViewSelector({
  workspaceId,
  currentConfig,
  onViewSelect,
}: SavedViewSelectorProps) {
  const queryClient = useQueryClient();
  const [saveDialogOpen, setSaveDialogOpen] = useState(false);
  const [newViewName, setNewViewName] = useState("");
  const [isShared, setIsShared] = useState(false);

  const { data: views = [] } = useQuery({
    queryKey: ["saved-views", workspaceId],
    queryFn: () => getSavedViews(workspaceId),
  });

  const createMutation = useMutation({
    mutationFn: (data: { name: string; isShared: boolean }) =>
      createSavedView(workspaceId, data.name, currentConfig, { isShared: data.isShared }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["saved-views"] });
      setSaveDialogOpen(false);
      setNewViewName("");
    },
  });

  const deleteMutation = useMutation({
    mutationFn: deleteSavedView,
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["saved-views"] }),
  });

  const setDefaultMutation = useMutation({
    mutationFn: (viewId: string) => setDefaultView(viewId, workspaceId),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["saved-views"] }),
  });

  const defaultView = views.find((v) => v.isDefault);

  return (
    <>
      <DropdownMenu>
        <DropdownMenuTrigger asChild>
          <Button variant="outline" size="sm" className="gap-2">
            {defaultView?.name ?? "Default View"}
            <ChevronDown className="h-3 w-3" />
          </Button>
        </DropdownMenuTrigger>
        <DropdownMenuContent align="start" className="w-56">
          <DropdownMenuItem onClick={() => onViewSelect({ columns: [], filters: {} })}>
            Default View
          </DropdownMenuItem>
          
          {views.length > 0 && <DropdownMenuSeparator />}
          
          {views.map((view) => (
            <DropdownMenuItem
              key={view.id}
              className="flex items-center justify-between"
            >
              <span
                className="flex-1 truncate"
                onClick={() => onViewSelect(view.config)}
              >
                {view.isDefault && <Star className="h-3 w-3 inline mr-1 text-yellow-500" />}
                {view.isShared && <Share2 className="h-3 w-3 inline mr-1 text-blue-500" />}
                {view.name}
              </span>
              <div className="flex items-center gap-1">
                <Button
                  variant="ghost"
                  size="icon"
                  className="h-6 w-6"
                  onClick={(e) => {
                    e.stopPropagation();
                    setDefaultMutation.mutate(view.id);
                  }}
                >
                  <Star className="h-3 w-3" />
                </Button>
                <Button
                  variant="ghost"
                  size="icon"
                  className="h-6 w-6 text-destructive"
                  onClick={(e) => {
                    e.stopPropagation();
                    deleteMutation.mutate(view.id);
                  }}
                >
                  <Trash2 className="h-3 w-3" />
                </Button>
              </div>
            </DropdownMenuItem>
          ))}
          
          <DropdownMenuSeparator />
          
          <DropdownMenuItem onClick={() => setSaveDialogOpen(true)}>
            <Plus className="h-4 w-4 mr-2" />
            Save Current View
          </DropdownMenuItem>
        </DropdownMenuContent>
      </DropdownMenu>

      <Dialog open={saveDialogOpen} onOpenChange={setSaveDialogOpen}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Save View</DialogTitle>
          </DialogHeader>
          <div className="space-y-4">
            <div className="space-y-2">
              <Label htmlFor="view-name">View Name</Label>
              <Input
                id="view-name"
                value={newViewName}
                onChange={(e) => setNewViewName(e.target.value)}
                placeholder="My Custom View"
              />
            </div>
            <div className="flex items-center justify-between">
              <Label htmlFor="share-view">Share with team</Label>
              <Switch
                id="share-view"
                checked={isShared}
                onCheckedChange={setIsShared}
              />
            </div>
          </div>
          <DialogFooter>
            <Button variant="outline" onClick={() => setSaveDialogOpen(false)}>
              Cancel
            </Button>
            <Button
              onClick={() => createMutation.mutate({ name: newViewName, isShared })}
              disabled={!newViewName.trim() || createMutation.isPending}
            >
              Save View
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </>
  );
}
```

### Task 5: Create ColumnCustomizer component

```tsx
"use client";

import { useState } from "react";
import {
  Button,
  Popover,
  PopoverContent,
  PopoverTrigger,
  Checkbox,
  Label,
} from "@tevero/ui";
import { Settings2, GripVertical } from "lucide-react";
import { DndContext, closestCenter, DragEndEvent } from "@dnd-kit/core";
import {
  SortableContext,
  useSortable,
  verticalListSortingStrategy,
  arrayMove,
} from "@dnd-kit/sortable";
import { CSS } from "@dnd-kit/utilities";
import { ALL_COLUMNS, type ViewConfig } from "@/types/saved-views";

interface ColumnCustomizerProps {
  config: ViewConfig;
  onConfigChange: (config: ViewConfig) => void;
}

function SortableColumn({
  column,
  isVisible,
  onToggle,
}: {
  column: { id: string; label: string; locked?: boolean };
  isVisible: boolean;
  onToggle: () => void;
}) {
  const { attributes, listeners, setNodeRef, transform, transition } =
    useSortable({ id: column.id, disabled: column.locked });

  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
  };

  return (
    <div
      ref={setNodeRef}
      style={style}
      className="flex items-center gap-2 py-1.5 px-2 rounded hover:bg-muted"
    >
      {!column.locked && (
        <GripVertical
          className="h-4 w-4 text-muted-foreground cursor-grab"
          {...attributes}
          {...listeners}
        />
      )}
      <Checkbox
        id={column.id}
        checked={isVisible}
        onCheckedChange={onToggle}
        disabled={column.locked}
      />
      <Label htmlFor={column.id} className="flex-1 cursor-pointer">
        {column.label}
      </Label>
    </div>
  );
}

export function ColumnCustomizer({ config, onConfigChange }: ColumnCustomizerProps) {
  const [columns, setColumns] = useState(
    config.columns.length > 0
      ? config.columns
      : ALL_COLUMNS.filter((c) => c.locked).map((c) => c.id)
  );

  const handleDragEnd = (event: DragEndEvent) => {
    const { active, over } = event;
    if (over && active.id !== over.id) {
      const oldIndex = columns.indexOf(active.id as string);
      const newIndex = columns.indexOf(over.id as string);
      const newColumns = arrayMove(columns, oldIndex, newIndex);
      setColumns(newColumns);
      onConfigChange({ ...config, columns: newColumns });
    }
  };

  const toggleColumn = (columnId: string) => {
    const newColumns = columns.includes(columnId)
      ? columns.filter((c) => c !== columnId)
      : [...columns, columnId];
    setColumns(newColumns);
    onConfigChange({ ...config, columns: newColumns });
  };

  return (
    <Popover>
      <PopoverTrigger asChild>
        <Button variant="outline" size="sm" className="gap-2">
          <Settings2 className="h-4 w-4" />
          Columns
        </Button>
      </PopoverTrigger>
      <PopoverContent align="end" className="w-64">
        <div className="space-y-1">
          <h4 className="font-medium text-sm mb-2">Toggle Columns</h4>
          <DndContext collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
            <SortableContext
              items={ALL_COLUMNS.map((c) => c.id)}
              strategy={verticalListSortingStrategy}
            >
              {ALL_COLUMNS.map((column) => (
                <SortableColumn
                  key={column.id}
                  column={column}
                  isVisible={columns.includes(column.id)}
                  onToggle={() => toggleColumn(column.id)}
                />
              ))}
            </SortableContext>
          </DndContext>
        </div>
      </PopoverContent>
    </Popover>
  );
}
```
