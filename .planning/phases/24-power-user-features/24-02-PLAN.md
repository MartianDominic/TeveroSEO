### Task 1: Create selection state hook

**File:** `apps/web/src/hooks/useRowSelection.ts`

```typescript
import { useState, useCallback, useMemo } from "react";

interface UseRowSelectionOptions<T> {
  data: T[];
  getRowId: (row: T) => string;
}

export function useRowSelection<T>({
  data,
  getRowId,
}: UseRowSelectionOptions<T>) {
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());
  const [lastSelectedIndex, setLastSelectedIndex] = useState<number | null>(null);

  const toggleRow = useCallback(
    (index: number, shiftKey = false) => {
      const rowId = getRowId(data[index]!);

      setSelectedIds((prev) => {
        const next = new Set(prev);

        if (shiftKey && lastSelectedIndex !== null) {
          // Range selection
          const start = Math.min(lastSelectedIndex, index);
          const end = Math.max(lastSelectedIndex, index);
          for (let i = start; i <= end; i++) {
            next.add(getRowId(data[i]!));
          }
        } else {
          // Toggle single
          if (next.has(rowId)) {
            next.delete(rowId);
          } else {
            next.add(rowId);
          }
        }

        return next;
      });

      setLastSelectedIndex(index);
    },
    [data, getRowId, lastSelectedIndex]
  );

  const selectAll = useCallback(() => {
    setSelectedIds(new Set(data.map(getRowId)));
  }, [data, getRowId]);

  const clearSelection = useCallback(() => {
    setSelectedIds(new Set());
    setLastSelectedIndex(null);
  }, []);

  const isSelected = useCallback(
    (row: T) => selectedIds.has(getRowId(row)),
    [selectedIds, getRowId]
  );

  const selectedRows = useMemo(
    () => data.filter((row) => selectedIds.has(getRowId(row))),
    [data, selectedIds, getRowId]
  );

  return {
    selectedIds,
    selectedRows,
    selectedCount: selectedIds.size,
    toggleRow,
    selectAll,
    clearSelection,
    isSelected,
    isAllSelected: selectedIds.size === data.length && data.length > 0,
    isSomeSelected: selectedIds.size > 0 && selectedIds.size < data.length,
  };
}
```

export type BulkActionType =
  | "run_audit"
  | "generate_report"
  | "reassign"
  | "add_tag"
  | "remove_tag"
  | "set_goals"
  | "archive"
  | "delete";

export interface BulkActionPayload {
  action: BulkActionType;
  clientIds: string[];
  params?: {
    assigneeId?: string;
    tagId?: string;
    tagName?: string;
    goalTemplateId?: string;
  };
}

export interface BulkActionResult {
  success: boolean;
  processed: number;
  failed: number;
  errors?: Array<{ clientId: string; error: string }>;
}

### Task 3: Create bulk actions server action

**File:** `apps/web/src/actions/clients/bulk-actions.ts`

```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { db } from "@/db";
import { clients, clientTags, audits } from "@/db/schema";
import { eq, inArray } from "drizzle-orm";
import { nanoid } from "nanoid";
import { auditQueue, reportQueue } from "@/lib/queues";
import type { BulkActionPayload, BulkActionResult } from "@/types/bulk-actions";

export async function executeBulkAction(
  payload: BulkActionPayload
): Promise<BulkActionResult> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const { action, clientIds, params } = payload;
  const errors: Array<{ clientId: string; error: string }> = [];
  let processed = 0;

  try {
    switch (action) {
      case "run_audit":
        for (const clientId of clientIds) {
          try {
            await auditQueue.add("audit", { clientId, userId });
            processed++;
          } catch (e) {
            errors.push({ clientId, error: String(e) });
          }
        }
        break;

      case "generate_report":
        for (const clientId of clientIds) {
          try {
            await reportQueue.add("generate", { clientId, userId });
            processed++;
          } catch (e) {
            errors.push({ clientId, error: String(e) });
          }
        }
        break;

      case "reassign":
        if (!params?.assigneeId) throw new Error("Assignee required");
        await db
          .update(clients)
          .set({ assignedTo: params.assigneeId })
          .where(inArray(clients.id, clientIds));
        processed = clientIds.length;
        break;

      case "add_tag":
        if (!params?.tagId && !params?.tagName) throw new Error("Tag required");
        const tagId = params.tagId ?? nanoid();
        for (const clientId of clientIds) {
          try {
            await db.insert(clientTags).values({
              id: nanoid(),
              clientId,
              tagId,
              tagName: params.tagName,
            }).onConflictDoNothing();
            processed++;
          } catch (e) {
            errors.push({ clientId, error: String(e) });
          }
        }
        break;

      case "remove_tag":
        if (!params?.tagId) throw new Error("Tag ID required");
        await db
          .delete(clientTags)
          .where(
            inArray(clientTags.clientId, clientIds)
          );
        processed = clientIds.length;
        break;

      case "archive":
        await db
          .update(clients)
          .set({ status: "archived", archivedAt: new Date() })
          .where(inArray(clients.id, clientIds));
        processed = clientIds.length;
        break;

      default:
        throw new Error(`Unknown action: ${action}`);
    }

    return {
      success: errors.length === 0,
      processed,
      failed: errors.length,
      errors: errors.length > 0 ? errors : undefined,
    };
  } catch (error) {
    return {
      success: false,
      processed,
      failed: clientIds.length - processed,
      errors: [{ clientId: "all", error: String(error) }],
    };
  }
}
```

### Task 4: Create BulkActionBar component

**File:** `apps/web/src/components/dashboard/BulkActionBar.tsx`

```tsx
"use client";

import { useState } from "react";
import {
  Button,
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuSeparator,
  DropdownMenuTrigger,
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from "@tevero/ui";
import {
  FileSearch,
  FileText,
  UserPlus,
  Tag,
  Archive,
  Trash2,
  ChevronDown,
  X,
  Target,
} from "lucide-react";
import { useMutation } from "@tanstack/react-query";
import { executeBulkAction } from "@/actions/clients/bulk-actions";
import type { BulkActionType } from "@/types/bulk-actions";
import { toast } from "sonner";

interface BulkActionBarProps {
  selectedCount: number;
  selectedIds: string[];
  onClear: () => void;
  teamMembers?: Array<{ id: string; name: string }>;
}

export function BulkActionBar({
  selectedCount,
  selectedIds,
  onClear,
  teamMembers = [],
}: BulkActionBarProps) {
  const [confirmAction, setConfirmAction] = useState<BulkActionType | null>(null);

  const mutation = useMutation({
    mutationFn: executeBulkAction,
    onSuccess: (result) => {
      if (result.success) {
        toast.success(`Successfully processed ${result.processed} clients`);
        onClear();
      } else {
        toast.error(`Failed: ${result.failed} errors`);
      }
      setConfirmAction(null);
    },
    onError: (error) => {
      toast.error(`Action failed: ${error.message}`);
    },
  });

  const handleAction = (action: BulkActionType, params?: object) => {
    if (action === "archive" || action === "delete") {
      setConfirmAction(action);
      return;
    }
    
    mutation.mutate({ action, clientIds: selectedIds, params });
  };

  const confirmDestructive = () => {
    if (confirmAction) {
      mutation.mutate({ action: confirmAction, clientIds: selectedIds });
    }
  };

  if (selectedCount === 0) return null;

  return (
    <>
      <div className="fixed bottom-4 left-1/2 -translate-x-1/2 z-50 flex items-center gap-2 bg-background border rounded-lg shadow-lg px-4 py-2">
        <span className="text-sm font-medium">
          {selectedCount} selected
        </span>

        <div className="h-4 w-px bg-border mx-2" />

        <Button
          variant="outline"
          size="sm"
          onClick={() => handleAction("run_audit")}
          disabled={mutation.isPending}
        >
          <FileSearch className="h-4 w-4 mr-1" />
          Run Audit
        </Button>

        <Button
          variant="outline"
          size="sm"
          onClick={() => handleAction("generate_report")}
          disabled={mutation.isPending}
        >
          <FileText className="h-4 w-4 mr-1" />
          Generate Report
        </Button>

        <DropdownMenu>
          <DropdownMenuTrigger asChild>
            <Button variant="outline" size="sm" disabled={mutation.isPending}>
              <UserPlus className="h-4 w-4 mr-1" />
              Reassign
              <ChevronDown className="h-3 w-3 ml-1" />
            </Button>
          </DropdownMenuTrigger>
          <DropdownMenuContent>
            {teamMembers.map((member) => (
              <DropdownMenuItem
                key={member.id}
                onClick={() =>
                  handleAction("reassign", { assigneeId: member.id })
                }
              >
                {member.name}
              </DropdownMenuItem>
            ))}
          </DropdownMenuContent>
        </DropdownMenu>

        <Button
          variant="outline"
          size="sm"
          onClick={() => handleAction("add_tag", { tagName: "bulk-selected" })}
          disabled={mutation.isPending}
        >
          <Tag className="h-4 w-4 mr-1" />
          Add Tag
        </Button>

        <DropdownMenu>
          <DropdownMenuTrigger asChild>
            <Button variant="outline" size="sm" disabled={mutation.isPending}>
              More
              <ChevronDown className="h-3 w-3 ml-1" />
            </Button>
          </DropdownMenuTrigger>
          <DropdownMenuContent>
            <DropdownMenuItem onClick={() => handleAction("set_goals")}>
              <Target className="h-4 w-4 mr-2" />
              Set Goals
            </DropdownMenuItem>
            <DropdownMenuSeparator />
            <DropdownMenuItem
              className="text-destructive"
              onClick={() => handleAction("archive")}
            >
              <Archive className="h-4 w-4 mr-2" />
              Archive
            </DropdownMenuItem>
          </DropdownMenuContent>
        </DropdownMenu>

        <div className="h-4 w-px bg-border mx-2" />

        <Button variant="ghost" size="sm" onClick={onClear}>
          <X className="h-4 w-4" />
        </Button>
      </div>

      {/* Confirmation Dialog */}
      <Dialog open={!!confirmAction} onOpenChange={() => setConfirmAction(null)}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>
              {confirmAction === "archive" ? "Archive" : "Delete"} {selectedCount} clients?
            </DialogTitle>
          </DialogHeader>
          <p className="text-sm text-muted-foreground">
            {confirmAction === "archive"
              ? "Archived clients can be restored later."
              : "This action cannot be undone."}
          </p>
          <DialogFooter>
            <Button variant="outline" onClick={() => setConfirmAction(null)}>
              Cancel
            </Button>
            <Button
              variant="destructive"
              onClick={confirmDestructive}
              disabled={mutation.isPending}
            >
              {mutation.isPending ? "Processing..." : "Confirm"}
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </>
  );
}
```

import { BulkActionBar } from "@/components/dashboard/BulkActionBar";
import { useRowSelection } from "@/hooks/useRowSelection";

export default function DashboardPage() {
  const { data: clients } = useClients();
  const { data: team } = useTeamMembers();
  
  const {
    selectedIds,
    selectedCount,
    toggleRow,
    selectAll,
    clearSelection,
    isSelected,
    isAllSelected,
    isSomeSelected,
  } = useRowSelection({
    data: clients ?? [],
    getRowId: (c) => c.id,
  });

  return (
    <div>
      {/* Table with selection */}
      <ClientPortfolioTable
        clients={clients}
        isSelected={isSelected}
        isAllSelected={isAllSelected}
        isSomeSelected={isSomeSelected}
        onToggleRow={toggleRow}
        onSelectAll={selectAll}
        onClearSelection={clearSelection}
      />

      {/* Bulk Action Bar */}
      <BulkActionBar
        selectedCount={selectedCount}
        selectedIds={Array.from(selectedIds)}
        onClear={clearSelection}
        teamMembers={team}
      />
    </div>
  );
}
