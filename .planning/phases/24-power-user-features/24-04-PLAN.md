### Task 2: Create CSV export utility

**File:** `apps/web/src/lib/export/csv.ts`

```typescript
import type { ExportColumn } from "@/types/export";

export function generateCSV<T>(
  data: T[],
  columns: ExportColumn[],
  options: { includeHeaders?: boolean } = {}
): string {
  const { includeHeaders = true } = options;
  const rows: string[] = [];

  // Header row
  if (includeHeaders) {
    rows.push(columns.map((col) => escapeCSV(col.label)).join(","));
  }

  // Data rows
  for (const row of data) {
    const values = columns.map((col) => escapeCSV(String(col.accessor(row))));
    rows.push(values.join(","));
  }

  return rows.join("\n");
}

function escapeCSV(value: string): string {
  if (value.includes(",") || value.includes('"') || value.includes("\n")) {
    return `"${value.replace(/"/g, '""')}"`;
  }
  return value;
}

export function downloadCSV(csv: string, filename: string): void {
  const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
  const url = URL.createObjectURL(blob);
  const link = document.createElement("a");
  link.href = url;
  link.download = filename.endsWith(".csv") ? filename : `${filename}.csv`;
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
  URL.revokeObjectURL(url);
}
```

### Task 3: Create PDF export utility

**File:** `apps/web/src/lib/export/pdf.ts`

```typescript
import { jsPDF } from "jspdf";
import autoTable from "jspdf-autotable";
import type { ExportColumn } from "@/types/export";

export async function generatePDF<T>(
  data: T[],
  columns: ExportColumn[],
  options: {
    title?: string;
    subtitle?: string;
    orientation?: "portrait" | "landscape";
  } = {}
): Promise<Blob> {
  const { title, subtitle, orientation = "landscape" } = options;

  const doc = new jsPDF({ orientation });

  // Title
  if (title) {
    doc.setFontSize(18);
    doc.text(title, 14, 20);
  }

  // Subtitle
  if (subtitle) {
    doc.setFontSize(11);
    doc.setTextColor(100);
    doc.text(subtitle, 14, 28);
  }

  // Table
  const tableColumns = columns.map((col) => ({
    header: col.label,
    dataKey: col.id,
  }));

  const tableData = data.map((row) => {
    const rowData: Record<string, string> = {};
    for (const col of columns) {
      rowData[col.id] = String(col.accessor(row));
    }
    return rowData;
  });

  autoTable(doc, {
    columns: tableColumns,
    body: tableData,
    startY: title ? 35 : 15,
    styles: {
      fontSize: 9,
      cellPadding: 3,
    },
    headStyles: {
      fillColor: [41, 128, 185],
      textColor: 255,
      fontStyle: "bold",
    },
    alternateRowStyles: {
      fillColor: [245, 245, 245],
    },
  });

  // Footer
  const pageCount = doc.getNumberOfPages();
  for (let i = 1; i <= pageCount; i++) {
    doc.setPage(i);
    doc.setFontSize(8);
    doc.setTextColor(150);
    doc.text(
      `Page ${i} of ${pageCount} • Generated ${new Date().toLocaleDateString()}`,
      doc.internal.pageSize.width / 2,
      doc.internal.pageSize.height - 10,
      { align: "center" }
    );
  }

  return doc.output("blob");
}

export function downloadPDF(blob: Blob, filename: string): void {
  const url = URL.createObjectURL(blob);
  const link = document.createElement("a");
  link.href = url;
  link.download = filename.endsWith(".pdf") ? filename : `${filename}.pdf`;
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
  URL.revokeObjectURL(url);
}
```

    label: "Priority Score",

### Task 5: Create ExportButton component

**File:** `apps/web/src/components/dashboard/ExportButton.tsx`

```tsx
"use client";

import { useState } from "react";
import {
  Button,
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
  Checkbox,
  Label,
  RadioGroup,
  RadioGroupItem,
} from "@tevero/ui";
import { Download, FileText, FileSpreadsheet, ChevronDown } from "lucide-react";
import { generateCSV, downloadCSV } from "@/lib/export/csv";
import { generatePDF, downloadPDF } from "@/lib/export/pdf";
import { EXPORT_COLUMNS, getExportColumns } from "@/lib/export/columns";
import type { ExportFormat } from "@/types/export";
import type { ClientDashboardRow } from "@/types/dashboard";

interface ExportButtonProps {
  data: ClientDashboardRow[];
  selectedColumns?: string[];
  filename?: string;
}

export function ExportButton({
  data,
  selectedColumns = [],
  filename = "portfolio-export",
}: ExportButtonProps) {
  const [dialogOpen, setDialogOpen] = useState(false);
  const [format, setFormat] = useState<ExportFormat>("csv");
  const [columns, setColumns] = useState<string[]>(
    selectedColumns.length > 0
      ? selectedColumns
      : EXPORT_COLUMNS.map((c) => c.id)
  );
  const [isExporting, setIsExporting] = useState(false);

  const handleExport = async () => {
    setIsExporting(true);

    try {
      const exportColumns = getExportColumns(columns);
      const timestamp = new Date().toISOString().split("T")[0];
      const exportFilename = `${filename}-${timestamp}`;

      if (format === "csv") {
        const csv = generateCSV(data, exportColumns);
        downloadCSV(csv, exportFilename);
      } else {
        const blob = await generatePDF(data, exportColumns, {
          title: "Portfolio Report",
          subtitle: `Generated on ${new Date().toLocaleDateString()} • ${data.length} clients`,
          orientation: columns.length > 6 ? "landscape" : "portrait",
        });
        downloadPDF(blob, exportFilename);
      }

      setDialogOpen(false);
    } finally {
      setIsExporting(false);
    }
  };

  const toggleColumn = (columnId: string) => {
    setColumns((prev) =>
      prev.includes(columnId)
        ? prev.filter((c) => c !== columnId)
        : [...prev, columnId]
    );
  };

  return (
    <>
      <DropdownMenu>
        <DropdownMenuTrigger asChild>
          <Button variant="outline" size="sm" data-export-btn className="gap-2">
            <Download className="h-4 w-4" />
            Export
            <ChevronDown className="h-3 w-3" />
          </Button>
        </DropdownMenuTrigger>
        <DropdownMenuContent align="end">
          <DropdownMenuItem onClick={() => {
            setFormat("csv");
            setDialogOpen(true);
          }}>
            <FileSpreadsheet className="h-4 w-4 mr-2" />
            Export as CSV
          </DropdownMenuItem>
          <DropdownMenuItem onClick={() => {
            setFormat("pdf");
            setDialogOpen(true);
          }}>
            <FileText className="h-4 w-4 mr-2" />
            Export as PDF
          </DropdownMenuItem>
        </DropdownMenuContent>
      </DropdownMenu>

      <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
        <DialogContent className="max-w-md">
          <DialogHeader>
            <DialogTitle>Export Options</DialogTitle>
          </DialogHeader>

          <div className="space-y-4">
            {/* Format Selection */}
            <div className="space-y-2">
              <Label>Format</Label>
              <RadioGroup value={format} onValueChange={(v) => setFormat(v as ExportFormat)}>
                <div className="flex items-center gap-4">
                  <div className="flex items-center gap-2">
                    <RadioGroupItem value="csv" id="csv" />
                    <Label htmlFor="csv">CSV</Label>
                  </div>
                  <div className="flex items-center gap-2">
                    <RadioGroupItem value="pdf" id="pdf" />
                    <Label htmlFor="pdf">PDF</Label>
                  </div>
                </div>
              </RadioGroup>
            </div>

            {/* Column Selection */}
            <div className="space-y-2">
              <Label>Columns to include</Label>
              <div className="grid grid-cols-2 gap-2 max-h-48 overflow-y-auto border rounded p-2">
                {EXPORT_COLUMNS.map((col) => (
                  <label
                    key={col.id}
                    className="flex items-center gap-2 text-sm cursor-pointer"
                  >
                    <Checkbox
                      checked={columns.includes(col.id)}
                      onCheckedChange={() => toggleColumn(col.id)}
                    />
                    {col.label}
                  </label>
                ))}
              </div>
              <div className="flex gap-2 text-xs">
                <button
                  type="button"
                  className="text-primary hover:underline"
                  onClick={() => setColumns(EXPORT_COLUMNS.map((c) => c.id))}
                >
                  Select all
                </button>
                <button
                  type="button"
                  className="text-primary hover:underline"
                  onClick={() => setColumns([])}
                >
                  Clear all
                </button>
              </div>
            </div>

            {/* Summary */}
            <div className="text-sm text-muted-foreground">
              Exporting {data.length} clients with {columns.length} columns
            </div>
          </div>

          <DialogFooter>
            <Button variant="outline" onClick={() => setDialogOpen(false)}>
              Cancel
            </Button>
            <Button
              onClick={handleExport}
              disabled={columns.length === 0 || isExporting}
            >
              {isExporting ? "Exporting..." : `Export ${format.toUpperCase()}`}
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </>
  );
}
```

### Task 6: Create server-side export endpoint (for large exports)

**File:** `apps/web/src/app/api/export/route.ts`

```typescript
import { NextResponse } from "next/server";
import { auth } from "@clerk/nextjs/server";
import { db } from "@/db";
import { clientDashboardMetrics, clients } from "@/db/schema";
import { eq } from "drizzle-orm";
import { generateCSV } from "@/lib/export/csv";
import { getExportColumns } from "@/lib/export/columns";

export async function POST(request: Request) {
  const { userId } = await auth();
  if (!userId) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const body = await request.json();
  const { workspaceId, columns, format } = body;

  // Fetch all data (for large exports)
  const data = await db
    .select({
      clientId: clientDashboardMetrics.clientId,
      clientName: clients.name,
      goalAttainmentPct: clientDashboardMetrics.goalAttainmentPct,
      goalsMetCount: clientDashboardMetrics.goalsMetCount,
      goalsTotalCount: clientDashboardMetrics.goalsTotalCount,
      primaryGoalName: clientDashboardMetrics.primaryGoalName,
      trafficTotal: clientDashboardMetrics.trafficTotal,
      trafficTrend: clientDashboardMetrics.trafficTrend,
      keywordsTop10: clientDashboardMetrics.keywordsTop10,
      keywordsTop3: clientDashboardMetrics.keywordsTop3,
      keywordsPosition1: clientDashboardMetrics.keywordsPosition1,
      alertsCritical: clientDashboardMetrics.alertsCritical,
      alertsWarning: clientDashboardMetrics.alertsWarning,
      priorityScore: clientDashboardMetrics.priorityScore,
    })
    .from(clientDashboardMetrics)
    .innerJoin(clients, eq(clients.id, clientDashboardMetrics.clientId))
    .where(eq(clientDashboardMetrics.workspaceId, workspaceId));

  const exportColumns = getExportColumns(columns);

  if (format === "csv") {
    const csv = generateCSV(data, exportColumns);
    return new NextResponse(csv, {
      headers: {
        "Content-Type": "text/csv",
        "Content-Disposition": `attachment; filename="export-${Date.now()}.csv"`,
      },
    });
  }

  // For PDF, return JSON and let client generate
  return NextResponse.json({ data, columns: exportColumns });
}
```
