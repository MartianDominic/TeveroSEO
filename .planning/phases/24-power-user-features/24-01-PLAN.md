### Task 2: Create keyboard navigation hook

**File:** `apps/web/src/hooks/useTableKeyboardNav.ts`

```typescript
import { useCallback, useEffect, useState } from "react";
import { useRouter } from "next/navigation";

interface UseTableKeyboardNavOptions {
  rowCount: number;
  onRowSelect?: (index: number) => void;
  onRowAction?: (index: number) => void;
  getRowId?: (index: number) => string;
}

export function useTableKeyboardNav({
  rowCount,
  onRowSelect,
  onRowAction,
  getRowId,
}: UseTableKeyboardNavOptions) {
  const router = useRouter();
  const [focusedIndex, setFocusedIndex] = useState(-1);
  const [isEnabled, setIsEnabled] = useState(true);

  const handleKeyDown = useCallback(
    (e: KeyboardEvent) => {
      if (!isEnabled) return;
      
      // Ignore if typing in input
      if (
        e.target instanceof HTMLInputElement ||
        e.target instanceof HTMLTextAreaElement
      ) {
        return;
      }

      switch (e.key) {
        case "j":
        case "ArrowDown":
          e.preventDefault();
          setFocusedIndex((prev) => Math.min(prev + 1, rowCount - 1));
          break;

        case "k":
        case "ArrowUp":
          e.preventDefault();
          setFocusedIndex((prev) => Math.max(prev - 1, 0));
          break;

        case "Enter":
          if (focusedIndex >= 0) {
            e.preventDefault();
            onRowAction?.(focusedIndex);
            if (getRowId) {
              router.push(`/clients/${getRowId(focusedIndex)}`);
            }
          }
          break;

        case " ":
          if (focusedIndex >= 0) {
            e.preventDefault();
            onRowSelect?.(focusedIndex);
          }
          break;

        case "/":
          e.preventDefault();
          // Focus search input
          document.querySelector<HTMLInputElement>('[data-search-input]')?.focus();
          break;

        case "Escape":
          setFocusedIndex(-1);
          break;

        case "g":
          // gg = go to top
          if (e.repeat) {
            setFocusedIndex(0);
          }
          break;

        case "G":
          // G = go to bottom
          setFocusedIndex(rowCount - 1);
          break;
      }
    },
    [isEnabled, rowCount, focusedIndex, onRowSelect, onRowAction, getRowId, router]
  );

  useEffect(() => {
    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [handleKeyDown]);

  return {
    focusedIndex,
    setFocusedIndex,
    setIsEnabled,
  };
}
```

### Task 3: Create Command Palette component

**File:** `apps/web/src/components/dashboard/CommandPalette.tsx`

```tsx
"use client";

import { useCallback, useEffect, useState } from "react";
import { useRouter } from "next/navigation";
import { Command } from "cmdk";
import {
  Dialog,
  DialogContent,
} from "@tevero/ui";
import {
  Search,
  Users,
  BarChart3,
  FileText,
  Settings,
  RefreshCw,
  Download,
  Plus,
  Target,
} from "lucide-react";
import { useClients } from "@/hooks/useClients";

interface CommandPaletteProps {
  workspaceId: string;
}

export function CommandPalette({ workspaceId }: CommandPaletteProps) {
  const router = useRouter();
  const [open, setOpen] = useState(false);
  const [search, setSearch] = useState("");
  const { data: clients } = useClients(workspaceId, { enabled: open });

  // Toggle with Cmd+K
  useEffect(() => {
    const down = (e: KeyboardEvent) => {
      if (e.key === "k" && (e.metaKey || e.ctrlKey)) {
        e.preventDefault();
        setOpen((prev) => !prev);
      }
    };
    
    window.addEventListener("keydown", down);
    return () => window.removeEventListener("keydown", down);
  }, []);

  const runCommand = useCallback((command: () => void) => {
    setOpen(false);
    command();
  }, []);

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogContent className="p-0 overflow-hidden max-w-lg">
        <Command className="[&_[cmdk-group-heading]]:px-2 [&_[cmdk-group-heading]]:font-medium [&_[cmdk-group-heading]]:text-muted-foreground">
          <div className="flex items-center border-b px-3">
            <Search className="mr-2 h-4 w-4 shrink-0 opacity-50" />
            <Command.Input
              value={search}
              onValueChange={setSearch}
              placeholder="Type a command or search..."
              className="flex h-11 w-full rounded-md bg-transparent py-3 text-sm outline-none placeholder:text-muted-foreground disabled:cursor-not-allowed disabled:opacity-50"
            />
          </div>
          <Command.List className="max-h-[300px] overflow-y-auto overflow-x-hidden p-2">
            <Command.Empty className="py-6 text-center text-sm text-muted-foreground">
              No results found.
            </Command.Empty>

            {/* Navigation Commands */}
            <Command.Group heading="Navigation">
              <CommandItem
                icon={<Users className="h-4 w-4" />}
                onSelect={() => runCommand(() => router.push("/dashboard"))}
              >
                Go to Dashboard
              </CommandItem>
              <CommandItem
                icon={<BarChart3 className="h-4 w-4" />}
                onSelect={() => runCommand(() => router.push("/analytics"))}
              >
                Go to Analytics
              </CommandItem>
              <CommandItem
                icon={<FileText className="h-4 w-4" />}
                onSelect={() => runCommand(() => router.push("/reports"))}
              >
                Go to Reports
              </CommandItem>
              <CommandItem
                icon={<Settings className="h-4 w-4" />}
                onSelect={() => runCommand(() => router.push("/settings"))}
              >
                Settings
              </CommandItem>
            </Command.Group>

            {/* Actions */}
            <Command.Group heading="Actions">
              <CommandItem
                icon={<Plus className="h-4 w-4" />}
                onSelect={() => runCommand(() => router.push("/clients/new"))}
              >
                Add New Client
              </CommandItem>
              <CommandItem
                icon={<RefreshCw className="h-4 w-4" />}
                onSelect={() => runCommand(() => window.location.reload())}
              >
                Refresh Dashboard
              </CommandItem>
              <CommandItem
                icon={<Download className="h-4 w-4" />}
                onSelect={() => runCommand(() => {
                  document.querySelector<HTMLButtonElement>('[data-export-btn]')?.click();
                })}
              >
                Export to CSV
              </CommandItem>
              <CommandItem
                icon={<Target className="h-4 w-4" />}
                onSelect={() => runCommand(() => router.push("/goals"))}
              >
                Manage Goals
              </CommandItem>
            </Command.Group>

            {/* Client Search Results */}
            {search && clients && (
              <Command.Group heading="Clients">
                {clients
                  .filter((c) =>
                    c.name.toLowerCase().includes(search.toLowerCase())
                  )
                  .slice(0, 5)
                  .map((client) => (
                    <CommandItem
                      key={client.id}
                      icon={<Users className="h-4 w-4" />}
                      onSelect={() =>
                        runCommand(() => router.push(`/clients/${client.id}`))
                      }
                    >
                      {client.name}
                    </CommandItem>
                  ))}
              </Command.Group>
            )}
          </Command.List>
          <div className="border-t p-2 text-xs text-muted-foreground">
            <kbd className="bg-muted px-1.5 py-0.5 rounded">↑↓</kbd> to navigate
            <kbd className="bg-muted px-1.5 py-0.5 rounded ml-2">Enter</kbd> to select
            <kbd className="bg-muted px-1.5 py-0.5 rounded ml-2">Esc</kbd> to close
          </div>
        </Command>
      </DialogContent>
    </Dialog>
  );
}

function CommandItem({
  children,
  icon,
  onSelect,
}: {
  children: React.ReactNode;
  icon: React.ReactNode;
  onSelect: () => void;
}) {
  return (
    <Command.Item
      onSelect={onSelect}
      className="flex items-center gap-2 px-2 py-1.5 text-sm rounded-md cursor-pointer aria-selected:bg-accent aria-selected:text-accent-foreground"
    >
      {icon}
      <span>{children}</span>
    </Command.Item>
  );
}
```

### Task 4: Create keyboard shortcuts help panel

**File:** `apps/web/src/components/dashboard/KeyboardShortcutsHelp.tsx`

```tsx
"use client";

import { useState, useEffect } from "react";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@tevero/ui";

const shortcuts = [
  { key: "⌘K", description: "Open command palette" },
  { key: "j / ↓", description: "Move down" },
  { key: "k / ↑", description: "Move up" },
  { key: "Enter", description: "Open client" },
  { key: "Space", description: "Select/deselect" },
  { key: "/", description: "Focus search" },
  { key: "Esc", description: "Clear selection" },
  { key: "gg", description: "Go to first row" },
  { key: "G", description: "Go to last row" },
  { key: "?", description: "Show this help" },
];

export function KeyboardShortcutsHelp() {
  const [open, setOpen] = useState(false);

  useEffect(() => {
    const down = (e: KeyboardEvent) => {
      if (e.key === "?" && !e.metaKey && !e.ctrlKey) {
        if (
          !(e.target instanceof HTMLInputElement) &&
          !(e.target instanceof HTMLTextAreaElement)
        ) {
          e.preventDefault();
          setOpen(true);
        }
      }
    };

    window.addEventListener("keydown", down);
    return () => window.removeEventListener("keydown", down);
  }, []);

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Keyboard Shortcuts</DialogTitle>
        </DialogHeader>
        <div className="grid gap-2">
          {shortcuts.map(({ key, description }) => (
            <div key={key} className="flex justify-between items-center py-1">
              <span className="text-muted-foreground">{description}</span>
              <kbd className="bg-muted px-2 py-1 rounded text-sm font-mono">
                {key}
              </kbd>
            </div>
          ))}
        </div>
      </DialogContent>
    </Dialog>
  );
}
```
