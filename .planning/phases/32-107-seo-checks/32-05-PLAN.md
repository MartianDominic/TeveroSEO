// Route: /clients/[clientId]/seo/[projectId]/audit/[pageId]
export default async function PageFindingsPage({ params, searchParams }) {
  const { clientId, projectId, pageId } = params;
  const auditId = searchParams.auditId;
  
  // Fetch findings and score
  const { findings, score } = await getPageFindings({
    projectId,
    clientId,
    auditId,
    pageId,
  });
  
  return (
    <div className="space-y-6">
      {/* Page URL and navigation */}
      <PageHeader title="Page SEO Analysis" />
      
      {/* Score overview */}
      <ScoreCard score={score.score} breakdown={score.breakdown} gates={score.gates} />
      
      {/* Stats summary */}
      <div className="grid grid-cols-4 gap-4">
        <StatCard title="Total Checks" value={findings.length} />
        <StatCard title="Passed" value={findings.filter(f => f.passed).length} variant="success" />
        <StatCard title="Failed" value={findings.filter(f => !f.passed).length} variant="destructive" />
        <StatCard title="Auto-Fixable" value={findings.filter(f => f.autoEditable).length} variant="warning" />
      </div>
      
      {/* Findings table with filters */}
      <FindingsTable findings={findings} onExport={() => exportFindings({ auditId, format: 'csv' })} />
    </div>
  );
}
