```typescript
import { createStartHandler, defaultStreamHandler } from "@tanstack/react-start/server";
import { validateEnv, REQUIRED_ENV_CORE } from "@/server/lib/runtime-env";
import { startAuditWorker, stopAuditWorker } from "@/server/workers/audit-worker";
import { closeRedis } from "@/server/lib/redis";
import { pool } from "@/db";

// Fail fast on missing required environment variables. Runs once per process.
validateEnv(REQUIRED_ENV_CORE);

// Start the BullMQ Worker as part of the HTTP server process. Plan 4 (Docker)
// may opt to run the worker in a separate container; at that time the startup
// of this function will move to a dedicated entry file. For Phase 3 we run
// Worker + HTTP in one process for dev simplicity.
startAuditWorker();

// Graceful shutdown: drain Worker (up to 25s), then close Redis connections,
// then close Postgres pool. Order matters — Worker may still write to DB/Redis
// during drain.
let shuttingDown = false;
async function shutdown(signal: string): Promise<void> {
  if (shuttingDown) return;
  shuttingDown = true;
  console.log(`[server] ${signal} received — shutting down`);
  try {
    await stopAuditWorker();
  } catch (err) {
    console.error("[server] stopAuditWorker failed:", err);
  }
  try {
    await closeRedis();
  } catch (err) {
    console.error("[server] closeRedis failed:", err);
  }
  try {
    await pool.end();
  } catch (err) {
    console.error("[server] pool.end failed:", err);
  }
  console.log("[server] shutdown complete");
  process.exit(0);
}

process.on("SIGTERM", () => {
  void shutdown("SIGTERM");
});
process.on("SIGINT", () => {
  void shutdown("SIGINT");
});

const fetch = createStartHandler(defaultStreamHandler);

export default {
  fetch,
};
```

## Trust Boundaries

| Boundary | Description |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-15 | Denial of service | Repeated startAudit for same id | mitigate | `jobId: auditId` dedupes at BullMQ layer; subsequent adds are no-ops |
| T-03-16 | Tampering | Orphaned DB row if Redis enqueue fails | mitigate | try/catch around `auditQueue.add` rolls back via `deleteAuditForProject` before throwing |
| T-03-17 | Denial of service | Slow shutdown leaves zombie worker | mitigate | 25s timeout inside `stopAuditWorker` (Plan 03) force-closes; `process.exit(0)` guarantees termination |
| T-03-18 | Information disclosure | Shutdown logs leak state | accept | Only signal name + failure classes logged; no job payloads |
| T-03-19 | Race condition | remove() vs active job | mitigate | `job.remove()` wrapped in try/catch; if Worker writes to deleted audit row, DB foreign-key / not-found errors surface gracefully in the Worker's on("failed") DLQ path |
</threat_model>
