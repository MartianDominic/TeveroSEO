/**
 * Cluster Labeler
 * Phase 86-04: Semantic Intelligence Pipeline
 * 
 * Generates human-readable labels for semantic clusters using:
 * 1. Centroid nearest: Label from keyword closest to centroid (fast, free)
 * 2. N-gram: Most frequent n-gram in cluster keywords (fast, free)
 * 3. LLM: Grok 4.1 Fast summarization (best quality, ~$0.03/analysis)
 */

import type {
  KeywordCluster,
  LabeledCluster,
  LabelingConfig,
} from './types';
import { DEFAULT_LABELING_CONFIG } from './types';
import { cosineSimilarity } from '@/server/lib/embeddings/UnifiedEmbeddingService';

/**
 * Cluster labeler with multiple strategies.
 */
export class ClusterLabeler {
  private config: LabelingConfig;

  constructor(config: Partial<LabelingConfig> = {}) {
    this.config = { ...DEFAULT_LABELING_CONFIG, ...config };
  }

  /**
   * Label all clusters using configured method.
   */
  async labelClusters(clusters: KeywordCluster[]): Promise<LabeledCluster[]> {
    const results: LabeledCluster[] = [];
    
    for (const cluster of clusters) {
      const labeled = await this.labelCluster(cluster);
      results.push(labeled);
    }
    
    return results;
  }

  /**
   * Label a single cluster.
   */
  private async labelCluster(cluster: KeywordCluster): Promise<LabeledCluster> {
    let labelLt: string;
    let labelConfidence: number;
    
    switch (this.config.method) {
      case 'centroid_nearest':
        ({ label: labelLt, confidence: labelConfidence } = this.centroidNearest(cluster));
        break;
      case 'ngram':
        ({ label: labelLt, confidence: labelConfidence } = this.ngramExtract(cluster));
        break;
      case 'llm':
        ({ label: labelLt, confidence: labelConfidence } = await this.llmSummarize(cluster));
        break;
      default:
        ({ label: labelLt, confidence: labelConfidence } = this.centroidNearest(cluster));
    }
    
    // Generate English label (simple title case for now)
    const labelEn = this.translateToEnglish(labelLt);
    
    // Generate URL slug
    const suggestedUrl = this.generateSlug(labelEn);
    
    return {
      ...cluster,
      labelLt,
      labelEn,
      suggestedUrl,
      labelConfidence,
    };
  }

  /**
   * Method 1: Label from keyword nearest to centroid.
   */
  private centroidNearest(cluster: KeywordCluster): { label: string; confidence: number } {
    if (cluster.keywords.length === 0) {
      return { label: 'Unknown', confidence: 0 };
    }
    
    let nearestKeyword = cluster.keywords[0];
    let maxSimilarity = -1;
    
    for (const keyword of cluster.keywords) {
      const similarity = cosineSimilarity(keyword.embedding, cluster.centroid);
      if (similarity > maxSimilarity) {
        maxSimilarity = similarity;
        nearestKeyword = keyword;
      }
    }
    
    // Clean up keyword for label
    const label = this.cleanLabel(nearestKeyword.keyword);
    
    return { label, confidence: maxSimilarity };
  }

  /**
   * Method 2: Most frequent n-gram in cluster keywords.
   */
  private ngramExtract(cluster: KeywordCluster): { label: string; confidence: number } {
    if (cluster.keywords.length === 0) {
      return { label: 'Unknown', confidence: 0 };
    }
    
    // Count bigrams
    const ngramCounts = new Map<string, number>();
    
    for (const keyword of cluster.keywords) {
      const words = keyword.keyword.toLowerCase().split(/\s+/);
      
      // Bigrams
      for (let i = 0; i < words.length - 1; i++) {
        const bigram = `${words[i]} ${words[i + 1]}`;
        ngramCounts.set(bigram, (ngramCounts.get(bigram) || 0) + 1);
      }
      
      // Unigrams as fallback
      for (const word of words) {
        if (word.length > 3) { // Skip short words
          ngramCounts.set(word, (ngramCounts.get(word) || 0) + 1);
        }
      }
    }
    
    // Find most frequent
    let bestNgram = '';
    let bestCount = 0;
    
    for (const [ngram, count] of ngramCounts) {
      if (count > bestCount) {
        bestCount = count;
        bestNgram = ngram;
      }
    }
    
    const label = this.cleanLabel(bestNgram);
    const confidence = bestCount / cluster.keywords.length;
    
    return { label, confidence };
  }

  /**
   * Method 3: LLM summarization (Grok 4.1 Fast).
   */
  private async llmSummarize(cluster: KeywordCluster): Promise<{ label: string; confidence: number }> {
    // TODO: Integrate with Grok API
    // For now, fallback to centroid nearest
    return this.centroidNearest(cluster);
  }

  /**
   * Clean up label string.
   */
  private cleanLabel(label: string): string {
    return label
      .trim()
      .replace(/\s+/g, ' ')
      .replace(/^./, c => c.toUpperCase()); // Capitalize first letter
  }

  /**
   * Simple LT->EN translation (placeholder).
   */
  private translateToEnglish(labelLt: string): string {
    // TODO: Use translation API or dictionary
    // For now, return original with title case
    return labelLt
      .split(' ')
      .map(w => w.charAt(0).toUpperCase() + w.slice(1).toLowerCase())
      .join(' ');
  }

  /**
   * Generate URL-safe slug from label.
   */
  private generateSlug(label: string): string {
    return label
      .toLowerCase()
      .normalize('NFD')
      .replace(/[̀-ͯ]/g, '') // Remove diacritics
      .replace(/[^a-z0-9]+/g, '-')
      .replace(/^-|-$/g, '');
  }
}

/**
 * Factory function for labeling.
 */
export async function labelClusters(
  clusters: KeywordCluster[],
  config?: Partial<LabelingConfig>
): Promise<LabeledCluster[]> {
  const labeler = new ClusterLabeler(config);
  return labeler.labelClusters(clusters);
}
