/**
 * Remove Cluster Operation
 * Phase 86-07: Proposal Editing
 * 
 * Removes entire cluster and replaces with backfill keywords.
 */

import type { ProposalState, RemoveClusterInput, RemoveClusterEdit } from '../types';
import { nanoid } from 'nanoid';

export function removeCluster(
  state: ProposalState,
  input: RemoveClusterInput
): { state: ProposalState; edit: RemoveClusterEdit } {
  const cluster = state.clusters.find(c => c.clusterId === input.clusterId);
  if (!cluster) {
    throw new Error(`Cluster ${input.clusterId} not found`);
  }
  
  // Remove cluster keywords from selection
  const removedKeywords = new Set(cluster.selectedKeywords.map(k => k.keyword));
  const newSelected = state.selected.filter(k => !removedKeywords.has(k.keyword));
  
  // Add from backfill to maintain count
  const needed = state.selected.length - newSelected.length;
  const replacements = state.backfillPool.slice(0, needed);
  const newBackfill = state.backfillPool.slice(needed);
  
  // Create new state (immutable)
  const newState: ProposalState = {
    ...state,
    version: state.version + 1,
    clusters: state.clusters.filter(c => c.clusterId !== input.clusterId),
    selected: [...newSelected, ...replacements],
    backfillPool: newBackfill,
    lastEditedAt: new Date(),
    editCount: state.editCount + 1,
  };
  
  const edit: RemoveClusterEdit = {
    id: nanoid(),
    proposalId: state.id,
    version: newState.version,
    type: 'remove_cluster',
    timestamp: new Date(),
    aiSummary: `Removed "${cluster.labelLt}" cluster (${cluster.selectedKeywords.length} keywords), replaced from backfill`,
    data: {
      clusterId: input.clusterId,
      clusterLabel: cluster.labelLt,
      keywordCount: cluster.selectedKeywords.length,
    },
  };
  
  return { state: newState, edit };
}

/**
 * Proposal History Manager
 * Phase 86-07: Undo/Redo
 * 
 * Manages immutable version snapshots for proposal editing.
 */

import type { ProposalState, ProposalEdit, ProposalSnapshot, ProposalHistory } from './types';

/**
 * History manager for undo/redo.
 */
export class ProposalHistoryManager {
  private history: ProposalHistory;
  private maxSnapshots: number;

  constructor(initialState: ProposalState, maxSnapshots: number = 50) {
    this.maxSnapshots = maxSnapshots;
    this.history = {
      proposalId: initialState.id,
      snapshots: [{
        version: initialState.version,
        state: initialState,
        timestamp: new Date(),
      }],
      currentVersion: initialState.version,
    };
  }

  /**
   * Record new state after edit.
   */
  push(state: ProposalState, edit: ProposalEdit): void {
    const snapshot: ProposalSnapshot = {
      version: state.version,
      state,
      edit,
      timestamp: new Date(),
    };
    
    this.history.snapshots.push(snapshot);
    this.history.currentVersion = state.version;
    
    // Trim old snapshots
    if (this.history.snapshots.length > this.maxSnapshots) {
      this.history.snapshots.shift();
    }
  }

  /**
   * Undo to previous version.
   */
  undo(): ProposalState | null {
    const currentIndex = this.history.snapshots.findIndex(
      s => s.version === this.history.currentVersion
    );
    
    if (currentIndex <= 0) return null;
    
    const previousSnapshot = this.history.snapshots[currentIndex - 1];
    this.history.currentVersion = previousSnapshot.version;
    
    return previousSnapshot.state;
  }

  /**
   * Redo to next version.
   */
  redo(): ProposalState | null {
    const currentIndex = this.history.snapshots.findIndex(
      s => s.version === this.history.currentVersion
    );
    
    if (currentIndex >= this.history.snapshots.length - 1) return null;
    
    const nextSnapshot = this.history.snapshots[currentIndex + 1];
    this.history.currentVersion = nextSnapshot.version;
    
    return nextSnapshot.state;
  }

  /**
   * Get current state.
   */
  getCurrent(): ProposalState {
    const snapshot = this.history.snapshots.find(
      s => s.version === this.history.currentVersion
    );
    return snapshot!.state;
  }

  /**
   * Check undo availability.
   */
  canUndo(): boolean {
    const currentIndex = this.history.snapshots.findIndex(
      s => s.version === this.history.currentVersion
    );
    return currentIndex > 0;
  }

  /**
   * Check redo availability.
   */
  canRedo(): boolean {
    const currentIndex = this.history.snapshots.findIndex(
      s => s.version === this.history.currentVersion
    );
    return currentIndex < this.history.snapshots.length - 1;
  }

  /**
   * Get edit history for display.
   */
  getEdits(): ProposalEdit[] {
    return this.history.snapshots
      .filter(s => s.edit)
      .map(s => s.edit!);
  }
}

/**

import { useCopilotAction } from '@copilotkit/react-core';
import type { 
  RemoveClusterInput, 
  AddKeywordInput, 
  RemoveKeywordInput,
  ChangeDistributionInput,
} from '@/server/features/keywords/proposal/types';

export function useProposalEditingActions(
  onRemoveCluster: (input: RemoveClusterInput) => Promise<void>,
  onAddKeyword: (input: AddKeywordInput) => Promise<void>,
  onRemoveKeyword: (input: RemoveKeywordInput) => Promise<void>,
  onChangeDistribution: (input: ChangeDistributionInput) => Promise<void>,
) {
  // Remove entire cluster
  useCopilotAction({
    name: 'removeCluster',
    description: 'Remove an entire growth area (cluster) from the proposal. The removed keywords will be replaced from the backfill pool.',
    parameters: [
      {
        name: 'clusterId',
        type: 'number',
        description: 'The ID of the cluster to remove',
        required: true,
      },
      {
        name: 'reason',
        type: 'string',
        description: 'Reason for removal (for learning)',
        required: false,
      },
    ],
    handler: async ({ clusterId }) => {
      await onRemoveCluster({ clusterId });
      return `Removed cluster ${clusterId} and replaced keywords from backfill pool.`;
    },
  });

  // Add keyword from backfill or suggestion
  useCopilotAction({
    name: 'addKeyword',
    description: 'Add a keyword to the proposal. Can come from the backfill pool or be a new suggestion.',
    parameters: [
      {
        name: 'keyword',
        type: 'string',
        description: 'The keyword to add',
        required: true,
      },
      {
        name: 'clusterId',
        type: 'number',
        description: 'Optional cluster to add keyword to',
        required: false,
      },
    ],
    handler: async ({ keyword, clusterId }) => {
      await onAddKeyword({ keyword, clusterId });
      return `Added "${keyword}" to the proposal.`;
    },
  });

  // Remove specific keyword
  useCopilotAction({
    name: 'removeKeyword',
    description: 'Remove a specific keyword from the proposal.',
    parameters: [
      {
        name: 'keyword',
        type: 'string',
        description: 'The keyword to remove',
        required: true,
      },
      {
        name: 'addToBlacklist',
        type: 'boolean',
        description: 'Whether to blacklist this keyword for future proposals',
        required: false,
      },
    ],
    handler: async ({ keyword, addToBlacklist }) => {
      await onRemoveKeyword({ keyword, addToBlacklist });
      return `Removed "${keyword}" from the proposal.`;
    },
  });

  // Change funnel distribution
  useCopilotAction({
    name: 'changeDistribution',
    description: 'Adjust the funnel stage distribution (BOFU/MOFU/TOFU percentages).',
    parameters: [
      {
        name: 'bofu',
        type: 'number',
        description: 'Target percentage for BOFU (buy-now) keywords (0-100)',
        required: false,
      },
      {
        name: 'mofu',
        type: 'number',
        description: 'Target percentage for MOFU (comparison) keywords (0-100)',
        required: false,
      },
      {
        name: 'tofu',
        type: 'number',
        description: 'Target percentage for TOFU (awareness) keywords (0-100)',
        required: false,
      },
    ],
    handler: async ({ bofu, mofu, tofu }) => {
      await onChangeDistribution({ bofu, mofu, tofu });
      return `Updated funnel distribution.`;
    },
  });
}
