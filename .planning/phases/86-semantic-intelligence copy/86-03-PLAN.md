import { describe, it, expect } from 'vitest';
import { IntentSplitter, splitByIntent } from './IntentSplitter';
import type { KeywordCluster, ClusteringInput } from './types';

describe('IntentSplitter', () => {
  function createKeyword(
    keyword: string,
    funnelStage: 'bofu' | 'mofu' | 'tofu'
  ): ClusteringInput {
    return {
      keyword,
      embedding: Array(768).fill(0.5),
      volume: 100,
      difficulty: 30,
      funnelStage,
      funnelConfidence: 0.8,
      geoCity: null,
      compositeScore: 0.7,
    };
  }

  function createCluster(
    id: number,
    keywords: ClusteringInput[]
  ): KeywordCluster {
    const funnelBreakdown = { bofu: 0, mofu: 0, tofu: 0 };
    for (const k of keywords) {
      funnelBreakdown[k.funnelStage]++;
    }
    const dominantFunnel = Object.entries(funnelBreakdown)
      .sort(([, a], [, b]) => b - a)[0][0] as 'bofu' | 'mofu' | 'tofu';

    return {
      clusterId: id,
      keywords,
      centroid: Array(768).fill(0.5),
      totalVolume: keywords.reduce((s, k) => s + k.volume, 0),
      averageDifficulty: 30,
      dominantFunnel,
      funnelBreakdown,
    };
  }

  describe('split', () => {
    it('should pass homogeneous cluster unchanged', () => {
      const splitter = new IntentSplitter();
      const cluster = createCluster(0, [
        createKeyword('a', 'bofu'),
        createKeyword('b', 'bofu'),
        createKeyword('c', 'bofu'),
      ]);

      const result = splitter.split([cluster]);

      expect(result).toHaveLength(1);
      expect(result[0].clusterId).toBe(0);
    });

    it('should split cluster with high funnel variance', () => {
      const splitter = new IntentSplitter({ funnelVarianceThreshold: 0.2 });
      const cluster = createCluster(0, [
        createKeyword('buy_a', 'bofu'),
        createKeyword('buy_b', 'bofu'),
        createKeyword('how_c', 'tofu'),
        createKeyword('how_d', 'tofu'),
      ]);

      const result = splitter.split([cluster]);

      expect(result.length).toBeGreaterThan(1);
    });

    it('should create sub-clusters with correct funnel stage', () => {
      const splitter = new IntentSplitter({ funnelVarianceThreshold: 0.1 });
      const cluster = createCluster(0, [
        createKeyword('buy_a', 'bofu'),
        createKeyword('buy_b', 'bofu'),
        createKeyword('how_c', 'tofu'),
        createKeyword('how_d', 'tofu'),
      ]);

      const result = splitter.split([cluster]);

      const bofuCluster = result.find(c => c.dominantFunnel === 'bofu');
      const tofuCluster = result.find(c => c.dominantFunnel === 'tofu');

      expect(bofuCluster).toBeDefined();
      expect(tofuCluster).toBeDefined();
      expect(bofuCluster!.keywords).toHaveLength(2);
      expect(tofuCluster!.keywords).toHaveLength(2);
    });

    it('should preserve sub-cluster IDs with suffix', () => {
      const splitter = new IntentSplitter({ funnelVarianceThreshold: 0.1 });
      const cluster = createCluster(5, [
        createKeyword('a', 'bofu'),
        createKeyword('b', 'tofu'),
      ]);

      const result = splitter.split([cluster]);

      // IDs should be derived from original
      expect(result.some(c => c.clusterId >= 500)).toBe(true);
    });

    it('should recalculate totalVolume for split clusters', () => {
      const splitter = new IntentSplitter({ funnelVarianceThreshold: 0.1 });
      const cluster = createCluster(0, [
        { ...createKeyword('a', 'bofu'), volume: 200 },
        { ...createKeyword('b', 'tofu'), volume: 100 },
      ]);

      const result = splitter.split([cluster]);

      const bofuCluster = result.find(c => c.dominantFunnel === 'bofu');
      const tofuCluster = result.find(c => c.dominantFunnel === 'tofu');

      expect(bofuCluster!.totalVolume).toBe(200);
      expect(tofuCluster!.totalVolume).toBe(100);
    });

    it('should handle multiple clusters', () => {
      const splitter = new IntentSplitter({ funnelVarianceThreshold: 0.1 });
      const clusters = [
        createCluster(0, [createKeyword('a', 'bofu'), createKeyword('b', 'bofu')]),
        createCluster(1, [createKeyword('c', 'bofu'), createKeyword('d', 'tofu')]),
      ];

      const result = splitter.split(clusters);

      // First cluster unchanged (homogeneous), second split
      expect(result.length).toBe(3);
    });

    it('should not split clusters below threshold', () => {
      const splitter = new IntentSplitter({ funnelVarianceThreshold: 0.5 });
      const cluster = createCluster(0, [
        createKeyword('a', 'bofu'),
        createKeyword('b', 'bofu'),
        createKeyword('c', 'mofu'), // 33% non-BOFU, below 50% threshold
      ]);

      const result = splitter.split([cluster]);

      expect(result).toHaveLength(1);
    });
  });

  describe('splitByIntent factory', () => {
    it('should work with default config', () => {
      const cluster = createCluster(0, [
        createKeyword('a', 'bofu'),
        createKeyword('b', 'bofu'),
      ]);

      const result = splitByIntent([cluster]);

      expect(result).toHaveLength(1);
    });
  });
});

/**
 * Intent Splitter
 * Phase 86-03: Semantic Intelligence Pipeline
 * 
 * Splits clusters with mixed funnel intent into homogeneous sub-clusters.
 * Preserves semantic grouping while separating BOFU/MOFU/TOFU content strategies.
 */

import type {
  KeywordCluster,
  ClusteringInput,
  IntentSplitConfig,
} from './types';
import { DEFAULT_INTENT_SPLIT_CONFIG } from './types';

/**
 * Intent-based cluster splitter.
 */
export class IntentSplitter {
  private config: IntentSplitConfig;

  constructor(config: Partial<IntentSplitConfig> = {}) {
    this.config = { ...DEFAULT_INTENT_SPLIT_CONFIG, ...config };
  }

  /**
   * Split clusters with high funnel variance.
   */
  split(clusters: KeywordCluster[]): KeywordCluster[] {
    const result: KeywordCluster[] = [];
    
    for (const cluster of clusters) {
      if (this.needsSplit(cluster)) {
        result.push(...this.splitCluster(cluster));
      } else {
        result.push(cluster);
      }
    }
    
    return result;
  }

  /**
   * Check if cluster needs splitting based on funnel variance.
   */
  private needsSplit(cluster: KeywordCluster): boolean {
    const total = cluster.keywords.length;
    if (total < 2) return false;
    
    const { bofu, mofu, tofu } = cluster.funnelBreakdown;
    
    // Calculate variance from dominant stage
    const dominant = Math.max(bofu, mofu, tofu);
    const nonDominant = total - dominant;
    const variance = nonDominant / total;
    
    return variance > this.config.funnelVarianceThreshold;
  }

  /**
   * Split cluster into funnel-specific sub-clusters.
   */
  private splitCluster(cluster: KeywordCluster): KeywordCluster[] {
    const byFunnel = {
      bofu: [] as ClusteringInput[],
      mofu: [] as ClusteringInput[],
      tofu: [] as ClusteringInput[],
    };
    
    // Group keywords by funnel stage
    for (const keyword of cluster.keywords) {
      byFunnel[keyword.funnelStage].push(keyword);
    }
    
    // Create sub-clusters for non-empty groups
    const subClusters: KeywordCluster[] = [];
    const stages = ['bofu', 'mofu', 'tofu'] as const;
    
    for (let i = 0; i < stages.length; i++) {
      const stage = stages[i];
      const keywords = byFunnel[stage];
      
      if (keywords.length > 0) {
        subClusters.push(this.createSubCluster(
          cluster,
          keywords,
          stage,
          i
        ));
      }
    }
    
    return subClusters;
  }

  /**
   * Create sub-cluster with recalculated metadata.
   */
  private createSubCluster(
    parent: KeywordCluster,
    keywords: ClusteringInput[],
    funnelStage: 'bofu' | 'mofu' | 'tofu',
    subIndex: number
  ): KeywordCluster {
    // Generate unique sub-cluster ID
    const clusterId = parent.clusterId * 100 + subIndex;
    
    // Recalculate centroid as average of keyword embeddings
    const centroid = this.calculateCentroid(keywords);
    
    // Calculate aggregates
    const totalVolume = keywords.reduce((sum, k) => sum + k.volume, 0);
    const averageDifficulty = keywords.reduce((sum, k) => sum + k.difficulty, 0) / keywords.length;
    
    // Funnel breakdown (all same stage for sub-cluster)
    const funnelBreakdown = { bofu: 0, mofu: 0, tofu: 0 };
    funnelBreakdown[funnelStage] = keywords.length;
    
    return {
      clusterId,
      keywords,
      centroid,
      totalVolume,
      averageDifficulty,
      dominantFunnel: funnelStage,
      funnelBreakdown,
    };
  }

  /**
   * Calculate centroid as mean of embeddings.
   */
  private calculateCentroid(keywords: ClusteringInput[]): number[] {
    if (keywords.length === 0) return [];
    
    const dim = keywords[0].embedding.length;
    const centroid = new Array(dim).fill(0);
    
    for (const keyword of keywords) {
      for (let i = 0; i < dim; i++) {
        centroid[i] += keyword.embedding[i];
      }
    }
    
    for (let i = 0; i < dim; i++) {
      centroid[i] /= keywords.length;
    }
    
    return centroid;
  }
}

/**
 * Factory function for intent splitting.
 */
export function splitByIntent(
  clusters: KeywordCluster[],
  config?: Partial<IntentSplitConfig>
): KeywordCluster[] {
  const splitter = new IntentSplitter(config);
  return splitter.split(clusters);
}
