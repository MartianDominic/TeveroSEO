  /** Status */
  status: 'draft' | 'running' | 'paused' | 'completed';
  
  /** Dates */
  startedAt?: Date;
  endedAt?: Date;
  
  /** Minimum sample size for significance */
  minSampleSize: number;
}

 * Variant definition.
 */
export interface Variant {
  id: string;
  name: string;
  description: string;
  
  /** Configuration applied when this variant is active */
  config: Record<string, unknown>;
}

/**
 * Stratification ensures balanced assignment across segments.
 */
export interface StratificationConfig {
  /** Field to stratify on (e.g., 'sophisticationScore') */
  field: string;
  
  /** Buckets for stratification */
  buckets: StratificationBucket[];
}

export interface StratificationBucket {
  name: string;
  min: number;
  max: number;
}

/**
 * Assignment record for a subject.
 */
export interface Assignment {
  experimentId: string;
  subjectId: string;
  variantId: string;
  assignedAt: Date;

  bucket?: string;
}

  experimentId: string;
  subjectId: string;
  variantId: string;

/**
 * A/B Experiment Manager
 * Phase 86-10: A/B Testing Infrastructure
 */

import type {
  Experiment,
  Assignment,
  ConversionEvent,
  ExperimentResults,
  VariantResult,
} from './types';

/**
 * A/B experiment manager.
 */
export class ABExperiment {
  private experiment: Experiment;
  private assignments: Map<string, Assignment> = new Map();
  private conversions: ConversionEvent[] = [];

  constructor(experiment: Experiment) {
    this.experiment = experiment;
  }

  /**
   * Assign subject to variant (deterministic by subject ID).
   */
  assignVariant(
    subjectId: string,
    stratificationValue?: number
  ): Assignment {
    // Check existing assignment
    const existing = this.assignments.get(subjectId);
    if (existing) return existing;
    
    // Determine bucket if stratified
    let bucket: string | undefined;
    if (this.experiment.stratification && stratificationValue !== undefined) {
      bucket = this.getBucket(stratificationValue);
    }
    
    // Deterministic assignment based on subject ID hash
    const variantId = this.selectVariant(subjectId, bucket);
    
    const assignment: Assignment = {
      experimentId: this.experiment.id,
      subjectId,
      variantId,
      assignedAt: new Date(),
      bucket,
    };
    
    this.assignments.set(subjectId, assignment);
    return assignment;
  }

  /**
   * Get bucket for stratification value.
   */
  private getBucket(value: number): string {
    if (!this.experiment.stratification) return 'default';
    
    for (const bucket of this.experiment.stratification.buckets) {
      if (value >= bucket.min && value <= bucket.max) {
        return bucket.name;
      }
    }
    
    return 'default';
  }

  /**
   * Select variant using consistent hashing.
   */
  private selectVariant(subjectId: string, bucket?: string): string {
    // Simple hash-based selection
    const hash = this.hashCode(subjectId + (bucket || ''));
    const normalized = Math.abs(hash) / 2147483647; // Normalize to 0-1
    
    let cumulative = 0;
    for (const [variantId, allocation] of Object.entries(this.experiment.allocation)) {
      cumulative += allocation;
      if (normalized < cumulative) {
        return variantId;
      }
    }
    
    // Fallback to first variant
    return this.experiment.variants[0].id;
  }

  /**
   * Simple hash function.
   */
  private hashCode(str: string): number {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return hash;
  }

  /**
   * Record conversion event.
   */
  recordConversion(
    subjectId: string,
    eventType: ConversionEvent['eventType'],
    metadata?: Record<string, unknown>
  ): void {
    const assignment = this.assignments.get(subjectId);
    if (!assignment) {
      throw new Error(`Subject ${subjectId} not assigned to experiment`);
    }
    
    this.conversions.push({
      experimentId: this.experiment.id,
      subjectId,
      variantId: assignment.variantId,
      eventType,
      timestamp: new Date(),
      metadata,
    });
  }

  /**
   * Calculate experiment results.
   */
  calculateResults(): ExperimentResults {
    const variantResults: VariantResult[] = [];
    
    for (const variant of this.experiment.variants) {
      const variantAssignments = Array.from(this.assignments.values())
        .filter(a => a.variantId === variant.id);
      
      const variantConversions = this.conversions
        .filter(c => c.variantId === variant.id && c.eventType === 'accepted');
      
      const sampleSize = variantAssignments.length;
      const conversions = variantConversions.length;
      const rate = sampleSize > 0 ? conversions / sampleSize : 0;
      
      // Calculate confidence interval (Wilson score)
      const [lower, upper] = this.wilsonScoreInterval(conversions, sampleSize);
      
      variantResults.push({
        variantId: variant.id,
        sampleSize,
        conversions,
        conversionRate: rate,
        confidenceInterval: [lower, upper],
      });
    }
    
    // Determine significance
    const totalSample = variantResults.reduce((s, v) => s + v.sampleSize, 0);
    const isSignificant = totalSample >= this.experiment.minSampleSize &&
      this.checkSignificance(variantResults);
    
    // Find winner if significant
    let winner: string | undefined;
    let confidenceLevel = 0;
    
    if (isSignificant) {
      const sorted = [...variantResults].sort((a, b) => b.conversionRate - a.conversionRate);
      winner = sorted[0].variantId;
      confidenceLevel = 0.95; // Assumed if significant
    }
    
    return {
      experimentId: this.experiment.id,
      variantResults,
      winner,
      confidenceLevel,
      isSignificant,
      sampleSize: totalSample,
    };
  }

  /**
   * Wilson score confidence interval.
   */
  private wilsonScoreInterval(conversions: number, total: number): [number, number] {
    if (total === 0) return [0, 0];
    
    const z = 1.96; // 95% confidence
    const p = conversions / total;
    const n = total;
    
    const denominator = 1 + z * z / n;
    const center = (p + z * z / (2 * n)) / denominator;
    const spread = (z / denominator) * Math.sqrt(p * (1 - p) / n + z * z / (4 * n * n));
    
    return [
      Math.max(0, center - spread),
      Math.min(1, center + spread),
    ];
  }

  /**
   * Check if results are statistically significant.
   */
  private checkSignificance(results: VariantResult[]): boolean {
    if (results.length < 2) return false;
    
    // Check if confidence intervals don't overlap
    const sorted = [...results].sort((a, b) => b.conversionRate - a.conversionRate);
    const best = sorted[0];
    const second = sorted[1];
    
    // Non-overlapping intervals indicate significance
    return best.confidenceInterval[0] > second.confidenceInterval[1];
  }
}

/**
 * Factory function for variant assignment.
 */
export function assignVariant(
  experiment: Experiment,
  subjectId: string,
  stratificationValue?: number
): Assignment {
  const ab = new ABExperiment(experiment);
  return ab.assignVariant(subjectId, stratificationValue);
}

import { pgTable, uuid, text, jsonb, timestamp, index } from 'drizzle-orm/pg-core';

export const abExperiments = pgTable(
  'ab_experiments',
  {
    id: uuid('id').defaultRandom().primaryKey(),
    name: text('name').notNull(),
    description: text('description'),
    variants: jsonb('variants').notNull(),
    allocation: jsonb('allocation').notNull(),
    stratification: jsonb('stratification'),
    status: text('status').notNull().default('draft'),
    startedAt: timestamp('started_at', { withTimezone: true }),
    endedAt: timestamp('ended_at', { withTimezone: true }),
    createdAt: timestamp('created_at', { withTimezone: true }).defaultNow(),
  },
  (table) => [
    index('ab_experiments_status_idx').on(table.status),
  ]
);

export const abAssignments = pgTable(
  'ab_assignments',
  {
    id: uuid('id').defaultRandom().primaryKey(),
    experimentId: uuid('experiment_id').notNull(),
    subjectId: uuid('subject_id').notNull(),
    variantId: text('variant_id').notNull(),
    bucket: text('bucket'),
    assignedAt: timestamp('assigned_at', { withTimezone: true }).defaultNow(),
  },
  (table) => [
    index('ab_assignments_experiment_idx').on(table.experimentId),
    index('ab_assignments_subject_idx').on(table.subjectId),
  ]
);

export const abConversions = pgTable(
  'ab_conversions',
  {
    id: uuid('id').defaultRandom().primaryKey(),
    experimentId: uuid('experiment_id').notNull(),
    subjectId: uuid('subject_id').notNull(),
    variantId: text('variant_id').notNull(),
    eventType: text('event_type').notNull(),
    metadata: jsonb('metadata'),
    timestamp: timestamp('timestamp', { withTimezone: true }).defaultNow(),
  },
  (table) => [
    index('ab_conversions_experiment_idx').on(table.experimentId),
    index('ab_conversions_event_type_idx').on(table.eventType),
  ]
);
