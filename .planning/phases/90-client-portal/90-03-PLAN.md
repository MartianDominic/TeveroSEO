<!-- API response types from 90-02 -->

```typescript
// Dashboard API response
interface DashboardResponse {
  success: true;
  data: {
    metrics: {
      clicks: number;
      impressions: number;
      avgPosition: number;
      top10Count: number;
      deltas: { clicks: number; impressions: number; avgPosition: number; top10Count: number };
    };
    recentWins: Array<{ keyword: string; position: number; previousPosition: number; date: string }>;
    needsAttention: Array<{ keyword: string; position: number; previousPosition: number; dropAmount: number }>;
    lastUpdated: string;
  };
}

// Keywords API response
interface KeywordsResponse {
  success: true;
  data: {
    keywords: Array<{
      keyword: string;
      position: number;
      previousPosition: number;
      change: number;
      clicks: number;
      impressions: number;
      volume: number | null;
      isEstimated: boolean;
    }>;
    pagination: { total: number; limit: number; offset: number; hasMore: boolean };
    summary: { top10: number; improving: number; declining: number; unchanged: number };
  };
}

// Activity API response
interface ActivityResponse {
  success: true;
  data: {
    activities: Array<{
      id: string;
      category: string;
      title: string;
      description: string | null;
      artifacts: Array<{ label: string; url: string }>;
      createdAt: string;
    }>;
    pagination: { total: number; limit: number; offset: number; hasMore: boolean };
  };
}
```

|----------|-------------|
| browser->API | All data flows through authenticated API |
| localStorage | Token stored in httpOnly cookie (not localStorage) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-90-11 | Information Disclosure | api.ts | mitigate | Never log tokens, use httpOnly cookies |
| T-90-12 | Spoofing | hooks.ts | mitigate | Token validation happens server-side, client just passes it |
| T-90-13 | Tampering | sw.js | mitigate | Cache only static assets, never cache API responses with sensitive data |
| T-90-14 | Denial of Service | KeywordTable | accept | Client-side pagination prevents large data loads |
</threat_model>
