/**
 * Chart snapshot service for PDF reports.
 * Renders chart data as PNG images for embedding in PDFs.
 * Uses Puppeteer to render HTML chart and capture screenshot.
 */
import puppeteer from "puppeteer";
import { createLogger } from "@/server/lib/logger";
import { REPORTS_DIR } from "@/server/lib/storage";
import { mkdir, writeFile } from "node:fs/promises";
import path from "node:path";

const log = createLogger({ module: "chart-snapshot" });

const PUPPETEER_WS_ENDPOINT = process.env.PUPPETEER_WS_ENDPOINT;
const SNAPSHOT_TIMEOUT_MS = 30_000;

export interface ChartData {
  type: "gsc" | "ga4";
  data: Array<Record<string, number | string>>;
  width?: number;
  height?: number;
}

/** RGB colors for chart rendering */
const CHART_COLORS = {
  primary: "#3b82f6",
  secondary: "#10b981",
  accent: "#f59e0b",
  grid: "#e5e7eb",
  text: "#374151",
};

/**
 * Generate HTML for a chart that can be rendered to PNG.
 * Uses inline SVG for reliable Puppeteer rendering.
 */
function generateChartHTML(chart: ChartData): string {
  const width = chart.width ?? 700;
  const height = chart.height ?? 280;
  
  // Calculate min/max for scaling
  const values = chart.data.flatMap((d) => 
    Object.values(d).filter((v): v is number => typeof v === "number")
  );
  const min = Math.min(...values);
  const max = Math.max(...values);
  
  // Generate SVG path for line chart
  const padding = 40;
  const chartWidth = width - padding * 2;
  const chartHeight = height - padding * 2;
  
  // Build SVG path points
  const points = chart.data.map((d, i) => {
    const x = padding + (i / (chart.data.length - 1)) * chartWidth;
    const primaryValue = chart.type === "gsc" ? (d.clicks as number) : (d.sessions as number);
    const y = padding + chartHeight - ((primaryValue - min) / (max - min)) * chartHeight;
    return `${x},${y}`;
  });
  
  const pathD = "M " + points.join(" L ");
  
  return `<!DOCTYPE html>
<html>
<head>
  <style>
    body { margin: 0; padding: 0; background: white; }
    svg { display: block; }
    .axis-line { stroke: ${CHART_COLORS.grid}; stroke-width: 1; }
    .chart-line { fill: none; stroke: ${CHART_COLORS.primary}; stroke-width: 2; }
    .axis-label { fill: ${CHART_COLORS.text}; font-size: 11px; font-family: system-ui, sans-serif; }
  </style>
</head>
<body>
  <svg width="${width}" height="${height}" viewBox="0 0 ${width} ${height}">
    <!-- Grid lines -->
    <line x1="${padding}" y1="${padding}" x2="${padding}" y2="${height - padding}" class="axis-line" />
    <line x1="${padding}" y1="${height - padding}" x2="${width - padding}" y2="${height - padding}" class="axis-line" />
    
    <!-- Data line -->
    <path d="${pathD}" class="chart-line" />
    
    <!-- Y-axis labels -->
    <text x="${padding - 8}" y="${padding + 4}" text-anchor="end" class="axis-label">${max.toLocaleString()}</text>
    <text x="${padding - 8}" y="${height - padding + 4}" text-anchor="end" class="axis-label">${min.toLocaleString()}</text>
    
    <!-- X-axis labels (first and last) -->
    <text x="${padding}" y="${height - padding + 16}" text-anchor="start" class="axis-label">${chart.data[0]?.date ?? ""}</text>
    <text x="${width - padding}" y="${height - padding + 16}" text-anchor="end" class="axis-label">${chart.data[chart.data.length - 1]?.date ?? ""}</text>
  </svg>
</body>
</html>`;
}

/**
 * Snapshot a chart to PNG using Puppeteer.
 * @returns Base64-encoded PNG data URL
 */
export async function snapshotChart(chart: ChartData): Promise<string> {
  if (!PUPPETEER_WS_ENDPOINT) {
    log.warn("PUPPETEER_WS_ENDPOINT not configured, using fallback");
    return ""; // Return empty, section renderer will use table fallback
  }

  const browser = await puppeteer.connect({
    browserWSEndpoint: PUPPETEER_WS_ENDPOINT,
  });

  const page = await browser.newPage();

  try {
    const width = chart.width ?? 700;
    const height = chart.height ?? 280;
    
    await page.setViewport({ width, height });
    
    const html = generateChartHTML(chart);
    await page.setContent(html, { waitUntil: "networkidle0" });

    const screenshot = await Promise.race([
      page.screenshot({ type: "png", encoding: "base64" }),
      new Promise<never>((_, reject) =>
        setTimeout(() => reject(new Error("Chart snapshot timed out")), SNAPSHOT_TIMEOUT_MS)
      ),
    ]);

    log.debug("Chart snapshot generated", { type: chart.type, width, height });

    return `data:image/png;base64,${screenshot}`;
  } finally {
    await page.close().catch((err) => log.error("Failed to close page", err));
    browser.disconnect();
  }
}

/**
 * Generate snapshots for multiple charts.
 * @returns Map of chart type to data URL
 */
export async function snapshotCharts(
  gscData?: ChartData["data"],
  ga4Data?: ChartData["data"],
): Promise<Record<string, string>> {
  const results: Record<string, string> = {};

  if (gscData && gscData.length > 0) {
    try {
      results.gsc = await snapshotChart({ type: "gsc", data: gscData });
    } catch (err) {
      log.error("GSC chart snapshot failed", err instanceof Error ? err : new Error(String(err)));
    }
  }

  if (ga4Data && ga4Data.length > 0) {
    try {
      results.ga4 = await snapshotChart({ type: "ga4", data: ga4Data });
    } catch (err) {
      log.error("GA4 chart snapshot failed", err instanceof Error ? err : new Error(String(err)));
    }
  }

  return results;
}

// Default sections if none specified
const DEFAULT_SECTIONS: ReportSection[] = [
  { type: "header", order: 0 },
  { type: "summary_stats", order: 1 },
  { type: "gsc_chart", order: 2 },
  { type: "ga4_chart", order: 3 },
  { type: "queries_table", order: 4 },
  { type: "footer", order: 5 },
];

export default async function processReportJob(
  job: Job<ReportJobData | ReportDLQJobData>,
): Promise<void> {
  // Skip DLQ jobs
  if (job.name.startsWith("dlq:")) {
    log.info("Skipping DLQ job", { jobId: job.id });
    return;
  }

  const data = job.data as ReportJobData;
  const { reportId, clientId, reportType, dateRange, locale, contentHash } = data;
  const sections = (data as ReportJobData & { sections?: ReportSection[] }).sections ?? DEFAULT_SECTIONS;

  const jobLog = createLogger({ module: "report-processor", reportId, clientId });

  try {
    // Update status to generating
    await db.update(reports).set({ status: "generating" }).where(eq(reports.id, reportId));

    // Fetch client branding
    const [branding] = await db
      .select()
      .from(clientBranding)
      .where(eq(clientBranding.clientId, clientId))
      .limit(1);

    // Fetch report data (GSC, GA4, queries)
    const reportData = await fetchReportData(clientId, dateRange, locale);

    // Generate chart snapshots if chart sections are included
    const hasGSCChart = sections.some((s) => s.type === "gsc_chart");
    const hasGA4Chart = sections.some((s) => s.type === "ga4_chart");
    const chartUrls = await snapshotCharts(
      hasGSCChart ? reportData.gscDaily : undefined,
      hasGA4Chart ? reportData.ga4Daily : undefined,
    );

    // Render HTML with selected sections
    const labels = getLabels(locale);
    const html = renderSectionsToHTML(sections, reportData, labels, branding ?? undefined, chartUrls);

    // Generate PDF
    const pdfBuffer = await generatePDF(html);

    // Save PDF to filesystem
    const pdfPath = await saveReport(clientId, reportId, reportType, pdfBuffer);

    // Update database with success
    await db
      .update(reports)
      .set({
        status: "complete",
        pdfPath,
        generatedAt: new Date(),
      })
      .where(eq(reports.id, reportId));

    jobLog.info("Report generated successfully", { pdfPath });
  } catch (err) {
    jobLog.error("Report generation failed", err instanceof Error ? err : new Error(String(err)));

    // Update database with failure
    await db
      .update(reports)
      .set({
        status: "failed",
        errorMessage: err instanceof Error ? err.message : "Unknown error",
      })
      .where(eq(reports.id, reportId));

    throw err; // Re-throw for BullMQ retry
  }

          const contentHash = createHash("sha256")
            .update(`${clientId}:${reportType}:${dateRange.start}:${dateRange.end}:${JSON.stringify(sections ?? [])}`)
            .digest("hex")
            .slice(0, 16);

          // Create report record
          const [report] = await db
            .insert(reports)
            .values({
              clientId,
              reportType,
              dateRangeStart: dateRange.start,
              dateRangeEnd: dateRange.end,
              locale,
              contentHash,
              status: "pending",
            })
            .returning({ id: reports.id });

          // Enqueue generation job
          await enqueueReportGeneration(report.id, {
            clientId,
            reportType,
            dateRange,
            locale,
            contentHash,
            sections,
          } as any); // sections added to job data

          log.info("Report generation queued", { reportId: report.id, clientId, reportType });

          return Response.json({ reportId: report.id }, { status: 202 });
        } catch (err) {
          if (err instanceof AppError) {
            const status = err.code === "UNAUTHENTICATED" ? 401 : err.code === "FORBIDDEN" ? 403 : 400;
            return Response.json({ error: err.message }, { status });
          }
          log.error("Generate report failed", err instanceof Error ? err : new Error(String(err)));
          return Response.json({ error: "Internal error" }, { status: 500 });
        }
      },
    },
  },
});
