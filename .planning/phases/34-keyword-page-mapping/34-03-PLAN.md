/**
 * Keyword aggregation service.
 * Merges keywords from multiple sources:
 * - GSC query snapshots (what's driving traffic)
 * - Saved keywords with metrics (researched keywords)
 * - Keyword rankings (tracked positions)
 * - Prospect analyses (if client was converted from prospect)
 *
 * Deduplicates by keyword text (case-insensitive).
 * Preserves source attribution and best available metrics.
 */
import { db } from "@/db";
import { eq, and, desc, gte, inArray } from "drizzle-orm";
import { gscQuerySnapshots } from "@/db/analytics-schema";
import { savedKeywords, keywordMetrics, projects } from "@/db/app.schema";
import { keywordRankings } from "@/db/ranking-schema";
import { prospectAnalyses, prospects, type OpportunityKeyword, type OrganicKeywordItem } from "@/db/prospect-schema";

export type KeywordSource = "gsc" | "saved" | "prospect_organic" | "prospect_opportunity" | "competitor";

export interface AggregatedKeyword {
  keyword: string;
  sources: KeywordSource[];
  searchVolume: number | null;
  cpc: number | null;
  difficulty: number | null;
  intent: string | null;
  currentPosition: number | null;
  currentUrl: string | null;
  gscClicks: number | null;
  gscImpressions: number | null;
  gscPosition: number | null;
}

interface AggregationOptions {
  projectId: string;
  clientId: string;
  includeGsc?: boolean;      // Default true
  includeSaved?: boolean;    // Default true
  includeProspect?: boolean; // Default true
  gscDaysBack?: number;      // Default 30
  minGscImpressions?: number; // Default 10
}

/**
 * Aggregate keywords from all sources for a project/client.
 */
export async function aggregateKeywords(
  options: AggregationOptions
): Promise<AggregatedKeyword[]> {
  const {
    projectId,
    clientId,
    includeGsc = true,
    includeSaved = true,
    includeProspect = true,
    gscDaysBack = 30,
    minGscImpressions = 10,
  } = options;

  // Map to deduplicate: keyword (lowercase) -> AggregatedKeyword
  const keywordMap = new Map<string, AggregatedKeyword>();

  // Helper to merge a keyword into the map
  const mergeKeyword = (
    keyword: string,
    source: KeywordSource,
    data: Partial<AggregatedKeyword>
  ) => {
    const key = keyword.toLowerCase().trim();
    if (!key) return;

    const existing = keywordMap.get(key);
    if (existing) {
      // Merge: add source, take best metrics
      if (!existing.sources.includes(source)) {
        existing.sources.push(source);
      }
      // Take highest search volume
      if (data.searchVolume && (!existing.searchVolume || data.searchVolume > existing.searchVolume)) {
        existing.searchVolume = data.searchVolume;
      }
      // Take CPC if not set
      if (data.cpc && !existing.cpc) {
        existing.cpc = data.cpc;
      }
      // Take difficulty if not set
      if (data.difficulty && !existing.difficulty) {
        existing.difficulty = data.difficulty;
      }
      // Take intent if not set
      if (data.intent && !existing.intent) {
        existing.intent = data.intent;
      }
      // Merge GSC data
      if (data.gscClicks) {
        existing.gscClicks = (existing.gscClicks ?? 0) + data.gscClicks;
      }
      if (data.gscImpressions) {
        existing.gscImpressions = (existing.gscImpressions ?? 0) + data.gscImpressions;
      }
      if (data.gscPosition && (!existing.gscPosition || data.gscPosition < existing.gscPosition)) {
        existing.gscPosition = data.gscPosition;
      }
      // Update position/URL if better data
      if (data.currentPosition && (!existing.currentPosition || data.currentPosition < existing.currentPosition)) {
        existing.currentPosition = data.currentPosition;
        existing.currentUrl = data.currentUrl ?? existing.currentUrl;
      }
    } else {
      // New keyword
      keywordMap.set(key, {
        keyword: keyword.trim(),  // Preserve original casing
        sources: [source],
        searchVolume: data.searchVolume ?? null,
        cpc: data.cpc ?? null,
        difficulty: data.difficulty ?? null,
        intent: data.intent ?? null,
        currentPosition: data.currentPosition ?? null,
        currentUrl: data.currentUrl ?? null,
        gscClicks: data.gscClicks ?? null,
        gscImpressions: data.gscImpressions ?? null,
        gscPosition: data.gscPosition ?? null,
      });
    }
  };

  // 1. GSC Query Snapshots (last N days, aggregated)
  if (includeGsc) {
    const gscCutoff = new Date();
    gscCutoff.setDate(gscCutoff.getDate() - gscDaysBack);

    const gscQueries = await db
      .select({
        query: gscQuerySnapshots.query,
        clicks: gscQuerySnapshots.clicks,
        impressions: gscQuerySnapshots.impressions,
        position: gscQuerySnapshots.position,
      })
      .from(gscQuerySnapshots)
      .where(
        and(
          eq(gscQuerySnapshots.clientId, clientId),
          gte(gscQuerySnapshots.date, gscCutoff.toISOString().split("T")[0])
        )
      );

    // Aggregate by query (sum clicks/impressions, avg position)
    const gscAggregated = new Map<string, { clicks: number; impressions: number; positions: number[] }>();
    for (const row of gscQueries) {
      const key = row.query.toLowerCase().trim();
      const existing = gscAggregated.get(key) ?? { clicks: 0, impressions: 0, positions: [] };
      existing.clicks += row.clicks ?? 0;
      existing.impressions += row.impressions ?? 0;
      if (row.position) existing.positions.push(row.position);
      gscAggregated.set(key, existing);
    }

    for (const [query, data] of gscAggregated) {
      // Filter by minimum impressions
      if (data.impressions < minGscImpressions) continue;

      const avgPosition = data.positions.length > 0
        ? Math.round(data.positions.reduce((a, b) => a + b, 0) / data.positions.length)
        : null;

      mergeKeyword(query, "gsc", {
        gscClicks: data.clicks,
        gscImpressions: data.impressions,
        gscPosition: avgPosition,
        currentPosition: avgPosition,
      });
    }
  }

  // 2. Saved Keywords with Metrics
  if (includeSaved) {
    const savedRows = await db
      .select({
        keyword: savedKeywords,
        metric: keywordMetrics,
      })
      .from(savedKeywords)
      .leftJoin(
        keywordMetrics,
        and(
          eq(keywordMetrics.keyword, savedKeywords.keyword),
          eq(keywordMetrics.projectId, savedKeywords.projectId),
          eq(keywordMetrics.locationCode, savedKeywords.locationCode),
          eq(keywordMetrics.languageCode, savedKeywords.languageCode)
        )
      )
      .where(eq(savedKeywords.projectId, projectId));

    for (const row of savedRows) {
      mergeKeyword(row.keyword.keyword, "saved", {
        searchVolume: row.metric?.searchVolume ?? null,
        cpc: row.metric?.cpc ?? null,
        difficulty: row.metric?.keywordDifficulty ?? null,
        intent: row.metric?.intent ?? null,
      });
    }

    // Get latest rankings for saved keywords
    const savedKeywordIds = savedRows.map(r => r.keyword.id);
    if (savedKeywordIds.length > 0) {
      // Get latest ranking for each keyword
      const latestRankings = await db
        .select()
        .from(keywordRankings)
        .where(inArray(keywordRankings.keywordId, savedKeywordIds))
        .orderBy(desc(keywordRankings.date));

      // Group by keywordId, take latest
      const rankingsByKeyword = new Map<string, typeof latestRankings[0]>();
      for (const ranking of latestRankings) {
        if (!rankingsByKeyword.has(ranking.keywordId)) {
          rankingsByKeyword.set(ranking.keywordId, ranking);
        }
      }

      // Update aggregated keywords with position data
      for (const row of savedRows) {
        const ranking = rankingsByKeyword.get(row.keyword.id);
        if (ranking) {
          const key = row.keyword.keyword.toLowerCase().trim();
          const existing = keywordMap.get(key);
          if (existing && (!existing.currentPosition || ranking.position < existing.currentPosition)) {
            existing.currentPosition = ranking.position;
            existing.currentUrl = ranking.url;
          }
        }
      }
    }
  }

  // 3. Prospect Analysis (if client was converted from prospect)
  if (includeProspect) {
    // Find if this client has a linked prospect
    const project = await db.query.projects.findFirst({
      where: eq(projects.id, projectId),
    });

    if (project?.domain) {
      // Look for prospect with matching domain
      const prospect = await db.query.prospects.findFirst({
        where: eq(prospects.domain, project.domain),
      });

      if (prospect) {
        // Get latest completed analysis
        const analysis = await db.query.prospectAnalyses.findFirst({
          where: and(
            eq(prospectAnalyses.prospectId, prospect.id),
            eq(prospectAnalyses.status, "completed")
          ),
          orderBy: desc(prospectAnalyses.completedAt),
        });

        if (analysis) {
          // Add organic keywords
          const organicKeywords = analysis.organicKeywords as OrganicKeywordItem[] | null;
          if (organicKeywords) {
            for (const kw of organicKeywords) {
              mergeKeyword(kw.keyword, "prospect_organic", {
                searchVolume: kw.searchVolume,
                cpc: kw.cpc ?? null,
                currentPosition: kw.position,
                currentUrl: kw.url ?? null,
              });
            }
          }

          // Add opportunity keywords
          const opportunityKeywords = analysis.opportunityKeywords as OpportunityKeyword[] | null;
          if (opportunityKeywords) {
            for (const kw of opportunityKeywords) {
              mergeKeyword(kw.keyword, "prospect_opportunity", {
                searchVolume: kw.searchVolume,
                cpc: kw.cpc,
                difficulty: kw.difficulty,
              });
            }
          }
        }
      }
    }
  }

  // Convert map to array and sort by search volume (nulls last)
  return Array.from(keywordMap.values()).sort((a, b) => {
    if (a.searchVolume === null && b.searchVolume === null) return 0;
    if (a.searchVolume === null) return 1;
    if (b.searchVolume === null) return -1;
    return b.searchVolume - a.searchVolume;
  });
}

/**
 * Get aggregation statistics.
 */
export interface AggregationStats {
  totalKeywords: number;
  bySource: Record<KeywordSource, number>;
  withSearchVolume: number;
  withPosition: number;
  uniqueFromGsc: number;
  uniqueFromSaved: number;
  uniqueFromProspect: number;
}

export function calculateAggregationStats(keywords: AggregatedKeyword[]): AggregationStats {
  const bySource: Record<KeywordSource, number> = {
    gsc: 0,
    saved: 0,
    prospect_organic: 0,
    prospect_opportunity: 0,
    competitor: 0,
  };

  for (const kw of keywords) {
    for (const source of kw.sources) {
      bySource[source]++;
    }
  }

  return {
    totalKeywords: keywords.length,
    bySource,
    withSearchVolume: keywords.filter(k => k.searchVolume !== null).length,
    withPosition: keywords.filter(k => k.currentPosition !== null).length,
    uniqueFromGsc: keywords.filter(k => k.sources.length === 1 && k.sources[0] === "gsc").length,
    uniqueFromSaved: keywords.filter(k => k.sources.length === 1 && k.sources[0] === "saved").length,
    uniqueFromProspect: keywords.filter(k => 
      k.sources.length === 1 && 
      (k.sources[0] === "prospect_organic" || k.sources[0] === "prospect_opportunity")
    ).length,
  };
}

export const KeywordAggregationService = {
  aggregateKeywords,
  calculateAggregationStats,
} as const;
