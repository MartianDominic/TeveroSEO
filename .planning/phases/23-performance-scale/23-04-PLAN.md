```typescript
import {
  pgTable,
  text,
  integer,
  numeric,
  timestamp,
  index,
} from "drizzle-orm/pg-core";

/**
 * Pre-computed portfolio-level aggregates.
 * Updated by worker every 5 minutes.
 */
export const portfolioAggregates = pgTable(
  "portfolio_aggregates",
  {
    id: text("id").primaryKey(),
    workspaceId: text("workspace_id").notNull().unique(),
    
    // Client counts
    totalClients: integer("total_clients").default(0),
    clientsOnTrack: integer("clients_on_track").default(0),       // Goal >= 80%
    clientsWatching: integer("clients_watching").default(0),      // Goal 60-79%
    clientsCritical: integer("clients_critical").default(0),      // Goal < 60%
    clientsNoGoals: integer("clients_no_goals").default(0),
    
    // Goal aggregates
    totalGoals: integer("total_goals").default(0),
    goalsMet: integer("goals_met").default(0),
    avgGoalAttainment: numeric("avg_goal_attainment"),
    avgGoalAttainmentTrend: numeric("avg_goal_attainment_trend"),
    
    // Traffic aggregates
    totalClicks30d: integer("total_clicks_30d").default(0),
    totalImpressions30d: integer("total_impressions_30d").default(0),
    avgCtr: numeric("avg_ctr"),
    totalClicksTrend: numeric("total_clicks_trend"),              // % change WoW
    
    // Keyword aggregates
    totalKeywordsTracked: integer("total_keywords_tracked").default(0),
    keywordsTop10: integer("keywords_top_10").default(0),
    keywordsTop3: integer("keywords_top_3").default(0),
    keywordsPosition1: integer("keywords_position_1").default(0),
    keywordsTop10Trend: integer("keywords_top_10_trend").default(0),
    
    // Alert aggregates
    alertsCriticalTotal: integer("alerts_critical_total").default(0),
    alertsWarningTotal: integer("alerts_warning_total").default(0),
    clientsWithCriticalAlerts: integer("clients_with_critical_alerts").default(0),
    
    // Team metrics
    unassignedClients: integer("unassigned_clients").default(0),
    avgDaysSinceTouch: numeric("avg_days_since_touch"),
    clientsNeglected: integer("clients_neglected").default(0),    // > 7 days
    
    // Computation metadata
    computedAt: timestamp("computed_at", { withTimezone: true }).defaultNow(),
    computationDurationMs: integer("computation_duration_ms"),
  },
  (table) => [
    index("idx_portfolio_aggregates_workspace").on(table.workspaceId),
  ]
);

export type PortfolioAggregatesSelect = typeof portfolioAggregates.$inferSelect;
export type PortfolioAggregatesInsert = typeof portfolioAggregates.$inferInsert;
```

### Task 3: Create aggregate computation function

**File:** `apps/worker/src/lib/compute-portfolio-aggregates.ts`

```typescript
import { db } from "@/db";
import { 
  portfolioAggregates,
  clientDashboardMetrics,
  clients,
  clientGoals,
  alerts,
} from "@/db/schema";
import { eq, sql, and, gte, lt, isNull, count, avg, sum } from "drizzle-orm";
import { nanoid } from "nanoid";

export async function computePortfolioAggregates(workspaceId: string): Promise<void> {
  const startTime = Date.now();

  // Get all client metrics for this workspace
  const clientMetrics = await db
    .select()
    .from(clientDashboardMetrics)
    .where(eq(clientDashboardMetrics.workspaceId, workspaceId));

  // Compute client distribution
  const totalClients = clientMetrics.length;
  const clientsOnTrack = clientMetrics.filter(c => Number(c.goalAttainmentPct) >= 80).length;
  const clientsWatching = clientMetrics.filter(c => {
    const pct = Number(c.goalAttainmentPct);
    return pct >= 60 && pct < 80;
  }).length;
  const clientsCritical = clientMetrics.filter(c => Number(c.goalAttainmentPct) < 60).length;
  const clientsNoGoals = clientMetrics.filter(c => c.goalsTotalCount === 0).length;

  // Goal aggregates
  const totalGoals = clientMetrics.reduce((sum, c) => sum + (c.goalsTotalCount ?? 0), 0);
  const goalsMet = clientMetrics.reduce((sum, c) => sum + (c.goalsMetCount ?? 0), 0);
  const avgGoalAttainment = totalClients > 0
    ? clientMetrics.reduce((sum, c) => sum + Number(c.goalAttainmentPct ?? 0), 0) / totalClients
    : 0;

  // Traffic aggregates
  const totalClicks30d = clientMetrics.reduce((sum, c) => sum + (c.trafficTotal ?? 0), 0);
  const totalImpressions30d = clientMetrics.reduce((sum, c) => sum + (c.impressionsTotal ?? 0), 0);
  const avgCtr = totalImpressions30d > 0 ? (totalClicks30d / totalImpressions30d) * 100 : 0;

  // Keyword aggregates
  const keywordsTop10 = clientMetrics.reduce((sum, c) => sum + (c.keywordsTop10 ?? 0), 0);
  const keywordsTop3 = clientMetrics.reduce((sum, c) => sum + (c.keywordsTop3 ?? 0), 0);
  const keywordsPosition1 = clientMetrics.reduce((sum, c) => sum + (c.keywordsPosition1 ?? 0), 0);

  // Alert aggregates
  const alertsCriticalTotal = clientMetrics.reduce((sum, c) => sum + (c.alertsCritical ?? 0), 0);
  const alertsWarningTotal = clientMetrics.reduce((sum, c) => sum + (c.alertsWarning ?? 0), 0);
  const clientsWithCriticalAlerts = clientMetrics.filter(c => (c.alertsCritical ?? 0) > 0).length;

  // Upsert aggregates
  await db
    .insert(portfolioAggregates)
    .values({
      id: `agg-${workspaceId}`,
      workspaceId,
      totalClients,
      clientsOnTrack,
      clientsWatching,
      clientsCritical,
      clientsNoGoals,
      totalGoals,
      goalsMet,
      avgGoalAttainment: String(avgGoalAttainment),
      totalClicks30d,
      totalImpressions30d,
      avgCtr: String(avgCtr),
      keywordsTop10,
      keywordsTop3,
      keywordsPosition1,
      alertsCriticalTotal,
      alertsWarningTotal,
      clientsWithCriticalAlerts,
      computedAt: new Date(),
      computationDurationMs: Date.now() - startTime,
    })
    .onConflictDoUpdate({
      target: portfolioAggregates.workspaceId,
      set: {
        totalClients,
        clientsOnTrack,
        clientsWatching,
        clientsCritical,
        clientsNoGoals,
        totalGoals,
        goalsMet,
        avgGoalAttainment: String(avgGoalAttainment),
        totalClicks30d,
        totalImpressions30d,
        avgCtr: String(avgCtr),
        keywordsTop10,
        keywordsTop3,
        keywordsPosition1,
        alertsCriticalTotal,
        alertsWarningTotal,
        clientsWithCriticalAlerts,
        computedAt: new Date(),
        computationDurationMs: Date.now() - startTime,
      },
    });
}
```

### Task 4: Add to worker schedule

**File:** `apps/worker/src/jobs/portfolio-aggregates.ts`

```typescript
import { Worker, Queue } from "bullmq";
import { redis } from "@/lib/redis";
import { computePortfolioAggregates } from "@/lib/compute-portfolio-aggregates";
import { db } from "@/db";
import { workspaces } from "@/db/schema";

const QUEUE_NAME = "portfolio-aggregates";

export const portfolioAggregatesQueue = new Queue(QUEUE_NAME, {
  connection: redis,
  defaultJobOptions: {
    removeOnComplete: 100,
    removeOnFail: 50,
  },
});

// Schedule for all workspaces every 5 minutes
export async function schedulePortfolioAggregates() {
  const allWorkspaces = await db.select({ id: workspaces.id }).from(workspaces);
  
  for (const workspace of allWorkspaces) {
    await portfolioAggregatesQueue.add(
      "compute",
      { workspaceId: workspace.id },
      {
        repeat: { every: 5 * 60 * 1000 }, // Every 5 minutes
        jobId: `portfolio-aggregates-${workspace.id}`,
      }
    );
  }
}

export const portfolioAggregatesWorker = new Worker(
  QUEUE_NAME,
  async (job) => {
    const { workspaceId } = job.data;
    await computePortfolioAggregates(workspaceId);
    return { success: true, workspaceId };
  },
  { connection: redis, concurrency: 5 }
);
```

### Task 6: Create server action

**File:** `apps/web/src/actions/dashboard/get-portfolio-aggregates.ts`

```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { db } from "@/db";
import { portfolioAggregates } from "@/db/schema";
import { eq } from "drizzle-orm";
import { cacheGet, cacheSet, cacheKeys, cacheTags } from "@/lib/cache";

export async function getPortfolioAggregates(workspaceId: string) {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  // Check cache first
  const cacheKey = cacheKeys.portfolioAggregates(workspaceId);
  const cached = await cacheGet(cacheKey);
  if (cached) return cached;

  const [aggregates] = await db
    .select()
    .from(portfolioAggregates)
    .where(eq(portfolioAggregates.workspaceId, workspaceId))
    .limit(1);

  if (!aggregates) {
    return null;
  }

  // Cache for 1 minute
  await cacheSet(cacheKey, aggregates, {
    ttl: 60,
    tags: [cacheTags.workspace(workspaceId)],
  });

  return aggregates;
}
```
