Create installment-reminder-processor.ts per D-17 to D-20:
```typescript
export default async function process(job: Job) {
  const today = new Date();
  
  // Find all workspaces with pending installments
  // For each workspace:
  
  // 1. Installments due in 3 days (not yet reminded)
  const upcoming = await PaymentScheduleRepository.getInstallmentsDueIn(3);
  for (const inst of upcoming.filter(i => !i.reminderSentAt)) {
    await sendReminder(inst, 'installment-reminder');
    await markReminderSent(inst.id, 'reminder');
  }
  
  // 2. Installments due today (not paid, not reminded today)
  const dueToday = await PaymentScheduleRepository.getInstallmentsDueIn(0);
  for (const inst of dueToday.filter(i => i.status === 'pending')) {
    await sendReminder(inst, 'installment-due-today');
    await markReminderSent(inst.id, 'due_today');
  }
  
  // 3. Overdue installments (1 day)
  const overdue1 = await PaymentScheduleRepository.getOverdueInstallments(1);
  for (const inst of overdue1) {
    await updateStatus(inst.id, 'overdue');
    await sendReminder(inst, 'installment-overdue');
  }
  
  // 4. Overdue installments (7 days)
  const overdue7 = await PaymentScheduleRepository.getOverdueInstallments(7);
  for (const inst of overdue7) {
    await sendReminder(inst, 'installment-overdue-urgent');
  }
}
```
