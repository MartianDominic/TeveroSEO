```typescript
// seo-chat.ts
import { pgTable, uuid, text, timestamp, jsonb, integer, index, uniqueIndex } from "drizzle-orm/pg-core";
import { relations } from "drizzle-orm";
import { workspaces } from "./workspace";
import { prospects } from "./prospect";

export const seoChatSessions = pgTable("seo_chat_sessions", {
  id: uuid("id").primaryKey().defaultRandom(),
  workspaceId: uuid("workspace_id").notNull().references(() => workspaces.id),
  prospectId: uuid("prospect_id").references(() => prospects.id),
  prospectDomain: text("prospect_domain"),
  title: text("title"),
  status: text("status").default("active"), // 'active' | 'archived' | 'converted'
  metadata: jsonb("metadata").default({}),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow(),
}, (table) => ({
  workspaceIdx: index("idx_sessions_workspace").on(table.workspaceId),
  prospectIdx: index("idx_sessions_prospect").on(table.prospectId),
}));

export const seoChatMessages = pgTable("seo_chat_messages", {
  id: uuid("id").primaryKey().defaultRandom(),
  sessionId: uuid("session_id").notNull().references(() => seoChatSessions.id, { onDelete: "cascade" }),
  role: text("role").notNull(), // 'user' | 'assistant' | 'system' | 'tool'
  content: text("content").notNull(),
  intent: text("intent"),
  extractedContext: jsonb("extracted_context").default({}),
  toolCalls: jsonb("tool_calls").default([]),
  tokenCount: integer("token_count"),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
}, (table) => ({
  sessionIdx: index("idx_messages_session").on(table.sessionId),
  createdIdx: index("idx_messages_created").on(table.createdAt),
}));

export const seoChatAnalyses = pgTable("seo_chat_analyses", {
  id: uuid("id").primaryKey().defaultRandom(),
  sessionId: uuid("session_id").notNull().references(() => seoChatSessions.id, { onDelete: "cascade" }),
  analysisType: text("analysis_type").notNull(),
  inputHash: text("input_hash").notNull(),
  result: jsonb("result").notNull(),
  costMicros: integer("cost_micros").default(0),
  durationMs: integer("duration_ms"),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
}, (table) => ({
  cacheIdx: uniqueIndex("idx_analyses_cache").on(table.sessionId, table.analysisType, table.inputHash),
}));
```
