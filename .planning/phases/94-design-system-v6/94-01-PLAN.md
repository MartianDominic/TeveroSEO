  font-weight: 400;
  line-height: 1.05;
  letter-spacing: -0.024em;
  color: var(--text-1);

.t-eyebrow {
  font-family: var(--font-sans);
  font-size: var(--type-tiny);

  line-height: 1.3;
  letter-spacing: 0.1em;
  text-transform: uppercase;
  color: var(--text-3);
}

.t-smallcaps {
  font-family: var(--font-display);
