  private client: OpenAI;
  private circuit: CircuitBreaker;

  constructor(apiKey?: string) {
    const key = apiKey ?? process.env.XAI_API_KEY;
    if (!key) {
      throw new Error("XAI_API_KEY not configured");
    }

    this.client = new OpenAI({
      apiKey: key,
      baseURL: GROK_CONFIG.baseURL,
    });

    this.circuit = new CircuitBreaker({
      name: "grok-classifier",
      failureThreshold: 3,
      resetTimeout: 60000,
    });
  }
}

```typescript
import { z } from "zod";

// 12 primary verticals from CONTEXT.md
export const VERTICALS = [
  "healthcare", "legal", "financial", "ecommerce", "saas",
  "real_estate", "home_services", "hospitality", "education",
  "professional", "manufacturing", "nonprofit", "general"
] as const;

export type Vertical = typeof VERTICALS[number];

// YMYL verticals requiring stricter quality gates
export const YMYL_VERTICALS: Vertical[] = ["healthcare", "legal", "financial"];

// Classification result
export interface Classification {
  vertical: Vertical;
  confidence: number;
  isYmyl: boolean;
  method: "schema" | "url-pattern" | "keyword" | "client-setting" | "llm";
}

// Classification Zod schema for LLM response validation
export const ClassificationResponseSchema = z.object({
  vertical: z.enum(VERTICALS),
  confidence: z.number().min(0).max(1),
  reasoning: z.string(),
});

// Schema.org type to vertical mapping (from COST-CONTROL.md)
export const SCHEMA_TO_VERTICAL: Record<string, Vertical> = {
  "LocalBusiness": "home_services",
  "MedicalBusiness": "healthcare",
  "MedicalOrganization": "healthcare",
  "Hospital": "healthcare",
  "Physician": "healthcare",
  "LegalService": "legal",
  "Attorney": "legal",
  "FinancialService": "financial",
  "BankOrCreditUnion": "financial",
  "InsuranceAgency": "financial",
  "Product": "ecommerce",
  "SoftwareApplication": "saas",
  "WebApplication": "saas",
  "RealEstateAgent": "real_estate",
  "RealEstateListing": "real_estate",
  "Restaurant": "hospitality",
  "Hotel": "hospitality",
  "LodgingBusiness": "hospitality",
  "EducationalOrganization": "education",
  "Course": "education",
  "ProfessionalService": "professional",
  "Organization": "nonprofit",
};

// URL patterns for vertical detection
export const URL_PATTERNS: [RegExp, Vertical][] = [
  [/\/products?\/|\/shop\/|\/cart|\/checkout/i, "ecommerce"],
  [/\/pricing|\/plans|\/features|\/integrations/i, "saas"],
  [/\/locations?\/|\/near-me|\/service-area/i, "home_services"],
  [/\/doctors?\/|\/health\/|\/medical|\/symptoms/i, "healthcare"],
  [/\/attorneys?\/|\/lawyers?\/|\/legal|\/law-firm/i, "legal"],
  [/\/invest|\/finance|\/loans?|\/mortgage|\/insurance/i, "financial"],
  [/\/properties|\/listings|\/real-estate|\/homes-for/i, "real_estate"],
  [/\/book|\/reservations|\/rooms|\/hotels/i, "hospitality"],
  [/\/courses|\/learn|\/training|\/education/i, "education"],
];

// YMYL keyword patterns for fallback detection
export const YMYL_KEYWORDS = {
  healthcare: /\b(diagnosis|treatment|symptoms|medication|surgery|cancer|disease|prescription)\b/i,
  legal: /\b(lawsuit|liability|damages|settlement|litigation|attorney|court|verdict)\b/i,
  financial: /\b(investment|retirement|401k|mortgage|bankruptcy|credit score|tax|debt)\b/i,
};

// On-page mastery context for check execution
export interface OnPageMasteryContext {
  url: string;
  html: string;
  vertical: Vertical;
  isYmyl: boolean;
  clientId: string;
  pageId?: string;
}

// Quality gate result
export interface GateResult {
  passed: boolean;
  score: number;
  message: string;
  method: "embedding" | "llm" | "llm-fallback" | "rule";
  confidence?: "high" | "medium" | "low";
  embeddingSimilarity?: number;
}
```

  - Check database cache first (verticalClassifications table)

  - Run classifyHeuristic
  - If confidence >= 0.90, cache and return
  - Otherwise call classifyLLM, cache and return
- extractPathPattern(path: string): string
  - Convert /product/123 to /product/*
  - Convert /blog/2024/01/title to /blog/*
