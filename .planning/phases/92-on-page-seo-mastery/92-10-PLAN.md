import { describe, it, expect, vi, beforeEach } from "vitest";
import { RuleEngineService } from "./RuleEngineService";

// Mock database
const mockDb = {
  select: vi.fn().mockReturnThis(),
  from: vi.fn().mockReturnThis(),
  where: vi.fn().mockResolvedValue([]),
};

vi.mock("@/db", () => ({
  getDb: vi.fn().mockResolvedValue(mockDb),
}));

describe("RuleEngineService", () => {
  let service: RuleEngineService;

  beforeEach(() => {
    service = new RuleEngineService(mockDb as any);
    vi.clearAllMocks();
  });

  describe("evaluateScorecard", () => {
    const baseContext = {
      url: "https://example.com/article",
      html: `
        <html>
          <head><title>Test Article</title></head>
          <body>
            <h1>Single H1 Heading</h1>
            <h2>Section One</h2>
            <p>${"word ".repeat(500)}</p>
            <h2>Section Two</h2>
            <p>${"word ".repeat(500)}</p>
          </body>
        </html>
      `,
      vertical: "general" as const,
      isYmyl: false,
      clientId: "client-1",
    };

    it("should return score 0-100", async () => {
      const result = await service.evaluateScorecard(baseContext);

      expect(result.score).toBeGreaterThanOrEqual(0);
      expect(result.score).toBeLessThanOrEqual(100);
    });

    it("should apply YMYL threshold of 85", async () => {
      const result = await service.evaluateScorecard({
        ...baseContext,
        vertical: "healthcare",
        isYmyl: true,
      });

      // With YMYL, passing requires score >= 85
      expect(result.isYmyl).toBe(true);
    });

    it("should apply standard threshold of 70 for non-YMYL", async () => {
      const result = await service.evaluateScorecard(baseContext);

      // Score 70+ should pass for non-YMYL
      if (result.score >= 70) {
        expect(result.passed).toBe(true);
      } else {
        expect(result.passed).toBe(false);
      }
    });

    it("should include passed and failed rules", async () => {
      const result = await service.evaluateScorecard(baseContext);

      expect(Array.isArray(result.passedRules)).toBe(true);
      expect(Array.isArray(result.failedRules)).toBe(true);
      expect(result.passedRules.length + result.failedRules.length).toBeGreaterThan(0);
    });
  });

  describe("rule override hierarchy", () => {
    it("should apply client overrides over vertical rules", async () => {
      // Set up mock to return client override
      mockDb.where.mockResolvedValueOnce([
        { ruleId: "R-01", weight: 0, enabled: false },
      ]);

      const result = await service.evaluateScorecard({
        url: "https://example.com",
        html: "<html><body></body></html>",
        vertical: "general",
        isYmyl: false,
        clientId: "client-1",
      });

      // R-01 should not appear in results since it's disabled
      const hasR01 = [...result.passedRules, ...result.failedRules]
        .some(r => r.id === "R-01");
      expect(hasR01).toBe(false);
    });
  });
});
