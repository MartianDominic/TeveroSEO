```typescript
"use server";

import { auth } from "@clerk/nextjs/server";
import { revalidatePath } from "next/cache";

const API_BASE = process.env.OPEN_SEO_API_URL || "http://localhost:3001";

export async function getServices() {
  const { orgId } = await auth();
  if (!orgId) return { success: false, error: "Unauthorized" };
  
  const res = await fetch(`${API_BASE}/api/services`, {
    headers: { "x-workspace-id": orgId },
    cache: "no-store",
  });
  
  if (!res.ok) return { success: false, error: "Failed to fetch services" };
  return { success: true, data: await res.json() };
}

export async function createService(data: CreateServiceInput) {
  const { orgId } = await auth();
  if (!orgId) return { success: false, error: "Unauthorized" };
  
  const res = await fetch(`${API_BASE}/api/services`, {
    method: "POST",
    headers: { "Content-Type": "application/json", "x-workspace-id": orgId },
    body: JSON.stringify(data),
  });
  
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    return { success: false, error: err.message || "Failed to create service" };
  }
  
  revalidatePath("/settings/services");
  return { success: true, data: await res.json() };
}
```

```tsx
"use client";

import { useState } from "react";
import { ServiceCard } from "./ServiceCard";
import { ServiceFormModal } from "./ServiceFormModal";
import { Button } from "@tevero/ui";
import { Plus } from "lucide-react";
import { useTranslations } from "next-intl";

interface ServiceTemplateSelect {
  id: string;
  workspaceId: string | null;
  category: 'seo_package' | 'addon' | 'one_time';
  name: string;
  nameEn?: string | null;
  nameLt?: string | null;
  description?: string | null;
  pricingType: 'monthly' | 'one_time' | 'per_unit';
  basePriceCents: number | null;
  setupFeeCents?: number | null;
  currency?: string | null;
  unitLabel?: string | null;
  inclusions?: string[] | null;
  icon?: string | null;
  displayOrder?: number | null;
  isActive: boolean;
}

interface Props {
  services: ServiceTemplateSelect[];
}

export function ServiceTable({ services }: Props) {
  const t = useTranslations("serviceCatalog");
  const [isCreateOpen, setIsCreateOpen] = useState(false);
  
  // Group services by category
  const seoPackages = services.filter(s => s.category === 'seo_package' && s.isActive);
  const addons = services.filter(s => s.category === 'addon' && s.isActive);
  const oneTime = services.filter(s => s.category === 'one_time' && s.isActive);
  
  const renderGroup = (title: string, items: ServiceTemplateSelect[]) => (
    <div className="mb-8">
      <h2 className="text-lg font-semibold mb-4">{title}</h2>
      {items.length === 0 ? (
        <p className="text-muted-foreground text-sm">No services in this category</p>
      ) : (
        <div className="space-y-2">
          {items.map(service => (
            <ServiceCard key={service.id} service={service} />
          ))}
        </div>
      )}
    </div>
  );
  
  return (
    <div>
      <div className="flex justify-end mb-6">
        <Button onClick={() => setIsCreateOpen(true)}>
          <Plus className="mr-2 h-4 w-4" />
          {t("addNew")}
        </Button>
      </div>
      
      {services.length === 0 ? (
        <div className="text-center py-12 text-muted-foreground">
          {t("empty")}
        </div>
      ) : (
        <>
          {renderGroup(t("categories.seo_package"), seoPackages)}
          {renderGroup(t("categories.addon"), addons)}
          {renderGroup(t("categories.one_time"), oneTime)}
        </>
      )}
      
      <ServiceFormModal 
        open={isCreateOpen} 
        onOpenChange={setIsCreateOpen}
        mode="create"
      />
    </div>
  );
}
```

```tsx
"use client";

import { useState, useTransition } from "react";
import { Card, CardContent, Button, DropdownMenu, DropdownMenuContent, DropdownMenuItem, DropdownMenuTrigger } from "@tevero/ui";
import { MoreHorizontal, Edit, Copy, Trash2, Lock } from "lucide-react";
import { ServiceFormModal } from "./ServiceFormModal";
import { deleteService, duplicateService } from "../actions";
import { useTranslations } from "next-intl";
import * as Icons from "lucide-react";

interface ServiceTemplateSelect { /* same as above */ }

interface Props {
  service: ServiceTemplateSelect;
}

export function ServiceCard({ service }: Props) {
  const t = useTranslations("serviceCatalog");
  const [isEditOpen, setIsEditOpen] = useState(false);
  const [isPending, startTransition] = useTransition();
  
  const isSystemTemplate = service.workspaceId === null;
  
  // Dynamic icon rendering
  const IconComponent = service.icon && Icons[service.icon as keyof typeof Icons] 
    ? Icons[service.icon as keyof typeof Icons] as React.ComponentType<{ className?: string }>
    : Icons.Package;
  
  const formatPrice = (cents: number | null) => {
    if (cents === null) return "-";
    return new Intl.NumberFormat("en-US", {
      style: "currency",
      currency: service.currency || "EUR",
    }).format(cents / 100);
  };
  
  const handleDuplicate = () => {
    startTransition(async () => {
      await duplicateService(service.id);
    });
  };
  
  const handleDelete = () => {
    if (!confirm(t("confirmDelete"))) return;
    startTransition(async () => {
      await deleteService(service.id);
    });
  };
  
  return (
    <>
      <Card className="flex items-center justify-between p-4">
        <div className="flex items-center gap-4">
          <div className="w-10 h-10 rounded-lg bg-muted flex items-center justify-center">
            <IconComponent className="h-5 w-5" />
          </div>
          <div>
            <div className="flex items-center gap-2">
              <span className="font-medium">{service.name}</span>
              {isSystemTemplate && (
                <Lock className="h-3 w-3 text-muted-foreground" title={t("systemTemplate")} />
              )}
            </div>
            <div className="text-sm text-muted-foreground">
              {formatPrice(service.basePriceCents)}
              {service.pricingType === "monthly" && "/mo"}
              {service.setupFeeCents && service.setupFeeCents > 0 && (
                <span className="ml-2">+ {formatPrice(service.setupFeeCents)} setup</span>
              )}
            </div>
          </div>
        </div>
        
        <DropdownMenu>
          <DropdownMenuTrigger asChild>
            <Button variant="ghost" size="icon" disabled={isPending}>
              <MoreHorizontal className="h-4 w-4" />
            </Button>
          </DropdownMenuTrigger>
          <DropdownMenuContent align="end">
            <DropdownMenuItem onClick={() => setIsEditOpen(true)} disabled={isSystemTemplate}>
              <Edit className="mr-2 h-4 w-4" />
              {t("edit")}
            </DropdownMenuItem>
            <DropdownMenuItem onClick={handleDuplicate}>
              <Copy className="mr-2 h-4 w-4" />
              {t("duplicate")}
            </DropdownMenuItem>
            <DropdownMenuItem 
              onClick={handleDelete} 
              disabled={isSystemTemplate}
              className="text-destructive"
            >
              <Trash2 className="mr-2 h-4 w-4" />
              {t("delete")}
            </DropdownMenuItem>
          </DropdownMenuContent>
        </DropdownMenu>
      </Card>
      
      <ServiceFormModal
        open={isEditOpen}
        onOpenChange={setIsEditOpen}
        mode="edit"
        service={service}
      />
    </>
  );
}
```

```tsx
"use client";

import { useState, useTransition, useEffect } from "react";
import {
  Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter,
  Button, Input, Label, Select, SelectContent, SelectItem, SelectTrigger, SelectValue,
  Textarea,
} from "@tevero/ui";
import { Loader2, Plus, X } from "lucide-react";
import { createService, updateService } from "../actions";
import { useTranslations } from "next-intl";

const SERVICE_CATEGORIES = ['seo_package', 'addon', 'one_time'] as const;
const PRICING_TYPES = ['monthly', 'one_time', 'per_unit'] as const;

const ICON_OPTIONS = [
  'Zap', 'TrendingUp', 'Building', 'MapPin', 'Star', 'Globe', 'Users', 'Calendar', 'Package'
];

interface ServiceTemplateSelect { /* full type */ }

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  mode: "create" | "edit";
  service?: ServiceTemplateSelect;
}

export function ServiceFormModal({ open, onOpenChange, mode, service }: Props) {
  const t = useTranslations("serviceCatalog");
  const tFields = useTranslations("serviceCatalog.fields");
  const tForm = useTranslations("serviceCatalog.form");
  const [isPending, startTransition] = useTransition();
  
  // Form state
  const [name, setName] = useState(service?.name || "");
  const [nameEn, setNameEn] = useState(service?.nameEn || "");
  const [nameLt, setNameLt] = useState(service?.nameLt || "");
  const [description, setDescription] = useState(service?.description || "");
  const [descriptionEn, setDescriptionEn] = useState(service?.descriptionEn || "");
  const [descriptionLt, setDescriptionLt] = useState(service?.descriptionLt || "");
  const [category, setCategory] = useState<typeof SERVICE_CATEGORIES[number]>(
    service?.category || "addon"
  );
  const [pricingType, setPricingType] = useState<typeof PRICING_TYPES[number]>(
    service?.pricingType || "monthly"
  );
  const [basePrice, setBasePrice] = useState(
    service?.basePriceCents ? String(service.basePriceCents / 100) : ""
  );
  const [setupFee, setSetupFee] = useState(
    service?.setupFeeCents ? String(service.setupFeeCents / 100) : ""
  );
  const [unitLabel, setUnitLabel] = useState(service?.unitLabel || "");
  const [inclusions, setInclusions] = useState<string[]>(service?.inclusions || []);
  const [newInclusion, setNewInclusion] = useState("");
  const [termsTemplate, setTermsTemplate] = useState(service?.termsTemplate || "");
  const [icon, setIcon] = useState(service?.icon || "Package");
  
  // Reset form when modal opens with different service
  useEffect(() => {
    if (open && service) {
      setName(service.name);
      setNameEn(service.nameEn || "");
      setNameLt(service.nameLt || "");
      setDescription(service.description || "");
      setCategory(service.category);
      setPricingType(service.pricingType);
      setBasePrice(service.basePriceCents ? String(service.basePriceCents / 100) : "");
      setSetupFee(service.setupFeeCents ? String(service.setupFeeCents / 100) : "");
      setUnitLabel(service.unitLabel || "");
      setInclusions(service.inclusions || []);
      setTermsTemplate(service.termsTemplate || "");
      setIcon(service.icon || "Package");
    } else if (open && !service) {
      // Reset for create mode
      setName("");
      setNameEn("");
      setNameLt("");
      setDescription("");
      setCategory("addon");
      setPricingType("monthly");
      setBasePrice("");
      setSetupFee("");
      setUnitLabel("");
      setInclusions([]);
      setTermsTemplate("");
      setIcon("Package");
    }
  }, [open, service]);
  
  const addInclusion = () => {
    if (newInclusion.trim()) {
      setInclusions([...inclusions, newInclusion.trim()]);
      setNewInclusion("");
    }
  };
  
  const removeInclusion = (index: number) => {
    setInclusions(inclusions.filter((_, i) => i !== index));
  };
  
  const handleSubmit = () => {
    const data = {
      name,
      nameEn: nameEn || undefined,
      nameLt: nameLt || undefined,
      description: description || undefined,
      descriptionEn: descriptionEn || undefined,
      descriptionLt: descriptionLt || undefined,
      category,
      pricingType,
      basePriceCents: Math.round(parseFloat(basePrice || "0") * 100),
      setupFeeCents: setupFee ? Math.round(parseFloat(setupFee) * 100) : undefined,
      unitLabel: pricingType === "per_unit" ? unitLabel : undefined,
      inclusions: inclusions.length > 0 ? inclusions : undefined,
      termsTemplate: termsTemplate || undefined,
      icon,
    };
    
    startTransition(async () => {
      const result = mode === "create"
        ? await createService(data)
        : await updateService(service!.id, data);
      
      if (result.success) {
        onOpenChange(false);
      } else {
        alert(result.error);
      }
    });
  };
  
  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>
            {mode === "create" ? tForm("create") : tForm("update")}
          </DialogTitle>
        </DialogHeader>
        
        <div className="space-y-6 py-4">
          {/* Name fields */}
          <div className="grid grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="nameEn">{tFields("name")} (EN)</Label>
              <Input id="nameEn" value={nameEn} onChange={e => { setNameEn(e.target.value); if (!name) setName(e.target.value); }} />
            </div>
            <div className="space-y-2">
              <Label htmlFor="nameLt">{tFields("name")} (LT)</Label>
              <Input id="nameLt" value={nameLt} onChange={e => setNameLt(e.target.value)} />
            </div>
          </div>
          
          {/* Category + Pricing Type */}
          <div className="grid grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label>{tFields("category")}</Label>
              <Select value={category} onValueChange={(v) => setCategory(v as typeof category)}>
                <SelectTrigger><SelectValue /></SelectTrigger>
                <SelectContent>
                  {SERVICE_CATEGORIES.map(cat => (
                    <SelectItem key={cat} value={cat}>
                      {t(`categories.${cat}`)}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
            <div className="space-y-2">
              <Label>{tFields("pricingType")}</Label>
              <Select value={pricingType} onValueChange={(v) => setPricingType(v as typeof pricingType)}>
                <SelectTrigger><SelectValue /></SelectTrigger>
                <SelectContent>
                  {PRICING_TYPES.map(pt => (
                    <SelectItem key={pt} value={pt}>
                      {t(`pricingTypes.${pt}`)}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
          </div>

            <div className="space-y-2">
              <Label htmlFor="basePrice">{tFields("basePrice")} (EUR)</Label>
              <Input id="basePrice" type="number" min="0" step="0.01" value={basePrice} onChange={e => setBasePrice(e.target.value)} />
            </div>
            {pricingType !== "per_unit" && (
              <div className="space-y-2">
                <Label htmlFor="setupFee">{tFields("setupFee")} (EUR)</Label>
                <Input id="setupFee" type="number" min="0" step="0.01" value={setupFee} onChange={e => setSetupFee(e.target.value)} />
              </div>
            )}
            {pricingType === "per_unit" && (
              <div className="space-y-2">
                <Label htmlFor="unitLabel">Unit Label</Label>
                <Input id="unitLabel" placeholder="e.g., per article" value={unitLabel} onChange={e => setUnitLabel(e.target.value)} />
              </div>
            )}
          </div>
          
          {/* Description */}
          <div className="space-y-2">
            <Label>{tFields("description")} (EN)</Label>
            <Textarea value={descriptionEn} onChange={e => { setDescriptionEn(e.target.value); if (!description) setDescription(e.target.value); }} rows={2} />
          </div>
          <div className="space-y-2">
            <Label>{tFields("description")} (LT)</Label>
            <Textarea value={descriptionLt} onChange={e => setDescriptionLt(e.target.value)} rows={2} />
          </div>
          
          {/* Inclusions */}
          <div className="space-y-2">
            <Label>{tFields("inclusions")}</Label>
            <div className="space-y-2">
              {inclusions.map((item, idx) => (
                <div key={idx} className="flex items-center gap-2">
                  <span className="flex-1 text-sm bg-muted px-3 py-1 rounded">{item}</span>
                  <Button variant="ghost" size="icon" onClick={() => removeInclusion(idx)}>
                    <X className="h-4 w-4" />
                  </Button>
                </div>
              ))}
              <div className="flex gap-2">
                <Input 
                  value={newInclusion} 
                  onChange={e => setNewInclusion(e.target.value)} 
                  placeholder="Add inclusion..."
                  onKeyDown={e => e.key === "Enter" && (e.preventDefault(), addInclusion())}
                />
                <Button variant="outline" size="icon" onClick={addInclusion}>
                  <Plus className="h-4 w-4" />
                </Button>
              </div>
            </div>
          </div>
          
          {/* Terms Template */}
          <div className="space-y-2">
            <Label>{tFields("terms")}</Label>
            <Textarea value={termsTemplate} onChange={e => setTermsTemplate(e.target.value)} rows={3} placeholder="Optional legal terms for this service..." />
          </div>
          
          {/* Icon */}
          <div className="space-y-2">
            <Label>{tFields("icon")}</Label>
            <Select value={icon} onValueChange={setIcon}>
              <SelectTrigger><SelectValue /></SelectTrigger>
              <SelectContent>
                {ICON_OPTIONS.map(iconName => (
                  <SelectItem key={iconName} value={iconName}>{iconName}</SelectItem>
                ))}
              </SelectContent>
            </Select>
          </div>
        </div>
        
        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>
            {tForm("cancel")}
          </Button>
          <Button onClick={handleSubmit} disabled={isPending || !name}>
            {isPending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
            {tForm("save")}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
```
