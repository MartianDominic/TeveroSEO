<objective>
Build the content intelligence layer for TeveroSEO: Topic Clusters with hub + spoke visualization and gap detection, Content Groups with folder-based auto-grouping and custom rules, and Index Coverage reporting with URL Inspection API integration and batch indexing requests.

Purpose: Content intelligence transforms raw page data into strategic insights. Agencies can visualize content architecture (clusters), organize pages for analysis (groups), and ensure content gets indexed (coverage). Topic clusters identify content gaps and internal linking opportunities. Index coverage prevents the common problem of published content never appearing in search results.

Output:
- Content Groups schema with folder-based, regex, and manual grouping
- Topic Cluster detection algorithm identifying hubs, spokes, coverage, and gaps
- URL Inspection API integration with priority queue and daily limit management
- Index Coverage dashboard with batch indexing request feature
- Hub + spoke cluster visualization using SVG force-directed layout
</objective>

/**
 * Content Intelligence Schema
 * Phase 96-04: Content Groups, Topic Clusters, Index Coverage
 * 
 * Design decisions:
 * - content_groups: folder-based, regex, and manual grouping types
 * - topic_clusters: hub + spoke relationship with coverage tracking
 * - page_index_status: URL Inspection API results with all Google fields
 * - indexing_requests: Track batch indexing with status and response
 */
import { pgTable, uuid, text, boolean, timestamp, jsonb, integer, real, index, unique } from "drizzle-orm/pg-core";
import { sites } from "./app.schema";

// ===== Content Groups =====

export const contentGroups = pgTable(
  "content_groups",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    siteId: uuid("site_id").notNull().references(() => sites.id, { onDelete: "cascade" }),
    name: text("name").notNull(),
    description: text("description"),
    matchType: text("match_type").notNull(), // 'folder', 'regex', 'manual'
    matchPattern: text("match_pattern"), // e.g., '/blog/' or '^/products/.*'
    color: text("color"), // Hex color for UI
    isAutoGenerated: boolean("is_auto_generated").default(false),
    createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    unique("uq_content_groups_site_name").on(table.siteId, table.name),
    index("idx_content_groups_site").on(table.siteId),
  ]
);

export const contentGroupPages = pgTable(
  "content_group_pages",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    groupId: uuid("group_id").notNull().references(() => contentGroups.id, { onDelete: "cascade" }),
    pageUrl: text("page_url").notNull(),
    manuallyAdded: boolean("manually_added").default(false),
    createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    unique("uq_content_group_pages").on(table.groupId, table.pageUrl),
    index("idx_content_group_pages_group").on(table.groupId),
    index("idx_content_group_pages_url").on(table.pageUrl),
  ]
);

// ===== Topic Clusters =====

export const topicClusters = pgTable(
  "topic_clusters",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    siteId: uuid("site_id").notNull().references(() => sites.id, { onDelete: "cascade" }),
    name: text("name").notNull(),
    hubPageUrl: text("hub_page_url").notNull(),
    hubTopic: text("hub_topic").notNull(),
    coverage: real("coverage").default(0), // 0-100 percentage
    totalClicks: integer("total_clicks").default(0),
    totalImpressions: integer("total_impressions").default(0),
    avgPosition: real("avg_position"),
    gaps: jsonb("gaps").$type<string[]>().default([]), // Missing subtopics
    lastAnalyzedAt: timestamp("last_analyzed_at", { withTimezone: true }),
    createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    unique("uq_topic_clusters_site_hub").on(table.siteId, table.hubPageUrl),
    index("idx_topic_clusters_site").on(table.siteId),
  ]
);

export const topicClusterPages = pgTable(
  "topic_cluster_pages",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    clusterId: uuid("cluster_id").notNull().references(() => topicClusters.id, { onDelete: "cascade" }),
    pageUrl: text("page_url").notNull(),
    pageTopic: text("page_topic"),
    isHub: boolean("is_hub").default(false),
    linksToHub: boolean("links_to_hub").default(false),
    internalLinkCount: integer("internal_link_count").default(0),
    clicks: integer("clicks").default(0),
    impressions: integer("impressions").default(0),
    position: real("position"),
    createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    unique("uq_topic_cluster_pages").on(table.clusterId, table.pageUrl),
    index("idx_topic_cluster_pages_cluster").on(table.clusterId),
  ]
);

// ===== Index Coverage =====

export const pageIndexStatus = pgTable(
  "page_index_status",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    siteId: uuid("site_id").notNull().references(() => sites.id, { onDelete: "cascade" }),
    pageUrl: text("page_url").notNull(),
    // Coverage states from GSC URL Inspection API
    coverageState: text("coverage_state"), // 'Submitted and indexed', 'Crawled - currently not indexed', 'Discovered - currently not indexed', 'URL is unknown to Google', 'Page with redirect', 'Not found (404)', 'Blocked by robots.txt', 'Blocked by noindex', etc.
    indexingState: text("indexing_state"), // 'Indexed', 'Not indexed'
    lastCrawlTime: timestamp("last_crawl_time", { withTimezone: true }),
    crawledAs: text("crawled_as"), // 'Googlebot Desktop', 'Googlebot Smartphone'
    robotsTxtState: text("robots_txt_state"), // 'ALLOWED', 'DISALLOWED', 'UNKNOWN'
    canonicalUrl: text("canonical_url"),
    isCanonical: boolean("is_canonical"),
    mobileUsability: text("mobile_usability"), // 'MOBILE_FRIENDLY', 'NOT_MOBILE_FRIENDLY', 'MOBILE_USABILITY_UNSPECIFIED'
    richResults: jsonb("rich_results").$type<{ detected: string[]; items?: unknown[] }>(),
    // Additional inspection metadata
    userDeclaredCanonical: text("user_declared_canonical"),
    googleSelectedCanonical: text("google_selected_canonical"),
    pageFetchState: text("page_fetch_state"), // 'SUCCESSFUL', 'SOFT_404', 'BLOCKED_ROBOTS_TXT', 'NOT_FOUND', 'ACCESS_DENIED', 'SERVER_ERROR', 'REDIRECT_ERROR', 'ACCESS_FORBIDDEN', 'BLOCKED_4XX', 'INTERNAL_CRAWL_ERROR', 'INVALID_URL'
    referringUrls: jsonb("referring_urls").$type<string[]>(),
    // Timestamps
    inspectionTime: timestamp("inspection_time", { withTimezone: true }).notNull(),
    createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    unique("uq_page_index_status").on(table.siteId, table.pageUrl),
    index("idx_page_index_coverage_state").on(table.siteId, table.coverageState),
    index("idx_page_index_indexing_state").on(table.siteId, table.indexingState),
    index("idx_page_index_crawl_time").on(table.siteId, table.lastCrawlTime),
  ]
);

export const indexingRequests = pgTable(
  "indexing_requests",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    siteId: uuid("site_id").notNull().references(() => sites.id, { onDelete: "cascade" }),
    pageUrl: text("page_url").notNull(),
    requestType: text("request_type").notNull(), // 'URL_UPDATED', 'URL_DELETED'
    status: text("status").notNull().default("pending"), // 'pending', 'submitted', 'success', 'failed', 'quota_exceeded'
    priority: integer("priority").default(0), // Higher = more urgent
    priorityReason: text("priority_reason"), // 'new_content', 'decaying', 'high_value', 'manual'
    submittedAt: timestamp("submitted_at", { withTimezone: true }),
    response: jsonb("response").$type<{ notificationType?: string; urlNotificationMetadata?: unknown; error?: string }>(),
    errorMessage: text("error_message"),
    retryCount: integer("retry_count").default(0),
    createdAt: timestamp("created_at", { withTimezone: true }).defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow(),
  },
  (table) => [
    index("idx_indexing_requests_site_status").on(table.siteId, table.status),
    index("idx_indexing_requests_priority").on(table.priority),
    index("idx_indexing_requests_submitted").on(table.submittedAt),
  ]
);

      GROUP BY folder

        .split(' ').map(w => w.charAt(0).toUpperCase() + w.slice(1)).join(' ');

      const group = await this.createGroup(siteId, {

        matchType: 'folder',
        matchPattern: folder,

      }

    await this.repo.clearGroupPages(groupId, true); // Keep manually added pages

    let patternSql: string;
    if (matchType === 'folder') {
      // Folder match: URL contains the folder path
      patternSql = `page_url LIKE '%${matchPattern}%'`;
    } else if (matchType === 'regex') {
      // Regex match: URL matches the pattern
      patternSql = `page_url ~ '${matchPattern}'`;
    } else {
      return;
    }

    const pages = await db.execute<{ page_url: string }>(sql`
      SELECT DISTINCT page_url
      FROM growing_pages_cagg
      WHERE site_id = ${siteId}
        AND page_url IS NOT NULL
        AND ${sql.raw(patternSql)}

  async getGroupWithMetrics(groupId: string, siteId: string): Promise<ContentGroupWithMetrics | null> {
    const group = await this.repo.findById(groupId);
    if (!group) return null;

    const pages = await this.repo.getGroupPages(groupId);
    if (pages.length === 0) {
      return {
        ...group,
        matchType: group.matchType as 'folder' | 'regex' | 'manual',
        pageCount: 0,
        metrics: { totalClicks: 0, totalImpressions: 0, avgPosition: 0, clicksChange: 0, impressionsChange: 0 },
      };
    }

    const pageUrls = pages.map(p => p.pageUrl);
    
    // Get metrics from continuous aggregate
    const metrics = await db.execute<{
      total_clicks: number;
      total_impressions: number;
      avg_position: number;
      prev_clicks: number;
      prev_impressions: number;
    }>(sql`
      WITH current_period AS (
        SELECT
          COALESCE(SUM(daily_clicks), 0) as total_clicks,
          COALESCE(SUM(daily_impressions), 0) as total_impressions,
          COALESCE(AVG(avg_position), 0) as avg_position
        FROM growing_pages_cagg
        WHERE site_id = ${siteId}
          AND page_url = ANY(${pageUrls})
          AND bucket >= NOW() - INTERVAL '21 days'
      ),
      previous_period AS (
        SELECT
          COALESCE(SUM(daily_clicks), 0) as total_clicks,
          COALESCE(SUM(daily_impressions), 0) as total_impressions
        FROM growing_pages_cagg
        WHERE site_id = ${siteId}
          AND page_url = ANY(${pageUrls})
          AND bucket >= NOW() - INTERVAL '42 days'
          AND bucket < NOW() - INTERVAL '21 days'
      )
      SELECT
        c.total_clicks,
        c.total_impressions,
        c.avg_position,
        p.total_clicks as prev_clicks,
        p.total_impressions as prev_impressions
      FROM current_period c, previous_period p
    `);

    const m = metrics.rows[0];
    const clicksChange = m.prev_clicks > 0 ? ((m.total_clicks - m.prev_clicks) / m.prev_clicks) * 100 : 0;
    const impressionsChange = m.prev_impressions > 0 ? ((m.total_impressions - m.prev_impressions) / m.prev_impressions) * 100 : 0;

    return {
      ...group,
      matchType: group.matchType as 'folder' | 'regex' | 'manual',
      pageCount: pages.length,
      metrics: {
        totalClicks: Number(m.total_clicks),
        totalImpressions: Number(m.total_impressions),
        avgPosition: Number(m.avg_position),
        clicksChange,
        impressionsChange,
      },
    };
  }

import { TopicClusterRepository } from "../repositories/TopicClusterRepository";
import { InternalLinkGraph, type LinkGraphPage } from "@/server/features/linking/services/InternalLinkGraph";
import { db } from "@/db";
import { sql } from "drizzle-orm";
import type { TopicClusterWithPages, ClusterDetectionOptions, ClusterCreateInput, ClusterGapAnalysis } from "../types";

export class TopicClusterService {
  constructor(
    private repo: TopicClusterRepository = new TopicClusterRepository(),
    private linkGraph: InternalLinkGraph = new InternalLinkGraph()
  ) {}

  async detectClusters(siteId: string, clientId: string, options: ClusterDetectionOptions = {}): Promise<TopicClusterWithPages[]> {
    const { minHubLinks = 10, minSpokeLinks = 1, maxSpokesPerHub = 50 } = options;

    // Step 1: Find potential hubs (pages with high incoming link count)
    const hubCandidates = await this.linkGraph.getPagesWithMostLinks(clientId, 50);
    const hubs = hubCandidates.filter(page => page.incomingLinks >= minHubLinks);

    const clusters: TopicClusterWithPages[] = [];

    for (const hub of hubs) {
      // Step 2: Find spoke pages that link TO this hub
      const incomingLinks = await this.linkGraph.getIncomingLinks(clientId, hub.url);
      const spokeUrls = incomingLinks
        .slice(0, maxSpokesPerHub)
        .map(link => link.sourceUrl);

      if (spokeUrls.length < 2) continue; // Need at least 2 spokes for a cluster

      // Step 3: Extract topic from hub URL/title
      const hubTopic = this.extractTopic(hub.url, hub.title);

      // Step 4: Create or update cluster
      const existingClusters = await this.repo.findBySiteId(siteId);
      let cluster = existingClusters.find(c => c.hubPageUrl === hub.url);

      if (!cluster) {
        cluster = await this.repo.create({
          siteId,
          name: `${hubTopic} Cluster`,
          hubPageUrl: hub.url,
          hubTopic,
          lastAnalyzedAt: new Date(),
        });
      }

      // Step 5: Clear and repopulate spoke pages
      await this.repo.clearClusterPages(cluster.id);
      
      // Add hub page
      await this.repo.addPageToCluster({
        clusterId: cluster.id,
        pageUrl: hub.url,
        pageTopic: hubTopic,
        isHub: true,
        linksToHub: false,
        internalLinkCount: hub.incomingLinks,
      });

      // Add spoke pages
      for (const link of incomingLinks.slice(0, maxSpokesPerHub)) {
        const spokeTopic = this.extractTopic(link.sourceUrl, link.anchorText);
        await this.repo.addPageToCluster({
          clusterId: cluster.id,
          pageUrl: link.sourceUrl,
          pageTopic: spokeTopic,
          isHub: false,
          linksToHub: true,
          internalLinkCount: 1, // Will be enhanced with full count
        });
      }

      // Step 6: Calculate coverage and gaps
      const clusterWithMetrics = await this.getClusterWithMetrics(cluster.id, siteId);
      if (clusterWithMetrics) {
        clusters.push(clusterWithMetrics);
      }
    }

    return clusters;
  }

  async getClusterWithMetrics(clusterId: string, siteId: string): Promise<TopicClusterWithPages | null> {
    const cluster = await this.repo.findById(clusterId);
    if (!cluster) return null;

    const pages = await this.repo.getClusterPages(clusterId);
    const hubPage = pages.find(p => p.isHub);
    const spokePages = pages.filter(p => !p.isHub);

    if (!hubPage) return null;

    // Get metrics from continuous aggregate
    const allUrls = pages.map(p => p.pageUrl);
    const metrics = await this.getPageMetrics(siteId, allUrls);

    const hubMetrics = metrics.get(hubPage.pageUrl) || { clicks: 0, impressions: 0, position: 0 };

    // Calculate coverage based on expected subtopics
    const coverage = await this.calculateCoverage(siteId, cluster.hubTopic, spokePages.length);
    const gaps = await this.identifyGaps(siteId, cluster.hubTopic, spokePages.map(p => p.pageTopic).filter(Boolean) as string[]);

    // Update cluster with metrics
    const totalClicks = Array.from(metrics.values()).reduce((sum, m) => sum + m.clicks, 0);
    const totalImpressions = Array.from(metrics.values()).reduce((sum, m) => sum + m.impressions, 0);
    const positions = Array.from(metrics.values()).map(m => m.position).filter(p => p > 0);
    const avgPosition = positions.length > 0 ? positions.reduce((a, b) => a + b, 0) / positions.length : 0;

    await this.repo.update(clusterId, {
      coverage,
      totalClicks,
      totalImpressions,
      avgPosition,
      gaps,
      lastAnalyzedAt: new Date(),

    if (!cluster) return null;

    return {
      clusterId: cluster.id,
      clusterName: cluster.name,
      coverage: cluster.coverage,
      gaps: cluster.gaps,

      })),

  private async getPageMetrics(siteId: string, urls: string[]): Promise<Map<string, { clicks: number; impressions: number; position: number }>> {
    if (urls.length === 0) return new Map();

    const result = await db.execute<{ page_url: string; clicks: number; impressions: number; position: number }>(sql`
      SELECT
        page_url,
        COALESCE(SUM(daily_clicks), 0) as clicks,
        COALESCE(SUM(daily_impressions), 0) as impressions,
        COALESCE(AVG(avg_position), 0) as position
      FROM growing_pages_cagg
      WHERE site_id = ${siteId}
        AND page_url = ANY(${urls})
        AND bucket >= NOW() - INTERVAL '21 days'
      GROUP BY page_url
    `);

    const map = new Map<string, { clicks: number; impressions: number; position: number }>();
    for (const row of result.rows) {
      map.set(row.page_url, {
        clicks: Number(row.clicks),
        impressions: Number(row.impressions),
        position: Number(row.position),
      });
    }
    return map;
  }

  private async calculateCoverage(siteId: string, hubTopic: string, spokeCount: number): Promise<number> {
    // Simple heuristic: estimate expected subtopics based on hub topic
    // In a real implementation, this would compare against keyword research data
    const expectedSubtopics = Math.max(5, Math.min(20, spokeCount + 3)); // Minimum 5, max 20
    return Math.min(100, Math.round((spokeCount / expectedSubtopics) * 100));
  }

  private async identifyGaps(siteId: string, hubTopic: string, existingTopics: string[]): Promise<string[]> {
    // Simple heuristic: suggest common subtopics not yet covered
    // In a real implementation, this would use keyword research data
    const commonSubtopics = this.getCommonSubtopics(hubTopic);
    const existingLower = new Set(existingTopics.map(t => t.toLowerCase()));
    return commonSubtopics.filter(st => !existingLower.has(st.toLowerCase()));

import { IndexStatusRepository } from "../repositories/IndexStatusRepository";
import { GscBridgeService } from "@/server/services/GscBridgeService";
import type { IndexCoverageStats, BatchIndexingRequest } from "../types";

const DAILY_INDEXING_LIMIT = 200;
const MAX_RETRIES = 3;

export class IndexCoverageService {
  constructor(
    private repo: IndexStatusRepository = new IndexStatusRepository(),
    private gscBridge: GscBridgeService = new GscBridgeService()
  ) {}

  async getCoverageStats(siteId: string): Promise<IndexCoverageStats> {
    const stats = await this.repo.getCoverageStats(siteId);

    const byState: Record<string, number> = {};
    let total = 0;
    let indexed = 0;
    let notIndexed = 0;

    for (const { state, count } of stats) {
      byState[state || 'Unknown'] = Number(count);
      total += Number(count);

      if (state?.includes('indexed') && !state.includes('not indexed')) {
        indexed += Number(count);
      } else {
        notIndexed += Number(count);
      }
    }

    return {
      total,
      indexed,
      notIndexed,
      byState,
      lastUpdated: new Date(),
    };
  }

  async requestIndexing(
    siteId: string,
    clientId: string,
    pageUrl: string,
    requestType: 'URL_UPDATED' | 'URL_DELETED' = 'URL_UPDATED',
    priority: number = 0,
    priorityReason: string = 'manual'
  ): Promise<{ success: boolean; message: string }> {
    // Check quota
    const usedToday = await this.repo.getTodayRequestCount(siteId);
    if (usedToday >= DAILY_INDEXING_LIMIT) {
      // Queue for later
      await this.repo.createIndexingRequest({
        siteId,
        pageUrl,
        requestType,
        status: 'pending',
        priority,
        priorityReason,
      });
      return { success: false, message: 'Daily quota exceeded. Request queued for tomorrow.' };
    }

    try {
      // Create request record
      const request = await this.repo.createIndexingRequest({
        siteId,
        pageUrl,
        requestType,
        status: 'submitted',
        priority,
        priorityReason,
        submittedAt: new Date(),
      });

      // Submit to GSC
      const result = await this.gscBridge.requestIndexing(clientId, pageUrl, requestType);

      // Update status
      await this.repo.updateRequestStatus(request.id, 'success', result);

      return { success: true, message: 'Indexing request submitted successfully.' };
    } catch (error) {
      return { success: false, message: `Indexing request failed: ${(error as Error).message}` };
    }
  }

  async requestBatchIndexing(
    siteId: string,
    clientId: string,
    urls: string[],
    requestType: 'URL_UPDATED' | 'URL_DELETED' = 'URL_UPDATED'
  ): Promise<{ submitted: number; queued: number; failed: number }> {
    const usedToday = await this.repo.getTodayRequestCount(siteId);
    const remaining = Math.max(0, DAILY_INDEXING_LIMIT - usedToday);

    let submitted = 0;
    let queued = 0;
    let failed = 0;

    for (let i = 0; i < urls.length; i++) {
      const url = urls[i];

      if (i < remaining) {
        // Submit immediately
        const result = await this.requestIndexing(siteId, clientId, url, requestType, urls.length - i, 'batch');
        if (result.success) {
          submitted++;
        } else if (result.message.includes('queued')) {
          queued++;
        } else {
          failed++;
        }
      } else {
        // Queue for later
        await this.repo.createIndexingRequest({
          siteId,
          pageUrl: url,
          requestType,
          status: 'pending',
          priority: urls.length - i,
          priorityReason: 'batch',
        });
        queued++;
      }

      // Rate limit: 1 request per 200ms for batch
      if (i < remaining) {
        await new Promise(resolve => setTimeout(resolve, 200));
      }
    }

    return { submitted, queued, failed };
  }

  async processPendingRequests(siteId: string, clientId: string): Promise<number> {
    const usedToday = await this.repo.getTodayRequestCount(siteId);
    const remaining = Math.max(0, DAILY_INDEXING_LIMIT - usedToday);

    if (remaining === 0) return 0;

    const pending = await this.repo.getPendingRequests(siteId, remaining);
    let processed = 0;

    for (const request of pending) {
      if (request.retryCount >= MAX_RETRIES) {
        await this.repo.updateRequestStatus(request.id, 'failed', undefined, 'Max retries exceeded');
        continue;
      }

      try {
        const result = await this.gscBridge.requestIndexing(clientId, request.pageUrl, request.requestType as 'URL_UPDATED' | 'URL_DELETED');
        await this.repo.updateRequestStatus(request.id, 'success', result);
        processed++;
      } catch (error) {
        await this.repo.incrementRetryCount(request.id);
        await this.repo.updateRequestStatus(request.id, 'pending', undefined, (error as Error).message);
      }

      // Rate limit
      await new Promise(resolve => setTimeout(resolve, 200));
    }

    return processed;
  }

  async getPageStatus(siteId: string, pageUrl: string) {
    return this.repo.findByUrl(siteId, pageUrl);
  }

  async getAllStatuses(siteId: string, limit = 1000) {
    return this.repo.findBySiteId(siteId, limit);
  }
}

    mutationFn: async (input: ContentGroupCreateInput) => {

interface TopicClusterVizProps {
  cluster: TopicClusterWithPages;
  width?: number;
  height?: number;
  onNodeClick?: (pageUrl: string) => void;
}

export function TopicClusterViz({ 
  cluster, 
  width = 600, 
  height = 400,
  onNodeClick 
}: TopicClusterVizProps) {
  const [hoveredNode, setHoveredNode] = useState<string | null>(null);

  // Calculate node positions using simple force-directed layout
  const { hubPosition, spokePositions } = useMemo(() => {
    const centerX = width / 2;
    const centerY = height / 2;
    const radius = Math.min(width, height) * 0.35;

    const spokeCount = cluster.spokePages.length;
    const spokePositions = cluster.spokePages.map((spoke, i) => {
      const angle = (2 * Math.PI * i) / spokeCount - Math.PI / 2;
      return {
        url: spoke.url,
        x: centerX + radius * Math.cos(angle),
        y: centerY + radius * Math.sin(angle),
        topic: spoke.topic,
        linksToHub: spoke.linksToHub,
        clicks: spoke.clicks,
      };
    });

    return {
      hubPosition: { x: centerX, y: centerY },
      spokePositions,
    };
  }, [cluster, width, height]);

  return (
    <svg 
      width={width} 
      height={height} 
      className="bg-surface-1 rounded-lg"
      viewBox={`0 0 ${width} ${height}`}
    >
      {/* Draw lines from spokes to hub */}
      {spokePositions.map((spoke) => (
        <line
          key={`line-${spoke.url}`}
          x1={spoke.x}
          y1={spoke.y}
          x2={hubPosition.x}
          y2={hubPosition.y}
          stroke={spoke.linksToHub ? "var(--success)" : "var(--text-4)"}
          strokeWidth={spoke.linksToHub ? 2 : 1}
          strokeDasharray={spoke.linksToHub ? undefined : "4,4"}
          opacity={hoveredNode && hoveredNode !== spoke.url ? 0.3 : 1}
        />
      ))}

      {/* Draw spoke nodes */}
      {spokePositions.map((spoke) => (
        <ClusterNode
          key={spoke.url}
          x={spoke.x}
          y={spoke.y}
          label={spoke.topic || spoke.url.split('/').pop() || 'Page'}
          clicks={spoke.clicks}
          isHub={false}
          isHighlighted={hoveredNode === spoke.url}
          onMouseEnter={() => setHoveredNode(spoke.url)}
          onMouseLeave={() => setHoveredNode(null)}
          onClick={() => onNodeClick?.(spoke.url)}
        />
      ))}

      {/* Draw hub node (on top) */}
      <ClusterNode
        x={hubPosition.x}
        y={hubPosition.y}
        label={cluster.hubPage.topic}
        clicks={cluster.hubPage.clicks}
        isHub={true}
        isHighlighted={hoveredNode === cluster.hubPage.url}

        onMouseLeave={() => setHoveredNode(null)}
        onClick={() => onNodeClick?.(cluster.hubPage.url)}

        x={20}
        y={height - 20}
        className="fill-text-3 text-sm"
      >
        Coverage: {cluster.coverage}%
      </text>
    </svg>
  );
