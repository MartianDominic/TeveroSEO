   ```typescript
   export class BriefRepository {
     async create(input: {
       mappingId: string;
       keyword: string;
       targetWordCount: number;
       voiceMode: VoiceMode;
       serpAnalysis: SerpAnalysisData;
     }): Promise<ContentBriefSelect> {
       const id = nanoid();
       const [brief] = await db.insert(contentBriefs).values({
         id,
         ...input,
         status: "draft",
       }).returning();
       return brief;
     }

     async findById(id: string): Promise<ContentBriefSelect | null> {
       const [brief] = await db.select().from(contentBriefs)
         .where(eq(contentBriefs.id, id));
       return brief ?? null;
     }

     async findByProjectId(projectId: string): Promise<ContentBriefSelect[]> {
       return db.select().from(contentBriefs)
         .innerJoin(keywordPageMapping, eq(contentBriefs.mappingId, keywordPageMapping.id))
         .where(eq(keywordPageMapping.projectId, projectId));
     }

     async findByMappingId(mappingId: string): Promise<ContentBriefSelect | null> {
       const [brief] = await db.select().from(contentBriefs)
         .where(eq(contentBriefs.mappingId, mappingId));
       return brief ?? null;
     }

     async updateStatus(id: string, status: BriefStatus): Promise<ContentBriefSelect | null> {
       const [updated] = await db.update(contentBriefs)
         .set({ status, updatedAt: new Date() })
         .where(eq(contentBriefs.id, id))
         .returning();
       return updated ?? null;
     }

     async updateArticleId(id: string, articleId: string): Promise<ContentBriefSelect | null> {
       const [updated] = await db.update(contentBriefs)
         .set({ articleId, updatedAt: new Date() })
         .where(eq(contentBriefs.id, id))
         .returning();
       return updated ?? null;
     }

     async delete(id: string): Promise<boolean> {
       const result = await db.delete(contentBriefs)
         .where(eq(contentBriefs.id, id));
       return result.rowCount > 0;
     }
   }
   ```

   ```typescript
   export interface BriefGeneratorInput {
     mappingId: string;
     voiceMode: VoiceMode;
     locationCode?: number;
   }

   export interface GeneratedBrief {
     brief: ContentBriefSelect;
     suggestedH2s: string[];
     paaQuestions: string[];
     competitorAvgWordCount: number;
   }

   export async function validateMapping(mappingId: string): Promise<KeywordPageMappingSelect> {
     const [mapping] = await db.select().from(keywordPageMapping)
       .where(eq(keywordPageMapping.id, mappingId));
     if (!mapping) {
       throw new AppError("NOT_FOUND", `Keyword mapping ${mappingId} not found`);
     }
     return mapping;
   }

   export async function previewSerp(
     mappingId: string,
     locationCode: number = 2840
   ): Promise<SerpAnalysisData> {
     const mapping = await validateMapping(mappingId);
     return analyzeSerpForKeyword(mappingId, mapping.keyword, locationCode);
   }

   export async function generateBrief(
     input: BriefGeneratorInput,
     repository: BriefRepository
   ): Promise<GeneratedBrief> {
     const mapping = await validateMapping(input.mappingId);
     
     // Get SERP analysis (uses cache if available)
     const serpAnalysis = await analyzeSerpForKeyword(
       input.mappingId,
       mapping.keyword,
       input.locationCode ?? 2840
     );
     
     // Calculate target word count: avg + 20% (to beat competitors)
     const avgWordCount = serpAnalysis.competitorWordCounts.length > 0
       ? Math.round(
           serpAnalysis.competitorWordCounts.reduce((a, b) => a + b, 0) /
           serpAnalysis.competitorWordCounts.length
         )
       : 1500; // Default if no data
     const targetWordCount = Math.round(avgWordCount * 1.2);
     
     // Create brief in database
     const brief = await repository.create({
       mappingId: input.mappingId,
       keyword: mapping.keyword,
       targetWordCount,
       voiceMode: input.voiceMode,
       serpAnalysis,
     });
     
     return {
       brief,
       suggestedH2s: serpAnalysis.commonH2s.map(h => h.heading),
       paaQuestions: serpAnalysis.paaQuestions,
       competitorAvgWordCount: avgWordCount,
     };
   }
   ```

   ```typescript
   import { Hono } from "hono";
   import { z } from "zod";
   import { zValidator } from "@hono/zod-validator";
   import { BriefRepository } from "../services/BriefRepository";
   import { generateBrief, previewSerp } from "../services/BriefGenerator";
   import { VOICE_MODES, BRIEF_STATUSES } from "@/db/brief-schema";

   const briefsRouter = new Hono();
   const repository = new BriefRepository();

   const createBriefSchema = z.object({
     mappingId: z.string().min(1),
     voiceMode: z.enum(VOICE_MODES),
     locationCode: z.number().int().positive().optional(),
   });

   const updateStatusSchema = z.object({
     status: z.enum(BRIEF_STATUSES),
   });

   // GET /api/seo/briefs?projectId=xxx
   briefsRouter.get("/", async (c) => {
     const projectId = c.req.query("projectId");
     if (!projectId) {
       return c.json({ error: "projectId required" }, 400);
     }
     const briefs = await repository.findByProjectId(projectId);
     return c.json({ data: briefs });
   });

   // GET /api/seo/briefs/:id
   briefsRouter.get("/:id", async (c) => {
     const brief = await repository.findById(c.req.param("id"));
     if (!brief) {
       return c.json({ error: "Brief not found" }, 404);
     }
     return c.json({ data: brief });
   });

   // POST /api/seo/briefs/analyze-serp/:mappingId
   briefsRouter.post("/analyze-serp/:mappingId", async (c) => {
     const mappingId = c.req.param("mappingId");
     const body = await c.req.json().catch(() => ({}));
     const locationCode = body.locationCode ?? 2840;
     
     const analysis = await previewSerp(mappingId, locationCode);
     return c.json({ data: analysis });
   });

   // POST /api/seo/briefs
   briefsRouter.post("/", zValidator("json", createBriefSchema), async (c) => {
     const input = c.req.valid("json");
     const result = await generateBrief(input, repository);
     return c.json({ data: result }, 201);
   });

   // PATCH /api/seo/briefs/:id/status
   briefsRouter.patch("/:id/status", zValidator("json", updateStatusSchema), async (c) => {
     const id = c.req.param("id");
     const { status } = c.req.valid("json");
     const updated = await repository.updateStatus(id, status);
     if (!updated) {
       return c.json({ error: "Brief not found" }, 404);
     }
     return c.json({ data: updated });
   });

   // DELETE /api/seo/briefs/:id
   briefsRouter.delete("/:id", async (c) => {
     const deleted = await repository.delete(c.req.param("id"));
     if (!deleted) {
       return c.json({ error: "Brief not found" }, 404);
     }
     return c.json({ success: true });
   });

   export { briefsRouter };
   ```
