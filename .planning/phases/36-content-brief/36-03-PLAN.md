"use server";

import { auth } from "@clerk/nextjs/server";
import { revalidatePath } from "next/cache";

const OPEN_SEO_API = process.env.OPEN_SEO_API_URL || "http://localhost:3001";

export interface Brief {
  id: string;
  mappingId: string;
  keyword: string;
  targetWordCount: number;
  voiceMode: "preservation" | "application" | "best_practices";
  status: "draft" | "ready" | "generating" | "published";
  serpAnalysis: SerpAnalysisData | null;
  articleId: string | null;
  createdAt: string;
  updatedAt: string;
}

  commonH2s: { heading: string; frequency: number }[];
  paaQuestions: string[];
  competitorWordCounts: number[];
  metaLengths: { title: number; description: number };
  analyzedAt: string;
  location: string;
}

export interface GeneratedBriefResult {
  brief: Brief;
  suggestedH2s: string[];
  paaQuestions: string[];
  competitorAvgWordCount: number;

export async function getBriefs(projectId: string): Promise<Brief[]> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(
    `${OPEN_SEO_API}/api/seo/briefs?projectId=${encodeURIComponent(projectId)}`,
    { headers: { "x-clerk-user-id": userId }, cache: "no-store" }
  );
  
  if (!response.ok) throw new Error("Failed to fetch briefs");
  const { data } = await response.json();
  return data;
}

export async function getBrief(briefId: string): Promise<Brief> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(
    `${OPEN_SEO_API}/api/seo/briefs/${briefId}`,
    { headers: { "x-clerk-user-id": userId }, cache: "no-store" }
  );
  
  if (!response.ok) throw new Error("Failed to fetch brief");
  const { data } = await response.json();
  return data;
}

export async function analyzeSerpForMapping(
  mappingId: string,
  locationCode: number = 2840
): Promise<SerpAnalysisData> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(
    `${OPEN_SEO_API}/api/seo/briefs/analyze-serp/${mappingId}`,
    {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "x-clerk-user-id": userId,
      },
      body: JSON.stringify({ locationCode }),
    }
  );
  
  if (!response.ok) throw new Error("Failed to analyze SERP");
  const { data } = await response.json();
  return data;
}

export async function createBrief(
  mappingId: string,
  voiceMode: "preservation" | "application" | "best_practices",
  clientId: string
): Promise<GeneratedBriefResult> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(`${OPEN_SEO_API}/api/seo/briefs`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "x-clerk-user-id": userId,
    },
    body: JSON.stringify({ mappingId, voiceMode }),
  });
  
  if (!response.ok) {
    const error = await response.json().catch(() => ({}));
    throw new Error(error.error || "Failed to create brief");
  }
  
  const { data } = await response.json();
  revalidatePath(`/clients/${clientId}/content-briefs`);
  return data;
}

export async function updateBriefStatus(
  briefId: string,
  status: "draft" | "ready" | "generating" | "published",
  clientId: string
): Promise<Brief> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(
    `${OPEN_SEO_API}/api/seo/briefs/${briefId}/status`,
    {
      method: "PATCH",
      headers: {
        "Content-Type": "application/json",
        "x-clerk-user-id": userId,
      },
      body: JSON.stringify({ status }),
    }
  );
  
  if (!response.ok) throw new Error("Failed to update brief status");
  const { data } = await response.json();
  revalidatePath(`/clients/${clientId}/content-briefs`);
  return data;
}

export async function deleteBrief(briefId: string, clientId: string): Promise<void> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(`${OPEN_SEO_API}/api/seo/briefs/${briefId}`, {
    method: "DELETE",
    headers: { "x-clerk-user-id": userId },
  });
  
  if (!response.ok) throw new Error("Failed to delete brief");
  revalidatePath(`/clients/${clientId}/content-briefs`);
}

```tsx
"use client";

import { useState, useTransition } from "react";
import { useRouter } from "next/navigation";
import { KeywordSelector } from "./KeywordSelector";
import { SerpPreview } from "./SerpPreview";
import { VoiceModeSelector } from "./VoiceModeSelector";
import { analyzeSerpForMapping, createBrief, type SerpAnalysisData } from "@/lib/server-actions/briefs";

interface KeywordMapping {
  id: string;
  keyword: string;
  targetUrl: string | null;
  action: "optimize" | "create";
  relevanceScore: number | null;
  searchVolume: number | null;
}

interface BriefWizardProps {
  clientId: string;
  projectId: string;
  mappings: KeywordMapping[];
}

type Step = 1 | 2 | 3;

export function BriefWizard({ clientId, projectId, mappings }: BriefWizardProps) {
  const router = useRouter();
  const [step, setStep] = useState<Step>(1);
  const [selectedMapping, setSelectedMapping] = useState<{ id: string; keyword: string } | null>(null);
  const [serpAnalysis, setSerpAnalysis] = useState<SerpAnalysisData | null>(null);
  const [isPending, startTransition] = useTransition();
  const [error, setError] = useState<string | null>(null);

  const handleKeywordSelect = async (mappingId: string, keyword: string) => {
    setSelectedMapping({ id: mappingId, keyword });
    setError(null);
    
    startTransition(async () => {
      try {
        const analysis = await analyzeSerpForMapping(mappingId);
        setSerpAnalysis(analysis);
        setStep(2);
      } catch (err) {
        setError(err instanceof Error ? err.message : "Failed to analyze SERP");
      }
    });
  };

  const handleSave = async (voiceMode: "preservation" | "application" | "best_practices") => {
    if (!selectedMapping) return;
    setError(null);
    
    startTransition(async () => {
      try {
        await createBrief(selectedMapping.id, voiceMode, clientId);
        router.push(`/clients/${clientId}/content-briefs`);
      } catch (err) {
        setError(err instanceof Error ? err.message : "Failed to create brief");
      }
    });
  };

  const targetWordCount = serpAnalysis
    ? Math.round(
        (serpAnalysis.competitorWordCounts.reduce((a, b) => a + b, 0) /
          Math.max(serpAnalysis.competitorWordCounts.length, 1)) *
          1.2
      )
    : 1500;

  return (
    <div className="max-w-3xl mx-auto">
      {error && (
        <div className="mb-4 p-4 bg-destructive/10 text-destructive rounded-lg">
          {error}
        </div>
      )}

      {step === 1 && (
        <KeywordSelector
          mappings={mappings}
          onSelect={handleKeywordSelect}
          isLoading={isPending}
        />
      )}

      {step === 2 && serpAnalysis && selectedMapping && (
        <SerpPreview
          keyword={selectedMapping.keyword}
          analysis={serpAnalysis}
          onContinue={() => setStep(3)}
          onBack={() => setStep(1)}
        />
      )}

      {step === 3 && selectedMapping && (
        <VoiceModeSelector
          keyword={selectedMapping.keyword}
          targetWordCount={targetWordCount}
          onSave={handleSave}
          onBack={() => setStep(2)}
          isSaving={isPending}
        />
      )}
    </div>
  );
}
```

import { Badge } from "@tevero/ui/badge";
import { PlusIcon } from "lucide-react";
import type { Brief } from "@/lib/server-actions/briefs";

interface BriefListProps {
  clientId: string;
  briefs: Brief[];
}

const STATUS_VARIANTS: Record<string, "default" | "secondary" | "destructive" | "outline"> = {
  draft: "secondary",
  ready: "outline",
  generating: "default",
  published: "default",
};

export function BriefList({ clientId, briefs }: BriefListProps) {
  return (
    <Card>
      <CardHeader className="flex flex-row items-center justify-between">
        <CardTitle>Content Briefs</CardTitle>
        <Link href={`/clients/${clientId}/content-briefs/create`}>
          <Button>
            <PlusIcon className="h-4 w-4 mr-2" />
            Create Brief
          </Button>
        </Link>
      </CardHeader>
      <CardContent>
        {briefs.length === 0 ? (
          <p className="text-muted-foreground text-center py-8">
            No content briefs yet. Create your first brief to get started.
          </p>
        ) : (
          <Table>
            <TableHeader>
              <TableRow>
                <TableHead>Keyword</TableHead>
                <TableHead>Target Words</TableHead>
                <TableHead>Voice Mode</TableHead>
                <TableHead>Status</TableHead>
                <TableHead>Created</TableHead>
                <TableHead></TableHead>
              </TableRow>
            </TableHeader>
            <TableBody>
              {briefs.map((brief) => (
                <TableRow key={brief.id}>
                  <TableCell className="font-medium">{brief.keyword}</TableCell>
                  <TableCell>{brief.targetWordCount.toLocaleString()}</TableCell>
                  <TableCell>
                    <Badge variant="outline">
                      {brief.voiceMode === "preservation"
                        ? "Match Existing"
                        : brief.voiceMode === "application"
                        ? "Brand Guidelines"
                        : "SEO Best Practices"}
                    </Badge>
                  </TableCell>
                  <TableCell>
                    <Badge variant={STATUS_VARIANTS[brief.status] ?? "secondary"}>
                      {brief.status}
                    </Badge>
                  </TableCell>
                  <TableCell className="text-muted-foreground">
                    {new Date(brief.createdAt).toLocaleDateString()}
                  </TableCell>
                  <TableCell>
                    {brief.status === "ready" && (
                      <Button size="sm" variant="outline">
                        Generate Content
                      </Button>
                    )}
                  </TableCell>
                </TableRow>
              ))}
            </TableBody>
          </Table>
        )}
      </CardContent>
    </Card>
  );
}

import { getBriefs } from "@/lib/server-actions/briefs";
import { BriefList } from "./components/BriefList";

interface PageProps {
  params: Promise<{ clientId: string }>;
  searchParams: Promise<{ projectId?: string }>;
}

export default async function ContentBriefsPage({ params, searchParams }: PageProps) {
  const { clientId } = await params;
  const { projectId } = await searchParams;
  
  if (!projectId) {
    return (
      <div className="p-6">
        <p className="text-muted-foreground">
          Select a project to view content briefs.
        </p>
      </div>
    );
  }

  const briefs = await getBriefs(projectId);

  return (
    <div className="p-6">
      <BriefList clientId={clientId} briefs={briefs} />
    </div>
  );
}

import { getKeywordMappings } from "@/lib/server-actions/mapping";
import { BriefWizard } from "../components/BriefWizard";

interface PageProps {
  params: Promise<{ clientId: string }>;
  searchParams: Promise<{ projectId?: string }>;
}

export default async function CreateBriefPage({ params, searchParams }: PageProps) {
  const { clientId } = await params;
  const { projectId } = await searchParams;
  
  if (!projectId) {
    return (
      <div className="p-6">
        <p className="text-muted-foreground">
          Select a project to create a content brief.
        </p>
      </div>
    );
  }

  // Fetch keyword mappings for the project
  const mappings = await getKeywordMappings(projectId);

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-6">Create Content Brief</h1>
      <BriefWizard clientId={clientId} projectId={projectId} mappings={mappings} />
    </div>
  );
}
