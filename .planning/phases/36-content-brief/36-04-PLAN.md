import { AppError } from "@/server/lib/errors";
import { createLogger } from "@/server/lib/logger";
import type { ContentBriefSelect, SerpAnalysisData } from "@/db/brief-schema";

const log = createLogger({ module: "AIWriterClient" });
const AI_WRITER_API = process.env.AI_WRITER_API_URL || "http://localhost:8000";

export interface ArticleCreatePayload {
  client_id: string;
  title: string;
  keyword: string;
  // Extended fields for brief integration
  brief_id?: string;
  target_word_count?: number;
  voice_mode?: string;
  suggested_h2s?: string[];
  paa_questions?: string[];
}

export interface ArticleResponse {
  id: string;
  client_id: string;
  title: string;
  keyword: string | null;
  status: string;
  meta_description: string | null;
  created_at: string;
  updated_at: string;
}

/**
 * Build an SEO-friendly title from keyword.
 * Capitalizes first letter of each word, appends year or context if helpful.
 */
export function buildArticleTitle(keyword: string): string {
  // Capitalize each word
  const title = keyword
    .split(" ")
    .map((word) => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
    .join(" ");
  
  // Append current year for evergreen topics
  const year = new Date().getFullYear();
  return `${title} - Complete Guide ${year}`;
}

/**
 * Create an article in AI-Writer from a content brief.
 */
export async function createArticleFromBrief(
  brief: ContentBriefSelect,
  clientId: string
): Promise<ArticleResponse> {
  const title = buildArticleTitle(brief.keyword);
  
  const payload: ArticleCreatePayload = {
    client_id: clientId,
    title,
    keyword: brief.keyword,
    brief_id: brief.id,
    target_word_count: brief.targetWordCount,
    voice_mode: brief.voiceMode,
    suggested_h2s: brief.serpAnalysis?.commonH2s.map((h) => h.heading) ?? [],
    paa_questions: brief.serpAnalysis?.paaQuestions ?? [],
  };

  log.info("Creating article from brief", { briefId: brief.id, keyword: brief.keyword });

  const response = await fetch(`${AI_WRITER_API}/api/articles`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
    },
    body: JSON.stringify(payload),
  });

  if (!response.ok) {
    const errorText = await response.text();
    log.error("AI-Writer article creation failed", {
      status: response.status,
      error: errorText,
    });
    throw new AppError(
      "INTERNAL_ERROR",
      `AI-Writer article creation failed: ${response.status}`
    );
  }

  const article: ArticleResponse = await response.json();
  log.info("Article created", { articleId: article.id, briefId: brief.id });
  return article;
}

/**
 * Get article status from AI-Writer.
 */
export async function getArticleStatus(articleId: string): Promise<string> {
  const response = await fetch(`${AI_WRITER_API}/api/articles/${articleId}`, {
    headers: {
      "Content-Type": "application/json",
    },
  });

  if (!response.ok) {
    throw new AppError("NOT_FOUND", `Article ${articleId} not found`);
  }

  const article: ArticleResponse = await response.json();
  return article.status;
}

/**
 * Trigger article generation in AI-Writer.
 * This changes article status from draft -> generating.
 */
export async function triggerArticleGeneration(articleId: string): Promise<void> {
  const response = await fetch(
    `${AI_WRITER_API}/api/articles/${articleId}`,
    {
      method: "PATCH",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ status: "generating" }),
    }
  );

  if (!response.ok) {
    const errorText = await response.text();
    throw new AppError(
      "INTERNAL_ERROR",
      `Failed to trigger article generation: ${errorText}`
    );
  }
}

// POST /api/seo/briefs/:id/generate
// Creates article in AI-Writer and triggers generation
briefsRouter.post("/:id/generate", async (c) => {
  const briefId = c.req.param("id");
  const body = await c.req.json().catch(() => ({}));
  const clientId = body.clientId;
  
  if (!clientId) {
    return c.json({ error: "clientId required" }, 400);
  }

  // Get the brief
  const brief = await repository.findById(briefId);
  if (!brief) {
    return c.json({ error: "Brief not found" }, 404);
  }

  // Validate brief is in ready state
  if (brief.status !== "ready" && brief.status !== "draft") {
    return c.json(
      { error: `Cannot generate from brief with status: ${brief.status}` },
      400
    );
  }

  try {
    // Create article in AI-Writer
    const article = await createArticleFromBrief(brief, clientId);
    
    // Update brief with article reference and status
    await repository.updateArticleId(briefId, article.id);
    await repository.updateStatus(briefId, "generating");
    
    // Trigger generation (async in AI-Writer)
    await triggerArticleGeneration(article.id);
    
    return c.json({
      data: {
        briefId: brief.id,
        articleId: article.id,
        status: "generating",
      },
    });
  } catch (err) {
    // Revert status if generation fails to start
    await repository.updateStatus(briefId, "ready");
    throw err;
  }
});

// GET /api/seo/briefs/:id/generation-status
// Get the current generation status (polls AI-Writer)
briefsRouter.get("/:id/generation-status", async (c) => {
  const briefId = c.req.param("id");
  
  const brief = await repository.findById(briefId);
  if (!brief) {
    return c.json({ error: "Brief not found" }, 404);
  }

  if (!brief.articleId) {
    return c.json({
      data: {
        briefStatus: brief.status,
        articleStatus: null,
        articleId: null,
      },
    });
  }

  try {
    const articleStatus = await getArticleStatus(brief.articleId);
    
    // Sync brief status with article status
    if (articleStatus === "generated" || articleStatus === "published") {
      await repository.updateStatus(briefId, "published");
    }
    
    return c.json({
      data: {
        briefStatus: brief.status,
        articleStatus,
        articleId: brief.articleId,
      },
    });
  } catch {
    return c.json({
      data: {
        briefStatus: brief.status,
        articleStatus: "unknown",
        articleId: brief.articleId,
      },
    });
  }
});

export async function generateContent(
  briefId: string,
  clientId: string
): Promise<{ briefId: string; articleId: string; status: string }> {
  const { userId } = await auth();
  if (!userId) throw new Error("Unauthorized");

  const response = await fetch(`${OPEN_SEO_API}/api/seo/briefs/${briefId}/generate`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "x-clerk-user-id": userId,
    },
    body: JSON.stringify({ clientId }),
  });
  
  if (!response.ok) {
    const error = await response.json().catch(() => ({}));
    throw new Error(error.error || "Failed to generate content");
  }
  
  const { data } = await response.json();
  revalidatePath(`/clients/${clientId}/content-briefs`);
  revalidatePath(`/clients/${clientId}/content-briefs/${briefId}`);
  return data;
}

  briefId: string
): Promise<{ briefStatus: string; articleStatus: string | null; articleId: string | null }> {
  const { userId } = await auth();

  const response = await fetch(
    `${OPEN_SEO_API}/api/seo/briefs/${briefId}/generation-status`,
    { headers: { "x-clerk-user-id": userId }, cache: "no-store" }
  );
  
  if (!response.ok) throw new Error("Failed to get generation status");

"use client";

import { useState, useTransition, useEffect } from "react";
import { Button } from "@tevero/ui/button";
import { Loader2Icon, SparklesIcon, CheckIcon, ExternalLinkIcon } from "lucide-react";
import { generateContent, getGenerationStatus } from "@/lib/server-actions/briefs";

interface GenerateButtonProps {
  briefId: string;
  clientId: string;
  initialStatus: string;
  articleId: string | null;
}

export function GenerateButton({
  briefId,
  clientId,
  initialStatus,
  articleId: initialArticleId,
}: GenerateButtonProps) {
  const [isPending, startTransition] = useTransition();
  const [status, setStatus] = useState(initialStatus);
  const [articleId, setArticleId] = useState(initialArticleId);
  const [error, setError] = useState<string | null>(null);

  // Poll for status updates when generating
  useEffect(() => {
    if (status !== "generating") return;

    const interval = setInterval(async () => {
      try {
        const result = await getGenerationStatus(briefId);
        if (result.articleStatus === "generated" || result.articleStatus === "published") {
          setStatus("published");
          setArticleId(result.articleId);
        } else if (result.articleStatus === "failed") {
          setStatus("failed");
          setError("Content generation failed");
        }
      } catch {
        // Ignore polling errors
      }
    }, 5000); // Poll every 5 seconds

    return () => clearInterval(interval);
  }, [status, briefId]);

  const handleGenerate = () => {
    setError(null);
    startTransition(async () => {
      try {
        const result = await generateContent(briefId, clientId);
        setStatus("generating");
        setArticleId(result.articleId);
      } catch (err) {
        setError(err instanceof Error ? err.message : "Generation failed");
      }
    });
  };

  if (status === "published" && articleId) {
    return (
      <a
        href={`/clients/${clientId}/articles/${articleId}`}
        className="inline-flex items-center"
      >
        <Button variant="outline">
          <CheckIcon className="h-4 w-4 mr-2" />
          View Article
          <ExternalLinkIcon className="h-4 w-4 ml-2" />
        </Button>
      </a>
    );
  }

  if (status === "generating") {
    return (
      <Button disabled>
        <Loader2Icon className="h-4 w-4 mr-2 animate-spin" />
        Generating...
      </Button>
    );
  }

  return (
    <div className="space-y-2">
      <Button onClick={handleGenerate} disabled={isPending}>
        {isPending ? (
          <Loader2Icon className="h-4 w-4 mr-2 animate-spin" />
        ) : (
          <SparklesIcon className="h-4 w-4 mr-2" />
        )}
        Generate Content
      </Button>
      {error && <p className="text-sm text-destructive">{error}</p>}
    </div>
  );
}
