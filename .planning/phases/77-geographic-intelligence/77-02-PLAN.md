```typescript
import { geoClassifier } from "../geo";
import { GeoConstraints, GeoClassification } from "../geo/types";

export interface ScoreWeights {
  volume: number;
  difficulty: number;
  cpc: number;
  quickWin: number;
  geo: number;        // NEW: default 0.15
}

export const DEFAULT_WEIGHTS: ScoreWeights = {
  volume: 0.30,
  difficulty: 0.25,
  cpc: 0.15,
  quickWin: 0.15,
  geo: 0.15,          // NEW: Phase 77
};

export interface PrioritizationResult {
  keyword: string;
  compositeScore: number;
  tier: KeywordTier;
  volume: number;
  difficulty: number;
  cpc: number;
  geoClassification?: GeoClassification;  // NEW
  // ... existing fields
}

export class PrioritizationService {
  prioritize(
    keywords: EnrichedKeyword[], 
    weights?: Partial<ScoreWeights>,
    geoConstraints?: Partial<GeoConstraints>
  ): PrioritizationResult {
    const w = { ...DEFAULT_WEIGHTS, ...weights };
    
    // Pre-classify all keywords for geo
    const geoResults = geoConstraints 
      ? geoClassifier.classifyBatch(keywords.map(k => k.keyword), geoConstraints)
      : null;
    
    const results = keywords.map((kw, i) => {
      const geo = geoResults?.[i];
      
      // If geo constraint provided and keyword fails, score = 0
      if (geo && !geo.passesGeoFilter) {
        return {
          ...this.buildBaseResult(kw),
          compositeScore: 0,
          tier: "excluded" as KeywordTier,
          geoClassification: geo,
        };
      }
      
      // Normal scoring with geo boost
      const baseScore = this.calculateBaseScore(kw, w);
      const geoBoost = geo ? geo.geoScore * w.geo : w.geo; // Default to full weight if no constraint
      const compositeScore = baseScore + geoBoost;
      
      return {
        ...this.buildBaseResult(kw),
        compositeScore,
        tier: this.calculateTier(compositeScore),
        geoClassification: geo,
      };
    });
    
    // Sort by compositeScore descending, excluded last
    return this.sortResults(results);
  }
}
```

Key implementation:
- Backward compatible: geoConstraints is optional
- Excluded keywords get compositeScore=0 and tier="excluded"
- geoScore contributes 15% to composite (configurable via weights.geo)
- geoClassification attached to each result for debugging

```typescript
import { describe, test, expect } from "vitest";
import { geoClassifier, GeoConstraints } from "./geo";
import { prioritizationService } from "./services/PrioritizationService";

describe("Geographic Intelligence Integration", () => {
  describe("End-to-end: Siauliai car wash", () => {
    const constraints: Partial<GeoConstraints> = {
      includeCities: ["siauliai"],
      nearMeAllowed: true,
      genericAllowed: true,
    };
    
    const keywords = [
      { keyword: "automobilu plovykla siauliuose", volume: 1000, difficulty: 30, cpc: 0.5 },
      { keyword: "plovykla kaune", volume: 2000, difficulty: 25, cpc: 0.6 },
      { keyword: "plovykla salia manes", volume: 500, difficulty: 20, cpc: 0.3 },
      { keyword: "automobilu plovykla", volume: 3000, difficulty: 40, cpc: 0.8 },
    ];
    
    test("correct keywords pass filter", () => {
      const results = prioritizationService.prioritize(keywords, {}, constraints);
      
      // Siauliai keyword passes with score 1.0
      const siauliai = results.find(r => r.keyword.includes("siauliuose"));
      expect(siauliai?.geoClassification?.passesGeoFilter).toBe(true);
      expect(siauliai?.geoClassification?.geoScore).toBe(1.0);
      expect(siauliai?.tier).not.toBe("excluded");
      
      // Kaunas keyword fails
      const kaunas = results.find(r => r.keyword.includes("kaune"));
      expect(kaunas?.geoClassification?.passesGeoFilter).toBe(false);
      expect(kaunas?.compositeScore).toBe(0);
      expect(kaunas?.tier).toBe("excluded");
      
      // Near-me passes with score 0.9
      const nearMe = results.find(r => r.keyword.includes("salia manes"));
      expect(nearMe?.geoClassification?.passesGeoFilter).toBe(true);
      expect(nearMe?.geoClassification?.geoScore).toBe(0.9);
      
      // Generic passes with score 0.5
      const generic = results.find(r => r.keyword === "automobilu plovykla");
      expect(generic?.geoClassification?.passesGeoFilter).toBe(true);
      expect(generic?.geoClassification?.geoScore).toBe(0.5);
    });
    
    test("excluded keywords sorted last", () => {
      const results = prioritizationService.prioritize(keywords, {}, constraints);
      const excludedIndex = results.findIndex(r => r.tier === "excluded");
      const nonExcludedAfter = results.slice(excludedIndex + 1).some(r => r.tier !== "excluded");
      expect(nonExcludedAfter).toBe(false);
    });
  });
  
  describe("End-to-end: Multi-city targeting", () => {
    const constraints: Partial<GeoConstraints> = {
      includeCities: ["vilnius", "kaunas"],
      nearMeAllowed: false,
      genericAllowed: false,
    };
    
    test("only target city keywords pass", () => {
      const keywords = [
        { keyword: "plovykla vilniuje", volume: 1000, difficulty: 30, cpc: 0.5 },
        { keyword: "plovykla kaune", volume: 800, difficulty: 25, cpc: 0.4 },
        { keyword: "plovykla siauliuose", volume: 600, difficulty: 20, cpc: 0.3 },
        { keyword: "plovykla salia manes", volume: 500, difficulty: 15, cpc: 0.2 },
        { keyword: "plovykla", volume: 2000, difficulty: 40, cpc: 0.8 },
      ];
      
      const results = prioritizationService.prioritize(keywords, {}, constraints);
      const passing = results.filter(r => r.tier !== "excluded");
      
      expect(passing.length).toBe(2); // vilnius + kaunas only
      expect(passing.every(r => 
        r.keyword.includes("vilniuje") || r.keyword.includes("kaune")
      )).toBe(true);
    });
  });
  
  describe("Backward compatibility", () => {
    test("prioritize without geoConstraints works as before", () => {
      const keywords = [
        { keyword: "plovykla vilniuje", volume: 1000, difficulty: 30, cpc: 0.5 },
        { keyword: "plovykla kaune", volume: 800, difficulty: 25, cpc: 0.4 },
      ];
      
      const results = prioritizationService.prioritize(keywords); // No geo constraints
      
      expect(results.every(r => r.geoClassification === undefined)).toBe(true);
      expect(results.every(r => r.compositeScore > 0)).toBe(true);
      expect(results.every(r => r.tier !== "excluded")).toBe(true);
    });
  });
});
```
