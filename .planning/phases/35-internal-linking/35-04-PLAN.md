  maxNewLinksPerPage: 3;      // Per day
  maxTotalLinksPerPage: 10;   // Ever

  maxNewLinksPerDay: 50;      // Per site
  maxNewLinksPerWeek: 200;

  maxPagesEditedPerDay: 20;

interface VelocityCheckResult {
  allowed: boolean;
  reason?: string;
}

export class VelocityService {
  constructor(private db: Database) {}

  async checkLinkVelocity(
    clientId: string,
    sourceUrl: string
  ): Promise<VelocityCheckResult> {
    const settings = this.getDefaultSettings();
    
    // Check page-specific limits
    const pageLinksToday = await this.countPageLinksToday(clientId, sourceUrl);
    if (pageLinksToday >= settings.maxNewLinksPerPage) {
      return {
        allowed: false,
        reason: `Page has reached daily limit of ${settings.maxNewLinksPerPage} new links`,
      };
    }
    
    // Check total links on page
    const totalLinksOnPage = await this.countTotalLinksOnPage(clientId, sourceUrl);
    if (totalLinksOnPage >= settings.maxTotalLinksPerPage) {
      return {
        allowed: false,
        reason: `Page has reached maximum of ${settings.maxTotalLinksPerPage} total links`,
      };
    }
    
    // Check site-wide limits
    const siteLinksToday = await this.countSiteLinksToday(clientId);
    if (siteLinksToday >= settings.maxNewLinksPerDay) {
      return {
        allowed: false,
        reason: `Site has reached daily limit of ${settings.maxNewLinksPerDay} new links`,
      };
    }
    
    // Check pages edited today
    const pagesEditedToday = await this.countPagesEditedToday(clientId);
    if (pagesEditedToday >= settings.maxPagesEditedPerDay) {
      return {
        allowed: false,
        reason: `Site has reached daily limit of ${settings.maxPagesEditedPerDay} pages edited`,
      };
    }
    
    return { allowed: true };
  }

export class LinkSuggestionService {
  constructor(
    private db: Database,
    private anchorSelector: typeof selectAnchorText
  ) {}

  async generateSuggestion(
    opportunity: LinkOpportunity,
    sourceContent: string,
    targetKeyword: string | null,
    targetTitle: string,
    brandName: string,
    currentDistribution: AnchorDistribution
  ): Promise<LinkSuggestionInsert> {
    // Select anchor text
    const anchor = this.anchorSelector({
      targetUrl: opportunity.targetUrl,
      targetPageId: opportunity.targetPageId,
      targetKeyword,
      targetTitle,
      brandName,
      currentAnchorDistribution: currentDistribution,
      sourceContent,
    });

    // Determine insertion method
    const insertionMethod = anchor.sourceText ? 'wrap_existing' : 'append_sentence';
    
    // Generate replacement text or new sentence
    const replacementText = anchor.sourceText
      ? `<a href="${opportunity.targetUrl}">${anchor.sourceText}</a>`
      : null;
    
    const newSentence = !anchor.sourceText
      ? this.generateLinkSentence(anchor.text, opportunity.targetUrl)
      : null;

    // Check if auto-applicable
    const isAutoApplicable = await this.isAutoApplicable({
      insertionMethod,
      confidence: anchor.confidence,
      sourceUrl: opportunity.sourceUrl,
      targetUrl: opportunity.targetUrl,
      clientId: opportunity.clientId,
    });

    return {
      id: crypto.randomUUID(),
      clientId: opportunity.clientId,
      opportunityId: opportunity.id,
      sourcePageId: opportunity.sourcePageId,
      sourceUrl: opportunity.sourceUrl,
      targetPageId: opportunity.targetPageId,
      targetUrl: opportunity.targetUrl,
      targetKeyword,
      anchorText: anchor.text,
      anchorType: anchor.type,
      insertionMethod,
      originalText: anchor.sourceText,
      replacementText,
      appendAfterParagraph: insertionMethod === 'append_sentence' ? 2 : null,
      newSentence,
      paragraphIndex: null, // Could be determined from opportunity
      surroundingContext: this.extractContext(sourceContent, anchor.sourceText),
      confidence: anchor.confidence,
      isAutoApplicable,
      status: 'pending',
    };
  }

  async isAutoApplicable(params: {
    insertionMethod: string;
    confidence: number;
    sourceUrl: string;
    targetUrl: string;
    clientId: string;
  }): Promise<boolean> {
    // Must have wrap_existing method (safest)
    if (params.insertionMethod !== 'wrap_existing') return false;
    
    // Must have high confidence
    if (params.confidence < 0.85) return false;
    
    // Check page link count
    const pageLinks = await this.getPageLinkCount(params.sourceUrl, params.clientId);
    if (pageLinks >= 10) return false;
    
    // Check if target in cannibalization set
    const isCannibalized = await this.isTargetCannibalized(params.targetUrl, params.clientId);
    if (isCannibalized) return false;
    
    return true;
  }

  private generateLinkSentence(anchorText: string, targetUrl: string): string {
    // Generate a natural sentence with the link
    return `Learn more about <a href="${targetUrl}">${anchorText}</a>.`;
  }
}

export class LinkApplyService {
  constructor(
    private db: Database,
    private velocityService: VelocityService,
    private connectionService: ConnectionService
  ) {}

  async applySuggestion(
    suggestion: LinkSuggestion,
    connectionId: string
  ): Promise<ApplyResult> {
    // 1. Check velocity limits
    const velocityCheck = await this.velocityService.checkLinkVelocity(
      suggestion.clientId,
      suggestion.sourceUrl
    );
    
    if (!velocityCheck.allowed) {
      return {
        success: false,
        error: velocityCheck.reason,
      };
    }

    // 2. Get current page content
    const currentContent = await this.connectionService.getPageContent(
      connectionId,
      suggestion.sourcePageId
    );

    // 3. Apply the link
    let newContent: string;
    
    if (suggestion.insertionMethod === 'wrap_existing') {
      // Find and replace text with linked version
      if (!currentContent.includes(suggestion.originalText!)) {
        return await this.handleContentChanged(suggestion);
      }
      newContent = currentContent.replace(
        suggestion.originalText!,
        suggestion.replacementText!
      );
    } else if (suggestion.insertionMethod === 'append_sentence') {
      // Insert new sentence after specified paragraph
      newContent = this.insertAfterParagraph(
        currentContent,
        suggestion.appendAfterParagraph!,
        suggestion.newSentence!
      );
    } else {
      return { success: false, error: 'Unknown insertion method' };
    }

    // 4. Create site_changes record
    const change = await this.db.insert(siteChanges).values({
      id: crypto.randomUUID(),
      clientId: suggestion.clientId,
      connectionId,
      changeType: 'internal_link',
      category: 'links',
      resourceType: 'post',
      resourceId: suggestion.sourcePageId,
      resourceUrl: suggestion.sourceUrl,
      field: 'content',
      beforeValue: currentContent,
      afterValue: newContent,
      triggeredBy: suggestion.isAutoApplicable ? 'audit' : 'manual',
      status: 'pending',
    }).returning();

    // 5. Apply via platform adapter
    const result = await this.connectionService.updatePageContent(
      connectionId,
      suggestion.sourcePageId,
      newContent
    );

    // 6. Update suggestion and change status
    if (result.success) {
      await this.db.update(linkSuggestions)
        .set({
          status: 'applied',
          appliedAt: new Date(),
          appliedChangeId: change[0].id,
        })
        .where(eq(linkSuggestions.id, suggestion.id));

      await this.db.update(siteChanges)
        .set({
          status: 'applied',
          appliedAt: new Date(),
        })
        .where(eq(siteChanges.id, change[0].id));

      // Update link graph
      await this.updateLinkGraph(suggestion);

      return { success: true, changeId: change[0].id };
    } else {
      await this.db.update(linkSuggestions)
        .set({
          status: 'failed',
          failureReason: result.error,
        })
        .where(eq(linkSuggestions.id, suggestion.id));

      return { success: false, error: result.error };
    }
  }

  private async handleContentChanged(suggestion: LinkSuggestion): Promise<ApplyResult> {
    await this.db.update(linkSuggestions)
      .set({
        status: 'failed',
        failureReason: 'Content changed since analysis - original text not found',
      })
      .where(eq(linkSuggestions.id, suggestion.id));

    return {
      success: false,
      error: 'Content changed since analysis',
    };
