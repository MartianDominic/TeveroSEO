export const linkOpportunities = pgTable("link_opportunities", {
  id: text("id").primaryKey(),
  clientId: text("client_id").notNull().references(() => clients.id, { onDelete: "cascade" }),
  auditId: text("audit_id").notNull(),
  
  // The opportunity
  sourcePageId: text("source_page_id").notNull(),
  sourceUrl: text("source_url").notNull(),
  targetPageId: text("target_page_id").notNull(),
  targetUrl: text("target_url").notNull(),
  
  // Why this is an opportunity
  opportunityType: text("opportunity_type").notNull(), // orphan_rescue, depth_reduction, topical_cluster, keyword_match, anchor_diversity, link_velocity, first_paragraph
  
  // Scoring
  relevanceScore: real("relevance_score").notNull(), // 0-1
  urgencyScore: real("urgency_score").notNull(),     // 0-1
  impactScore: real("impact_score").notNull(),       // 0-1
  combinedScore: real("combined_score").notNull(),   // Weighted combination

  detectionMethod: text("detection_method").notNull(), // keyword_scan, topic_model, link_count, depth_analysis
  matchedKeywords: jsonb("matched_keywords").$type<string[]>(),
  matchedTopics: jsonb("matched_topics").$type<string[]>(),

  suggestedParagraph: integer("suggested_paragraph"),

  suggestedAnchor: text("suggested_anchor"),
  
  // Status
  status: text("status").notNull().default("detected"), // detected, approved, rejected, applied, auto_applied
  
  // If applied
  appliedAt: timestamp("applied_at", { withTimezone: true, mode: "date" }),
  appliedChangeId: text("applied_change_id"),
  
  detectedAt: timestamp("detected_at", { withTimezone: true, mode: "date" }).notNull().defaultNow(),
  reviewedAt: timestamp("reviewed_at", { withTimezone: true, mode: "date" }),

  index("ix_link_opps_combined_score").on(table.combinedScore),
  index("ix_link_opps_status").on(table.status),
  index("ix_link_opps_type").on(table.opportunityType),

  <action>
Create click-depth.ts with BFS algorithm per design doc:

1. computeClickDepth(auditId, clientId)
   - Fetch all links from link_graph for this audit
   - Build adjacency list: Map<sourceUrl, Set<targetUrl>>
   - Find homepage URL (path "/" or empty)
   - BFS from homepage:
     - Queue: [url, depth][]
     - Visited: Map<url, depth>
     - Process until queue empty or 10k iterations
     - Cap depth at 10 (DoS protection)
   - Update page_links.clickDepthFromHome for each URL
   - Find pages with depth > 3
   - Call createDepthReductionOpportunity for each

2. createDepthReductionOpportunity(auditId, clientId, targetUrl, currentDepth)
   - Find candidate source pages at depth 1 or 2
   - Filter out pages already linking to target
   - Score relevance using keyword overlap (computeTopicalRelevance)
   - Insert into link_opportunities with:
     - opportunityType: 'depth_reduction'
     - urgencyScore: min(1, (currentDepth - 3) / 3)
     - relevanceScore: from keyword overlap
     - impactScore: 0.5 (medium)
     - combinedScore: weighted combination

3. computeTopicalRelevance(sourceUrl, targetUrl, auditId) - helper
   - Get page titles/keywords from audit_pages
   - Calculate keyword overlap using extractKeywordsFromContent
   - Return 0-1 relevance score

4. findHomepageUrl(auditId) - helper
   - Query audit_pages for path "/" or empty path
   - Return URL or null if not found
  </action>

Create opportunity-detector.ts with detection methods per design doc:

1. detectLinkOpportunities(auditId, clientId) - main entry
   - Call all detection methods in sequence
   - All methods are token-free (rule-based)
   ```typescript
   await detectOrphanRescueOpportunities(auditId, clientId);
   await detectLinkVelocityOpportunities(auditId, clientId);
   await detectAnchorDiversityOpportunities(auditId, clientId);
   await detectKeywordMentionOpportunities(auditId, clientId);
   await detectFirstParagraphOpportunities(auditId, clientId);
   ```

2. detectOrphanRescueOpportunities(auditId, clientId)
   - Get orphan_pages with status='detected'
   - For each orphan, find relevant source pages
   - Score: urgencyScore=1.0 (highest), relevance from content
   - Insert with opportunityType='orphan_rescue'

3. detectLinkVelocityOpportunities(auditId, clientId)
   - Get page_links where inboundTotal < 40
   - Calculate linksNeeded = 40 - inboundTotal
   - urgencyScore = min(1, linksNeeded / 40)
   - Find candidate sources using findRelevantSourcePages
   - Insert with opportunityType='link_velocity'

4. detectAnchorDiversityOpportunities(auditId, clientId)
   - Get page_links where inboundExactMatch = 0
   - These pages need an exact-match anchor
   - Find sources that mention the page's target keyword
   - urgencyScore = 0.8 (high priority)
   - Insert with opportunityType='anchor_diversity'

5. detectKeywordMentionOpportunities(auditId, clientId)
   - Get saved_keywords with their target URLs
   - Build keyword -> targetPage map
   - For each page, check if content mentions keywords
   - If keyword mentioned but not linked, create opportunity
   - relevanceScore = 0.9 (exact keyword match)
   - Insert with opportunityType='keyword_match'

6. detectFirstParagraphOpportunities(auditId, clientId)
   - Get high-value pages (high traffic or target keywords)
   - Check if they have first-paragraph inbound links
   - If not, create opportunity to add one
   - urgencyScore = 0.7
   - Insert with opportunityType='first_paragraph'

7. findRelevantSourcePages(auditId, targetUrl, targetPageId, limit)
   - Query pages that don't already link to target
   - Score by keyword overlap with target
   - Return top N candidates with relevance scores

Export all from index.ts.
