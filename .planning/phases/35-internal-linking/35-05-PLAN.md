```typescript
  id: text("id").primaryKey(),

    url: string;
    title: string;

    gscClicks: number | null;
    inboundLinks: number;
    hasExactMatchAnchor: boolean;
  }>>(),
```

Create CannibalizationService.ts per design doc:

```typescript
export class CannibalizationService {
  constructor(private db: Database) {}

  async detectKeywordCannibalization(clientId: string): Promise<void> {
    // Get keyword -> page mappings from GSC data
    const keywordPages = await this.db.select({
      keyword: gscKeywords.keyword,
      pageUrl: gscKeywords.url,
      position: gscKeywords.avgPosition,
      clicks: gscKeywords.clicks,
    })
    .from(gscKeywords)
    .where(eq(gscKeywords.clientId, clientId));

    // Group by keyword
    const keywordGroups = new Map<string, typeof keywordPages>();
    for (const row of keywordPages) {
      const key = row.keyword.toLowerCase();
      if (!keywordGroups.has(key)) {
        keywordGroups.set(key, []);
      }
      keywordGroups.get(key)!.push(row);
    }

    // Check for cannibalization
    for (const [keyword, pages] of keywordGroups) {
      if (pages.length < 2) continue;

      // Filter to pages actually ranking
      const competing = pages
        .filter(p => p.position <= 100)
        .sort((a, b) => a.position - b.position);

      if (competing.length < 2) continue;

      // Determine severity
      const positionDiff = competing[1].position - competing[0].position;
      let severity: 'critical' | 'high' | 'medium' | 'low';
      
      if (positionDiff < 5) severity = 'critical';
      else if (positionDiff < 10) severity = 'high';
      else if (positionDiff < 20) severity = 'medium';
      else severity = 'low';

      // Recommend primary (most clicks)
      const recommended = [...competing].sort((a, b) => {
        if (b.clicks !== a.clicks) return b.clicks - a.clicks;
        return a.position - b.position;
      })[0];

      // Get link counts for competing pages
      const pageLinkData = await this.getPageLinkCounts(
        clientId,
        competing.map(p => p.pageUrl)
      );

      // Insert/update cannibalization record
      await this.db.insert(keywordCannibalization).values({
        id: crypto.randomUUID(),
        clientId,
        keyword,
        keywordLower: keyword.toLowerCase(),
        competingPages: competing.map(p => ({
          pageId: '', // lookup
          url: p.pageUrl,
          title: '', // lookup
          gscPosition: p.position,
          gscClicks: p.clicks,
          inboundLinks: pageLinkData.get(p.pageUrl)?.inboundTotal || 0,
          hasExactMatchAnchor: (pageLinkData.get(p.pageUrl)?.inboundExactMatch || 0) > 0,
        })),
        severity,
        recommendedPrimary: recommended.pageUrl,
        reasoning: `${competing.length} pages competing. Position gap: ${positionDiff}. ${recommended.pageUrl} has most clicks.`,
      }).onConflictDoUpdate({
        target: [keywordCannibalization.clientId, keywordCannibalization.keywordLower],
        set: { severity, competingPages: /* updated */ },
      });
    }
  }
}
```

```typescript
import { Hono } from 'hono';
import { authMiddleware } from '../middleware/auth';

export const linkRoutes = new Hono();

// Link health overview
linkRoutes.get('/health/:clientId', authMiddleware, async (c) => {
  const { clientId } = c.req.param();
  
  // Get page_links aggregates
  const metrics = await db.select({
    totalPages: count(),
    orphanPages: count(/* where inboundTotal = 0 */),
    avgInboundLinks: avg(pageLinks.inboundTotal),
    deepPages: count(/* where clickDepthFromHome > 3 */),
  })
  .from(pageLinks)
  .where(eq(pageLinks.clientId, clientId));

  // Get link distribution
  const distribution = await db.select({
    bucket: sql`CASE WHEN inbound_total = 0 THEN '0' WHEN inbound_total <= 10 THEN '1-10' ... END`,
    count: count(),
  })
  .from(pageLinks)
  .groupBy(sql`bucket`);

  // Get opportunity counts
  const opportunities = await db.select({
    total: count(),
    byType: sql`jsonb_object_agg(opportunity_type, count)`,
  })
  .from(linkOpportunities)
  .where(and(
    eq(linkOpportunities.clientId, clientId),
    eq(linkOpportunities.status, 'detected')
  ));

  return c.json({
    success: true,
    data: { metrics, distribution, opportunities },
  });
});

// List opportunities with filters
linkRoutes.get('/opportunities/:clientId', authMiddleware, async (c) => {
  const { clientId } = c.req.param();
  const { type, status, page = '1', limit = '20' } = c.req.query();
  
  const offset = (parseInt(page) - 1) * parseInt(limit);
  
  const opportunities = await db.select()
    .from(linkOpportunities)
    .where(and(
      eq(linkOpportunities.clientId, clientId),
      type ? eq(linkOpportunities.opportunityType, type) : undefined,
      status ? eq(linkOpportunities.status, status) : eq(linkOpportunities.status, 'detected'),
    ))
    .orderBy(desc(linkOpportunities.combinedScore))
    .limit(parseInt(limit))
    .offset(offset);

  const total = await db.select({ count: count() })
    .from(linkOpportunities)
    .where(/* same conditions */);

  return c.json({
    success: true,
    data: opportunities,
    meta: { total: total[0].count, page: parseInt(page), limit: parseInt(limit) },
  });
});

// Approve opportunity
linkRoutes.post('/opportunities/:id/approve', authMiddleware, async (c) => {
  const { id } = c.req.param();
  
  await db.update(linkOpportunities)
    .set({
      status: 'approved',
      reviewedAt: new Date(),
      reviewedBy: c.get('userId'),
    })
    .where(eq(linkOpportunities.id, id));

  return c.json({ success: true });
});

// Reject opportunity
linkRoutes.post('/opportunities/:id/reject', authMiddleware, async (c) => {
  const { id } = c.req.param();
  
  await db.update(linkOpportunities)
    .set({
      status: 'rejected',
      reviewedAt: new Date(),
      reviewedBy: c.get('userId'),
    })
    .where(eq(linkOpportunities.id, id));

  return c.json({ success: true });
});

// Apply suggestion
linkRoutes.post('/suggestions/:id/apply', authMiddleware, async (c) => {
  const { id } = c.req.param();
  const suggestion = await db.select().from(linkSuggestions).where(eq(linkSuggestions.id, id)).limit(1);
  
  if (!suggestion[0]) {
    return c.json({ success: false, error: 'Suggestion not found' }, 404);
  }

  const result = await linkApplyService.applySuggestion(suggestion[0], connectionId);
  return c.json(result);
});

// Cannibalization issues
linkRoutes.get('/cannibalization/:clientId', authMiddleware, async (c) => {
  const { clientId } = c.req.param();
  
  const issues = await db.select()
    .from(keywordCannibalization)
    .where(and(
      eq(keywordCannibalization.clientId, clientId),
      inArray(keywordCannibalization.status, ['detected', 'monitoring'])
    ))
    .orderBy(
      sql`CASE severity WHEN 'critical' THEN 1 WHEN 'high' THEN 2 WHEN 'medium' THEN 3 ELSE 4 END`
    );

  return c.json({ success: true, data: issues });
});

// Batch apply safe suggestions
linkRoutes.post('/batch/apply-safe', authMiddleware, async (c) => {
  const { clientId } = await c.req.json();
  
  // Get all auto-applicable suggestions
  const suggestions = await db.select()
    .from(linkSuggestions)
    .where(and(
      eq(linkSuggestions.clientId, clientId),
      eq(linkSuggestions.isAutoApplicable, true),
      eq(linkSuggestions.status, 'pending')
    ))
    .limit(50); // Respect velocity

  const results = [];
  for (const suggestion of suggestions) {
    const result = await linkApplyService.applySuggestion(suggestion, connectionId);
    results.push({ id: suggestion.id, ...result });
    if (!result.success && result.error?.includes('limit')) break; // Hit velocity limit
  }

  return c.json({ success: true, data: { applied: results.filter(r => r.success).length, results } });
});
```
