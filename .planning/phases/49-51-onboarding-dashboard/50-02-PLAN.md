```typescript
import { eq, and, isNull, lte, gte } from "drizzle-orm";
import { db } from "@/db";
import { tasks, type TaskInsert, type TaskSelect, type TaskPriority } from "@/db/tasks-schema";
import { nanoid } from "nanoid";

export async function createTask(
  data: Omit<TaskInsert, "id" | "createdAt" | "updatedAt">
): Promise<TaskSelect> {
  const [task] = await db
    .insert(tasks)
    .values({
      ...data,
      id: nanoid(),
      source: data.source || "manual",
    })
    .returning();
  return task;
}

export async function completeTask(
  taskId: string,
  completedBy: string
): Promise<TaskSelect | undefined> {
  const [task] = await db
    .update(tasks)
    .set({
      completedAt: new Date(),
      completedBy,
    })
    .where(eq(tasks.id, taskId))
    .returning();
  return task;
}

export async function pinTask(taskId: string): Promise<TaskSelect | undefined> {
  const [task] = await db
    .update(tasks)
    .set({ pinnedAt: new Date() })
    .where(eq(tasks.id, taskId))
    .returning();
  return task;
}

export async function unpinTask(taskId: string): Promise<TaskSelect | undefined> {
  const [task] = await db
    .update(tasks)
    .set({ pinnedAt: null })
    .where(eq(tasks.id, taskId))
    .returning();
  return task;
}

export async function snoozeTask(
  taskId: string,
  until: Date
): Promise<TaskSelect | undefined> {
  const [task] = await db
    .update(tasks)
    .set({ snoozedUntil: until })
    .where(eq(tasks.id, taskId))
    .returning();
  return task;
}

export async function updatePriority(
  taskId: string,
  priority: TaskPriority
): Promise<TaskSelect | undefined> {
  const [task] = await db
    .update(tasks)
    .set({ priority })
    .where(eq(tasks.id, taskId))
    .returning();
  return task;
}

export async function getTasksByWorkspace(
  workspaceId: string,
  options?: {
    includeCompleted?: boolean;
    assignedTo?: string;
    clientId?: string;
  }
): Promise<TaskSelect[]> {
  const conditions = [eq(tasks.workspaceId, workspaceId)];
  
  if (!options?.includeCompleted) {
    conditions.push(isNull(tasks.completedAt));
  }
  
  if (options?.assignedTo) {
    conditions.push(eq(tasks.assignedTo, options.assignedTo));
  }
  
  if (options?.clientId) {
    conditions.push(eq(tasks.clientId, options.clientId));
  }
  
  return db.select().from(tasks).where(and(...conditions));
}

export const TaskService = {
  createTask,
  completeTask,
  pinTask,
  unpinTask,
  snoozeTask,
  updatePriority,
  getTasksByWorkspace,
};
```

Create TaskAggregationService.ts implementing D-09 sources and D-11 urgency algorithm:
```typescript
import { differenceInDays, isToday, isBefore } from "date-fns";
import { db } from "@/db";
import { tasks, type TaskSelect, type TaskPriority } from "@/db/tasks-schema";
import { onboardingChecklists, type ChecklistItem } from "@/db/onboarding-schema";
import { prospects } from "@/db/prospect-schema";
import { contracts } from "@/db/contract-schema";
import { proposals } from "@/db/proposal-schema";
import { eq, and, isNull, lt, lte, gte } from "drizzle-orm";

// D-11 Layer 1: Priority weights
const PRIORITY_WEIGHTS: Record<TaskPriority, number> = {
  high: 75,
  medium: 50,
  low: 25,
};

// Stale threshold for pipeline cards
const STALE_DAYS_THRESHOLD = 7;

export interface AggregatedTask {
  id: string;
  source: "checklist" | "pipeline" | "follow_up" | "expiring" | "seo" | "manual";
  entityType: string | null;
  entityId: string | null;
  title: string;
  description: string | null;
  dueAt: Date | null;
  urgencyScore: number;
  priority: TaskPriority | null;
  clientId: string | null;
  clientName: string | null;
  pinnedAt: Date | null;
  snoozedUntil: Date | null;
  // For deal value scoring
  dealValueCents: number | null;
  // For stale scoring
  daysInStage: number | null;
}

// D-11 Layer 1: Smart urgency score algorithm
export function calculateUrgencyScore(task: {
  dueAt: Date | null;
  dealValueCents: number | null;
  daysInStage: number | null;
  priority: TaskPriority | null;
}): number {
  let score = 0;
  const now = new Date();
  
  if (task.dueAt) {
    // Overdue: +20 per day overdue
    if (isBefore(task.dueAt, now)) {
      const daysOverdue = Math.max(0, differenceInDays(now, task.dueAt));
      score += daysOverdue * 20;
    }
    
    // Due today: +50
    if (isToday(task.dueAt)) {
      score += 50;
    }
  }
  
  // Deal value: +1 per 1000 cents (10 currency units)
  if (task.dealValueCents) {
    score += Math.floor(task.dealValueCents / 1000);
  }
  
  // Days stale in stage: +3 per day
  if (task.daysInStage) {
    score += task.daysInStage * 3;
  }
  
  // Manual priority: +25/50/75
  if (task.priority) {
    score += PRIORITY_WEIGHTS[task.priority];
  }
  
  return score;
}

export async function aggregateTasks(
  workspaceId: string,
  userId: string
): Promise<AggregatedTask[]> {
  const now = new Date();
  
  // Parallel fetch from all D-09 sources
  const [
    overdueChecklists,
    stalePipeline,
    followUps,
    expiringContracts,
    // seoTasks - would query SEO system
    manualTasks,
  ] = await Promise.all([
    getOverdueChecklistItems(workspaceId),
    getStalePipelineCards(workspaceId, STALE_DAYS_THRESHOLD),
    getScheduledFollowUps(workspaceId, userId),
    getExpiringContracts(workspaceId, 30), // 30 day window
    // getStuckSeoTasks(workspaceId), // TODO: integrate with SEO system
    getManualTasks(workspaceId, userId),
  ]);
  
  // Map each source to AggregatedTask
  const allTasks: AggregatedTask[] = [
    ...overdueChecklists.map(toChecklistTask),
    ...stalePipeline.map(toPipelineTask),
    ...followUps.map(toFollowUpTask),
    ...expiringContracts.map(toExpiringTask),
    ...manualTasks.map(toManualTask),
  ];
  
  // Filter out snoozed tasks (D-11 Layer 2)
  const activeTasks = allTasks.filter(
    (t) => !t.snoozedUntil || isBefore(t.snoozedUntil, now)
  );
  
  // Calculate urgency scores (D-11 Layer 1)
  for (const task of activeTasks) {
    task.urgencyScore = calculateUrgencyScore(task);
  }
  
  // Sort by urgencyScore descending
  activeTasks.sort((a, b) => b.urgencyScore - a.urgencyScore);
  
  return activeTasks;
}

// Helper query functions
async function getOverdueChecklistItems(workspaceId: string) {
  // Query checklists with incomplete items
  const checklists = await db
    .select()
    .from(onboardingChecklists)
    .where(eq(onboardingChecklists.workspaceId, workspaceId));
  
  const overdue: Array<{ checklist: typeof checklists[0]; item: ChecklistItem }> = [];
  for (const checklist of checklists) {
    for (const item of checklist.items) {
      if (!item.completedAt) {
        overdue.push({ checklist, item });
      }
    }
  }
  return overdue;
}

async function getStalePipelineCards(workspaceId: string, staleDays: number) {
  const cutoff = new Date();
  cutoff.setDate(cutoff.getDate() - staleDays);
  
  return db
    .select()
    .from(prospects)
    .where(and(
      eq(prospects.workspaceId, workspaceId),
      lt(prospects.updatedAt, cutoff),
      // Exclude converted/archived
    ));
}

async function getScheduledFollowUps(workspaceId: string, userId: string) {
  return db
    .select()
    .from(tasks)
    .where(and(
      eq(tasks.workspaceId, workspaceId),
      eq(tasks.source, "follow_up"),
      isNull(tasks.completedAt),
    ));
}

async function getExpiringContracts(workspaceId: string, windowDays: number) {
  const cutoff = new Date();
  cutoff.setDate(cutoff.getDate() + windowDays);
  
  return db
    .select()
    .from(contracts)
    .where(and(
      eq(contracts.workspaceId, workspaceId),
      eq(contracts.status, "executed"),
      lte(contracts.expiresAt, cutoff),
    ));
}

async function getManualTasks(workspaceId: string, userId: string) {
  return db
    .select()
    .from(tasks)
    .where(and(
      eq(tasks.workspaceId, workspaceId),
      eq(tasks.source, "manual"),
      isNull(tasks.completedAt),
    ));
}
```

```typescript
import { TaskItem } from "./TaskItem";
import type { AggregatedTask } from "@/types/tasks";

type SortMode = "smart" | "due_date" | "deal_value" | "client_name";

interface TodaysFeedProps {
  tasks: AggregatedTask[];
  onComplete: (taskId: string) => Promise<void>;
  onPin: (taskId: string) => Promise<void>;
  onUnpin: (taskId: string) => Promise<void>;
  onSnooze: (taskId: string, until: Date) => Promise<void>;
  onSetPriority: (taskId: string, priority: "high" | "medium" | "low") => Promise<void>;
  onClick: (taskId: string) => void;
  onAddTask: () => void;
}

export function TodaysFeed({
  tasks,
  onComplete,
  onPin,
  onUnpin,
  onSnooze,
  onSetPriority,
  onClick,
  onAddTask,
}: TodaysFeedProps) {
  // D-11 Layer 3: Sort mode toggle
  const [sortMode, setSortMode] = useState<SortMode>("smart");
  
  // Sort tasks based on mode
  const sortedTasks = [...tasks].sort((a, b) => {
    switch (sortMode) {
      case "smart":
        return b.urgencyScore - a.urgencyScore;
      case "due_date":
        if (!a.dueAt && !b.dueAt) return 0;
        if (!a.dueAt) return 1;
        if (!b.dueAt) return -1;
        return a.dueAt.getTime() - b.dueAt.getTime();
      case "deal_value":
        return (b.dealValueCents || 0) - (a.dealValueCents || 0);
      case "client_name":
        return (a.clientName || "").localeCompare(b.clientName || "");
      default:
        return 0;
    }
  });
  
  // Separate pinned (for My Focus) and unpinned
  const unpinnedTasks = sortedTasks.filter((t) => !t.pinnedAt);
  
  const sortLabels: Record<SortMode, string> = {
    smart: "Smart Priority",
    due_date: "Due Date",
    deal_value: "Deal Value",
    client_name: "Client Name",
  };
  
  return (
    <div className="space-y-4">
      {/* D-11 Layer 5: My Focus section */}
      <MyFocusSection
        tasks={tasks}
        onComplete={onComplete}
        onPin={onPin}
        onUnpin={onUnpin}
        onSnooze={onSnooze}
        onSetPriority={onSetPriority}
        onClick={onClick}
      />
      
      {/* Main task list */}
      <Card>
        <CardHeader className="flex flex-row items-center justify-between">
          <CardTitle className="text-base">Today's Tasks</CardTitle>
          <div className="flex items-center gap-2">
            {/* D-11 Layer 3: Sort mode toggle */}
            <DropdownMenu>
              <DropdownMenuTrigger asChild>
                <Button variant="ghost" size="sm">
                  <ArrowUpDown className="mr-2 h-4 w-4" />
                  {sortLabels[sortMode]}
                </Button>
              </DropdownMenuTrigger>
              <DropdownMenuContent align="end">
                {Object.entries(sortLabels).map(([mode, label]) => (
                  <DropdownMenuItem
                    key={mode}
                    onClick={() => setSortMode(mode as SortMode)}
                  >
                    {label}
                  </DropdownMenuItem>
                ))}
              </DropdownMenuContent>
            </DropdownMenu>
            
            <Button size="sm" onClick={onAddTask}>
              <Plus className="mr-2 h-4 w-4" />
              Add Task
            </Button>
          </div>
        </CardHeader>
        <CardContent className="space-y-1">
          {unpinnedTasks.length === 0 ? (

          )}
        </CardContent>
      </Card>
    </div>
  );
```
