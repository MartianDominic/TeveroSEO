import { eq } from "drizzle-orm";
import { db } from "@/db";
import { pipelineConfigs, DEFAULT_PIPELINE_STAGES, type PipelineStageConfig } from "@/db/pipeline-config-schema";
import { nanoid } from "nanoid";

export async function getOrCreateConfig(workspaceId: string) {
  const [existing] = await db
    .select()
    .from(pipelineConfigs)
    .where(eq(pipelineConfigs.workspaceId, workspaceId))
    .limit(1);
  
  if (existing) return existing;
  
  // D-06: Create default stages for new workspace
  const [created] = await db
    .insert(pipelineConfigs)
    .values({
      id: nanoid(),
      workspaceId,
      stages: DEFAULT_PIPELINE_STAGES,
    })
    .returning();
  
  return created;
}

export async function updateStages(workspaceId: string, stages: PipelineStageConfig[]) {
  // Validate: each stage has unique id and sequential order
  const ids = stages.map((s) => s.id);
  if (new Set(ids).size !== ids.length) {
    throw new Error("Duplicate stage IDs");
  }
  
  const [updated] = await db
    .update(pipelineConfigs)
    .set({ stages })
    .where(eq(pipelineConfigs.workspaceId, workspaceId))
    .returning();
  
  return updated;
}

export const PipelineConfigService = { getOrCreateConfig, updateStages };

import { eq, and } from "drizzle-orm";
import { db } from "@/db";
import { prospects, type ProspectSelect } from "@/db/prospect-schema";
import { PipelineConfigService } from "./PipelineConfigService";
import { ActivityRepository } from "../../contracts/repositories/ActivityRepository";
import { nanoid } from "nanoid";

export async function moveProspectToStage(
  workspaceId: string,
  prospectId: string,
  targetStage: string
): Promise<ProspectSelect> {
  // Validate target stage exists in workspace config
  const config = await PipelineConfigService.getOrCreateConfig(workspaceId);
  const validStages = config.stages.map((s) => s.id);
  
  if (!validStages.includes(targetStage)) {
    throw new Error(`Invalid stage: ${targetStage}`);
  }
  
  const [updated] = await db
    .update(prospects)
    .set({ pipelineStage: targetStage, updatedAt: new Date() })
    .where(and(
      eq(prospects.id, prospectId),
      eq(prospects.workspaceId, workspaceId)
    ))
    .returning();
  
  if (!updated) {
    throw new Error("Prospect not found");
  }
  
  // Log activity
  await ActivityRepository.insertActivity({
    id: nanoid(),
    workspaceId,
    entityType: "prospect",
    entityId: prospectId,
    activityType: "status_changed",
    activityData: { newStage: targetStage },
  });
  
  return updated;
}

export async function getProspectsGroupedByStage(
  workspaceId: string
): Promise<Record<string, ProspectSelect[]>> {
  const config = await PipelineConfigService.getOrCreateConfig(workspaceId);
  const allProspects = await db
    .select()
    .from(prospects)
    .where(eq(prospects.workspaceId, workspaceId));
  
  // Initialize all stages with empty arrays
  const grouped: Record<string, ProspectSelect[]> = {};
  for (const stage of config.stages) {
    grouped[stage.id] = [];
  }
  
  // Group prospects
  for (const prospect of allProspects) {
    const stage = prospect.pipelineStage;
    if (grouped[stage]) {
      grouped[stage].push(prospect);
    } else {
      // Prospect has stage not in config (stale data)
      grouped["new"] = grouped["new"] || [];
      grouped["new"].push(prospect);
    }
  }
  
  return grouped;
}

export const PipelineService = { moveProspectToStage, getProspectsGroupedByStage };

"use client";

import { useState, useRef, useCallback } from "react";
import { DragDropProvider } from "@dnd-kit/react";
import { useSortable, isSortable } from "@dnd-kit/react/sortable";
import { KanbanColumn } from "@tevero/ui";
import { PipelineCard } from "./PipelineCard";
import type { ProspectSelect } from "@/types/prospect";
import type { PipelineStageConfig } from "@/types/pipeline";

interface PipelineKanbanProps {
  stages: PipelineStageConfig[];
  initialState: Record<string, ProspectSelect[]>;
  onMoveProspect: (prospectId: string, targetStage: string) => Promise<void>;
  onViewProspect: (prospectId: string) => void;
  onArchiveProspect: (prospectId: string) => void;
}

// Sortable card wrapper
function SortableCard({
  prospect,
  stages,
  onViewDetails,
  onMoveToStage,
  onArchive,
}: {
  prospect: ProspectSelect;
  stages: PipelineStageConfig[];
  onViewDetails: () => void;
  onMoveToStage: (stageId: string) => void;
  onArchive: () => void;
}) {
  const { ref } = useSortable({ id: prospect.id, group: prospect.pipelineStage });
  
  return (
    <div ref={ref}>
      <PipelineCard
        id={prospect.id}
        domain={prospect.domain}
        companyName={prospect.companyName}
        dealValueCents={null} // TODO: Join with proposal
        stageEnteredAt={prospect.updatedAt}
        assignedTo={prospect.assignedTo}
        firstContactAt={prospect.createdAt}
        stages={stages}
        onViewDetails={onViewDetails}
        onMoveToStage={onMoveToStage}
        onArchive={onArchive}
        draggable
      />
    </div>
  );
}

export function PipelineKanban({
  stages,
  initialState,
  onMoveProspect,
  onViewProspect,
  onArchiveProspect,
}: PipelineKanbanProps) {
  const [items, setItems] = useState(initialState);
  const snapshot = useRef(structuredClone(initialState));
  
  const handleDragEnd = useCallback(async (event: any) => {
    if (event.canceled) {
      setItems(snapshot.current);
      return;
    }
    
    const { source } = event.operation;
    if (!isSortable(source)) return;
    
    const { initialIndex, index, initialGroup, group } = source;
    if (initialGroup == null || group == null) return;
    if (initialGroup === group && initialIndex === index) return;
    
    // Get prospect ID
    const prospectId = items[initialGroup][initialIndex].id;
    
    // Optimistic update
    setItems((prev) => {
      if (initialGroup === group) {
        // Reorder within same column
        const groupItems = [...prev[group]];
        const [removed] = groupItems.splice(initialIndex, 1);
        groupItems.splice(index, 0, removed);
        return { ...prev, [group]: groupItems };
      }
      
      // Move between columns
      const sourceItems = [...prev[initialGroup]];
      const [removed] = sourceItems.splice(initialIndex, 1);
      removed.pipelineStage = group as string;
      const targetItems = [...prev[group]];
      targetItems.splice(index, 0, removed);
      return { ...prev, [initialGroup]: sourceItems, [group]: targetItems };
    });
    
    // Persist to backend
    try {
      await onMoveProspect(prospectId, group as string);
    } catch (error) {
      // Rollback on error
      setItems(snapshot.current);
    }
  }, [items, onMoveProspect]);
  
  return (
    <DragDropProvider
      onDragStart={() => {
        snapshot.current = structuredClone(items);
      }}
      onDragEnd={handleDragEnd}
    >
      <div className="flex gap-4 overflow-x-auto pb-4">
        {stages.map((stage) => (
          <KanbanColumn
            key={stage.id}
            title={stage.name}
            count={items[stage.id]?.length || 0}
            status={stage.id}
            statusConfig={{ color: stage.color, pulse: false }}
          >
            {items[stage.id]?.map((prospect) => (
              <SortableCard
                key={prospect.id}
                prospect={prospect}
                stages={stages}
                onViewDetails={() => onViewProspect(prospect.id)}
                onMoveToStage={(stageId) => {
                  onMoveProspect(prospect.id, stageId);
                }}
                onArchive={() => onArchiveProspect(prospect.id)}
              />
            ))}
          </KanbanColumn>
        ))}
      </div>
    </DragDropProvider>
  );
}

"use client";

import { useState } from "react";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from "@tevero/ui";
import { Button } from "@tevero/ui";
import { Input } from "@tevero/ui";
import { Plus, Trash2, GripVertical } from "lucide-react";

interface PipelineStageConfig {
  id: string;
  name: string;
  order: number;
  color: string;
}

export interface StageConfigDialogProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  stages: PipelineStageConfig[];
  onSave: (stages: PipelineStageConfig[]) => Promise<void>;
}

export function StageConfigDialog({
  open,
  onOpenChange,
  stages: initialStages,
  onSave,
}: StageConfigDialogProps) {
  const [stages, setStages] = useState(initialStages);
  const [saving, setSaving] = useState(false);
  
  const handleAddStage = () => {
    const newId = `stage_${Date.now()}`;
    setStages([
      ...stages,
      {
        id: newId,
        name: "New Stage",
        order: stages.length,
        color: "#6b7280",
      },
    ]);
  };
  
  const handleRemoveStage = (id: string) => {
    // D-06: Can remove stages (prospects will need migration)
    setStages(stages.filter((s) => s.id !== id).map((s, i) => ({ ...s, order: i })));
  };
  
  const handleUpdateStage = (id: string, updates: Partial<PipelineStageConfig>) => {
    setStages(stages.map((s) => (s.id === id ? { ...s, ...updates } : s)));
  };
  
  const handleSave = async () => {
    setSaving(true);
    try {
      await onSave(stages);
      onOpenChange(false);
    } finally {
      setSaving(false);
    }
  };
  
  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-lg">
        <DialogHeader>
          <DialogTitle>Configure Pipeline Stages</DialogTitle>
        </DialogHeader>
        
        <div className="space-y-2 max-h-96 overflow-y-auto">
          {stages.map((stage, index) => (
            <div key={stage.id} className="flex items-center gap-2 p-2 bg-surface-2 rounded">
              <GripVertical className="h-4 w-4 text-text-4 cursor-grab" />
              <Input
                value={stage.name}
                onChange={(e) => handleUpdateStage(stage.id, { name: e.target.value })}
                className="flex-1"
              />
              <input
                type="color"
                value={stage.color}
                onChange={(e) => handleUpdateStage(stage.id, { color: e.target.value })}
                className="h-8 w-8 rounded border-0 cursor-pointer"
              />
              <Button
                size="icon-sm"
                variant="ghost"
                onClick={() => handleRemoveStage(stage.id)}
                disabled={stages.length <= 2}
              >
                <Trash2 className="h-4 w-4 text-error" />
              </Button>
            </div>
          ))}
        </div>
        
        <Button variant="outline" onClick={handleAddStage} className="w-full">
          <Plus className="mr-2 h-4 w-4" />
          Add Stage
        </Button>
        
        <DialogFooter>
          <Button variant="ghost" onClick={() => onOpenChange(false)}>
            Cancel
          </Button>
          <Button onClick={handleSave} disabled={saving}>
            {saving ? "Saving..." : "Save Changes"}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
