```typescript
import { db } from "@/db";
import { voiceTemplates, type VoiceTemplateSelect, type VoiceTemplateInsert } from "@/db/voice-schema";
import { eq, and, isNull, asc } from "drizzle-orm";
import { nanoid } from "nanoid";
import { INDUSTRY_TEMPLATES } from "../templates/industryTemplates";

export const voiceTemplateService = {
  async listAll(): Promise<VoiceTemplateSelect[]> {
    return db.select().from(voiceTemplates).orderBy(asc(voiceTemplates.name));
  },

  async listByIndustry(industry: string): Promise<VoiceTemplateSelect[]> {
    return db.select().from(voiceTemplates)
      .where(eq(voiceTemplates.industry, industry))
      .orderBy(asc(voiceTemplates.name));
  },

  async getById(id: string): Promise<VoiceTemplateSelect | null> {
    const [result] = await db.select().from(voiceTemplates)
      .where(eq(voiceTemplates.id, id)).limit(1);
    return result ?? null;
  },

  async create(data: Omit<VoiceTemplateInsert, "id">): Promise<VoiceTemplateSelect> {
    const id = nanoid();
    const [result] = await db.insert(voiceTemplates)
      .values({ id, ...data })
      .returning();
    return result;
  },

  async incrementUsage(id: string): Promise<void> {
    await db.update(voiceTemplates)
      .set({ usageCount: db.raw`usage_count + 1` })
      .where(eq(voiceTemplates.id, id));
  },

  async delete(id: string): Promise<void> {
    const template = await this.getById(id);
    if (!template) return;
    if (template.isSystem) {
      throw new Error("Cannot delete system template");
    }
    await db.delete(voiceTemplates).where(eq(voiceTemplates.id, id));
  },

  async seedSystemTemplates(): Promise<void> {
    for (const template of INDUSTRY_TEMPLATES) {
      const existing = await this.getById(template.id);
      if (existing) continue;
      await db.insert(voiceTemplates).values({
        id: template.id,
        name: template.name,
        description: template.description,
        industry: template.id,
        isSystem: true,
        templateConfig: template.defaults,
        usageCount: 0,
      });
    }
  },
};
```

```typescript
import { z } from "zod";

const analyzeSchema = z.object({
  urls: z.array(z.string().url()).min(1).max(10),
});

export default defineEventHandler(async (event) => {
  const clientId = getRouterParam(event, "clientId");
  if (!clientId) {
    throw createError({ statusCode: 400, message: "Missing clientId" });
  }

  const body = await readBody(event);
  const parsed = analyzeSchema.safeParse(body);
  if (!parsed.success) {
    throw createError({ statusCode: 400, message: parsed.error.message });
  }

  // Ensure profile exists
  let profile = await voiceProfileService.getByClientId(clientId);
  if (!profile) {
    profile = await voiceProfileService.upsert(clientId, { mode: "best_practices" });
  }

  // Queue analysis job
  const job = await voiceAnalysisQueue.add("voice-analysis", {
    clientId,
    profileId: profile.id,
    urls: parsed.data.urls,
  });

  return {
    success: true,
    data: {
      jobId: job.id,
      profileId: profile.id,
      urlCount: parsed.data.urls.length,
    },
  };
});
```

```typescript
import { defineEventHandler, readBody, createError, getRouterParam, getMethod, getQuery } from "h3";
import { voiceProfileService, protectionRulesService } from "@/server/features/voice";
import { z } from "zod";

const createRuleSchema = z.object({
  ruleType: z.enum(["page", "section", "pattern"]),
  target: z.string().min(1),
  reason: z.string().optional(),
  expiresAt: z.string().datetime().optional(),
});

export default defineEventHandler(async (event) => {
  const clientId = getRouterParam(event, "clientId");
  if (!clientId) {
    throw createError({ statusCode: 400, message: "Missing clientId" });
  }

  const profile = await voiceProfileService.getByClientId(clientId);
  if (!profile) {
    throw createError({ statusCode: 404, message: "Voice profile not found" });
  }

  const method = getMethod(event);

  if (method === "GET") {
    const rules = await protectionRulesService.list(profile.id);
    return { success: true, data: rules };
  }

  if (method === "POST") {
    const body = await readBody(event);
    const parsed = createRuleSchema.safeParse(body);
    if (!parsed.success) {
      throw createError({ statusCode: 400, message: parsed.error.message });
    }
    const rule = await protectionRulesService.create({
      profileId: profile.id,
      ruleType: parsed.data.ruleType,
      target: parsed.data.target,
      reason: parsed.data.reason,
      expiresAt: parsed.data.expiresAt ? new Date(parsed.data.expiresAt) : null,
      createdBy: "user", // TODO: get from auth context
    });
    return { success: true, data: rule };
  }

  if (method === "DELETE") {
    const query = getQuery(event);
    const ruleId = query.ruleId as string;
    if (!ruleId) {
      throw createError({ statusCode: 400, message: "Missing ruleId" });
    }
    await protectionRulesService.delete(ruleId);
    return { success: true };
  }

  throw createError({ statusCode: 405, message: "Method not allowed" });
});
```
