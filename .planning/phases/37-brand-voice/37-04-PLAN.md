```typescript
import { defineEventHandler, readBody, createError, getRouterParam } from "h3";
import { voiceProfileService, buildVoiceConstraints } from "@/server/features/voice";
import { z } from "zod";

const previewSchema = z.object({
  previewType: z.enum(["headline", "paragraph", "cta"]),
  topic: z.string().min(1).max(500),
  keywords: z.array(z.string()).optional(),
});

export default defineEventHandler(async (event) => {
  const clientId = getRouterParam(event, "clientId");
  if (!clientId) {
    throw createError({ statusCode: 400, message: "Missing clientId" });
  }

  const body = await readBody(event);
  const parsed = previewSchema.safeParse(body);
  if (!parsed.success) {
    throw createError({ statusCode: 400, message: parsed.error.message });
  }

  const profile = await voiceProfileService.getByClientId(clientId);
  if (!profile) {
    throw createError({ statusCode: 404, message: "Voice profile not found" });
  }

  // Build voice constraints for AI
  const voiceConstraints = buildVoiceConstraints({
    profile,
    templateBlend: profile.voiceBlendWeight ?? 0,
    templateId: profile.voiceTemplateId ?? undefined,
  });

  // Call AI-Writer preview API
  const aiWriterUrl = process.env.AI_WRITER_API_URL || "http://localhost:8000";
  const response = await fetch(`${aiWriterUrl}/api/voice/preview`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      preview_type: parsed.data.previewType,
      topic: parsed.data.topic,
      keywords: parsed.data.keywords || [],
      voice_constraints: voiceConstraints,
      voice_profile: {
        mode: profile.mode,
        primary_tone: profile.primaryTone,
        formality_level: profile.formalityLevel,
        personality_traits: profile.personalityTraits,
        signature_phrases: profile.signaturePhrases,
        forbidden_phrases: profile.forbiddenPhrases,
      },
    }),
  });

  if (!response.ok) {
    throw createError({ statusCode: 500, message: "Preview generation failed" });
  }

  const result = await response.json();
  return { success: true, data: result };
});
```

```python
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from typing import Optional
import logging

from services.gemini_client import GeminiClient

router = APIRouter(prefix="/api/voice", tags=["voice"])
logger = logging.getLogger(__name__)

class VoiceProfile(BaseModel):
    mode: str
    primary_tone: Optional[str] = None
    formality_level: Optional[int] = None
    personality_traits: Optional[list[str]] = None
    signature_phrases: Optional[list[str]] = None
    forbidden_phrases: Optional[list[str]] = None

class PreviewRequest(BaseModel):
    preview_type: str  # "headline" | "paragraph" | "cta"
    topic: str
    keywords: list[str] = []
    voice_constraints: str
    voice_profile: VoiceProfile

class ComplianceScore(BaseModel):
    overall: float
    tone_consistency: float
    vocabulary_alignment: float
    structure_compliance: float
    issues: list[dict]

class PreviewResponse(BaseModel):
    content: str
    compliance: ComplianceScore

@router.post("/preview", response_model=PreviewResponse)
async def generate_preview(request: PreviewRequest) -> PreviewResponse:
    """Generate a voice preview sample with compliance scoring."""
    
    gemini = GeminiClient()
    
    # Build prompt based on preview type
    if request.preview_type == "headline":
        word_target = "5-10 words"
        structure = "a compelling headline"
    elif request.preview_type == "paragraph":
        word_target = "50-80 words"
        structure = "an engaging paragraph"
    else:  # cta
        word_target = "10-20 words"
        structure = "a call-to-action"
    
    generation_prompt = f"""Generate {structure} about the following topic.

TOPIC: {request.topic}

KEYWORDS TO INCLUDE (naturally): {', '.join(request.keywords) if request.keywords else 'None'}

TARGET LENGTH: {word_target}

VOICE REQUIREMENTS:
{request.voice_constraints}

Generate ONLY the content, no explanations or meta-commentary."""

    try:
        content = await gemini.generate(generation_prompt, max_tokens=200)
    except Exception as e:
        logger.error(f"Generation failed: {e}")
        raise HTTPException(status_code=500, detail="Content generation failed")
    
    # Score compliance
    compliance = await _score_compliance(content, request.voice_profile)
    
    return PreviewResponse(content=content.strip(), compliance=compliance)


async def _score_compliance(content: str, profile: VoiceProfile) -> ComplianceScore:
    """Score generated content against voice profile."""
    
    # Basic compliance checks (production would use AI for nuanced scoring)
    issues = []
    tone_score = 80.0  # Base score
    vocab_score = 80.0
    structure_score = 80.0
    
    # Check forbidden phrases
    content_lower = content.lower()
    for phrase in (profile.forbidden_phrases or []):
        if phrase.lower() in content_lower:
            issues.append({
                "type": "vocabulary",
                "severity": "warning",
                "message": f"Contains forbidden phrase: '{phrase}'",
                "suggestion": f"Remove or rephrase to avoid '{phrase}'"
            })
            vocab_score -= 15
    
    # Check signature phrases (bonus for inclusion)
    signature_found = 0
    for phrase in (profile.signature_phrases or []):
        if phrase.lower() in content_lower:
            signature_found += 1
    if profile.signature_phrases and len(profile.signature_phrases) > 0:
        vocab_score += (signature_found / len(profile.signature_phrases)) * 10
    
    # Clamp scores
    tone_score = max(0, min(100, tone_score))
    vocab_score = max(0, min(100, vocab_score))
    structure_score = max(0, min(100, structure_score))
    
    overall = (tone_score * 0.4 + vocab_score * 0.35 + structure_score * 0.25)
    
    return ComplianceScore(
        overall=round(overall, 1),
        tone_consistency=round(tone_score, 1),
        vocabulary_alignment=round(vocab_score, 1),
        structure_compliance=round(structure_score, 1),
        issues=issues
    )
```

```typescript
import { Badge } from "@tevero/ui";
import { cn } from "@tevero/ui/lib/utils";
import { CheckCircle, AlertTriangle, XCircle } from "lucide-react";

interface ComplianceScoreBadgeProps {
  score: number;
  showLabel?: boolean;
}

export function ComplianceScoreBadge({ score, showLabel = true }: ComplianceScoreBadgeProps) {
  const getConfig = () => {
    if (score >= 70) {
      return {
        icon: CheckCircle,
        label: "Matches voice profile",
        variant: "default" as const,
        className: "bg-green-500/10 text-green-600 border-green-500/20",
      };

    if (score >= 50) {
      return {
        icon: AlertTriangle,
        label: "Partial match - review suggestions",
        variant: "outline" as const,
        className: "bg-amber-500/10 text-amber-600 border-amber-500/20",
      };
    }
    return {
      icon: XCircle,
      label: "Does not match - adjust settings",
      variant: "destructive" as const,
      className: "bg-red-500/10 text-red-600 border-red-500/20",
    };
  };

  const config = getConfig();
  const Icon = config.icon;

  return (
    <div className="flex items-center gap-2">
      <Badge className={cn("gap-1.5", config.className)}>
        <Icon className="h-3.5 w-3.5" />
        {score.toFixed(0)}%
      </Badge>
      {showLabel && (
        <span className="text-xs text-muted-foreground">{config.label}</span>
      )}
    </div>
  );
}
```

```python
def build_voice_constraints_from_profile(profile: dict) -> str:

Write in a professional, SEO-optimized style with clear, actionable content.

Maintain existing content style. Only modify for technical SEO (meta tags, alt text).
```
