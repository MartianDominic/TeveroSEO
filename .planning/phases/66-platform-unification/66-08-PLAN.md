interface PixelAnalyticsDaily {
  id: string;
  installationId: string;
  date: string; // YYYY-MM-DD
  pageviews: number;
  sessions: number;
  uniqueVisitors: number;
  avgTimeOnPage: number; // seconds
  bounceRate: number; // 0-100
  lcpP75: number; // ms
  clsP75: number; // unitless
  inpP75: number; // ms
  topPages: Array<{
    url: string;
    views: number;
  }>;
}

interface AnalyticsQuery {
  siteId: string;
  startDate: string;
  endDate: string;
  granularity?: 'daily' | 'weekly' | 'monthly';
}

interface AnalyticsResponse {
  summary: {
    totalPageviews: number;
    totalSessions: number;
    totalUniqueVisitors: number;
    avgTimeOnPage: number;
    bounceRate: number;
  };
  cwv: {
    lcp: { p75: number; rating: 'good' | 'needs-improvement' | 'poor' };
    cls: { p75: number; rating: 'good' | 'needs-improvement' | 'poor' };
    inp: { p75: number; rating: 'good' | 'needs-improvement' | 'poor' };
  };
  timeseries: Array<{
    date: string;
    pageviews: number;
    sessions: number;
    uniqueVisitors: number;
  }>;
  topPages: Array<{
    url: string;
    views: number;
    avgTimeOnPage: number;
  }>;
}
```

export class PixelAnalyticsService {
  // Get analytics summary and timeseries
  async getAnalytics(query: AnalyticsQuery): Promise<AnalyticsResponse>;
  
  // Get top pages for a date range
  async getTopPages(siteId: string, startDate: string, endDate: string, limit?: number): Promise<TopPage[]>;
  
  // Get CWV trend over time
  async getCwvTrend(siteId: string, startDate: string, endDate: string): Promise<CwvTrend[]>;
  
  // Calculate CWV rating based on Google thresholds
  private getCwvRating(metric: 'lcp' | 'cls' | 'inp', value: number): 'good' | 'needs-improvement' | 'poor';
}
```

**CWV thresholds (Google standards):**
- LCP: good < 2500ms, needs-improvement < 4000ms, poor >= 4000ms
- CLS: good < 0.1, needs-improvement < 0.25, poor >= 0.25
- INP: good < 200ms, needs-improvement < 500ms, poor >= 500ms

**Aggregation logic:**
- Sum pageviews, sessions, uniqueVisitors
- Average timeOnPage, bounceRate
- p75 for CWV (already aggregated per day)
- Merge topPages arrays, sum views per URL
