```typescript
export class PlatformIntegrationFacade {
  constructor(
    private pixelService: PixelScriptService,
    private oauthService: OAuthService, // From P61
    private writeAdapters: WriteAdapterRegistry, // From P31/33
    private cmsPublishers: CmsPublisherRegistry, // From P39
  ) {}

  // CONNECTION STATUS
  async getConnectionStatus(siteId: string): Promise<ConnectionStatus>;
  async getAvailableIntegrations(siteId: string): Promise<Integration[]>;
  
  // READ OPERATIONS (routes to best available source)
  async getAnalytics(siteId: string, type: 'traffic' | 'rankings' | 'cwv'): Promise<Analytics>;
  async getContent(siteId: string, platform: Platform): Promise<Content[]>;
  
  // WRITE OPERATIONS (routes to OAuth adapter or pixel)
  async updateSeoField(siteId: string, resourceId: string, field: SeoField, value: string): Promise<Result>;
  async createRedirect(siteId: string, from: string, to: string): Promise<Result>;
  
  // PUBLISH OPERATIONS (routes to P39 CMS Publishers)
  async publishContent(siteId: string, content: Content): Promise<Result>;
  async scheduleContent(siteId: string, content: Content, publishAt: Date): Promise<Result>;
  
  // PIXEL OPERATIONS
  async getPixelStatus(siteId: string): Promise<PixelStatus>;
  async getPixelAnalytics(siteId: string, range: DateRange): Promise<PixelAnalytics>;
  async queueDomChange(siteId: string, change: DomChange): Promise<Result>;
  async approveDomChange(changeId: string): Promise<Result>;
}
```

**Routing logic:**
- `getAnalytics('traffic')`: Use GA OAuth if available, else pixel analytics
- `getAnalytics('rankings')`: Requires GSC OAuth, return unavailable if not connected
- `getAnalytics('cwv')`: Use pixel analytics (always available after pixel install)
- `updateSeoField`: Use OAuth write adapter if available, else queue pixel DOM change
- `publishContent`: Requires OAuth with write scopes

Update `index.ts` to export facade.
