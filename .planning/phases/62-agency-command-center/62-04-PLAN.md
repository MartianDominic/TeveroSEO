```typescript
export class MetricsService {
  constructor(
    private readonly metricsRepo: PipelineMetricsRepository,
    private readonly db: DrizzleClient
  ) {}

  // Get cached metrics for dashboard
  async getMetrics(workspaceId: string): Promise<PipelineMetricsSelect | null>;

  // Compute all metrics for a workspace
  async computeWorkspaceMetrics(workspaceId: string): Promise<PipelineMetricsSelect> {
    const startTime = Date.now();

    // Parallel queries for counts
    const [
      prospectCounts,
      proposalCounts,
      contractCounts,
      invoiceCounts,
      financialMetrics,
      conversionRates,
      cycleTimes
    ] = await Promise.all([
      this.countProspects(workspaceId),
      this.countProposals(workspaceId),
      this.countContracts(workspaceId),
      this.countInvoices(workspaceId),
      this.computeFinancials(workspaceId),
      this.computeConversionRates(workspaceId),
      this.computeCycleTimes(workspaceId)
    ]);

    const metrics = {
      workspaceId,
      ...prospectCounts,
      ...proposalCounts,
      ...contractCounts,
      ...invoiceCounts,
      ...financialMetrics,
      ...conversionRates,
      ...cycleTimes,
      computedAt: new Date(),
      computationDurationMs: Date.now() - startTime
    };

    await this.metricsRepo.upsert(workspaceId, metrics);
    return metrics;
  }

  // Individual computation methods
  private async countProspects(workspaceId: string): Promise<ProspectCounts>;
  private async countProposals(workspaceId: string): Promise<ProposalCounts>;
  private async countContracts(workspaceId: string): Promise<ContractCounts>;
  private async countInvoices(workspaceId: string): Promise<InvoiceCounts>;
  private async computeFinancials(workspaceId: string): Promise<FinancialMetrics>;
  private async computeConversionRates(workspaceId: string): Promise<ConversionRates>;
  private async computeCycleTimes(workspaceId: string): Promise<CycleTimes>;
}
```

```sql
FROM prospects

GROUP BY status;

SELECT COUNT(*) FROM prospects
WHERE workspace_id = $1 AND status = 'converted' AND updated_at > NOW() - INTERVAL '30 days';

-- Financial metrics
SELECT
  SUM(CASE WHEN status = 'draft' THEN total_value_cents ELSE 0 END) as pipeline_value_draft_cents,
  SUM(CASE WHEN status = 'sent' THEN total_value_cents ELSE 0 END) as pipeline_value_sent_cents
FROM proposals WHERE workspace_id = $1;

-- Win rate
SELECT
  (COUNT(*) FILTER (WHERE outcome = 'won') * 10000 / NULLIF(COUNT(*), 0))::int as win_rate_pct
FROM deal_outcomes WHERE workspace_id = $1 AND outcome_at > NOW() - INTERVAL '90 days';
```

```typescript
export interface PipelineMetricsJobData {
  type: 'compute_workspace' | 'compute_all_stale';
  workspaceId?: string;
}

export const pipelineMetricsQueue = new Queue<PipelineMetricsJobData>('pipeline-metrics', {
  connection: redisConnection,
  defaultJobOptions: {
    attempts: 2,
    backoff: { type: 'fixed', delay: 5000 }
  }
});

// Schedule refresh every 5 minutes
export async function schedulePipelineMetricsRefresh() {
  await pipelineMetricsQueue.add('refresh-all', { type: 'compute_all_stale' }, {
    repeat: { every: 5 * 60 * 1000 },
    jobId: 'pipeline-metrics-refresh'
  });
}
```

**pipeline-metrics-processor.ts:**

```typescript
export async function processPipelineMetrics(job: Job<PipelineMetricsJobData>) {
  const logger = createLogger({ module: 'pipeline-metrics-processor', jobId: job.id });

  if (job.data.type === 'compute_workspace' && job.data.workspaceId) {
    logger.info(`Computing metrics for workspace ${job.data.workspaceId}`);
    await metricsService.computeWorkspaceMetrics(job.data.workspaceId);
    return;
  }

  if (job.data.type === 'compute_all_stale') {
    // Find workspaces with stale or missing metrics (> 5 min old)
    const staleWorkspaces = await metricsRepo.getStale(5);
    logger.info(`Found ${staleWorkspaces.length} workspaces needing refresh`);

    // Process in batches to avoid overwhelming DB
    for (const workspaceId of staleWorkspaces) {
      try {
        await metricsService.computeWorkspaceMetrics(workspaceId);
      } catch (error) {
        logger.error(`Failed to compute metrics for ${workspaceId}`, error);
      }
    }
  }
}
```

```typescript
export async function getDashboardMetrics(workspaceId: string): Promise<DashboardMetricsResponse> {
  const metrics = await metricsService.getMetrics(workspaceId);

  if (!metrics) {
    // Trigger computation and return partial data
    await pipelineMetricsQueue.add('compute', { type: 'compute_workspace', workspaceId });
    return { pending: true, metrics: null };
  }

  // Check if metrics are stale (> 10 minutes old)
  const isStale = Date.now() - metrics.computedAt.getTime() > 10 * 60 * 1000;
  if (isStale) {
    // Trigger refresh in background, return current data
    await pipelineMetricsQueue.add('compute', { type: 'compute_workspace', workspaceId });
  }

  return {
    pending: false,
    metrics: formatMetricsForDashboard(metrics),
    computedAt: metrics.computedAt,
    isStale
  };
}

function formatMetricsForDashboard(metrics: PipelineMetricsSelect): DashboardMetrics {
  return {
    today: {
      overdue: metrics.invoicesOverdue + metrics.contractsExpiring7d,
      dueToday: 0, // Computed from follow_ups
      awaitingYou: metrics.proposalsViewed + metrics.contractsPendingSignature,
      new: metrics.prospectsNew
    },
    pipeline: {
      prospects: { new: metrics.prospectsNew, qualified: metrics.prospectsQualified, ... },
      proposals: { draft: metrics.proposalsDraft, sent: metrics.proposalsSent, ... },
      agreements: { draft: metrics.contractsDraft, pending: metrics.contractsPendingSignature, ... },
      payments: { sent: metrics.invoicesSent, overdue: metrics.invoicesOverdue, ... }
    },
    revenue: {
      thisMonth: metrics.revenueThisMonthCents,
      lastMonth: metrics.revenueLastMonthCents,
      outstanding: metrics.outstandingCents,
      overdue: metrics.overdueAmountCents
    },
    conversions: {
      winRate: metrics.winRatePct / 100,
      avgCycleDays: metrics.avgCycleDays
    }
  };
}
```
