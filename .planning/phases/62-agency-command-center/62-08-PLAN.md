```typescript
export class AnalyticsService {
  constructor(
    private readonly dealOutcomeRepo: DealOutcomeRepository,
    private readonly db: DrizzleClient
  ) {}

  async getWinLossAnalytics(workspaceId: string, dateRange?: DateRange): Promise<WinLossAnalytics> {
    const [outcomes, lossReasons, competitors, avgCycle] = await Promise.all([
      this.dealOutcomeRepo.countByOutcome(workspaceId, dateRange),
      this.dealOutcomeRepo.groupByLossReason(workspaceId),
      this.dealOutcomeRepo.getTopCompetitors(workspaceId, 5),
      this.getAvgCycleTime(workspaceId, dateRange)
    ]);

    const total = outcomes.won + outcomes.lost;
    const winRate = total > 0 ? (outcomes.won / total) * 100 : 0;

    return {
      summary: {
        totalDeals: total,
        won: outcomes.won,
        lost: outcomes.lost,
        winRate,
        avgCycleDays: avgCycle
      },
      lossReasons: lossReasons.map(r => ({
        reason: r.reason,
        count: r.count,
        percentage: outcomes.lost > 0 ? (r.count / outcomes.lost) * 100 : 0
      })),
      topCompetitors: competitors
    };
  }

  async getAvgCycleTime(workspaceId: string, dateRange?: DateRange): Promise<number> {
    const result = await this.db
      .select({ avg: sql<number>`AVG(${dealOutcomes.cycleDays})` })
      .from(dealOutcomes)
      .where(and(
        eq(dealOutcomes.workspaceId, workspaceId),
        eq(dealOutcomes.outcome, 'won'),
        dateRange ? gte(dealOutcomes.outcomeAt, dateRange.from) : undefined
      ));

    return Math.round(result[0]?.avg ?? 0);
  }
}
```

```typescript
'use client';

import { PieChart, Pie, Cell, ResponsiveContainer, Tooltip, Legend } from 'recharts';
import { useTranslation } from '@/hooks/useTranslation';

const COLORS = ['#ef4444', '#f97316', '#eab308', '#22c55e', '#3b82f6', '#8b5cf6', '#ec4899', '#6b7280'];

export function LossReasonChart({ data }: { data: { reason: string; count: number; percentage: number }[] }) {
  const { t } = useTranslation('command-center');

  const chartData = data.map(d => ({
    name: t(`lossReasons.${d.reason}`),
    value: d.count,
    percentage: d.percentage

    <ResponsiveContainer width="100%" height={250}>
      <PieChart>
        <Pie
          data={chartData}
          cx="50%"
          cy="50%"
          innerRadius={60}
          outerRadius={80}
          dataKey="value"
          label={({ name, percentage }) => `${name}: ${percentage.toFixed(0)}%`}

            <Cell key={`cell-${index}`} fill={COLORS[index % COLORS.length]} />
          ))}
        </Pie>
        <Tooltip formatter={(value: number) => value} />
      </PieChart>
    </ResponsiveContainer>
  );
}
```
