export class WorkflowExecutor {
  constructor(
    private readonly workflowRepo: WorkflowRepository,
    private readonly engagementService: EngagementService,
    private readonly followUpService: FollowUpService,
    private readonly emailService: EmailService,
    private readonly alertService: AlertService,
    private readonly queue: Queue<WorkflowJobData>
  ) {}

  async executeStep(instanceId: string): Promise<StepResult> {
    const instance = await this.workflowRepo.findById(instanceId);
    if (!instance || instance.status !== 'active') return { skipped: true };

    const template = await this.workflowRepo.getTemplateById(instance.templateId);
    const steps = template.steps as WorkflowStep[];
    const currentStep = steps[instance.currentStep];

    if (!currentStep) {
      // All steps completed
      await this.engagementService.completeWorkflow(instanceId, 'completed');
      return { completed: true };
    }

    // Check anti-annoyance before touch steps
    if (['email', 'task', 'alert'].includes(currentStep.type)) {
      const canTouch = await this.engagementService.canExecuteTouch(instance);
      if (!canTouch) {
        // Reschedule for later (after cooldown)
        await this.scheduleWithDelay(instanceId, template.cooldownHours * 60 * 60 * 1000);
        return { rescheduled: true, reason: 'anti_annoyance' };
      }
    }

    const result = await this.executeStepByType(currentStep, instance, template);
    await this.workflowRepo.logEvent(instanceId, {
      eventType: 'step_executed',
      stepIndex: instance.currentStep,
      actionTaken: currentStep.type,
      result
    });

    return result;
  }

  private async executeStepByType(step: WorkflowStep, instance, template): Promise<StepResult>;
  private async executeWaitStep(config: WaitConfig, instanceId: string): Promise<StepResult>;
  private async executeEmailStep(config: EmailConfig, instance, context): Promise<StepResult>;
  private async executeTaskStep(config: TaskConfig, instance): Promise<StepResult>;
  private async executeConditionStep(config: ConditionConfig, instance): Promise<StepResult>;
  private async executeAlertStep(config: AlertConfig, instance): Promise<StepResult>;
  private async executeWebhookStep(config: WebhookConfig, instance): Promise<StepResult>;

  // Template interpolation: {{client.name}}, {{proposal.id}}, etc.
  interpolateTemplate(template: string, context: Record<string, unknown>): string;
}
```

export interface WorkflowJobData {
  type: 'execute_step' | 'unsnooze' | 'reset_weekly_touches' | 'start_from_trigger';
  instanceId?: string;
  workspaceId?: string;
  templateId?: string;
  entityType?: EntityType;
  entityId?: string;
}

export const workflowQueue = new Queue<WorkflowJobData>('workflow', {
  connection: redisConnection,
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 2000 }
  }
});

// Schedule weekly touch reset (Mondays at midnight)
export async function scheduleWorkflowMaintenance() {
  await workflowQueue.add('reset-touches', { type: 'reset_weekly_touches' }, {
    repeat: { pattern: '0 0 * * 1' }, // Every Monday at midnight
    jobId: 'workflow-weekly-reset'
  });
}
```
