interface EntityEvent {
  workspaceId: string;
  entityType: EntityType;
  entityId: string;
  eventType: 'status_changed' | 'created' | 'updated' | 'overdue';
  previousStatus?: string;
  currentStatus?: string;
  metadata?: Record<string, unknown>;
}

export class RulesEngine {
  constructor(
    private readonly rulesRepo: FollowUpRulesRepository,
    private readonly followUpService: FollowUpService,
    private readonly queue: Queue<FollowUpJobData>
  ) {}

  // Evaluate a single rule's trigger conditions against an event
  evaluateTriggerConditions(
    conditions: TriggerConditions,
    event: EntityEvent,
    entityData: Record<string, unknown>
  ): boolean;

  // Process an entity event and create follow-ups for matching rules
  async processEntityEvent(event: EntityEvent): Promise<FollowUp[]>;

  // Schedule a delayed follow-up based on rule config
  async scheduleDelayedFollowUp(
    rule: FollowUpRuleSelect,
    event: EntityEvent
  ): Promise<void>;
}
```

Condition evaluation logic:
- `status_changed_to`: Compare event.currentStatus
- `status_equals`: Compare entityData.status
- `days_since`: Calculate days from entityData.created_at or last_updated
- `days_overdue_gte`: For invoices, calculate days past due_date
- `value_gte_cents`: Compare entityData.total_value_cents or deal_value_cents

When rule matches:
1. If delay_hours = 0, create follow-up immediately
2. If delay_hours > 0, schedule BullMQ job with delay
3. Log rule execution for debugging

export interface FollowUpJobData {
  type: 'create_scheduled' | 'process_due' | 'evaluate_rules';
  workspaceId: string;
  ruleId?: string;
  entityType?: EntityType;
  entityId?: string;
  followUpData?: Partial<FollowUpInsert>;
}

export const followUpQueue = new Queue<FollowUpJobData>('follow-up', {
  connection: redisConnection,
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 1000 }
  }
});

// Schedule recurring job to process due follow-ups every 5 minutes
export async function scheduleFollowUpProcessing() {
  await followUpQueue.add('process-due', { type: 'process_due', workspaceId: 'all' }, {
    repeat: { every: 5 * 60 * 1000 },
    jobId: 'follow-up-due-processor'
  });
}
```

**follow-up-processor.ts:**
- `processFollowUpJob(job: Job<FollowUpJobData>)`: Main processor
- Handle 'create_scheduled': Create follow-up from rule after delay
- Handle 'process_due': Find all due follow-ups and send notifications
- Handle 'evaluate_rules': Re-evaluate rules for an entity (after status change)
