```typescript
export const ALERT_RULES: AlertRule[] = [
  {
    type: 'high_value_stuck',
    name: 'High-value deal stuck',
    severity: 'high',
    detectFn: async (metrics, workspace, db) => {
      const stuckDeals = await db.query.proposals.findMany({
        where: and(
          eq(proposals.workspaceId, workspace.id),
          inArray(proposals.status, ['sent', 'viewed']),
          gte(proposals.totalValueCents, 500000), // 5000 EUR
          lte(proposals.updatedAt, subDays(new Date(), 7))
        ),
        orderBy: desc(proposals.totalValueCents),
        limit: 1
      });

      if (stuckDeals.length === 0) return null;

      const deal = stuckDeals[0];
      const daysSinceUpdate = Math.floor((Date.now() - deal.updatedAt.getTime()) / (1000 * 60 * 60 * 24));

      return {
        workspaceId: workspace.id,
        alertType: 'high_value_stuck',
        severity: 'high',
        title: 'High-value deal stuck',
        description: `Proposal for ${deal.clientName} (${formatCurrency(deal.totalValueCents)}) has no activity in ${daysSinceUpdate} days`,
        entityType: 'proposal',
        entityId: deal.id,
        metricCurrent: daysSinceUpdate,
        metricUnit: 'days',
        suggestedAction: 'Review and follow up',
        actionUrl: `/proposals/${deal.id}`
      };
    }
  },
  {
    type: 'win_rate_declining',
    name: 'Win rate declining',
    severity: 'medium',
    detectFn: async (metrics, workspace) => {
      // Compare current 30d win rate to previous 30d
      if (!metrics.winRatePreviousPct) return null;
      if (metrics.winRatePct >= metrics.winRatePreviousPct - 500) return null; // 5% threshold

      return {
        workspaceId: workspace.id,
        alertType: 'win_rate_declining',
        severity: 'medium',
        title: 'Win rate declining',
        description: `Win rate dropped from ${(metrics.winRatePreviousPct / 100).toFixed(1)}% to ${(metrics.winRatePct / 100).toFixed(1)}%`,
        metricCurrent: metrics.winRatePct / 100,
        metricPrevious: metrics.winRatePreviousPct / 100,
        metricUnit: '%',
        suggestedAction: 'Review recent lost deals for patterns'
      };
    }
  },
  {
    type: 'contract_expiring_soon',
    name: 'Contracts expiring soon',
    severity: 'high',
    detectFn: async (metrics, workspace, db) => {
      const expiring = await db.query.contracts.findMany({
        where: and(
          eq(contracts.workspaceId, workspace.id),
          eq(contracts.status, 'executed'),
          lte(contracts.expiresAt, addDays(new Date(), 14)),
          gte(contracts.expiresAt, new Date())
        )
      });

      if (expiring.length === 0) return null;

      return {
        workspaceId: workspace.id,
        alertType: 'contract_expiring_soon',
        severity: 'high',
        title: `${expiring.length} contract(s) expiring soon`,
        description: `${expiring.length} contracts will expire in the next 14 days`,
        metricCurrent: expiring.length,
        metricUnit: 'contracts',
        suggestedAction: 'Review and send renewal proposals'
      };
    }
  },
  {
    type: 'unassigned_prospects',
    name: 'Unassigned prospects',
    severity: 'low',
    detectFn: async (metrics, workspace, db) => {
      const unassigned = await db.query.prospects.findMany({
        where: and(
          eq(prospects.workspaceId, workspace.id),
          isNull(prospects.assignedTo),
          gte(prospects.createdAt, subDays(new Date(), 2))
        )
      });

      if (unassigned.length < 3) return null;

      return {
        workspaceId: workspace.id,
        alertType: 'unassigned_prospects',
        severity: 'low',
        title: 'Unassigned prospects',
        description: `${unassigned.length} prospects without an owner`,
        metricCurrent: unassigned.length,
        metricUnit: 'prospects',
        suggestedAction: 'Assign team members to new prospects'
      };
    }
  },
  {
    type: 'collection_velocity_drop',
    name: 'Payment velocity drop',
    severity: 'medium',
    detectFn: async (metrics, workspace) => {
      if (!metrics.avgCollectionDaysHistorical) return null;
      if (metrics.avgCollectionDays <= metrics.avgCollectionDaysHistorical + 5) return null;

      return {
        workspaceId: workspace.id,
        alertType: 'collection_velocity_drop',
        severity: 'medium',
        title: 'Payment velocity drop',
        description: `Average collection time increased from ${metrics.avgCollectionDaysHistorical} to ${metrics.avgCollectionDays} days`,
        metricCurrent: metrics.avgCollectionDays,
        metricPrevious: metrics.avgCollectionDaysHistorical,
        metricUnit: 'days',
        suggestedAction: 'Review overdue invoices and follow up'
      };
    }
  }

export class AlertDetectionService {
  constructor(
    private readonly alertRepo: SmartAlertRepository,
    private readonly db: DrizzleClient,
    private readonly notificationService: NotificationService
  ) {}

  async detectAlerts(workspaceId: string): Promise<void> {
    const [workspace, metrics] = await Promise.all([
      this.db.query.organizations.findFirst({ where: eq(organizations.id, workspaceId) }),
      this.db.query.pipelineMetrics.findFirst({ where: eq(pipelineMetrics.workspaceId, workspaceId) })
    ]);

    if (!workspace || !metrics) return;

    for (const rule of ALERT_RULES) {
      await this.evaluateRule(rule, metrics, workspace);
    }
  }

  private async evaluateRule(rule: AlertRule, metrics: PipelineMetricsSelect, workspace: Organization) {
    const existing = await this.alertRepo.findActiveByType(workspace.id, rule.type);
    const newAlert = await rule.detectFn(metrics, workspace, this.db);

    if (newAlert && !existing) {
      await this.alertRepo.create(newAlert);
      await this.notificationService.sendAlertNotification(workspace.id, newAlert);
    } else if (!newAlert && existing) {
      await this.alertRepo.resolve(existing.id);
    }
  }
}
```

```typescript
export async function processAlertDetection(job: Job<AlertDetectionJobData>) {
  const logger = createLogger({ module: 'alert-detection', jobId: job.id });

  if (job.data.type === 'detect_workspace' && job.data.workspaceId) {
    await alertService.detectAlerts(job.data.workspaceId);
    return;
  }

  if (job.data.type === 'detect_all') {
    const workspaces = await db.query.organizations.findMany({
      where: eq(organizations.isActive, true)
    });

    for (const workspace of workspaces) {
      try {
        await alertService.detectAlerts(workspace.id);
      } catch (error) {
        logger.error(`Alert detection failed for ${workspace.id}`, error);
      }
    }
  }
}
```

```typescript
import { Server } from 'socket.io';

let io: Server | null = null;

export function initActivityFeed(server: Server) {
  io = server;

  io.on('connection', (socket) => {
    const workspaceId = socket.handshake.query.workspaceId as string;
    if (workspaceId) {
      socket.join(`workspace:${workspaceId}`);
    }

    socket.on('disconnect', () => {
      // Cleanup if needed
    });
  });
}

export interface ActivityEvent {
  id: string;
  type: string;
  entityType: string;
  entityId: string;
  title: string;
  description?: string;
  timestamp: Date;
  userId?: string;
  userName?: string;
}

export function emitActivityEvent(workspaceId: string, event: Omit<ActivityEvent, 'id'>) {
  if (!io) return;

  const fullEvent: ActivityEvent = {
    ...event,
    id: nanoid(),
    timestamp: event.timestamp || new Date()
  };

  io.to(`workspace:${workspaceId}`).emit('activity:new', fullEvent);
}
```

```typescript
'use client';

import { useEffect, useState } from 'react';
import { io, Socket } from 'socket.io-client';

export function useActivityFeed(workspaceId: string) {
  const [activities, setActivities] = useState<ActivityEvent[]>([]);
  const [isConnected, setIsConnected] = useState(false);

  useEffect(() => {
    const socket: Socket = io(process.env.NEXT_PUBLIC_WS_URL!, {
      query: { workspaceId },
      transports: ['websocket']
    });

    socket.on('connect', () => setIsConnected(true));
    socket.on('disconnect', () => setIsConnected(false));

    socket.on('activity:new', (event: ActivityEvent) => {
      setActivities(prev => [event, ...prev.slice(0, 49)]); // Keep last 50
    });

    return () => { socket.disconnect(); };
  }, [workspaceId]);

  return { activities, isConnected };
}
```

```typescript
'use client';

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { dismissAlert } from '../actions';

export function useSmartAlerts(workspaceId: string) {
  const queryClient = useQueryClient();

  const query = useQuery({
    queryKey: ['smart-alerts', workspaceId],
    queryFn: async () => {
      const res = await fetch('/api/command-center/alerts', {
        headers: { 'X-Workspace-Id': workspaceId }
      });
      return res.json() as Promise<{ alerts: SmartAlertSelect[] }>;
    },
    refetchInterval: 60 * 1000
  });

  const dismiss = useMutation({
    mutationFn: dismissAlert,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['smart-alerts', workspaceId] });
    }
  });

  return { ...query, dismiss };
}
```
