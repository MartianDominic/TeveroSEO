```typescript
import { useQuery } from '@tanstack/react-query';

export function useNeedsAttention(workspaceId: string) {
  return useQuery({
    queryKey: ['needs-attention', workspaceId],
    queryFn: async () => {
      const res = await fetch('/api/command-center/needs-attention', {
        headers: { 'X-Workspace-Id': workspaceId }
      });
      if (!res.ok) throw new Error('Failed to fetch');
      return res.json() as Promise<{ items: AttentionItem[] }>;
    },
    refetchInterval: 60 * 1000, // Refresh every minute
  });
}
```

```typescript
'use server';

import { revalidatePath } from 'next/cache';
import { z } from 'zod';

const sendReminderSchema = z.object({
  entityType: z.enum(['prospect', 'proposal', 'contract', 'invoice']),
  entityId: z.string(),
  message: z.string().optional()
});

export async function sendReminder(data: z.infer<typeof sendReminderSchema>) {
  const validated = sendReminderSchema.parse(data);

  const response = await fetch(`${process.env.OPEN_SEO_API_URL}/api/command-center/actions/send-reminder`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(validated)
  });

  if (!response.ok) {
    const error = await response.text();
    throw new Error(error || 'Failed to send reminder');
  }

  revalidatePath('/command-center');
  return { success: true };
}

const snoozeSchema = z.object({
  entityType: z.enum(['prospect', 'proposal', 'contract', 'invoice', 'follow_up']),
  entityId: z.string(),
  snoozedUntil: z.string().datetime(),
  reason: z.string().optional()
});

export async function snoozeFollowUp(data: z.infer<typeof snoozeSchema>) {
  const validated = snoozeSchema.parse(data);

  const response = await fetch(`${process.env.OPEN_SEO_API_URL}/api/command-center/actions/snooze`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(validated)
  });

  if (!response.ok) throw new Error('Failed to snooze');

  revalidatePath('/command-center');
  return { success: true };
}

const markLostSchema = z.object({
  entityType: z.enum(['prospect', 'proposal']),
  entityId: z.string(),
  reason: z.enum([
    'too_expensive', 'budget_cut', 'competitor_cheaper',
    'bad_timing', 'project_delayed', 'internal_changes',
    'wrong_fit', 'scope_mismatch', 'different_direction',
    'chose_competitor', 'went_internal', 'found_alternative',
    'unresponsive', 'ghosted', 'decision_maker_left',
    'unknown', 'other'
  ]),
  notes: z.string().optional(),

});

export async function markAsLost(data: z.infer<typeof markLostSchema>) {
  const validated = markLostSchema.parse(data);

    headers: { 'Content-Type': 'application/json' },

  });

  if (!response.ok) throw new Error('Failed to mark as lost');

  revalidatePath('/command-center');
```

```typescript
'use client';

import { useState, useTransition } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Textarea } from '@/components/ui/textarea';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
import { Calendar } from '@/components/ui/calendar';
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
import { format } from 'date-fns';
import { CalendarIcon, Loader2 } from 'lucide-react';
import { sendReminder, snoozeFollowUp, markAsLost, addNote } from '../actions';
import { toast } from 'sonner';

const LOSS_REASONS = [
  { value: 'too_expensive', label: 'Too expensive' },
  { value: 'budget_cut', label: 'Budget cut' },
  { value: 'competitor_cheaper', label: 'Competitor cheaper' },
  { value: 'bad_timing', label: 'Bad timing' },
  { value: 'chose_competitor', label: 'Chose competitor' },
  { value: 'unresponsive', label: 'Unresponsive' },
  { value: 'ghosted', label: 'Ghosted' },
  { value: 'other', label: 'Other' },
];

interface Props {
  item: AttentionItem;
  actionType: 'reminder' | 'lost' | 'snooze' | 'note';
  onClose: () => void;
}

export function QuickActionDialog({ item, actionType, onClose }: Props) {
  const [isPending, startTransition] = useTransition();
  const [message, setMessage] = useState('');
  const [snoozeDate, setSnoozeDate] = useState<Date | undefined>();
  const [lossReason, setLossReason] = useState('');
  const [competitorName, setCompetitorName] = useState('');

  const handleSubmit = () => {
    startTransition(async () => {
      try {
        switch (actionType) {
          case 'reminder':
            await sendReminder({ entityType: item.entityType, entityId: item.entityId, message });
            toast.success('Reminder sent');
            break;
          case 'snooze':
            if (!snoozeDate) return toast.error('Select a date');
            await snoozeFollowUp({
              entityType: item.entityType,
              entityId: item.entityId,
              snoozedUntil: snoozeDate.toISOString(),
              reason: message
            });
            toast.success(`Snoozed until ${format(snoozeDate, 'PPP')}`);
            break;
          case 'lost':
            if (!lossReason) return toast.error('Select a reason');
            await markAsLost({
              entityType: item.entityType as 'prospect' | 'proposal',
              entityId: item.entityId,
              reason: lossReason as any,
              notes: message,
              competitorName: competitorName || undefined
            });
            toast.success('Marked as lost');
            break;
          case 'note':
            if (!message.trim()) return toast.error('Enter a note');
            await addNote({ entityType: item.entityType, entityId: item.entityId, note: message });
            toast.success('Note added');
            break;
        }
        onClose();
      } catch (error) {
        toast.error(error instanceof Error ? error.message : 'Action failed');
      }
    });
  };

  const titles = {
    reminder: 'Send Reminder',
    snooze: 'Snooze Follow-up',
    lost: 'Mark as Lost',
    note: 'Add Note'
  };

  return (
    <Dialog open onOpenChange={() => onClose()}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>{titles[actionType]}</DialogTitle>
        </DialogHeader>

        <div className="space-y-4 py-4">
          {actionType === 'snooze' && (
            <div className="space-y-2">
              <label className="text-sm font-medium">Follow up on</label>
              <Popover>
                <PopoverTrigger asChild>
                  <Button variant="outline" className="w-full justify-start">
                    <CalendarIcon className="mr-2 h-4 w-4" />
                    {snoozeDate ? format(snoozeDate, 'PPP') : 'Pick a date'}
                  </Button>
                </PopoverTrigger>
                <PopoverContent className="w-auto p-0">
                  <Calendar mode="single" selected={snoozeDate} onSelect={setSnoozeDate} disabled={(date) => date < new Date()} />
                </PopoverContent>
              </Popover>
            </div>
          )}

          {actionType === 'lost' && (
            <>
              <div className="space-y-2">
                <label className="text-sm font-medium">Reason</label>
                <Select value={lossReason} onValueChange={setLossReason}>
                  <SelectTrigger><SelectValue placeholder="Select reason" /></SelectTrigger>
                  <SelectContent>
                    {LOSS_REASONS.map(r => (
                      <SelectItem key={r.value} value={r.value}>{r.label}</SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>
              {lossReason === 'chose_competitor' && (
                <Input placeholder="Competitor name" value={competitorName} onChange={e => setCompetitorName(e.target.value)} />
              )}
            </>
          )}

          <div className="space-y-2">
            <label className="text-sm font-medium">
              {actionType === 'note' ? 'Note' : 'Message (optional)'}
            </label>
            <Textarea
              value={message}
              onChange={e => setMessage(e.target.value)}
              placeholder={actionType === 'reminder' ? 'Custom message for reminder...' : 'Add details...'}
              rows={3}
            />
          </div>
        </div>

        <DialogFooter>
          <Button variant="outline" onClick={onClose} disabled={isPending}>Cancel</Button>
          <Button onClick={handleSubmit} disabled={isPending}>
            {isPending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
            {actionType === 'reminder' ? 'Send' : actionType === 'lost' ? 'Mark Lost' : 'Save'}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

```typescript
export class QuickActionService {
  constructor(
    private readonly db: DrizzleClient,
    private readonly emailService: EmailService,
    private readonly followUpService: FollowUpService,
    private readonly engagementService: EngagementService
  ) {}

  async sendReminder(workspaceId: string, entityType: EntityType, entityId: string, message?: string) {
    // Get entity details
    const entity = await this.getEntity(entityType, entityId);
    if (!entity || entity.workspaceId !== workspaceId) throw new Error('Entity not found');

    // Get contact email
    const contactEmail = await this.getContactEmail(entity);
    if (!contactEmail) throw new Error('No contact email');

    // Send reminder email
    await this.emailService.sendReminder({
      to: contactEmail,
      entityType,
      entityId,
      entityTitle: entity.title || entity.name,
      customMessage: message
    });

    // Log activity
    await this.logActivity(workspaceId, 'reminder_sent', entityType, entityId);
  }

  async snooze(workspaceId: string, entityType: EntityType, entityId: string, snoozedUntil: Date, reason?: string) {
    // If it's a workflow instance, snooze the workflow
    const workflow = await this.db.query.workflowInstances.findFirst({
      where: and(
        eq(workflowInstances.entityType, entityType),
        eq(workflowInstances.entityId, entityId),
        eq(workflowInstances.status, 'active')
      )
    });

    if (workflow) {
      await this.engagementService.snoozeWorkflow(workflow.id, snoozedUntil, reason);
    }

    // Also snooze any pending follow-ups for this entity
    await this.db.update(followUps)
      .set({ status: 'snoozed', snoozedUntil, updatedAt: new Date() })
      .where(and(
        eq(followUps.entityType, entityType),
        eq(followUps.entityId, entityId),
        eq(followUps.status, 'pending')
      ));

    await this.logActivity(workspaceId, 'snoozed', entityType, entityId, { snoozedUntil, reason });
  }

  async markAsLost(
    workspaceId: string,
    userId: string,
    entityType: 'prospect' | 'proposal',
    entityId: string,
    reason: LossReason,
    notes?: string,
    competitorName?: string
  ) {
    // Get entity for value tracking
    const entity = await this.getEntity(entityType, entityId);
    if (!entity || entity.workspaceId !== workspaceId) throw new Error('Entity not found');

    // Create deal outcome record
    await this.db.insert(dealOutcomes).values({
      id: nanoid(),
      workspaceId,
      entityType,
      entityId,
      outcome: 'lost',
      lossReason: reason,
      lossReasonDetail: notes,
      competitorName,
      dealValueCents: entity.totalValueCents,
      outcomeAt: new Date(),
      ownerId: userId
    });

    // Update entity status
    if (entityType === 'prospect') {
      await this.db.update(prospects).set({ status: 'archived' }).where(eq(prospects.id, entityId));
    } else {
      await this.db.update(proposals).set({ status: 'declined' }).where(eq(proposals.id, entityId));
    }

    // Cancel any active workflows
    const workflow = await this.db.query.workflowInstances.findFirst({
      where: and(
        eq(workflowInstances.entityType, entityType),
        eq(workflowInstances.entityId, entityId),
        inArray(workflowInstances.status, ['active', 'paused', 'snoozed'])
      )
    });

    if (workflow) {
      await this.engagementService.completeWorkflow(workflow.id, 'lost', reason);
    }

    await this.logActivity(workspaceId, 'marked_lost', entityType, entityId, { reason, notes });
  }

  async addNote(workspaceId: string, userId: string, entityType: EntityType, entityId: string, note: string) {
    // Add to entity's notes/activity log
    // Implementation depends on how notes are stored per entity type
    await this.logActivity(workspaceId, 'note_added', entityType, entityId, { note, addedBy: userId });
  }

  private async logActivity(workspaceId: string, type: string, entityType: EntityType, entityId: string, data?: unknown) {
    // Emit to activity feed via Socket.IO
    await emitActivityEvent(workspaceId, { type, entityType, entityId, data, timestamp: new Date() });
  }
}
```
