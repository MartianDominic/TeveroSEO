# Plan 73-03: Retrieval Quality Enhancement

## Goal

Improve retrieval quality through two complementary approaches:
1. **BGE Reranker** - Cross-encoder reranking for +3-8 recall@10
2. **Hierarchical Retrieval** - Category routing for better relevance

## Research Prescription

From `docs/infra-research/cpu-only-rag-graph.md`:

### BGE Reranker
> Add `bge-reranker-v2-m3` on top-50 candidates for an extra +3-8 points recall@10 at ~80 ms per (query, candidate) pair on AVX2 CPU.

### Hierarchical Retrieval
> Two-stage retrieval — category routing first, in-category dense search second — is the published industry pattern (Best Buy, JD.com UniERF, Taobao ULIM, eBay CoT-BFS, the CHARM paper arXiv 2501.18707).

```python
"""
BGE Reranker v2 M3 for cross-encoder reranking.
Adds +3-8 recall@10 at ~80ms per (query, candidate) pair.
"""
from sentence_transformers import CrossEncoder
from typing import List, Tuple
import torch

# Model: BAAI/bge-reranker-v2-m3 (multilingual, good for Lithuanian)
MODEL_NAME = "BAAI/bge-reranker-v2-m3"

class BGEReranker:
    def __init__(self, device: str = "cpu"):
        self.model = CrossEncoder(MODEL_NAME, device=device)
        self.device = device
    
    def rerank(
        self, 
        query: str, 
        candidates: List[str], 
        top_k: int = 10
    ) -> List[Tuple[int, float]]:
        """
        Rerank candidates by relevance to query.
        
        Args:
            query: Search query
            candidates: List of candidate texts (e.g., top-50 from vector search)
            top_k: Number of results to return
            
        Returns:
            List of (original_index, score) tuples, sorted by score descending
        """
        if not candidates:
            return []
        
        # Create query-candidate pairs
        pairs = [[query, candidate] for candidate in candidates]
        
        # Get relevance scores
        scores = self.model.predict(pairs, show_progress_bar=False)
        
        # Sort by score and return top_k
        indexed_scores = list(enumerate(scores))
        indexed_scores.sort(key=lambda x: x[1], reverse=True)
        
        return indexed_scores[:top_k]
    
    def rerank_with_metadata(
        self,
        query: str,
        candidates: List[dict],  # [{"text": str, "id": str, ...}, ...]
        text_key: str = "text",
        top_k: int = 10
    ) -> List[dict]:
        """
        Rerank candidates preserving metadata.
        """
        texts = [c[text_key] for c in candidates]
        reranked = self.rerank(query, texts, top_k)
        
        return [
            {**candidates[idx], "rerank_score": float(score)}
            for idx, score in reranked
        ]

# Singleton for reuse
_reranker = None

def get_reranker() -> BGEReranker:
    global _reranker
    if _reranker is None:
        _reranker = BGEReranker(device="cpu")
    return _reranker
```

```python
    start = time.perf_counter()
    
    reranker = get_reranker()
    results = reranker.rerank_with_metadata(
        query=request.query,
        candidates=request.candidates,
        text_key=request.text_key,
        top_k=request.top_k
    )
    
    latency_ms = (time.perf_counter() - start) * 1000
    return RerankResponse(results=results, latency_ms=latency_ms)
```

```typescript
import { embeddingService } from '../embeddings/embedding-service';

interface CategoryCentroid {
  categoryId: string;
  categoryName: string;
  centroid: Float32Array;  // Mean embedding of category products
  productCount: number;
}

export class CategoryRouter {
  private centroids: Map<string, CategoryCentroid> = new Map();
  
  /**
   * Build category centroids from product embeddings.
   * Call this periodically (e.g., daily) to update.
   */
  async buildCentroids(tenantId: string): Promise<void> {
    // Query all products grouped by category
    const categoryProducts = await db.query.productEmbeddings.findMany({
      where: eq(productEmbeddings.tenantId, tenantId),
      columns: { categoryId: true, embedding: true }
    });
    
    // Group by category
    const grouped = new Map<string, Float32Array[]>();
    for (const p of categoryProducts) {
      const list = grouped.get(p.categoryId) || [];
      list.push(new Float32Array(p.embedding));
      grouped.set(p.categoryId, list);
    }
    
    // Compute centroids
    for (const [categoryId, embeddings] of grouped) {
      const centroid = this.computeMeanEmbedding(embeddings);
      this.centroids.set(categoryId, {
        categoryId,
        categoryName: await this.getCategoryName(categoryId),
        centroid,
        productCount: embeddings.length
      });
    }
  }
  
  /**
   * Route a query to top-k most relevant categories.
   */
  async route(query: string, k: number = 3): Promise<string[]> {
    const queryEmbedding = await embeddingService.embed(query, 'query');
    
    const scores: { categoryId: string; score: number }[] = [];
    for (const [categoryId, cat] of this.centroids) {
      const score = this.cosineSimilarity(queryEmbedding, cat.centroid);
      scores.push({ categoryId, score });
    }
    
    // Sort by score descending
    scores.sort((a, b) => b.score - a.score);
    
    return scores.slice(0, k).map(s => s.categoryId);
  }
  
  private computeMeanEmbedding(embeddings: Float32Array[]): Float32Array {
    if (embeddings.length === 0) return new Float32Array(768);
    
    const mean = new Float32Array(embeddings[0].length);
    for (const emb of embeddings) {
      for (let i = 0; i < emb.length; i++) {
        mean[i] += emb[i];
      }
    }
    for (let i = 0; i < mean.length; i++) {
      mean[i] /= embeddings.length;
    }
    
    // Normalize
    const norm = Math.sqrt(mean.reduce((s, v) => s + v * v, 0));
    for (let i = 0; i < mean.length; i++) {
      mean[i] /= norm || 1;
    }
    
    return mean;
  }
  
  private cosineSimilarity(a: Float32Array, b: Float32Array): number {
    let dot = 0;
    for (let i = 0; i < a.length; i++) {
      dot += a[i] * b[i];
    }
    return dot; // Vectors are already normalized
  }
}
```

```typescript
import { CategoryRouter } from '../retrieval/category-router';
import { rerankCandidates } from '../retrieval/reranker-client';

export async function hierarchicalSearch(
  tenantId: string,
  query: string,
  options: {
    topCategories?: number;  // Default 3
    candidatesPerCategory?: number;  // Default 20
    finalTopK?: number;  // Default 10
    useReranker?: boolean;  // Default true
  } = {}
): Promise<SearchResult[]> {
  const {
    topCategories = 3,
    candidatesPerCategory = 20,
    finalTopK = 10,
    useReranker = true
  } = options;
  
  // Step 1: Route to top categories
  const router = getCategoryRouter(tenantId);
  const categoryIds = await router.route(query, topCategories);
  
  // Step 2: Search within each category
  const candidates: SearchResult[] = [];
  for (const categoryId of categoryIds) {
    const results = await vectorSearchInCategory(
      tenantId,
      query,
      categoryId,
      candidatesPerCategory
    );
    candidates.push(...results);
  }
  
  // Step 3: Rerank all candidates
  if (useReranker && candidates.length > finalTopK) {
    const reranked = await rerankCandidates(
      query,
      candidates.map(c => ({ ...c, text: c.content })),
      finalTopK
    );
    return reranked;
  }
  
  // Fallback: sort by vector score
  return candidates
    .sort((a, b) => b.score - a.score)
    .slice(0, finalTopK);
}
```
