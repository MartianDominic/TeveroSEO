# Plan 73-01: Deficit Round Robin Fair Queuing

## Goal

Implement weighted fair queuing so no single client can monopolize queue resources. A client with 100 jobs should not starve a client with 10 jobs.

## Research Prescription

From `docs/infra-research/crawling-10-5000-tasks-day.md`:

> Fair queuing across `client_id` buckets uses Deficit Round Robin — provably gives each client `weight_i / Σ weights` of throughput, O(1) dequeue, and lets you tune priority at runtime.

## Current State

- `priority-score.ts` exists but handles client priority scoring, not queue scheduling
- BullMQ handles queuing but has no weighted fairness mechanism
- One heavy client could theoretically consume all worker slots

```typescript
/**
 * Deficit Round Robin queue manager for multi-tenant fairness.
 * 
 * Each client bucket gets a "deficit counter" that accumulates.
 * When a bucket's deficit >= job cost, a job is dequeued.
 * This ensures proportional throughput across clients.
 */

interface DRRBucket {
  clientId: string;
  weight: number;       // Default 1.0, can be tuned per client
  deficit: number;      // Accumulated deficit counter
  pendingJobs: number;  // Jobs waiting in this bucket
}

interface DRRConfig {
  quantum: number;           // Deficit increment per round (default: 100)
  maxDeficit: number;        // Prevent runaway accumulation (default: 1000)
  heavyClientThreshold: number; // % of daily volume to trigger weight reduction
}

export class DRRQueueManager {
  private buckets: Map<string, DRRBucket> = new Map();
  private redis: Redis;
  
  constructor(redis: Redis, config?: Partial<DRRConfig>) {
    this.config = { quantum: 100, maxDeficit: 1000, heavyClientThreshold: 0.3, ...config };
    this.redis = redis;
  }
  
  /**
   * Register a job for a client. Creates bucket if needed.
   */
  async registerJob(clientId: string, jobId: string, cost: number = 100): Promise<void> {
    const bucket = this.getOrCreateBucket(clientId);
    bucket.pendingJobs++;
    await this.redis.zadd(`drr:pending:${clientId}`, Date.now(), jobId);
  }
  
  /**
   * Get next job to process using DRR algorithm.
   * Returns null if no jobs available.
   */
  async getNextJob(): Promise<{ clientId: string; jobId: string } | null> {
    // Round-robin through buckets, incrementing deficit
    for (const [clientId, bucket] of this.buckets) {
      if (bucket.pendingJobs === 0) continue;
      
      bucket.deficit += this.config.quantum * bucket.weight;
      bucket.deficit = Math.min(bucket.deficit, this.config.maxDeficit);
      
      if (bucket.deficit >= 100) { // Job cost
        bucket.deficit -= 100;
        bucket.pendingJobs--;
        const jobId = await this.redis.zpopmin(`drr:pending:${clientId}`);
        if (jobId) return { clientId, jobId: jobId[0] };
      }
    }
    return null;
  }
  
  /**
   * Adjust weight for a client (e.g., reduce for heavy users)
   */
  setWeight(clientId: string, weight: number): void {
    const bucket = this.buckets.get(clientId);
    if (bucket) bucket.weight = Math.max(0.1, Math.min(2.0, weight));
  }
  
  /**
   * Check if any client exceeds heavy threshold and auto-reduce weight
   */
  async enforceHeavyClientLimits(): Promise<string[]> {
    const dailyVolume = await this.getDailyVolumeByClient();
    const total = Object.values(dailyVolume).reduce((a, b) => a + b, 0);
    const reduced: string[] = [];
    
    for (const [clientId, volume] of Object.entries(dailyVolume)) {
      if (volume / total > this.config.heavyClientThreshold) {
        this.setWeight(clientId, 0.5); // Reduce to half weight
        reduced.push(clientId);
      }
    }
    return reduced;
  }
}
```
