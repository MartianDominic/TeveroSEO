```python
@router.get("/{client_id}/full", response_model=AnalyticsData)
async def get_client_analytics(
    client_id: UUID,
    days: int = Query(30, ge=7, le=90),
    db: Session = Depends(get_shared_db),
):
    """
    Get full analytics data for a specific client.
    
    Returns:
    - gsc_daily: Daily GSC metrics for the period
    - gsc_summary: Aggregated GSC metrics
    - ga4_daily: Daily GA4 metrics for the period
    - ga4_summary: Aggregated GA4 metrics
    - top_queries: Top 10 queries with position WoW change
    """
    client_id_str = str(client_id)
    
    # GSC daily data
    gsc_daily_query = text("""
        SELECT 
            date::text,
            clicks,
            impressions,
            ctr,
            position
        FROM gsc_snapshots
        WHERE client_id = :client_id
          AND date >= NOW() - INTERVAL ':days days'
        ORDER BY date ASC
    """.replace(":days", str(days)))
    
    gsc_rows = db.execute(gsc_daily_query, {"client_id": client_id_str}).fetchall()
    gsc_daily = [
        GSCDataPoint(
            date=row.date,
            clicks=row.clicks,
            impressions=row.impressions,
            ctr=round(row.ctr, 4),
            position=round(row.position, 1),
        )
        for row in gsc_rows
    ]
    
    # GSC summary
    gsc_summary_query = text("""
        SELECT 
            COALESCE(SUM(clicks), 0)::int as clicks,
            COALESCE(SUM(impressions), 0)::int as impressions,
            COALESCE(AVG(ctr), 0)::float as ctr,
            COALESCE(AVG(position), 0)::float as position
        FROM gsc_snapshots
        WHERE client_id = :client_id
          AND date >= NOW() - INTERVAL ':days days'
    """.replace(":days", str(days)))
    
    gsc_sum_row = db.execute(gsc_summary_query, {"client_id": client_id_str}).fetchone()
    gsc_summary = GSCSummary(
        clicks=gsc_sum_row.clicks,
        impressions=gsc_sum_row.impressions,
        ctr=round(gsc_sum_row.ctr, 4),
        position=round(gsc_sum_row.position, 1),
    )
    
    # GA4 daily data
    ga4_daily_query = text("""
        SELECT 
            date::text,
            sessions,
            users,
            bounce_rate
        FROM ga4_snapshots
        WHERE client_id = :client_id
          AND date >= NOW() - INTERVAL ':days days'
        ORDER BY date ASC
    """.replace(":days", str(days)))
    
    ga4_rows = db.execute(ga4_daily_query, {"client_id": client_id_str}).fetchall()
    ga4_daily = [
        GA4DataPoint(
            date=row.date,
            sessions=row.sessions,
            users=row.users,
            bounce_rate=round(row.bounce_rate, 2),
        )
        for row in ga4_rows
    ]
    
    # GA4 summary
    ga4_summary_query = text("""
        SELECT 
            COALESCE(SUM(sessions), 0)::int as sessions,
            COALESCE(SUM(users), 0)::int as users,
            COALESCE(SUM(conversions), 0)::int as conversions,
            COALESCE(AVG(bounce_rate), 0)::float as bounce_rate
        FROM ga4_snapshots
        WHERE client_id = :client_id
          AND date >= NOW() - INTERVAL ':days days'
    """.replace(":days", str(days)))
    
    ga4_sum_row = db.execute(ga4_summary_query, {"client_id": client_id_str}).fetchone()
    ga4_summary = GA4Summary(
        sessions=ga4_sum_row.sessions,
        users=ga4_sum_row.users,
        conversions=ga4_sum_row.conversions,
        bounce_rate=round(ga4_sum_row.bounce_rate, 2),
    )
    
    # Top queries with position delta (comparing last 7d to prior 7d)
    top_queries_query = text("""
        WITH recent AS (
            SELECT 
                query,
                SUM(clicks) as clicks,
                SUM(impressions) as impressions,
                AVG(ctr) as ctr,
                AVG(position) as position
            FROM gsc_query_snapshots
            WHERE client_id = :client_id
              AND date >= NOW() - INTERVAL '7 days'
            GROUP BY query
        ),
        prior AS (
            SELECT 
                query,
                AVG(position) as position
            FROM gsc_query_snapshots
            WHERE client_id = :client_id
              AND date >= NOW() - INTERVAL '14 days'
              AND date < NOW() - INTERVAL '7 days'
            GROUP BY query
        )
        SELECT 
            r.query,
            r.clicks::int,
            r.impressions::int,
            r.ctr::float,
            r.position::float,
            COALESCE(r.position - p.position, 0)::float as position_delta
        FROM recent r
        LEFT JOIN prior p ON r.query = p.query
        ORDER BY r.clicks DESC
        LIMIT 10
    """)
    
    query_rows = db.execute(top_queries_query, {"client_id": client_id_str}).fetchall()
    top_queries = [
        TopQuery(
            query=row.query,
            clicks=row.clicks,
            impressions=row.impressions,
            ctr=round(row.ctr, 4),
            position=round(row.position, 1),
            position_delta=round(row.position_delta, 1),
        )
        for row in query_rows
    ]
    
    return AnalyticsData(
        gsc_daily=gsc_daily,
        gsc_summary=gsc_summary,
        ga4_daily=ga4_daily,
        ga4_summary=ga4_summary,
        top_queries=top_queries,
    )
```

import {

  ResponsiveContainer,

import type { GSCDataPoint } from "@/lib/analytics/types";

interface GSCChartProps {
  data: GSCDataPoint[];
}

export function GSCChart({ data }: GSCChartProps) {
  const formatDate = (dateStr: string) => {
    const date = new Date(dateStr);
    return date.toLocaleDateString("en-US", { month: "short", day: "numeric" });

  };

  return (

      <LineChart data={data} margin={{ top: 5, right: 30, left: 20, bottom: 5 }}>
        <CartesianGrid strokeDasharray="3 3" className="stroke-muted" />

          dataKey="date"
          tick={{ fontSize: 12 }}
          tickFormatter={formatDate}
          className="text-muted-foreground"
        />

          yAxisId="left"
          orientation="left"
          tick={{ fontSize: 12 }}
          tickFormatter={formatNumber}

"use client";

import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@tevero/ui";
import { TrendingUp, TrendingDown, Minus } from "lucide-react";
import type { TopQuery } from "@/lib/analytics/types";

interface QueriesTableProps {
  queries: TopQuery[];
}

function PositionDelta({ delta }: { delta: number }) {
  // Negative delta = improved (position went from e.g. 10 to 5)
  // Positive delta = worsened (position went from e.g. 5 to 10)
  if (Math.abs(delta) < 0.1) {
    return <Minus className="h-4 w-4 text-muted-foreground" />;
  }
  if (delta < 0) {
    return (
      <span className="flex items-center text-emerald-600">
        <TrendingUp className="h-4 w-4 mr-1" />
        {Math.abs(delta).toFixed(1)}
      </span>
    );
  }
  return (
    <span className="flex items-center text-red-600">
      <TrendingDown className="h-4 w-4 mr-1" />
      {delta.toFixed(1)}
    </span>
  );
}

export function QueriesTable({ queries }: QueriesTableProps) {
  if (queries.length === 0) {
    return (
      <div className="text-center py-8 text-muted-foreground">
        No query data available
      </div>
    );
  }

  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead className="w-[250px]">Query</TableHead>
          <TableHead className="text-right">Clicks</TableHead>
          <TableHead className="text-right">Impressions</TableHead>
          <TableHead className="text-right">CTR</TableHead>
          <TableHead className="text-right">Position</TableHead>
          <TableHead className="text-right">WoW</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {queries.map((q) => (
          <TableRow key={q.query}>
            <TableCell
              className="font-medium max-w-[250px] truncate"
              title={q.query}
            >
              {q.query}
            </TableCell>
            <TableCell className="text-right tabular-nums">
              {q.clicks.toLocaleString()}
            </TableCell>
            <TableCell className="text-right tabular-nums">
              {q.impressions.toLocaleString()}
            </TableCell>
            <TableCell className="text-right tabular-nums">
              {(q.ctr * 100).toFixed(1)}%
            </TableCell>
            <TableCell className="text-right tabular-nums">
              {q.position.toFixed(1)}
            </TableCell>
            <TableCell className="text-right">
              <PositionDelta delta={q.position_delta} />
            </TableCell>
          </TableRow>
        ))}
      </TableBody>
    </Table>
  );
}
