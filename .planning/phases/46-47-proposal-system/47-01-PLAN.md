// Add to existing actions.ts

export interface AIRecommendations {
  awarenessLevel: AwarenessLevel;
  confidence: number;
  hookStrategy: string;
  recommendedApproach: {
    openingAngle: string;
    primaryCialdini: string;
    objectionsToAddress: string[];
  };
  keywordHighlights: {
    totalKeywords: number;
    quickWins: number;
    highValueKeywords: Array<{
      keyword: string;
      volume: number;
      potential: number;
    }>;
    topCompetitor: string | null;
    estimatedTrafficGain: number;
  };
}

export async function getAIRecommendations(
  prospectId: string
): Promise<{ success: boolean; data?: AIRecommendations; error?: string }> {
  const openSeoUrl = process.env.OPEN_SEO_API_URL || "http://localhost:3001";

  try {
    const response = await fetch(
      `${openSeoUrl}/api/prospects/${prospectId}/recommendations`,
      { cache: "no-store" }
    );

    if (!response.ok) {
      // Fallback if recommendations endpoint doesn't exist yet
      return {
        success: true,
        data: {
          awarenessLevel: "solution-aware",
          confidence: 0.7,
          hookStrategy: "Show immediate value through quick wins",
          recommendedApproach: {
            openingAngle: "Your competitors are ranking for keywords you're missing",
            primaryCialdini: "Social proof + scarcity",
            objectionsToAddress: ["Cost", "Timeline", "Results guarantee"],
          },
          keywordHighlights: {
            totalKeywords: 0,
            quickWins: 0,
            highValueKeywords: [],
            topCompetitor: null,
            estimatedTrafficGain: 0,
          },
        },
      };
    }

    const result = await response.json();
    return { success: true, data: result.data };
  } catch (error) {
    return { success: false, error: "Failed to fetch recommendations" };
  }
}

import { Lightbulb, TrendingUp, Target, Users, Loader2 } from "lucide-react";
import { getAIRecommendations, type AIRecommendations, type AwarenessLevel } from "../actions";

interface RecommendationsPanelProps {
  prospectId: string;
  onAwarenessDetected?: (level: AwarenessLevel) => void;
}

const AWARENESS_COLORS: Record<AwarenessLevel, { bg: string; text: string }> = {
  unaware: { bg: "bg-gray-100", text: "text-gray-700" },
  "problem-aware": { bg: "bg-amber-50", text: "text-amber-700" },
  "solution-aware": { bg: "bg-blue-50", text: "text-blue-700" },

  "most-aware": { bg: "bg-emerald-50", text: "text-emerald-700" },
};

const AWARENESS_LABELS: Record<AwarenessLevel, string> = {
  unaware: "Neinformuotas",

  "solution-aware": "Iesko sprendimu",
  "product-aware": "Lygina pasiulymus",
  "most-aware": "Pasiruoses pirkti",
};

export function RecommendationsPanel({
  prospectId,
  onAwarenessDetected,
}: RecommendationsPanelProps) {
  const [recommendations, setRecommendations] = useState<AIRecommendations | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchRecommendations() {
      setIsLoading(true);
      const result = await getAIRecommendations(prospectId);
      
      if (result.success && result.data) {
        setRecommendations(result.data);
        onAwarenessDetected?.(result.data.awarenessLevel);
      } else {
        setError(result.error || "Failed to load");
      }
      setIsLoading(false);
    }

    fetchRecommendations();
  }, [prospectId, onAwarenessDetected]);

  if (isLoading) {
    return (
      <Card>
        <CardContent className="py-8 text-center">
          <Loader2 className="h-8 w-8 animate-spin mx-auto text-[#93939a]" />
          <p className="mt-2 text-sm text-[#93939a]">Analizuojame perspektyva...</p>
        </CardContent>
      </Card>
    );
  }

  if (error || !recommendations) {
    return null; // Silently fail, recommendations are optional

          </div>
        </div>
      </CardHeader>
      <CardContent className="space-y-4">
        {/* Hook Strategy */}
        <div className="p-3 bg-[#eaf1ed] rounded-lg">
          <p className="text-sm font-medium text-[#093528] mb-1">Rekomenduojama strategija</p>
          <p className="text-sm text-[#54545a]">{hookStrategy}</p>
        </div>

        {/* Opening Angle */}
        <div className="space-y-2">
          <div className="flex items-center gap-2">
            <Target className="h-4 w-4 text-[#93939a]" />
            <span className="text-xs font-medium text-[#93939a] uppercase tracking-wider">
              Pradinis kampas
            </span>
          </div>
          <p className="text-sm text-[#14141a]">{recommendedApproach.openingAngle}</p>
