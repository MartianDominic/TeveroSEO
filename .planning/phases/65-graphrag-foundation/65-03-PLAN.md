// open-seo-main/src/server/features/rag/retrieval-service.ts
import { db } from "@/db";
import { vectorChunks } from "@/db/vector-schema";
import { generateEmbedding } from "@/server/lib/embeddings";
import { GraphService } from "@/server/features/graph/graph-service";
import { sql, eq, and } from "drizzle-orm";

export type RetrievalMode = "vector" | "graph" | "hybrid";

export interface RetrievalResult {
  id: string;
  docId: string;
  docType: string;
  content: string;
  score: number;
  metadata?: Record<string, unknown>;
}

export interface RetrievalOptions {
  mode?: RetrievalMode;
  topK?: number;
  graphDepth?: number;
  rerank?: boolean;
}

const DEFAULT_TOP_K = 10;
const QUERY_TIMEOUT_MS = 500;

export const RetrievalService = {
  /**
   * Pure vector similarity search using pgvector DiskANN.
   */
  async vectorSearch(
    workspaceId: string,
    query: string,
    topK: number = DEFAULT_TOP_K
  ): Promise<RetrievalResult[]> {
    if (!query.trim()) return [];

    const queryEmbedding = await generateEmbedding(query);
    
    // Use raw SQL for pgvector cosine distance with rescore
    const results = await db.execute(sql`
      SET LOCAL diskann.query_rescore = 50;
      SET LOCAL statement_timeout = ${QUERY_TIMEOUT_MS};
      
      SELECT 
        id,
        doc_id as "docId",
        doc_type as "docType",
        content,
        1 - (embedding <=> ${sql.raw(`'[${queryEmbedding.join(",")}]'::halfvec`)}) as score,
        metadata
      FROM vector_chunks
      WHERE workspace_id = ${workspaceId}
      ORDER BY embedding <=> ${sql.raw(`'[${queryEmbedding.join(",")}]'::halfvec`)}
      LIMIT ${topK}
    `);

    return results.rows.map((row: any) => ({
      id: row.id,
      docId: row.docId,
      docType: row.docType,
      content: row.content,
      score: row.score,
      metadata: row.metadata ? JSON.parse(row.metadata) : undefined,
    }));
  },

  /**
   * Main retrieval entry point supporting multiple modes.
   */
  async retrieve(
    workspaceId: string,
    query: string,
    options: RetrievalOptions = {}
  ): Promise<RetrievalResult[]> {
    const { mode = "hybrid", topK = DEFAULT_TOP_K, graphDepth = 2, rerank = true } = options;

    switch (mode) {
      case "vector":
        return this.vectorSearch(workspaceId, query, topK);
      case "graph":
        return this.graphSearch(workspaceId, query, topK, graphDepth);
      case "hybrid":
        return this.hybridSearch(workspaceId, query, topK, graphDepth, rerank);
      default:
        throw new Error(`Unknown retrieval mode: ${mode}`);
    }
  },
};

// Add to retrieval-service.ts

export const RetrievalService = {
  // ... vectorSearch from Task 1

  /**
   * Graph-only retrieval: extract entities from query, traverse graph, find linked chunks.
   */
  async graphSearch(
    workspaceId: string,
    query: string,
    topK: number = DEFAULT_TOP_K,
    depth: number = 2
  ): Promise<RetrievalResult[]> {
    // 1. Extract entities from query
    const extraction = await EntityExtractor.extract(query);
    if (extraction.entities.length === 0) {
      // Fallback to vector search if no entities found
      return this.vectorSearch(workspaceId, query, topK);
    }

    // 2. Find subgraphs for each entity
    const relatedDocIds = new Set<string>();
    for (const entity of extraction.entities.slice(0, 5)) {
      const nodeId = `${entity.type}:${entity.name}`;
      const subgraph = await GraphService.getSubgraph(workspaceId, nodeId, depth);
      
      // Collect document IDs from graph nodes
      for (const node of subgraph.nodes) {
        if (node.label === "Page" && node.id.startsWith("Doc:")) {
          relatedDocIds.add(node.id.replace("Doc:", ""));
        }
      }
    }

    if (relatedDocIds.size === 0) {
      return [];
    }

    // 3. Fetch chunks for related documents
    const results = await db
      .select()
      .from(vectorChunks)
      .where(
        and(
          eq(vectorChunks.workspaceId, workspaceId),
          sql`doc_id = ANY(${Array.from(relatedDocIds)})`
        )
      )
      .limit(topK);

    return results.map((row, i) => ({
      id: row.id,
      docId: row.docId,
      docType: row.docType,
      content: row.content,
      score: 1 - i * 0.1, // Decay score by position
      metadata: row.metadata,
    }));
  },

  /**
   * Hybrid retrieval: vector candidates -> graph expansion -> fusion -> rerank.
   */
  async hybridSearch(
    workspaceId: string,
    query: string,
    topK: number = DEFAULT_TOP_K,
    depth: number = 2,
    shouldRerank: boolean = true
  ): Promise<RetrievalResult[]> {
    // 1. Get initial vector candidates (2x topK for expansion)
    const vectorResults = await this.vectorSearch(workspaceId, query, topK * 2);
    
    // 2. Graph expansion: get neighbors of top vector hits
    const expandedDocIds = new Set<string>();
    for (const result of vectorResults.slice(0, 5)) {
      const docNodeId = `Doc:${result.docId}`;
      try {
        const subgraph = await GraphService.getSubgraph(workspaceId, docNodeId, depth);
        for (const node of subgraph.nodes) {
          if (node.label === "Page" && node.id.startsWith("Doc:")) {
            expandedDocIds.add(node.id.replace("Doc:", ""));
          }
        }
      } catch {
        // Node may not exist in graph; continue
      }
    }

    // 3. Fetch graph-expanded chunks not in vector results
    const vectorDocIds = new Set(vectorResults.map(r => r.docId));
    const newDocIds = Array.from(expandedDocIds).filter(id => !vectorDocIds.has(id));
    
    let graphResults: RetrievalResult[] = [];
    if (newDocIds.length > 0) {
      const graphChunks = await db
        .select()
        .from(vectorChunks)
        .where(
          and(
            eq(vectorChunks.workspaceId, workspaceId),
            sql`doc_id = ANY(${newDocIds})`
          )
        )
        .limit(topK);
      
      graphResults = graphChunks.map((row, i) => ({
        id: row.id,
        docId: row.docId,
        docType: row.docType,
        content: row.content,
        score: 0.5 - i * 0.05, // Lower base score for graph-only results
        metadata: row.metadata,
      }));
    }

    // 4. Fusion: combine and deduplicate
    const seen = new Set<string>();
    const combined: RetrievalResult[] = [];
    
    for (const result of vectorResults) {
      if (!seen.has(result.id)) {
        seen.add(result.id);
        combined.push({ ...result, score: result.score * 0.7 }); // Vector weight
      }
    }
    
    for (const result of graphResults) {
      if (!seen.has(result.id)) {
        seen.add(result.id);
        combined.push({ ...result, score: result.score * 0.3 }); // Graph weight
      }
    }

    // 5. Sort by fused score
    combined.sort((a, b) => b.score - a.score);
    const candidates = combined.slice(0, topK * 2);

    // 6. Rerank if enabled
    if (shouldRerank && candidates.length > 0) {
      return Reranker.rerank(query, candidates, topK);
    }

    return candidates.slice(0, topK);
  },
};

// open-seo-main/src/server/features/rag/reranker.ts
import type { RetrievalResult } from "./retrieval-service";

const JINA_RERANK_URL = "https://api.jina.ai/v1/rerank";
const RERANK_MODEL = "jina-reranker-v2-base-multilingual";
const BATCH_SIZE = 50;
const TIMEOUT_MS = 200;

function getApiKey(): string {
  const key = process.env.JINA_API_KEY;
  if (!key) {
    throw new Error("JINA_API_KEY environment variable required for reranking");
  }
  return key;
}

export const Reranker = {
  /**
   * Rerank candidates using cross-encoder model.
   * Falls back to original order on error.
   */
  async rerank(
    query: string,
    candidates: RetrievalResult[],
    topK: number
  ): Promise<RetrievalResult[]> {
    if (candidates.length === 0) return [];
    if (candidates.length <= topK) {
      // No need to rerank if fewer candidates than requested
      return candidates;
    }

    try {
      // Batch candidates
      const batch = candidates.slice(0, BATCH_SIZE);
      
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), TIMEOUT_MS);

      const response = await fetch(JINA_RERANK_URL, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${getApiKey()}`,
        },
        body: JSON.stringify({
          model: RERANK_MODEL,
          query,
          documents: batch.map(c => c.content),
          top_n: topK,
        }),
        signal: controller.signal,
      });

      clearTimeout(timeoutId);

      if (!response.ok) {
        console.warn(`Reranker API error: ${response.status}, falling back to original order`);
        return candidates.slice(0, topK);
      }

      const data = await response.json();
      
      // Map reranker scores back to candidates
      const reranked: RetrievalResult[] = data.results.map((result: any) => ({
        ...batch[result.index],
        score: result.relevance_score,
      }));

      // Sort by reranker score, use original score as tiebreaker
      reranked.sort((a, b) => {
        const scoreDiff = b.score - a.score;
        if (Math.abs(scoreDiff) < 0.001) {
          // Tiebreaker: original score
          const origA = candidates.find(c => c.id === a.id)?.score ?? 0;
          const origB = candidates.find(c => c.id === b.id)?.score ?? 0;
          return origB - origA;
        }
        return scoreDiff;
      });

      return reranked.slice(0, topK);
    } catch (error) {
      if (error instanceof Error && error.name === "AbortError") {
        console.warn("Reranker timeout, falling back to original order");
      } else {
        console.error("Reranker error:", error);
      }
      return candidates.slice(0, topK);
    }
  },
};

// Convenience export
export async function rerank(
  query: string,
  candidates: RetrievalResult[],
  topK: number
): Promise<RetrievalResult[]> {
  return Reranker.rerank(query, candidates, topK);
}
