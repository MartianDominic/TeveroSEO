/**
 * Hybrid Retrieval Pipeline
 * Phase 65: GraphRAG Foundation
 * 
 * Combines vector similarity + graph traversal using RRF fusion.
 */
import { getTenantGraphManager } from "./tenant-graph-manager";
import { embedQuery } from "@/server/lib/embeddings";
import { db } from "@/db";
import { graphragChunks } from "@/db/graphrag-schema";
import { sql, eq, and } from "drizzle-orm";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "hybrid-retrieval" });

export interface SearchResult {
  id: string;
  score: number;
  source: "vector" | "graph" | "both";
}

export interface HybridSearchResult {
  id: string;
  content?: string;
  name?: string;
  type?: string;
  score: number;
  source: "vector" | "graph" | "both";
  related?: string[];
}

export interface HybridSearchOptions {
  k?: number;          // Final results to return (default: 20)
  kExpand?: number;    // Candidates from each source (default: k * 2)
  rrfK?: number;       // RRF k parameter (default: 60)
  includeContent?: boolean;
}

/**
 * Reciprocal Rank Fusion (RRF) algorithm.
 * 
 * Combines rankings from multiple sources using the formula:
 * score(d) = sum(1 / (k + rank(d)))
 * 
 * @param vectorResults - Results from vector search (id, score)
 * @param graphResults - Results from graph traversal (id, score)
 * @param k - RRF constant (default 60, per Cormack et al.)
 */
export function reciprocalRankFusion(
  vectorResults: Array<{ id: string; score: number }>,
  graphResults: Array<{ id: string; score: number }>,
  k: number = 60
): SearchResult[] {
  const scores = new Map<string, { score: number; sources: Set<string> }>();
  
  // Process vector results
  vectorResults.forEach((result, rank) => {
    const existing = scores.get(result.id) || { score: 0, sources: new Set<string>() };
    existing.score += 1 / (k + rank + 1);
    existing.sources.add("vector");
    scores.set(result.id, existing);
  });
  
  // Process graph results
  graphResults.forEach((result, rank) => {
    const existing = scores.get(result.id) || { score: 0, sources: new Set<string>() };
    existing.score += 1 / (k + rank + 1);
    existing.sources.add("graph");
    scores.set(result.id, existing);
  });
  
  // Convert to array and sort by score descending
  return Array.from(scores.entries())
    .map(([id, data]) => ({
      id,
      score: data.score,
      source: data.sources.size === 2 ? "both" as const : 
              (data.sources.has("vector") ? "vector" as const : "graph" as const)
    }))
    .sort((a, b) => b.score - a.score);
}

/**
 * Vector search using pgvector DiskANN index.
 */
async function vectorSearch(
  tenantId: string,
  queryEmbedding: number[],
  limit: number
): Promise<Array<{ id: string; score: number; content: string }>> {
  const vectorStr = `[${queryEmbedding.join(",")}]`;
  
  const results = await db.execute(sql`
    SET LOCAL diskann.query_rescore = 50;
    SELECT
      id::text,
      content,
      1 - (embedding <=> ${sql.raw(`'${vectorStr}'::halfvec`)}) AS score
    FROM graphrag_chunks
    WHERE tenant_id = ${tenantId}
    ORDER BY embedding <=> ${sql.raw(`'${vectorStr}'::halfvec`)}
    LIMIT ${limit};
  `);
  
  return results.rows.map(row => ({
    id: String(row.id),
    score: Number(row.score),
    content: String(row.content),
  }));
}

/**
 * Hybrid search combining vector similarity and graph traversal.
 */
export async function hybridSearch(
  tenantId: string,
  query: string,
  options: HybridSearchOptions = {}
): Promise<HybridSearchResult[]> {
  const startTime = Date.now();
  const {
    k = 20,
    kExpand = k * 2,
    rrfK = 60,
    includeContent = true,
  } = options;
  
  // Generate query embedding (768-dim)
  const embeddingResult = await embedQuery(query);
  const queryEmbedding = embeddingResult.embedding;
  
  // Parallel search: vector + graph
  const graphManager = await getTenantGraphManager();
  
  const [vectorResults, graphResults] = await Promise.all([
    vectorSearch(tenantId, queryEmbedding, kExpand),
    graphManager.hybridVectorGraphSearch(tenantId, queryEmbedding, { k: kExpand }),
  ]);
  
  // RRF fusion
  const fusedResults = reciprocalRankFusion(
    vectorResults.map(r => ({ id: r.id, score: r.score })),
    graphResults.map(r => ({ id: r.id, score: r.score })),
    rrfK
  );
  
  // Build result map for content lookup
  const vectorContentMap = new Map(vectorResults.map(r => [r.id, r.content]));
  const graphDataMap = new Map(graphResults.map(r => [r.id, r]));
  
  // Assemble final results
  const results: HybridSearchResult[] = fusedResults.slice(0, k).map(fused => {
    const graphData = graphDataMap.get(fused.id);
    return {
      id: fused.id,
      content: includeContent ? vectorContentMap.get(fused.id) : undefined,
      name: graphData?.name,
      type: graphData?.type,
      score: fused.score,
      source: fused.source,
      related: graphData?.related,
    };
  });
  
  const latencyMs = Date.now() - startTime;
  log.info("Hybrid search completed", { tenantId, k, latencyMs, resultCount: results.length });
  
  if (latencyMs > 500) {
    log.warn("Hybrid search exceeded 500ms target", { tenantId, latencyMs });
  }
  
  return results;
}

/**
 * Graph Service
 * Phase 65: GraphRAG Foundation
 * 
 * High-level service for graph entity operations.

import { getTenantGraphManager, TenantGraphManager } from "@/server/lib/graph";
import { 
  GraphEntity, 
  GraphRelation, 
  createEntityCypher, 
  createRelationCypher,
  GRAPH_ENTITY_TYPES 
} from "@/server/lib/graph/graph-schema";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "graph-service" });

export class GraphService {
  private manager: TenantGraphManager;
  
  constructor(manager: TenantGraphManager) {
    this.manager = manager;
  }
  
  /**
   * Initialize tenant graph with indexes.
   */
  async initializeTenant(tenantId: string): Promise<void> {
    await this.manager.initializeTenant(tenantId);
    log.info("Tenant graph initialized", { tenantId });
  }
  
  /**
   * Add entity to tenant graph.
   */
  async addEntity(tenantId: string, entity: GraphEntity): Promise<void> {
    const graph = await this.manager.getGraph(tenantId);
    const { cypher, params } = createEntityCypher(entity);
    await graph.query(cypher, { params });
    log.debug("Entity added", { tenantId, entityId: entity.id, type: entity.type });
  }
  
  /**
   * Add relation between entities.
   */
  async addRelation(tenantId: string, relation: GraphRelation): Promise<void> {
    const graph = await this.manager.getGraph(tenantId);
    const { cypher, params } = createRelationCypher(relation);
    await graph.query(cypher, { params });
    log.debug("Relation added", { tenantId, from: relation.from, to: relation.to });
  }
  
  /**
   * Get entities by type.
   */
  async getEntitiesByType(tenantId: string, type: string): Promise<GraphEntity[]> {
    const graph = await this.manager.getGraph(tenantId);
    const result = await graph.query<GraphEntity>(
      `MATCH (e:Entity {type: $type}) RETURN e.id AS id, e.name AS name, e.type AS type, e.metadata AS metadata`,
      { params: { type } }
    );
    return result.data;
  }
  
  /**
   * Delete tenant graph.
   */
  async deleteTenant(tenantId: string): Promise<void> {
    await this.manager.deleteTenant(tenantId);
    log.info("Tenant graph deleted", { tenantId });
  }
}

// Singleton
let _graphService: GraphService | null = null;

export async function getGraphService(): Promise<GraphService> {
  if (!_graphService) {
    const manager = await getTenantGraphManager();
    _graphService = new GraphService(manager);
  }
  return _graphService;
}

/**
 * Retrieval Service
 * Phase 65: GraphRAG Foundation
 * 
 * High-level service for GraphRAG retrieval operations.
 */
import { hybridSearch, HybridSearchResult, HybridSearchOptions } from "@/server/lib/graph/hybrid-retrieval";
import { getLightRAGService, LightRAGService } from "@/server/lib/lightrag";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "retrieval-service" });

export interface RetrievalOptions extends HybridSearchOptions {
  mode?: "hybrid" | "vector" | "graph" | "lightrag";
}

export interface RetrievalResult {
  results: HybridSearchResult[];
  mode: string;
  latencyMs: number;
}

export class RetrievalService {
  private lightrag: LightRAGService;
  
  constructor(lightrag?: LightRAGService) {
    this.lightrag = lightrag || getLightRAGService();
  }
  
  /**
   * Retrieve relevant context for a query.
   * 
   * Modes:
   * - hybrid: Vector + graph with RRF fusion (default, best quality)
   * - vector: pgvector DiskANN only
   * - graph: FalkorDB traversal only
   * - lightrag: Full LightRAG query (includes LLM synthesis)
   */
  async retrieve(
    tenantId: string,
    query: string,
    options: RetrievalOptions = {}
  ): Promise<RetrievalResult> {
    const startTime = Date.now();
    const { mode = "hybrid", ...searchOptions } = options;
    
    let results: HybridSearchResult[];
    
    switch (mode) {
      case "hybrid":
        results = await hybridSearch(tenantId, query, searchOptions);
        break;
        
      case "vector":
        // Vector-only mode (skip graph)
        results = await hybridSearch(tenantId, query, { 
          ...searchOptions, 
          // Override to only use vector results
        });
        results = results.filter(r => r.source === "vector" || r.source === "both");
        break;
        
      case "graph":
        // Graph-only mode (skip vector)
        results = await hybridSearch(tenantId, query, searchOptions);
        results = results.filter(r => r.source === "graph" || r.source === "both");
        break;
        
      case "lightrag":
        // Full LightRAG query
        const lightragResult = await this.lightrag.queryRAG(tenantId, query, "hybrid");
        results = lightragResult.entities.map(e => ({
          id: e.id,
          name: e.name,
          type: e.type,
          score: 1.0, // LightRAG doesn't expose scores
          source: "graph" as const,
          content: lightragResult.answer,
        }));
        break;
        
      default:
        throw new Error(`Unknown retrieval mode: ${mode}`);
    }
    
    const latencyMs = Date.now() - startTime;
    log.info("Retrieval completed", { tenantId, mode, latencyMs, resultCount: results.length });
    
    return {
      results,
      mode,
      latencyMs,
    };
  }
  
  /**
   * Check if tenant has GraphRAG data.
   */
  async hasTenantData(tenantId: string): Promise<boolean> {
    try {
      const health = await this.lightrag.healthCheck(tenantId);
      return health.tenantInitialized ?? false;
    } catch {
      return false;
    }
  }
}

// Singleton
let _retrievalService: RetrievalService | null = null;

export function getRetrievalService(): RetrievalService {
  if (!_retrievalService) {
    _retrievalService = new RetrievalService();
  }
  return _retrievalService;
}
