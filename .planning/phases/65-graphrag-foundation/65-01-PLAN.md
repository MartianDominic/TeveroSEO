import { FalkorDB, Graph } from "falkordb";
import { getSharedBullMQConnection } from "./redis";
import { safeTenantGraphName } from "@/lib/db/safe-query";

const GRAPH_PREFIX = "kg";

// Graph instance cache (workspaceId -> Graph)
const graphCache = new Map<string, Graph>();

let falkorInstance: FalkorDB | null = null;

export function getFalkorDB(): FalkorDB {
  if (!falkorInstance) {
    const connection = getSharedBullMQConnection("falkordb:main");
    falkorInstance = new FalkorDB({ connection });
  }
  return falkorInstance;
}

export function getTenantGraph(workspaceId: string): Graph {
  const cached = graphCache.get(workspaceId);
  if (cached) return cached;
  
  const graphName = safeTenantGraphName(GRAPH_PREFIX, workspaceId);
  const graph = getFalkorDB().selectGraph(graphName);
  graphCache.set(workspaceId, graph);
  return graph;
}

  // Create test graph and run simple query to verify module loaded
}

export async function closeFalkorDB(): Promise<void> {
  graphCache.clear();
  falkorInstance = null;
}

// open-seo-main/src/server/features/graph/types.ts
export type NodeLabel = "Product" | "Category" | "Content" | "Keyword" | "Page" | "Entity";

export interface GraphNode {
  id: string;
  label: NodeLabel;
  properties: Record<string, unknown>;
}

export interface GraphEdge {
  fromId: string;
  toId: string;
  type: string;
  properties?: Record<string, unknown>;
}

export interface SubgraphResult {
  nodes: GraphNode[];
  edges: GraphEdge[];
}

// open-seo-main/src/server/features/graph/graph-service.ts
import { getTenantGraph } from "@/server/lib/falkordb";
import { validateCypherQuery } from "@/lib/db/safe-query";
import { db } from "@/db";
import { graphMetadata } from "@/db/graph-schema";
import { eq } from "drizzle-orm";
import type { GraphNode, GraphEdge, SubgraphResult, NodeLabel } from "./types";

export const GraphService = {
  async initializeGraph(workspaceId: string): Promise<void> {
    const graph = getTenantGraph(workspaceId);
    // Create indexes for each node label
    const labels: NodeLabel[] = ["Product", "Category", "Content", "Keyword", "Page", "Entity"];
    for (const label of labels) {
      await graph.query(`CREATE INDEX FOR (n:${label}) ON (n.id)`);
    }
    // Initialize metadata record
    await db.insert(graphMetadata).values({ workspaceId }).onConflictDoNothing();
  },

  async addNode(workspaceId: string, node: GraphNode): Promise<void> {
    const graph = getTenantGraph(workspaceId);
    const cypher = `CREATE (n:${node.label} $props)`;
    validateCypherQuery(cypher);
    await graph.query(cypher, { props: { id: node.id, ...node.properties } });
    await this.updateStats(workspaceId);
  },

  async addEdge(workspaceId: string, edge: GraphEdge): Promise<void> {
    const graph = getTenantGraph(workspaceId);
    const cypher = `
      MATCH (a {id: $fromId}), (b {id: $toId})
      CREATE (a)-[r:${edge.type}]->(b)
    `;
    validateCypherQuery(cypher);
    await graph.query(cypher, { fromId: edge.fromId, toId: edge.toId });
    await this.updateStats(workspaceId);
  },

  async getSubgraph(workspaceId: string, nodeId: string, depth: number = 2): Promise<SubgraphResult> {
    const graph = getTenantGraph(workspaceId);
    // BFS traversal within depth limit
    const cypher = `
      MATCH path = (start {id: $nodeId})-[*1..${Math.min(depth, 5)}]-(connected)
      RETURN nodes(path) as nodes, relationships(path) as rels
    `;
    validateCypherQuery(cypher);
    const result = await graph.query(cypher, { nodeId });
    // Parse result into GraphNode[] and GraphEdge[]
    return { nodes: [], edges: [] }; // Implement parsing
  },

  async updateStats(workspaceId: string): Promise<void> {
    const graph = getTenantGraph(workspaceId);
    const nodeResult = await graph.query("MATCH (n) RETURN count(n) as count");
    const edgeResult = await graph.query("MATCH ()-[r]->() RETURN count(r) as count");
    
    await db.update(graphMetadata)
      .set({
        nodeCount: nodeResult.data[0]?.count ?? 0,
        edgeCount: edgeResult.data[0]?.count ?? 0,
        lastSyncAt: new Date(),
        updatedAt: new Date(),
      })
      .where(eq(graphMetadata.workspaceId, workspaceId));
  },

  async deleteGraph(workspaceId: string): Promise<void> {
    const graph = getTenantGraph(workspaceId);
    await graph.delete();
    await db.delete(graphMetadata).where(eq(graphMetadata.workspaceId, workspaceId));
  },
};
