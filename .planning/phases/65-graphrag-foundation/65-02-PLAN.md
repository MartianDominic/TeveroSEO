// open-seo-main/src/server/lib/embeddings.ts
import { redis } from "./redis";
import { createHash } from "node:crypto";

const JINA_API_URL = "https://api.jina.ai/v1/embeddings";
const JINA_MODEL = "jina-embeddings-v3";
const EMBEDDING_DIM = 768;
const BATCH_SIZE = 32;
const CACHE_TTL = 7 * 24 * 60 * 60; // 7 days
const CACHE_PREFIX = "emb:";

function getApiKey(): string {
  const key = process.env.JINA_API_KEY;
  if (!key) {
    throw new Error("JINA_API_KEY environment variable required");
  }
  return key;
}

function contentHash(text: string): string {
  return createHash("sha256").update(text).digest("hex").slice(0, 16);
}

export async function generateEmbedding(text: string): Promise<number[]> {
  const [embedding] = await generateEmbeddings([text]);
  return embedding;
}

export async function generateEmbeddings(texts: string[]): Promise<number[][]> {
  const results: number[][] = new Array(texts.length);
  const uncachedIndices: number[] = [];
  const uncachedTexts: string[] = [];

  // Check cache first
  for (let i = 0; i < texts.length; i++) {
    const hash = contentHash(texts[i]);
    const cached = await redis.get(`${CACHE_PREFIX}${hash}`);
    if (cached) {
      results[i] = JSON.parse(cached);
    } else {
      uncachedIndices.push(i);
      uncachedTexts.push(texts[i]);
    }
  }

  // Batch API calls for uncached texts
  for (let i = 0; i < uncachedTexts.length; i += BATCH_SIZE) {
    const batch = uncachedTexts.slice(i, i + BATCH_SIZE);
    const batchIndices = uncachedIndices.slice(i, i + BATCH_SIZE);
    
    const response = await fetch(JINA_API_URL, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${getApiKey()}`,
      },
      body: JSON.stringify({
        model: JINA_MODEL,
        input: batch,
        task: "retrieval.passage", // Optimized for retrieval
      }),
    });

    if (!response.ok) {
      if (response.status === 429) {
        // Rate limited - implement backoff
        await new Promise(r => setTimeout(r, 1000));
        i -= BATCH_SIZE; // Retry batch
        continue;
      }
      throw new Error(`Jina API error: ${response.status}`);
    }

    const data = await response.json();
    
    // Store results and cache
    for (let j = 0; j < data.data.length; j++) {
      const embedding = data.data[j].embedding;
      const originalIndex = batchIndices[j];
      results[originalIndex] = embedding;
      
      // Cache the embedding
      const hash = contentHash(uncachedTexts[i + j]);
      await redis.setex(`${CACHE_PREFIX}${hash}`, CACHE_TTL, JSON.stringify(embedding));
    }
  }

  return results;
}

export class EmbeddingCache {
  static async get(text: string): Promise<number[] | null> {
    const hash = contentHash(text);
    const cached = await redis.get(`${CACHE_PREFIX}${hash}`);
    return cached ? JSON.parse(cached) : null;
  }

  static async set(text: string, embedding: number[]): Promise<void> {
    const hash = contentHash(text);
    await redis.setex(`${CACHE_PREFIX}${hash}`, CACHE_TTL, JSON.stringify(embedding));
  }

  static async invalidate(text: string): Promise<void> {
    const hash = contentHash(text);
    await redis.del(`${CACHE_PREFIX}${hash}`);
  }
}

// open-seo-main/src/db/vector-schema.ts
import { pgTable, text, integer, timestamp, index, customType } from "drizzle-orm/pg-core";
import { createInsertSchema, createSelectSchema } from "drizzle-zod";
import { sql } from "drizzle-orm";

// Custom halfvec type for pgvector (768 dimensions, half precision)
const halfvec = customType<{ data: number[]; driverData: string }>({
  dataType() {
    return "halfvec(768)";
  },
  toDriver(value: number[]): string {
    return `[${value.join(",")}]`;
  },
  fromDriver(value: string): number[] {
    return JSON.parse(value.replace(/[\[\]]/g, (m) => m === "[" ? "[" : "]"));
  },
});

export const vectorChunks = pgTable(
  "vector_chunks",
  {
    id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
    workspaceId: text("workspace_id").notNull(),
    docId: text("doc_id").notNull(), // Reference to source document
    docType: text("doc_type").notNull(), // 'page', 'product', 'content', etc.
    content: text("content").notNull(),
    embedding: halfvec("embedding").notNull(),
    chunkIndex: integer("chunk_index").notNull().default(0),
    metadata: text("metadata").$type<Record<string, unknown>>(), // JSON metadata
    createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => [
    index("vector_chunks_workspace_doc_idx").on(table.workspaceId, table.docId),
    index("vector_chunks_workspace_type_idx").on(table.workspaceId, table.docType),
    // DiskANN index for vector similarity search (created via raw SQL migration)
  ]
);

export const insertVectorChunkSchema = createInsertSchema(vectorChunks);
export const selectVectorChunkSchema = createSelectSchema(vectorChunks);
export type VectorChunk = typeof vectorChunks.$inferSelect;
export type InsertVectorChunk = typeof vectorChunks.$inferInsert;

// open-seo-main/src/server/features/rag/types.ts
export interface Document {
  id: string;
  type: "page" | "product" | "content" | "keyword";
  title: string;
  content: string;
  url?: string;
  metadata?: Record<string, unknown>;
}

export interface ExtractedEntity {
  name: string;
  type: "Product" | "Category" | "Keyword" | "Page" | "Entity" | "Brand";
  confidence: number;
}

export interface ExtractedRelation {
  from: string;
  to: string;
  type: string;

export interface ExtractionResult {
  entities: ExtractedEntity[];
  relations: ExtractedRelation[];

// open-seo-main/src/server/lib/lightrag.ts
import { generateEmbeddings } from "./embeddings";
import { GraphService } from "@/server/features/graph/graph-service";
import { db } from "@/db";
import { vectorChunks } from "@/db/vector-schema";
import type { Document, ExtractionResult, ExtractedEntity } from "@/server/features/rag/types";

const CHUNK_SIZE = 1200; // tokens
const CHUNK_OVERLAP = 100; // tokens
const ENTITY_TYPES = ["Product", "Category", "Keyword", "Page", "Entity", "Brand"] as const;

export class EntityExtractor {
  /**
   * Extract entities and relations from text using LLM.
   * Uses GPT-4o-mini for cost efficiency (~$8-14 per 10k pages).
   */
  static async extract(text: string): Promise<ExtractionResult> {
    // Use existing LLM infrastructure
    const prompt = `Extract entities and relationships from this text.
    
Entity types allowed: ${ENTITY_TYPES.join(", ")}

Return JSON:
{
  "entities": [{"name": "...", "type": "...", "confidence": 0.0-1.0}],
  "relations": [{"from": "...", "to": "...", "type": "..."}]
}

Text:
${text.slice(0, 4000)}`;

    // Call LLM and parse response
    // Implementation uses existing anthropic.ts or openai.ts
    return { entities: [], relations: [] };
  }
}

export class LightRAGOrchestrator {
  /**
   * Ingest a document: chunk, embed, extract entities, update graph.
   */
  static async ingest(workspaceId: string, doc: Document): Promise<void> {
    // 1. Chunk the document
    const chunks = this.chunkText(doc.content, CHUNK_SIZE, CHUNK_OVERLAP);
    
    // 2. Generate embeddings for all chunks
    const embeddings = await generateEmbeddings(chunks);
    
    // 3. Store chunks with embeddings
    const chunkRecords = chunks.map((content, i) => ({
      workspaceId,
      docId: doc.id,
      docType: doc.type,
      content,
      embedding: embeddings[i],
      chunkIndex: i,
      metadata: doc.metadata,
    }));
    
    await db.insert(vectorChunks).values(chunkRecords);
    
    // 4. Extract entities from first few chunks (for efficiency)
    const extractionText = chunks.slice(0, 3).join("\n\n");
    const extraction = await EntityExtractor.extract(extractionText);
    
    // 5. Add entities as nodes to graph
    for (const entity of extraction.entities) {
      await GraphService.addNode(workspaceId, {
        id: `${entity.type}:${entity.name}`,
        label: entity.type as any,
        properties: { name: entity.name, confidence: entity.confidence },
      });
    }
    
    // 6. Add document node linked to entities
    await GraphService.addNode(workspaceId, {
      id: `Doc:${doc.id}`,
      label: "Page",
      properties: { title: doc.title, url: doc.url, type: doc.type },
    });
    
    for (const entity of extraction.entities) {
      await GraphService.addEdge(workspaceId, {
        fromId: `Doc:${doc.id}`,
        toId: `${entity.type}:${entity.name}`,
        type: "MENTIONS",
      });
    }
    
    // 7. Add extracted relations as edges
    for (const rel of extraction.relations) {
      await GraphService.addEdge(workspaceId, {
        fromId: rel.from,
        toId: rel.to,
        type: rel.type,
      });
    }
  }

  /**
   * Chunk text with overlap for context preservation.
   */
  private static chunkText(text: string, chunkSize: number, overlap: number): string[] {
    // Simple word-based chunking (production would use tiktoken)
    const words = text.split(/\s+/);
    const chunks: string[] = [];
    
    for (let i = 0; i < words.length; i += chunkSize - overlap) {
      const chunk = words.slice(i, i + chunkSize).join(" ");
      if (chunk.trim()) chunks.push(chunk);
    }
    
    return chunks;
  }
}
