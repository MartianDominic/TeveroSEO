```typescript
/**
 * GraphRAG Query Endpoint
 * Phase 65: GraphRAG Foundation
 * 
 * POST /api/graphrag/query
 * Returns hybrid search results for a natural language query.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { z } from "zod";
import { getRetrievalService } from "@/server/features/graph";
import { withAuth, getWorkspaceId } from "@/server/lib/clerk-jwt";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "api-graphrag-query" });

const querySchema = z.object({
  query: z.string().min(1).max(2000),
  mode: z.enum(["hybrid", "vector", "graph", "lightrag"]).default("hybrid"),
  k: z.number().int().min(1).max(100).default(20),
  includeContent: z.boolean().default(true),
});

export const Route = createAPIFileRoute("/api/graphrag/query")({
  POST: withAuth(async ({ request }) => {
    const workspaceId = getWorkspaceId(request);
    const body = await request.json();
    
    const parsed = querySchema.safeParse(body);
    if (!parsed.success) {
      return Response.json(
        { success: false, error: parsed.error.message },
        { status: 400 }
      );
    }
    
    const { query, mode, k, includeContent } = parsed.data;
    
    try {
      const retrieval = getRetrievalService();
      const result = await retrieval.retrieve(workspaceId, query, {
        mode,
        k,
        includeContent,
      });
      
      log.info("GraphRAG query completed", {
        workspaceId,
        mode,
        latencyMs: result.latencyMs,
        resultCount: result.results.length,
      });
      
      return Response.json({
        success: true,
        data: {
          results: result.results,
          mode: result.mode,
          latencyMs: result.latencyMs,
        },
      });
    } catch (error) {
      log.error("GraphRAG query failed", error instanceof Error ? error : new Error(String(error)));
      return Response.json(
        { success: false, error: "Query failed" },
        { status: 500 }
      );
    }
  }),
});
```

```typescript
/**
 * GraphRAG Ingest Endpoint
 * Phase 65: GraphRAG Foundation
 * 
 * POST /api/graphrag/ingest
 * Ingests documents into tenant's knowledge graph.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { z } from "zod";
import { getLightRAGService } from "@/server/lib/lightrag";
import { getGraphService } from "@/server/features/graph";
import { withAuth, getWorkspaceId } from "@/server/lib/clerk-jwt";
import { createLogger } from "@/server/lib/logger";

const log = createLogger({ module: "api-graphrag-ingest" });

const ingestSchema = z.object({
  documents: z.array(z.object({
    id: z.string(),
    content: z.string().min(1).max(100000),
    url: z.string().url().optional(),
    metadata: z.record(z.unknown()).optional(),
  })).min(1).max(50),
});

export const Route = createAPIFileRoute("/api/graphrag/ingest")({
  POST: withAuth(async ({ request }) => {
    const workspaceId = getWorkspaceId(request);
    const body = await request.json();
    
    const parsed = ingestSchema.safeParse(body);
    if (!parsed.success) {
      return Response.json(
        { success: false, error: parsed.error.message },
        { status: 400 }
      );
    }
    
    const { documents } = parsed.data;
    
    try {
      // Ensure tenant graph is initialized
      const graphService = await getGraphService();
      await graphService.initializeTenant(workspaceId);
      
      // Insert documents via LightRAG
      const lightrag = getLightRAGService();
      const results = await lightrag.insertDocuments(
        workspaceId,
        documents.map(d => ({
          id: d.id,
          content: d.content,
          url: d.url || "",
        }))
      );
      
      log.info("GraphRAG ingest completed", {
        workspaceId,
        documentCount: documents.length,
        entitiesExtracted: results.reduce((sum, r) => sum + r.entitiesExtracted, 0),
      });
      
      return Response.json({
        success: true,
        data: {
          processed: results.length,
          results: results.map(r => ({
            documentId: r.documentId,
            chunksProcessed: r.chunksProcessed,
            entitiesExtracted: r.entitiesExtracted,
          })),
        },
      });
    } catch (error) {
      log.error("GraphRAG ingest failed", error instanceof Error ? error : new Error(String(error)));
      return Response.json(
        { success: false, error: "Ingest failed" },
        { status: 500 }
      );
    }
  }),
});
```

```typescript
/**
 * GraphRAG Status Endpoint
 * Phase 65: GraphRAG Foundation
 * 
 * GET /api/graphrag/status
 * Returns tenant GraphRAG health and statistics.
 */
import { createAPIFileRoute } from "@tanstack/start/api";
import { getLightRAGService } from "@/server/lib/lightrag";
import { getTenantGraphManager } from "@/server/lib/graph";
import { withAuth, getWorkspaceId } from "@/server/lib/clerk-jwt";

export const Route = createAPIFileRoute("/api/graphrag/status")({
  GET: withAuth(async ({ request }) => {
    const workspaceId = getWorkspaceId(request);
    
    try {
      const lightrag = getLightRAGService();
      const graphManager = await getTenantGraphManager();
      
      const [health, memoryUsage] = await Promise.all([
        lightrag.healthCheck(workspaceId),
        graphManager.getGraphMemoryUsage?.(workspaceId).catch(() => 0) || Promise.resolve(0),
      ]);
      
      return Response.json({
        success: true,
        data: {
          healthy: health.healthy,
          tenantInitialized: health.tenantInitialized ?? false,
          graphMemoryBytes: memoryUsage,
        },
      });
    } catch (error) {
      return Response.json({
        success: true,
        data: {
          healthy: false,
          tenantInitialized: false,
          graphMemoryBytes: 0,
        },
      });
    }
  }),
});
```

```python
"""GraphRAG FastAPI Router.

Phase 65: GraphRAG Foundation
Exposes LightRAG operations via REST API.
"""
from fastapi import APIRouter, HTTPException, Depends
from pydantic import BaseModel, Field
from typing import List, Optional
import logging

from lib.graphrag import get_tenant_rag, query_lightrag, JinaEmbeddingService
from auth.clerk_auth import get_current_workspace

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/graphrag", tags=["graphrag"])


class DocumentInput(BaseModel):
    id: str
    content: str = Field(..., min_length=1, max_length=100000)
    url: Optional[str] = None


class IngestRequest(BaseModel):
    documents: List[DocumentInput] = Field(..., min_items=1, max_items=50)


class IngestResult(BaseModel):
    document_id: str
    chunks_processed: int
    entities_extracted: int


class IngestResponse(BaseModel):
    success: bool
    processed: int
    results: List[IngestResult]


class QueryRequest(BaseModel):
    query: str = Field(..., min_length=1, max_length=2000)
    mode: str = Field(default="hybrid", pattern="^(hybrid|local|global|naive|mix)$")
    top_k: int = Field(default=20, ge=1, le=100)


class QueryResponse(BaseModel):
    success: bool
    answer: str
    mode: str


class StatusResponse(BaseModel):
    success: bool
    healthy: bool
    tenant_initialized: bool


@router.post("/ingest", response_model=IngestResponse)
async def ingest_documents(
    request: IngestRequest,
    workspace_id: str = Depends(get_current_workspace)
):
    """Ingest documents into tenant's knowledge graph."""
    try:
        rag = get_tenant_rag(workspace_id)
        
        results = []
        for doc in request.documents:
            # Insert into LightRAG
            await rag.ainsert(doc.content)
            results.append(IngestResult(
                document_id=doc.id,
                chunks_processed=1,  # LightRAG handles chunking internally
                entities_extracted=0,  # Not exposed by LightRAG
            ))
        
        logger.info(f"Ingested {len(results)} documents for workspace {workspace_id}")
        
        return IngestResponse(
            success=True,
            processed=len(results),
            results=results,
        )
    except Exception as e:
        logger.error(f"Ingest failed for workspace {workspace_id}: {e}")
        raise HTTPException(status_code=500, detail=str(e))


@router.post("/query", response_model=QueryResponse)
async def query_graphrag(
    request: QueryRequest,
    workspace_id: str = Depends(get_current_workspace)
):
    """Query tenant's knowledge graph."""
    try:
        answer = await query_lightrag(
            workspace_id,
            request.query,
            mode=request.mode
        )
        
        logger.info(f"Query completed for workspace {workspace_id}, mode={request.mode}")
        
        return QueryResponse(
            success=True,
            answer=answer,
            mode=request.mode,
        )
    except Exception as e:
        logger.error(f"Query failed for workspace {workspace_id}: {e}")
        raise HTTPException(status_code=500, detail=str(e))


@router.get("/status", response_model=StatusResponse)
async def get_status(workspace_id: str = Depends(get_current_workspace)):
    """Get GraphRAG status for tenant."""
    try:
        rag = get_tenant_rag(workspace_id)
        # Check if RAG has data by attempting a simple operation
        initialized = rag is not None
        
        return StatusResponse(
            success=True,
            healthy=True,
            tenant_initialized=initialized,
        )
    except Exception:
        return StatusResponse(
            success=True,
            healthy=False,
            tenant_initialized=False,
        )
```

```typescript
    });
  });
```
