// open-seo-main/src/server/workers/rag-worker.ts
import { Queue, Worker, Job } from "bullmq";
import { getSharedBullMQConnection } from "@/server/lib/redis";
import { LightRAGOrchestrator } from "@/server/lib/lightrag";
import type { Document } from "@/server/features/rag/types";

const QUEUE_NAME = "rag:ingest";
const DLQ_NAME = "rag:ingest:dlq";
const CONCURRENCY = 5;
const MAX_RETRIES = 3;

export interface RAGIngestJobData {
  workspaceId: string;
  document: Document;
}

// Create queue
export const ragQueue = new Queue<RAGIngestJobData>(QUEUE_NAME, {
  connection: getSharedBullMQConnection("queue:rag"),
  defaultJobOptions: {
    attempts: MAX_RETRIES,
    backoff: {
      type: "exponential",
      delay: 1000, // 1s, 2s, 4s
    },
    removeOnComplete: { count: 1000 },
    removeOnFail: false, // Keep for DLQ
  },
});

// Create dead-letter queue
export const ragDLQ = new Queue<RAGIngestJobData>(DLQ_NAME, {
  connection: getSharedBullMQConnection("queue:rag-dlq"),
});

// Create worker
export const ragWorker = new Worker<RAGIngestJobData>(
  QUEUE_NAME,
  async (job: Job<RAGIngestJobData>) => {
    const { workspaceId, document } = job.data;
    
    job.log(`Starting ingestion for doc ${document.id} in workspace ${workspaceId}`);
    await job.updateProgress(10);
    
    try {
      await LightRAGOrchestrator.ingest(workspaceId, document);
      await job.updateProgress(100);
      job.log(`Ingestion complete for doc ${document.id}`);
      return { success: true, docId: document.id };
    } catch (error) {
      const message = error instanceof Error ? error.message : String(error);
      job.log(`Ingestion failed: ${message}`);
      throw error;
    }
  },
  {
    connection: getSharedBullMQConnection("worker:rag"),
    concurrency: CONCURRENCY,
    limiter: {
      max: 10,
      duration: 1000, // 10 jobs per second max
    },
  }
);

// Move exhausted jobs to DLQ
ragWorker.on("failed", async (job, err) => {
  if (job && job.attemptsMade >= MAX_RETRIES) {
    await ragDLQ.add("exhausted", job.data, {
      removeOnComplete: false,
    });
    console.error(`[RAG] Job ${job.id} moved to DLQ after ${job.attemptsMade} attempts:`, err.message);
  }
});

ragWorker.on("completed", (job) => {
  console.log(`[RAG] Job ${job.id} completed`);
});

ragWorker.on("error", (err) => {
  console.error("[RAG] Worker error:", err.message);
});

/**
 * Queue a document for ingestion.
 */
export async function queueIngestion(workspaceId: string, document: Document): Promise<string> {
  const job = await ragQueue.add("ingest", { workspaceId, document });
  return job.id ?? "unknown";
}

/**
 * Get job status.
 */
export async function getJobStatus(jobId: string) {
  const job = await ragQueue.getJob(jobId);
  if (!job) return null;
  
  const state = await job.getState();
  return {
    id: job.id,
    state,
    progress: job.progress,
    attemptsMade: job.attemptsMade,
    data: job.data,
    returnvalue: job.returnvalue,
    failedReason: job.failedReason,
  };
}

/**
 * Graceful shutdown.
 */
export async function closeRAGWorker(): Promise<void> {
  await ragWorker.close();
  await ragQueue.close();
  await ragDLQ.close();
}

// open-seo-main/src/routes/api/rag/query.ts
import { createAPIFileRoute } from "@tanstack/start/api";
import { z } from "zod";
import { RetrievalService } from "@/server/features/rag/retrieval-service";
import { getWorkspaceIdFromRequest } from "@/server/lib/auth-session";

const querySchema = z.object({
  query: z.string().min(1).max(1000),
  mode: z.enum(["vector", "graph", "hybrid"]).default("hybrid"),
  topK: z.number().int().min(1).max(50).default(10),
  rerank: z.boolean().default(true),
});

export const Route = createAPIFileRoute("/api/rag/query")({
  POST: async ({ request }) => {
    const workspaceId = await getWorkspaceIdFromRequest(request);
    if (!workspaceId) {
      return new Response(JSON.stringify({ error: "Unauthorized" }), { status: 401 });
    }

    const body = await request.json();
    const parsed = querySchema.safeParse(body);
    if (!parsed.success) {
      return new Response(JSON.stringify({ error: parsed.error.message }), { status: 400 });
    }

    const { query, mode, topK, rerank } = parsed.data;

    try {
      const results = await RetrievalService.retrieve(workspaceId, query, {
        mode,
        topK,
        rerank,
      });

      return new Response(JSON.stringify({ results }), {
        status: 200,
        headers: { "Content-Type": "application/json" },
      });
    } catch (error) {
      console.error("[RAG] Query error:", error);
      return new Response(JSON.stringify({ error: "Retrieval failed" }), { status: 500 });
    }
  },
});

// open-seo-main/src/routes/api/rag/ingest.ts
import { createAPIFileRoute } from "@tanstack/start/api";
import { z } from "zod";
import { queueIngestion } from "@/server/workers/rag-worker";
import { getWorkspaceIdFromRequest } from "@/server/lib/auth-session";

const ingestSchema = z.object({
  document: z.object({
    id: z.string(),
    type: z.enum(["page", "product", "content", "keyword"]),
    title: z.string(),
    content: z.string(),
    url: z.string().url().optional(),
    metadata: z.record(z.unknown()).optional(),
  }),
});

export const Route = createAPIFileRoute("/api/rag/ingest")({
  POST: async ({ request }) => {
    const workspaceId = await getWorkspaceIdFromRequest(request);
    if (!workspaceId) {
      return new Response(JSON.stringify({ error: "Unauthorized" }), { status: 401 });
    }

    const body = await request.json();
    const parsed = ingestSchema.safeParse(body);
    if (!parsed.success) {
      return new Response(JSON.stringify({ error: parsed.error.message }), { status: 400 });
    }

    try {
      const jobId = await queueIngestion(workspaceId, parsed.data.document);
      return new Response(JSON.stringify({ jobId, status: "queued" }), {
        status: 202,
        headers: { "Content-Type": "application/json" },
      });
    } catch (error) {
      console.error("[RAG] Ingest error:", error);
      return new Response(JSON.stringify({ error: "Ingestion failed" }), { status: 500 });
    }
  },
});

// open-seo-main/src/routes/api/rag/status.$jobId.ts
import { createAPIFileRoute } from "@tanstack/start/api";
import { getJobStatus } from "@/server/workers/rag-worker";
import { getWorkspaceIdFromRequest } from "@/server/lib/auth-session";

export const Route = createAPIFileRoute("/api/rag/status/$jobId")({
  GET: async ({ request, params }) => {
    const workspaceId = await getWorkspaceIdFromRequest(request);
    if (!workspaceId) {
      return new Response(JSON.stringify({ error: "Unauthorized" }), { status: 401 });
    }

    const status = await getJobStatus(params.jobId);
    if (!status) {
      return new Response(JSON.stringify({ error: "Job not found" }), { status: 404 });
    }

    // Verify workspace ownership
    if (status.data.workspaceId !== workspaceId) {
      return new Response(JSON.stringify({ error: "Forbidden" }), { status: 403 });
    }

    return new Response(JSON.stringify(status), {
      status: 200,
      headers: { "Content-Type": "application/json" },
    });
  },
});
