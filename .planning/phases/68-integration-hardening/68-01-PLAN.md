```typescript
import { verifyToken } from '@clerk/backend';

interface ClerkClaims {
  userId: string;
  orgId?: string;
  sessionId: string;
}

export async function verifyClerkToken(token: string): Promise<ClerkClaims> {
  const secretKey = process.env.CLERK_SECRET_KEY;
  if (!secretKey) {
    throw new AppError('CLERK_SECRET_KEY not configured', 500);
  }
  
  try {
    const payload = await verifyToken(token, { secretKey });
    return {
      userId: payload.sub,
      orgId: payload.org_id,
      sessionId: payload.sid,
    };
  } catch (err) {
    throw new AppError('Invalid or expired token', 401);
  }
}
```

```typescript
export async function resolveClientId(request: Request): Promise<ResolvedContext> {
  const authHeader = request.headers.get('Authorization');
  const internalToken = request.headers.get('X-Internal-Token');
  
  // Internal service bypass (service-to-service)
  if (internalToken === process.env.INTERNAL_SERVICE_TOKEN) {
    return resolveFromHeaders(request);
  }
  
  // JWT validation required for external requests
  if (!authHeader?.startsWith('Bearer ')) {
    throw new AppError('Missing authorization header', 401);
  }
  
  const token = authHeader.slice(7);
  const claims = await verifyClerkToken(token);
  
  // Now safe to use X-Client-ID
  const clientId = request.headers.get('X-Client-ID');
  if (!clientId) {
    throw new AppError('Missing X-Client-ID header', 400);
  }
  
  return { userId: claims.userId, clientId, orgId: claims.orgId };
}
```
