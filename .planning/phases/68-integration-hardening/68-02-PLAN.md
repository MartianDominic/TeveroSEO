```typescript
import { AppError } from '../server/lib/errors';

export function requireClientContext(request: Request): string {
  const clientId = request.headers.get('X-Client-ID');
  
  if (!clientId || clientId.trim() === '') {
    throw new AppError('X-Client-ID header is required', 400, 'MISSING_CLIENT_ID');
  }
  
  // Validate UUID format
  const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
  if (!uuidRegex.test(clientId)) {
    throw new AppError('Invalid X-Client-ID format', 400, 'INVALID_CLIENT_ID');
  }
  
  return clientId;
}
```

```typescript
class AbortManager {
  private controllers = new Map<string, AbortController>();
  
  getController(clientId: string): AbortController {
    let controller = this.controllers.get(clientId);
    if (!controller) {
      controller = new AbortController();
      this.controllers.set(clientId, controller);
    }
    return controller;
  }
  
  abortClient(clientId: string): void {
    const controller = this.controllers.get(clientId);
    if (controller) {
      controller.abort();
      this.controllers.delete(clientId);
    }
  }
  
  abortAll(): void {
    this.controllers.forEach(controller => controller.abort());
    this.controllers.clear();
  }
}

export const abortManager = new AbortManager();
```

```typescript
import { redis, pubsub } from './redis';

const OWNERSHIP_CHANNEL = 'tevero:ownership:changes';
const ownershipCache = new Map<string, { userId: string; expiresAt: number }>();

export async function subscribeToOwnershipChanges(): Promise<void> {
  await pubsub.subscribe(OWNERSHIP_CHANNEL, (message) => {
    const { clientId, action } = JSON.parse(message);
    if (action === 'revoked') {
      ownershipCache.delete(clientId);
    }
  });
}

export async function publishOwnershipChange(clientId: string, action: 'granted' | 'revoked'): Promise<void> {
  await redis.publish(OWNERSHIP_CHANNEL, JSON.stringify({ clientId, action, timestamp: Date.now() }));
}

export function getCachedOwnership(clientId: string): { userId: string } | null {
  const entry = ownershipCache.get(clientId);
  if (!entry || entry.expiresAt < Date.now()) {
    ownershipCache.delete(clientId);
    return null;
  }
  return { userId: entry.userId };
}
```

```typescript
export function validateClientAccess(clientId: string | null): string {
  if (!clientId || clientId.trim() === '') {
    throw new AppError('Client ID is required', 400);
  }
  
  // Validate UUID format
  const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
  if (!uuidRegex.test(clientId)) {
    throw new AppError('Invalid client ID format', 400);
  }
  
  return clientId;
}

export async function checkClientOwnership(
  clientId: string,
  userId: string
): Promise<boolean> {
  // Pre-check ownership before expensive backend calls
  const response = await fetch(`/api/clients/${clientId}/access`, {
    method: 'HEAD',
    headers: { 'X-User-ID': userId },
  });
  
  if (response.status === 403) {
    throw new AppError('Access denied to client', 403);
  }
  
  return response.ok;
}
```
