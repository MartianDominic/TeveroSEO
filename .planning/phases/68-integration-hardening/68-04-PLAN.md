import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useClientStore } from '@/stores/clientStore';

const CLIENTS_KEY = ['clients'] as const;
const CLIENT_KEY = (id: string) => ['clients', id] as const;

export function useClients() {
  return useQuery({
    queryKey: CLIENTS_KEY,
    queryFn: async () => {
      const response = await fetch('/api/clients');
      if (!response.ok) throw new Error('Failed to fetch clients');
      return response.json();
    },
    staleTime: 5 * 60 * 1000, // 5 minutes
    gcTime: 10 * 60 * 1000, // 10 minutes
  });
}

export function useActiveClient() {
  const activeClientId = useClientStore((state) => state.activeClientId);
  const { data: clients } = useClients();
  
  return clients?.find((c: Client) => c.id === activeClientId) ?? null;
}

export function useSetActiveClient() {
  const queryClient = useQueryClient();
  const setActiveClient = useClientStore((state) => state.setActiveClient);
  
  return (clientId: string) => {
    setActiveClient(clientId);
    // Invalidate client-specific queries
    queryClient.invalidateQueries({ queryKey: ['audits', clientId] });
  };
}
```

const CHANNEL_NAME = 'tevero-state-sync';

type SyncMessage = 
  | { type: 'CLIENT_CHANGED'; clientId: string }
  | { type: 'LOGOUT' }
  | { type: 'CACHE_INVALIDATE'; keys: string[] };

class BroadcastSync {
  private channel: BroadcastChannel | null = null;
  private handlers = new Map<string, (msg: SyncMessage) => void>();
  
  init() {
    if (typeof window === 'undefined') return;
    
    this.channel = new BroadcastChannel(CHANNEL_NAME);
    this.channel.onmessage = (event) => {
      const message = event.data as SyncMessage;
      this.handlers.forEach((handler) => handler(message));
    };
  }
  
  subscribe(id: string, handler: (msg: SyncMessage) => void) {
    this.handlers.set(id, handler);
    return () => this.handlers.delete(id);
  }
  
  broadcast(message: SyncMessage) {
    this.channel?.postMessage(message);
  }
  
  broadcastClientChange(clientId: string) {
    this.broadcast({ type: 'CLIENT_CHANGED', clientId });
  }
  
  broadcastLogout() {
    this.broadcast({ type: 'LOGOUT' });
  }
}

export const broadcastSync = new BroadcastSync();
```
