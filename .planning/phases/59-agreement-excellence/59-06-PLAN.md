"use server";

import { db } from "@/lib/db";
import { eq, and } from "drizzle-orm";
import { agreementSigners, clientAgreements } from "@/db/schema";
import { nanoid } from "nanoid";
import { revalidatePath } from "next/cache";
import { SignerNotificationService } from "@/server/services/signer-notification-service";

export interface SignerData {
  id: number;
  name: string;
  email: string;
  phone: string | null;
  role: string;
  signingOrder: number;
  status: string;
  viewedAt: Date | null;
  signedAt: Date | null;
}

export interface PreSigningData {
  agreement: {
    id: number;
    title: string;
    status: string;
    signingMode: "sequential" | "parallel";
  };
  signers: SignerData[];
  canSend: boolean;
  sendMessage?: string;
}

export async function getPreSigningData(
  clientId: number,
  agreementId: number
): Promise<PreSigningData | { error: string }> {
  const agreement = await db.query.clientAgreements.findFirst({
    where: eq(clientAgreements.id, agreementId),
  });
  
  if (!agreement) return { error: "Agreement not found" };
  if (agreement.clientId !== clientId) return { error: "Agreement not found" };
  
  const signers = await db.query.agreementSigners.findMany({
    where: eq(agreementSigners.agreementId, agreementId),
    orderBy: (s, { asc }) => [asc(s.signingOrder)],
  });
  
  const canSend = signers.length > 0 && agreement.status === "draft";
  const sendMessage = !canSend
    ? signers.length === 0 
      ? "Add at least one signer"
      : "Agreement must be in draft status"
    : undefined;
  
  return {
    agreement: {
      id: agreement.id,
      title: agreement.title,
      status: agreement.status,
      signingMode: (agreement.signingMode as "sequential" | "parallel") || "sequential",
    },
    signers: signers.map(s => ({
      id: s.id,
      name: s.name,
      email: s.email,
      phone: s.phone,
      role: s.role || "signer",
      signingOrder: s.signingOrder || 1,
      status: s.status || "pending",
      viewedAt: s.viewedAt,
      signedAt: s.signedAt,
    })),
    canSend,
    sendMessage,
  };
}

export async function addSigner(
  agreementId: number,
  data: { name: string; email: string; phone?: string; role?: string }
): Promise<{ success: boolean; signer?: SignerData; error?: string }> {
  try {
    // Get current max signing order
    const existing = await db.query.agreementSigners.findMany({
      where: eq(agreementSigners.agreementId, agreementId),
    });
    const maxOrder = Math.max(0, ...existing.map(s => s.signingOrder || 0));
    
    const [signer] = await db.insert(agreementSigners)
      .values({
        agreementId,
        name: data.name,
        email: data.email,
        phone: data.phone || null,
        role: data.role || "signer",
        signingOrder: maxOrder + 1,
        status: "pending",
      })
      .returning();
    
    return {
      success: true,
      signer: {
        id: signer.id,
        name: signer.name,
        email: signer.email,
        phone: signer.phone,
        role: signer.role || "signer",
        signingOrder: signer.signingOrder || 1,
        status: signer.status || "pending",
        viewedAt: signer.viewedAt,
        signedAt: signer.signedAt,
      },
    };
  } catch (e) {
    return { success: false, error: "Failed to add signer" };
  }
}

export async function removeSigner(signerId: number): Promise<{ success: boolean }> {
  try {
    await db.delete(agreementSigners).where(eq(agreementSigners.id, signerId));
    return { success: true };
  } catch (e) {
    return { success: false };
  }
}

export async function updateSigningOrder(
  agreementId: number,
  signerIds: number[]
): Promise<{ success: boolean }> {
  try {
    // Update each signer's order based on array position
    await Promise.all(
      signerIds.map((signerId, index) =>
        db.update(agreementSigners)
          .set({ signingOrder: index + 1, updatedAt: new Date() })
          .where(eq(agreementSigners.id, signerId))
      )
    );
    return { success: true };
  } catch (e) {
    return { success: false };
  }
}

export async function updateSigningMode(
  agreementId: number,
  mode: "sequential" | "parallel"
): Promise<{ success: boolean }> {
  try {
    await db.update(clientAgreements)
      .set({ signingMode: mode, updatedAt: new Date() })
      .where(eq(clientAgreements.id, agreementId));
    return { success: true };
  } catch (e) {
    return { success: false };
  }
}

export async function sendInvitations(
  agreementId: number
): Promise<{ success: boolean; error?: string }> {
  try {
    const agreement = await db.query.clientAgreements.findFirst({
      where: eq(clientAgreements.id, agreementId),
    });
    
    if (!agreement || agreement.status !== "draft") {
      return { success: false, error: "Agreement must be in draft status" };
    }
    
    const signers = await db.query.agreementSigners.findMany({
      where: eq(agreementSigners.agreementId, agreementId),
    });
    
    if (signers.length === 0) {
      return { success: false, error: "No signers configured" };
    }
    
    // Generate tokens and update signers
    const tokenExpiry = new Date();
    tokenExpiry.setDate(tokenExpiry.getDate() + 14); // 14-day expiry per D-06
    
    const updatedSigners = await Promise.all(
      signers.map(async (signer) => {
        const accessToken = nanoid(32); // 32-char token per D-06
        
        await db.update(agreementSigners)
          .set({
            accessToken,
            tokenExpiresAt: tokenExpiry,
            status: "notified",
            updatedAt: new Date(),
          })
          .where(eq(agreementSigners.id, signer.id));
        
        return { ...signer, accessToken };
      })
    );
    
    // Update agreement status
    await db.update(clientAgreements)
      .set({ status: "pending_signatures", updatedAt: new Date() })
      .where(eq(clientAgreements.id, agreementId));
    
    // Send notification emails
    const notificationService = new SignerNotificationService();
    
    const signingMode = (agreement.signingMode as "sequential" | "parallel") || "sequential";
    
    if (signingMode === "parallel") {
      // Send to all signers immediately
      await Promise.all(
        updatedSigners.map(signer =>
          notificationService.sendSigningInvitation({
            signerEmail: signer.email,
            signerName: signer.name,
            agreementTitle: agreement.title,
            accessToken: signer.accessToken!,
          })
        )
      );
    } else {
      // Sequential: only send to first signer (order = 1)
      const firstSigner = updatedSigners.find(s => s.signingOrder === 1);
      if (firstSigner) {
        await notificationService.sendSigningInvitation({
          signerEmail: firstSigner.email,
          signerName: firstSigner.name,
          agreementTitle: agreement.title,
          accessToken: firstSigner.accessToken!,
        });
      }
    }
    
    return { success: true };
  } catch (e) {
    return { success: false, error: "Failed to send invitations" };
  }
}

import { Resend } from "resend";

const resend = new Resend(process.env.RESEND_API_KEY);

export interface SigningInvitationParams {
  signerEmail: string;
  signerName: string;
  agreementTitle: string;
  accessToken: string;
  locale?: string;
}

export interface ReminderParams extends SigningInvitationParams {
  daysRemaining: number;
}

export class SignerNotificationService {
  private baseUrl: string;
  
  constructor() {
    this.baseUrl = process.env.NEXT_PUBLIC_APP_URL || "http://localhost:3000";
  }
  
  private getMagicLink(token: string, locale: string = "en"): string {
    return `${this.baseUrl}/${locale}/c/${token}`;
  }
  
  async sendSigningInvitation(params: SigningInvitationParams): Promise<{ success: boolean; error?: string }> {
    const { signerEmail, signerName, agreementTitle, accessToken, locale = "en" } = params;
    const magicLink = this.getMagicLink(accessToken, locale);
    
    try {
      await resend.emails.send({
        from: "TeveroSEO <noreply@teveroseo.com>",
        to: signerEmail,
        subject: `Action Required: Sign "${agreementTitle}"`,
        html: this.buildInvitationHtml({
          signerName,
          agreementTitle,
          magicLink,
          locale,
        }),
      });
      
      return { success: true };
    } catch (error) {
      console.error("Failed to send signing invitation:", error);
      return { success: false, error: "Email delivery failed" };
    }
  }
  
  async sendReminder(params: ReminderParams): Promise<{ success: boolean; error?: string }> {
    const { signerEmail, signerName, agreementTitle, accessToken, daysRemaining, locale = "en" } = params;
    const magicLink = this.getMagicLink(accessToken, locale);
    
    try {
      await resend.emails.send({
        from: "TeveroSEO <noreply@teveroseo.com>",
        to: signerEmail,
        subject: `Reminder: "${agreementTitle}" awaits your signature`,
        html: this.buildReminderHtml({
          signerName,
          agreementTitle,
          magicLink,
          daysRemaining,

      <p className="text-gray-600 mb-8">Configure signers and send for signatures</p>

    </div>
  );
}

    const result = await addSigner(data.agreement.id, signerData);
    if (result.success && result.signer) {
      setData(prev => ({
        ...prev,
        signers: [...prev.signers, result.signer!],
        canSend: true,
        sendMessage: undefined,
      }));
    }
    setIsAddDialogOpen(false);
  };
  
  const handleRemoveSigner = async (signerId: number) => {
    const result = await removeSigner(signerId);
    if (result.success) {
      setData(prev => ({
        ...prev,
        signers: prev.signers.filter(s => s.id !== signerId),
        canSend: prev.signers.length > 1,
        sendMessage: prev.signers.length <= 1 ? "Add at least one signer" : undefined,
      }));
    }
  };
  
  const handleReorder = async (newOrder: number[]) => {
    await updateSigningOrder(data.agreement.id, newOrder);
    setData(prev => ({
      ...prev,
      signers: newOrder.map((id, idx) => {
        const signer = prev.signers.find(s => s.id === id)!;
        return { ...signer, signingOrder: idx + 1 };
      }),
    }));
  };
  
  const handleModeChange = async (mode: "sequential" | "parallel") => {
    await updateSigningMode(data.agreement.id, mode);
    setData(prev => ({
      ...prev,
      agreement: { ...prev.agreement, signingMode: mode },
    }));
  };
  
  const handleSendInvitations = async () => {
    setIsSending(true);
    setSendError(null);
    
    const result = await sendInvitations(data.agreement.id);
    
    if (result.success) {
      setSendSuccess(true);
    } else {
      setSendError(result.error || "Failed to send invitations");
    }
    
    setIsSending(false);
  };
  
  if (sendSuccess) {
    return (
      <Card className="border-green-200 bg-green-50">
        <CardContent className="py-8 text-center">
          <h2 className="text-xl font-semibold text-green-800 mb-2">
            {t("preSigning.invitationsSent")}
          </h2>
          <p className="text-green-700">
            {data.agreement.signingMode === "sequential"
              ? t("preSigning.sequentialSentMessage")
              : t("preSigning.parallelSentMessage", { count: data.signers.length })}
          </p>
        </CardContent>
      </Card>
    );
  }
  
  return (
    <div className="space-y-6">
      {/* Signing Mode Selection */}
      <Card>
        <CardHeader>
          <CardTitle>{t("preSigning.signingMode")}</CardTitle>
        </CardHeader>
        <CardContent>
          <RadioGroup
            value={data.agreement.signingMode}
            onValueChange={(v) => handleModeChange(v as "sequential" | "parallel")}
          >
            <div className="flex items-center space-x-2">
              <RadioGroupItem value="sequential" id="sequential" />
              <Label htmlFor="sequential" className="cursor-pointer">
                <span className="font-medium">{t("preSigning.sequential")}</span>
                <span className="text-sm text-gray-500 block">
                  {t("preSigning.sequentialDescription")}
                </span>
              </Label>
            </div>
            <div className="flex items-center space-x-2 mt-4">
              <RadioGroupItem value="parallel" id="parallel" />
              <Label htmlFor="parallel" className="cursor-pointer">
                <span className="font-medium">{t("preSigning.parallel")}</span>
                <span className="text-sm text-gray-500 block">
                  {t("preSigning.parallelDescription")}
                </span>
              </Label>
            </div>
          </RadioGroup>
        </CardContent>
      </Card>
      
      {/* Signers List */}
      <Card>
        <CardHeader className="flex flex-row items-center justify-between">
          <CardTitle>{t("preSigning.signers")}</CardTitle>
          <Button size="sm" variant="outline" onClick={() => setIsAddDialogOpen(true)}>
            <Plus className="w-4 h-4 mr-2" />
            {t("preSigning.addSigner")}
          </Button>
        </CardHeader>
        <CardContent>
          {data.signers.length === 0 ? (
            <p className="text-gray-500 text-center py-4">{t("preSigning.noSigners")}</p>
          ) : (
            <SignerList
              signers={data.signers}
              mode={data.agreement.signingMode}
              onRemove={handleRemoveSigner}
              onReorder={handleReorder}
            />
          )}
        </CardContent>
      </Card>
      
      {/* Send Button */}
      <div className="flex flex-col items-end gap-2">
        {sendError && (
          <Alert variant="destructive" className="w-full">
            <AlertDescription>{sendError}</AlertDescription>
          </Alert>
        )}
        
        <Button
          size="lg"
          onClick={handleSendInvitations}
          disabled={!data.canSend || isSending}
        >
          <Send className="w-4 h-4 mr-2" />
          {isSending ? t("preSigning.sending") : t("preSigning.sendInvitations")}
        </Button>
        
        {data.sendMessage && (
          <p className="text-sm text-gray-500">{data.sendMessage}</p>
        )}
      </div>
      
      <AddSignerDialog
        open={isAddDialogOpen}
        onOpenChange={setIsAddDialogOpen}
        onAdd={handleAddSigner}
      />
    </div>
  );
}

"use client";

import { DndContext, closestCenter, DragEndEvent } from "@dnd-kit/core";
import { SortableContext, useSortable, verticalListSortingStrategy, arrayMove } from "@dnd-kit/sortable";
import { CSS } from "@dnd-kit/utilities";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { GripVertical, Trash2, Mail, Eye, CheckCircle } from "lucide-react";
import { SignerData } from "@/app/[locale]/(dashboard)/clients/[clientId]/agreements/[agreementId]/pre-sign/actions";

interface SignerListProps {
  signers: SignerData[];
  mode: "sequential" | "parallel";
  onRemove: (signerId: number) => void;
  onReorder: (newOrder: number[]) => void;
}

export function SignerList({ signers, mode, onRemove, onReorder }: SignerListProps) {
  const handleDragEnd = (event: DragEndEvent) => {
    const { active, over } = event;
    if (over && active.id !== over.id) {
      const oldIndex = signers.findIndex(s => s.id === active.id);
      const newIndex = signers.findIndex(s => s.id === over.id);
      const reordered = arrayMove(signers, oldIndex, newIndex);
      onReorder(reordered.map(s => s.id));
    }
  };
  
  const sortedSigners = [...signers].sort((a, b) => a.signingOrder - b.signingOrder);
  
  return (
    <DndContext collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
      <SortableContext
        items={sortedSigners.map(s => s.id)}
        strategy={verticalListSortingStrategy}
        disabled={mode === "parallel"}
      >
        <div className="space-y-2">
          {sortedSigners.map(signer => (
            <SortableSignerItem
              key={signer.id}
              signer={signer}
              mode={mode}
              onRemove={() => onRemove(signer.id)}
            />
          ))}
        </div>
      </SortableContext>
    </DndContext>
  );
}

function SortableSignerItem({
  signer,
  mode,
  onRemove,
}: {
  signer: SignerData;
  mode: "sequential" | "parallel";
  onRemove: () => void;
}) {
  const {
    attributes,
    listeners,
    setNodeRef,
    transform,
    transition,
    isDragging,
  } = useSortable({ id: signer.id, disabled: mode === "parallel" });
  
  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
    opacity: isDragging ? 0.5 : 1,
  };
  
  const statusIcon = {
    pending: null,
    notified: <Mail className="w-4 h-4 text-blue-500" />,
    viewed: <Eye className="w-4 h-4 text-amber-500" />,
    signed: <CheckCircle className="w-4 h-4 text-green-500" />,
  }[signer.status] || null;
  
  const statusBadge = {
    pending: <Badge variant="secondary">Pending</Badge>,
    notified: <Badge variant="outline" className="border-blue-500 text-blue-500">Notified</Badge>,
    viewed: <Badge variant="outline" className="border-amber-500 text-amber-500">Viewed</Badge>,
    signed: <Badge className="bg-green-500">Signed</Badge>,
  }[signer.status] || <Badge variant="secondary">{signer.status}</Badge>;
  
  return (
    <div
      ref={setNodeRef}
      style={style}
      className="flex items-center gap-3 p-3 bg-gray-50 rounded-lg"
    >
      {mode === "sequential" && (
        <button
          {...attributes}
          {...listeners}
          className="cursor-grab text-gray-400 hover:text-gray-600"
        >
          <GripVertical className="w-5 h-5" />
        </button>
      )}
      
      {mode === "sequential" && (
        <span className="w-6 h-6 rounded-full bg-gray-200 flex items-center justify-center text-sm font-medium">
          {signer.signingOrder}
        </span>
      )}
      
      <div className="flex-1 min-w-0">
        <p className="font-medium truncate">{signer.name}</p>
        <p className="text-sm text-gray-500 truncate">{signer.email}</p>
      </div>
      
      <div className="flex items-center gap-2">
        {statusIcon}
        {statusBadge}
      </div>
      
      <Button
        variant="ghost"
        size="sm"
        onClick={onRemove}
        className="text-gray-400 hover:text-red-500"
        disabled={signer.status === "signed"}
      >
        <Trash2 className="w-4 h-4" />
      </Button>
    </div>
  );
}
