**ProgressIndicator.tsx** - Shows signing progress:
```typescript
"use client";

import { useTranslations } from "next-intl";
import { CheckCircle, Circle, Clock } from "lucide-react";

interface ProgressIndicatorProps {
  mode: "sequential" | "parallel";
  totalSigners: number;
  signedCount: number;
  currentOrder: number;
}

export function ProgressIndicator({ mode, totalSigners, signedCount, currentOrder }: ProgressIndicatorProps) {
  const t = useTranslations("contract");
  
  if (mode === "parallel") {
    return (
      <div className="mb-6 p-4 bg-blue-50 rounded-lg">
        <p className="text-sm text-blue-800">
          {t("progress.parallel", { signed: signedCount, total: totalSigners })}
        </p>
      </div>
    );
  }
  
  // Sequential mode - show steps
  return (
    <div className="mb-6">
      <div className="flex items-center gap-2">
        {Array.from({ length: totalSigners }, (_, i) => {
          const order = i + 1;
          const isSigned = order <= signedCount;
          const isCurrent = order === currentOrder;
          
          return (
            <div key={order} className="flex items-center">
              {isSigned ? (
                <CheckCircle className="w-6 h-6 text-green-500" />
              ) : isCurrent ? (
                <Clock className="w-6 h-6 text-blue-500" />
              ) : (
                <Circle className="w-6 h-6 text-gray-300" />
              )}
              {order < totalSigners && (
                <div className={`w-8 h-0.5 ${isSigned ? "bg-green-500" : "bg-gray-300"}`} />
              )}
            </div>
          );
        })}
      </div>
      <p className="text-sm text-gray-600 mt-2">
        {t("progress.sequential", { current: currentOrder, total: totalSigners })}
      </p>
    </div>
  );
}
```

**SigningButtons.tsx** - Smart-ID and Mobile-ID buttons:
```typescript
"use client";

import { useState } from "react";
import { useTranslations } from "next-intl";
import { Button } from "@/components/ui/button";
import { Loader2 } from "lucide-react";
import { initiateSigning } from "@/app/[locale]/c/[token]/actions";

interface SigningButtonsProps {
  token: string;
  disabled: boolean;
}

export function SigningButtons({ token, disabled }: SigningButtonsProps) {
  const t = useTranslations("contract");
  const [isLoading, setIsLoading] = useState<"smart-id" | "mobile-id" | null>(null);
  const [error, setError] = useState<string | null>(null);
  
  const handleSign = async (method: "smart-id" | "mobile-id") => {
    setIsLoading(method);
    setError(null);
    
    try {
      const result = await initiateSigning(token);
      
      if ("error" in result) {
        setError(result.error);
        setIsLoading(null);
        return;
      }
      
      // Redirect to Dokobit
      window.location.href = result.redirectUrl;
    } catch (e) {
      setError(t("error.signingFailed"));
      setIsLoading(null);
    }
  };
  
  return (
    <div className="space-y-3">
      <Button
        className="w-full"
        onClick={() => handleSign("smart-id")}
        disabled={disabled || isLoading !== null}
      >
        {isLoading === "smart-id" && <Loader2 className="w-4 h-4 mr-2 animate-spin" />}
        {t("signature.smartId")}
      </Button>
      
      <Button
        variant="outline"
        className="w-full"
        onClick={() => handleSign("mobile-id")}
        disabled={disabled || isLoading !== null}
      >
        {isLoading === "mobile-id" && <Loader2 className="w-4 h-4 mr-2 animate-spin" />}
        {t("signature.mobileId")}
      </Button>
      
      {error && (
        <p className="text-sm text-red-600 text-center">{error}</p>
      )}
    </div>
  );
}
```
