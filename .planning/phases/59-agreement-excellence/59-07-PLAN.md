```typescript
import { db } from "@/lib/db";
import { eq } from "drizzle-orm";
import { workspaces } from "@/db/schema";

export interface BrandingConfig {
  logoUrl: string | null;
  primaryColor: string;
  secondaryColor: string;
  companyName: string;
  companyAddress: string | null;
  companyEmail: string | null;
  companyPhone: string | null;
  footerText: string | null;
}

const DEFAULT_BRANDING: BrandingConfig = {
  logoUrl: null,
  primaryColor: "#2563eb", // Blue-600
  secondaryColor: "#64748b", // Slate-500
  companyName: "TeveroSEO",
  companyAddress: null,
  companyEmail: null,
  companyPhone: null,
  footerText: null,
};

export class PdfBrandingService {
  async getBrandingConfig(workspaceId: number): Promise<BrandingConfig> {
    const workspace = await db.query.workspaces.findFirst({
      where: eq(workspaces.id, workspaceId),
    });
    
    if (!workspace) {
      return DEFAULT_BRANDING;
    }
    
    // Extract branding from workspace settings
    const settings = (workspace.settings as Record<string, unknown>) || {};
    const branding = (settings.branding as Record<string, unknown>) || {};
    
    return {
      logoUrl: (branding.logoUrl as string) || null,
      primaryColor: (branding.primaryColor as string) || DEFAULT_BRANDING.primaryColor,
      secondaryColor: (branding.secondaryColor as string) || DEFAULT_BRANDING.secondaryColor,
      companyName: workspace.name || DEFAULT_BRANDING.companyName,
      companyAddress: (branding.address as string) || null,
      companyEmail: (branding.email as string) || null,
      companyPhone: (branding.phone as string) || null,
      footerText: (branding.footerText as string) || null,
    };
  }
  
  hexToRgb(hex: string): { r: number; g: number; b: number } {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    if (!result) {
      return { r: 37, g: 99, b: 235 }; // Default blue
    }
    return {
      r: parseInt(result[1], 16) / 255,
      g: parseInt(result[2], 16) / 255,
      b: parseInt(result[3], 16) / 255,
    };
  }
  
  async fetchLogoBytes(logoUrl: string): Promise<Uint8Array | null> {
    try {
      const response = await fetch(logoUrl);
      if (!response.ok) return null;
      
      const arrayBuffer = await response.arrayBuffer();
      return new Uint8Array(arrayBuffer);
    } catch {
      return null;
    }
  }
}
```

Create PdfGenerationService:

```typescript
import { PDFDocument, rgb, StandardFonts } from "pdf-lib";
import fontkit from "@pdf-lib/fontkit";
import { db } from "@/lib/db";
import { eq } from "drizzle-orm";
import { clientAgreements, agreementTemplates, agreementSigners } from "@/db/schema";
import { AgreementVariableService } from "./agreement-variable-service";
import { PdfBrandingService, BrandingConfig } from "./pdf-branding-service";
import * as fs from "fs/promises";
import * as path from "path";

interface TemplateClause {
  id: string;
  title: string;
  content: string;
  order: number;
}

export interface PdfGenerationOptions {
  includeSignatures?: boolean;
  locale?: string;
}

export class PdfGenerationService {
  private variableService: AgreementVariableService;
  private brandingService: PdfBrandingService;
  
  constructor() {
    this.variableService = new AgreementVariableService();
    this.brandingService = new PdfBrandingService();
  }
  
  async generateAgreementPdf(
    agreementId: number,
    options: PdfGenerationOptions = {}
  ): Promise<Uint8Array> {
    const { includeSignatures = true, locale = "en" } = options;
    
    // Load agreement data
    const agreement = await db.query.clientAgreements.findFirst({
      where: eq(clientAgreements.id, agreementId),
      with: {
        template: true,
        client: true,
      },
    });
    
    if (!agreement || !agreement.template) {
      throw new Error("Agreement or template not found");
    }
    
    // Load signers if including signatures
    const signers = includeSignatures
      ? await db.query.agreementSigners.findMany({
          where: eq(agreementSigners.agreementId, agreementId),
          orderBy: (s, { asc }) => [asc(s.signingOrder)],
        })
      : [];
    
    // Resolve variables
    const variableContext = await this.variableService.loadContext(agreementId);
    
    // Get branding
    const branding = await this.brandingService.getBrandingConfig(agreement.workspaceId);
    
    // Create PDF
    const pdfDoc = await PDFDocument.create();
    pdfDoc.registerFontkit(fontkit);
    
    // Try to load custom font, fallback to Helvetica
    let regularFont;
    let boldFont;
    
    try {
      const fontPath = path.join(process.cwd(), "public/fonts/Inter-Regular.ttf");
      const boldFontPath = path.join(process.cwd(), "public/fonts/Inter-Bold.ttf");
      
      const regularFontBytes = await fs.readFile(fontPath);
      const boldFontBytes = await fs.readFile(boldFontPath);
      
      regularFont = await pdfDoc.embedFont(regularFontBytes);
      boldFont = await pdfDoc.embedFont(boldFontBytes);
    } catch {
      // Fallback to standard fonts if Inter not available
      regularFont = await pdfDoc.embedFont(StandardFonts.Helvetica);
      boldFont = await pdfDoc.embedFont(StandardFonts.HelveticaBold);
    }
    
    // Parse clauses
    const content = agreement.template.content as { clauses: TemplateClause[] };
    const clauses = content.clauses || [];
    const clauseOrder = (agreement.template.clauseOrder as string[]) || clauses.map(c => c.id);
    const orderedClauses = clauseOrder
      .map(id => clauses.find(c => c.id === id))
      .filter((c): c is TemplateClause => c !== undefined);
    
    // Page settings
    const pageWidth = 595.28; // A4 width in points
    const pageHeight = 841.89; // A4 height in points
    const margin = 50;
    const lineHeight = 14;
    const titleSize = 18;
    const headingSize = 14;
    const bodySize = 11;
    
    let currentPage = pdfDoc.addPage([pageWidth, pageHeight]);
    let yPosition = pageHeight - margin;
    
    const primaryRgb = this.brandingService.hexToRgb(branding.primaryColor);
    const secondaryRgb = this.brandingService.hexToRgb(branding.secondaryColor);
    
    // Header with logo and company info
    if (branding.logoUrl) {
      const logoBytes = await this.brandingService.fetchLogoBytes(branding.logoUrl);
      if (logoBytes) {
        try {
          const logoImage = await pdfDoc.embedPng(logoBytes);
          const logoDims = logoImage.scale(0.5);
          currentPage.drawImage(logoImage, {
            x: margin,
            y: yPosition - logoDims.height,
            width: logoDims.width,
            height: logoDims.height,
          });
          yPosition -= logoDims.height + 20;
        } catch {
          // Logo embed failed, continue without
        }
      }
    }
    
    // Company name
    currentPage.drawText(branding.companyName, {
      x: margin,
      y: yPosition,
      size: titleSize,
      font: boldFont,
      color: rgb(primaryRgb.r, primaryRgb.g, primaryRgb.b),
    });
    yPosition -= lineHeight * 2;
    
    // Agreement title
    currentPage.drawText(agreement.title, {
      x: margin,
      y: yPosition,
      size: headingSize,
      font: boldFont,
      color: rgb(0, 0, 0),
    });
    yPosition -= lineHeight * 2;
    
    // Horizontal line
    currentPage.drawLine({
      start: { x: margin, y: yPosition },
      end: { x: pageWidth - margin, y: yPosition },
      thickness: 1,
      color: rgb(secondaryRgb.r, secondaryRgb.g, secondaryRgb.b),
    });
    yPosition -= lineHeight * 2;
    
    // Render clauses
    for (const clause of orderedClauses) {
      // Check if we need a new page
      if (yPosition < margin + 100) {
        currentPage = pdfDoc.addPage([pageWidth, pageHeight]);
        yPosition = pageHeight - margin;
      }
      
      // Clause title
      currentPage.drawText(clause.title, {
        x: margin,
        y: yPosition,
        size: headingSize,
        font: boldFont,
        color: rgb(0, 0, 0),
      });
      yPosition -= lineHeight * 1.5;
      
      // Resolve variables in content
      const resolvedContent = this.variableService.replaceInText(clause.content, variableContext);
      
      // Strip HTML tags for PDF (basic conversion)
      const plainText = this.stripHtml(resolvedContent);
      
      // Wrap and draw text
      const lines = this.wrapText(plainText, pageWidth - (margin * 2), regularFont, bodySize);
      
      for (const line of lines) {
        if (yPosition < margin + 20) {
          currentPage = pdfDoc.addPage([pageWidth, pageHeight]);
          yPosition = pageHeight - margin;
        }
        
        currentPage.drawText(line, {
          x: margin,
          y: yPosition,
          size: bodySize,
          font: regularFont,
          color: rgb(0, 0, 0),
        });
        yPosition -= lineHeight;
      }
      
      yPosition -= lineHeight; // Extra space between clauses
    }
    
    // Signature section
    if (includeSignatures && signers.length > 0) {
      // Check if we need a new page for signatures
      if (yPosition < margin + 150) {
        currentPage = pdfDoc.addPage([pageWidth, pageHeight]);
        yPosition = pageHeight - margin;
      }
      
      yPosition -= lineHeight * 2;
      
      currentPage.drawText(locale === "lt" ? "Pasirasymai" : "Signatures", {
        x: margin,
        y: yPosition,
        size: headingSize,
        font: boldFont,
        color: rgb(0, 0, 0),
      });
      yPosition -= lineHeight * 2;
      
      for (const signer of signers) {
        if (signer.status === "signed" && signer.signedAt) {
          currentPage.drawText(signer.name + " (" + (signer.role || "Signer") + ")", {
            x: margin,
            y: yPosition,
            size: bodySize,
            font: boldFont,
            color: rgb(0, 0, 0),
          });
          yPosition -= lineHeight;
          
          const signedDate = new Date(signer.signedAt).toLocaleDateString(
            locale === "lt" ? "lt-LT" : "en-US",
            { year: "numeric", month: "long", day: "numeric", hour: "2-digit", minute: "2-digit" }
          );
          
          const signedLabel = locale === "lt" ? "Pasirašyta: " : "Signed: ";
          currentPage.drawText(signedLabel + signedDate, {
            x: margin,
            y: yPosition,
            size: bodySize - 1,
            font: regularFont,
            color: rgb(secondaryRgb.r, secondaryRgb.g, secondaryRgb.b),
          });
          yPosition -= lineHeight;
          
          // Show signing method
          const sigData = signer.signatureData as { method?: string } | null;
          if (sigData?.method) {
            currentPage.drawText("Method: " + sigData.method, {
              x: margin,
              y: yPosition,
              size: bodySize - 2,
              font: regularFont,
              color: rgb(secondaryRgb.r, secondaryRgb.g, secondaryRgb.b),
            });
            yPosition -= lineHeight;
          }
          
          yPosition -= lineHeight; // Space between signers
        } else {
          // Pending signature placeholder
          const pendingLabel = locale === "lt" ? " - Laukiama parašo" : " - Awaiting signature";
          currentPage.drawText(signer.name + pendingLabel, {
            x: margin,
            y: yPosition,
            size: bodySize,
            font: regularFont,
            color: rgb(secondaryRgb.r, secondaryRgb.g, secondaryRgb.b),
          });
          yPosition -= lineHeight * 2;
        }
      }
    }
    
    // Footer on each page
    const pages = pdfDoc.getPages();
    for (let i = 0; i < pages.length; i++) {
      const page = pages[i];
      const footerY = 30;
      
      // Page number
      page.drawText((i + 1) + " / " + pages.length, {
        x: pageWidth - margin - 30,
        y: footerY,
        size: 9,
        font: regularFont,
        color: rgb(secondaryRgb.r, secondaryRgb.g, secondaryRgb.b),
      });
      
      // Footer text
      if (branding.footerText) {
        page.drawText(branding.footerText, {
          x: margin,
          y: footerY,
          size: 9,
          font: regularFont,
          color: rgb(secondaryRgb.r, secondaryRgb.g, secondaryRgb.b),
        });
      }
    }
    
    // Save and return
    const pdfBytes = await pdfDoc.save();
    return pdfBytes;
  }
  
  private stripHtml(html: string): string {
    return html
      .replace(/<br\s*\/?>/gi, "\n")
      .replace(/<\/p>/gi, "\n\n")
      .replace(/<[^>]+>/g, "")
      .replace(/&nbsp;/g, " ")
      .replace(/&amp;/g, "&")
      .replace(/&lt;/g, "<")
      .replace(/&gt;/g, ">")
      .replace(/&quot;/g, '"')
      .trim();
  }
  
  private wrapText(text: string, maxWidth: number, font: any, fontSize: number): string[] {
    const lines: string[] = [];
    const paragraphs = text.split("\n");
    
    for (const paragraph of paragraphs) {
      if (paragraph.trim() === "") {
        lines.push("");
        continue;
      }
      
      const words = paragraph.split(" ");
      let currentLine = "";
      
      for (const word of words) {
        const testLine = currentLine ? currentLine + " " + word : word;
        const testWidth = font.widthOfTextAtSize(testLine, fontSize);
        
        if (testWidth > maxWidth && currentLine) {
          lines.push(currentLine);
          currentLine = word;
        } else {
          currentLine = testLine;
        }
      }
      
      if (currentLine) {
        lines.push(currentLine);
      }
    }
    
    return lines;
  }
}
```

Create PDF download API route:

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { db } from "@/lib/db";
import { eq } from "drizzle-orm";
import { clientAgreements } from "@/db/schema";
import { PdfGenerationService } from "@/server/services/pdf-generation-service";

export async function GET(
  request: NextRequest,
  { params }: { params: { agreementId: string } }
) {
  try {
    // Auth check
    const session = await auth();
    if (!session?.user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }
    
    const agreementId = parseInt(params.agreementId, 10);
    if (isNaN(agreementId)) {
      return NextResponse.json({ error: "Invalid agreement ID" }, { status: 400 });
    }
    
    // Verify access to agreement
    const agreement = await db.query.clientAgreements.findFirst({
      where: eq(clientAgreements.id, agreementId),
    });
    
    if (!agreement) {
      return NextResponse.json({ error: "Agreement not found" }, { status: 404 });
    }
    
    // Get query params
    const searchParams = request.nextUrl.searchParams;
    const includeSignatures = searchParams.get("signatures") !== "false";
    const locale = searchParams.get("locale") || "en";
    
    // Generate PDF
    const pdfService = new PdfGenerationService();
    const pdfBytes = await pdfService.generateAgreementPdf(agreementId, {
      includeSignatures,
      locale,
    });
    
    // Create filename - sanitize title for filename
    const safeTitle = agreement.title.replace(/[^a-z0-9]/gi, "-").substring(0, 50);
    const filename = safeTitle + "-" + agreementId + ".pdf";
    
    // Return PDF response
    return new NextResponse(pdfBytes, {
      status: 200,
      headers: {
        "Content-Type": "application/pdf",
        "Content-Disposition": "attachment; filename=\"" + filename + "\"",
        "Content-Length": pdfBytes.length.toString(),
        "Cache-Control": "no-store",
      },
    });
  } catch (error) {
    console.error("PDF generation error:", error);
    return NextResponse.json(
      { error: "Failed to generate PDF" },
      { status: 500 }
    );
  }
}
```
