```typescript
async function getSignerData(token: string) {
  const signer = await db.query.agreementSigners.findFirst({
    where: and(
      eq(agreementSigners.accessToken, token),
      gt(agreementSigners.tokenExpiresAt, new Date())
    ),
  });
  
  if (!signer) return null;
  
  const agreement = await db.query.clientAgreements.findFirst({
    where: eq(clientAgreements.id, signer.agreementId),
  });
  
  if (!agreement) return null;
  
  // Get all signers to show progress
  const allSigners = await db.query.agreementSigners.findMany({
    where: eq(agreementSigners.agreementId, signer.agreementId),
    orderBy: (s, { asc }) => [asc(s.signingOrder)],
  });
  
  const signedCount = allSigners.filter(s => s.status === "signed").length;
  const isComplete = signedCount === allSigners.length;
  
  return {
    signer: {
      name: signer.name,
      signedAt: signer.signedAt,
      status: signer.status,
    },
    agreement: {
      id: agreement.id,
      title: agreement.title,
      status: agreement.status,
    },
    progress: {
      signed: signedCount,
      total: allSigners.length,
      isComplete,
    },
  };
}
```

```typescript
// page.tsx
import { notFound } from "next/navigation";
import { db } from "@/lib/db";
import { eq } from "drizzle-orm";
import { clientAgreements, agreementSigners, agreementTemplates } from "@/db/schema";
import { AgreementStatusTracker } from "@/components/agreement/AgreementStatusTracker";
import { SignerStatusList } from "@/components/agreement/SignerStatusList";
import { AgreementActions } from "@/components/agreement/AgreementActions";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";

interface Props {
  params: { clientId: string; agreementId: string };
}

async function getAgreementData(clientId: number, agreementId: number) {
  const agreement = await db.query.clientAgreements.findFirst({
    where: eq(clientAgreements.id, agreementId),
  });
  
  if (!agreement || agreement.clientId !== clientId) return null;
  
  const template = await db.query.agreementTemplates.findFirst({
    where: eq(agreementTemplates.id, agreement.templateId),
  });
  
  const signers = await db.query.agreementSigners.findMany({
    where: eq(agreementSigners.agreementId, agreementId),
    orderBy: (s, { asc }) => [asc(s.signingOrder)],
  });
  
  return {
    agreement: {
      id: agreement.id,
      title: agreement.title,
      status: agreement.status,
      signingMode: (agreement.signingMode as "sequential" | "parallel") || "sequential",
      createdAt: agreement.createdAt,
      updatedAt: agreement.updatedAt,
    },
    template: template ? {
      id: template.id,
      name: template.name,
    } : null,
    signers: signers.map(s => ({
      id: s.id,
      name: s.name,
      email: s.email,
      role: s.role || "signer",
      signingOrder: s.signingOrder || 1,
      status: s.status || "pending",
      viewedAt: s.viewedAt,
      signedAt: s.signedAt,
      tokenExpiresAt: s.tokenExpiresAt,
    })),
  };
}

export default async function AgreementDetailPage({ params }: Props) {
  const clientId = parseInt(params.clientId, 10);
  const agreementId = parseInt(params.agreementId, 10);
  
  if (isNaN(clientId) || isNaN(agreementId)) notFound();
  
  const data = await getAgreementData(clientId, agreementId);
  if (!data) notFound();
  
  const { agreement, template, signers } = data;
  
  const statusVariant: Record<string, "default" | "secondary" | "destructive" | "outline"> = {
    draft: "secondary",
    pending_signatures: "outline",
    partially_signed: "outline",
    completed: "default",
    expired: "destructive",
    cancelled: "destructive",
  };
  
  const statusLabel: Record<string, string> = {
    draft: "Draft",
    pending_signatures: "Pending Signatures",
    partially_signed: "Partially Signed",
    completed: "Completed",
    expired: "Expired",
    cancelled: "Cancelled",
  };
  
  return (
    <div className="container py-8 max-w-4xl">
      {/* Header */}
      <div className="flex items-start justify-between mb-8">
        <div>
          <h1 className="text-2xl font-bold text-gray-900">{agreement.title}</h1>
          {template && (
            <p className="text-gray-500 mt-1">Template: {template.name}</p>
          )}
        </div>
        <Badge variant={statusVariant[agreement.status] || "secondary"}>
          {statusLabel[agreement.status] || agreement.status}
        </Badge>
      </div>
      
      {/* Status Tracker */}
      <Card className="mb-6">
        <CardHeader>
          <CardTitle>Signing Progress</CardTitle>
        </CardHeader>
        <CardContent>
          <AgreementStatusTracker
            signers={signers}
            mode={agreement.signingMode}
          />
        </CardContent>
      </Card>
      
      {/* Signers List */}
      <Card className="mb-6">
        <CardHeader className="flex flex-row items-center justify-between">
          <CardTitle>Signers</CardTitle>
          <span className="text-sm text-gray-500">
            {agreement.signingMode === "sequential" ? "Sequential" : "Parallel"} signing
          </span>
        </CardHeader>
        <CardContent>
          <SignerStatusList
            signers={signers}
            agreementId={agreement.id}
            agreementStatus={agreement.status}
          />
        </CardContent>
      </Card>
      
      {/* Actions */}
      <AgreementActions
        agreementId={agreement.id}
        clientId={clientId}
        status={agreement.status}
        hasSigners={signers.length > 0}
      />
    </div>
  );
}
```

```typescript
"use client";

import { CheckCircle, Circle, Clock, Eye, XCircle } from "lucide-react";

interface Signer {
  id: number;
  name: string;
  status: string;
  signingOrder: number;
}

interface AgreementStatusTrackerProps {
  signers: Signer[];
  mode: "sequential" | "parallel";
}

export function AgreementStatusTracker({ signers, mode }: AgreementStatusTrackerProps) {
  const sortedSigners = [...signers].sort((a, b) => a.signingOrder - b.signingOrder);
  const signedCount = signers.filter(s => s.status === "signed").length;
  const totalCount = signers.length;
  const progressPercent = totalCount > 0 ? (signedCount / totalCount) * 100 : 0;
  
  const getStatusIcon = (status: string) => {
    switch (status) {
      case "signed":
        return <CheckCircle className="w-6 h-6 text-green-500" />;
      case "viewed":
        return <Eye className="w-6 h-6 text-amber-500" />;
      case "notified":
        return <Clock className="w-6 h-6 text-blue-500" />;
      case "declined":
        return <XCircle className="w-6 h-6 text-red-500" />;
      default:
        return <Circle className="w-6 h-6 text-gray-300" />;
    }
  };
  
  if (mode === "parallel") {
    return (
      <div>
        {/* Progress bar */}
        <div className="mb-4">
          <div className="flex justify-between text-sm mb-2">
            <span className="text-gray-600">Overall Progress</span>
            <span className="font-medium">{signedCount} of {totalCount} signed</span>
          </div>
          <div className="h-3 bg-gray-200 rounded-full overflow-hidden">
            <div
              className="h-full bg-green-500 transition-all duration-300"
              style={{ width: progressPercent + "%" }}
            />
          </div>
        </div>
        
        {/* Grid of signers */}
        <div className="grid grid-cols-2 md:grid-cols-3 gap-3">
          {sortedSigners.map(signer => (
            <div
              key={signer.id}
              className="flex items-center gap-2 p-3 bg-gray-50 rounded-lg"
            >
              {getStatusIcon(signer.status)}
              <span className="text-sm font-medium truncate">{signer.name}</span>
            </div>
          ))}
        </div>
      </div>
    );
  }
  
  // Sequential mode - show as timeline
  return (
    <div>
      {/* Progress bar */}
      <div className="mb-6">
        <div className="flex justify-between text-sm mb-2">
          <span className="text-gray-600">Overall Progress</span>
          <span className="font-medium">{signedCount} of {totalCount} signed</span>
        </div>
        <div className="h-3 bg-gray-200 rounded-full overflow-hidden">
          <div
            className="h-full bg-green-500 transition-all duration-300"
            style={{ width: progressPercent + "%" }}
          />
        </div>
      </div>
      
      {/* Timeline */}
      <div className="relative">
        {sortedSigners.map((signer, index) => {
          const isLast = index === sortedSigners.length - 1;
          const isCurrent = signer.status === "notified" || signer.status === "viewed";
          
          return (
            <div key={signer.id} className="flex gap-4">
              {/* Timeline line and dot */}
              <div className="flex flex-col items-center">
                <div className={"w-8 h-8 rounded-full flex items-center justify-center " + 
                  (signer.status === "signed" ? "bg-green-100" : 
                   isCurrent ? "bg-blue-100" : "bg-gray-100")}>
                  {getStatusIcon(signer.status)}
                </div>
                {!isLast && (
                  <div className={"w-0.5 flex-1 my-1 " + 
                    (signer.status === "signed" ? "bg-green-300" : "bg-gray-200")} />
                )}
              </div>
              
              {/* Content */}
              <div className={"pb-6 " + (isLast ? "" : "")}>
                <p className="font-medium text-gray-900">{signer.name}</p>
                <p className="text-sm text-gray-500">Step {signer.signingOrder}</p>
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
}
```

```typescript
"use client";

import { useState } from "react";
import { useTranslations } from "next-intl";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table";
import { Mail, RotateCcw, CheckCircle, Clock, Eye, AlertCircle } from "lucide-react";

interface Signer {
  id: number;
  name: string;
  email: string;
  role: string;
  signingOrder: number;
  status: string;
  viewedAt: Date | null;
  signedAt: Date | null;
  tokenExpiresAt: Date | null;
}

interface SignerStatusListProps {
  signers: Signer[];
  agreementId: number;
  agreementStatus: string;
}

export function SignerStatusList({ signers, agreementId, agreementStatus }: SignerStatusListProps) {
  const t = useTranslations("agreement");
  const [resending, setResending] = useState<number | null>(null);
  
  const handleResend = async (signerId: number) => {
    setResending(signerId);
    // TODO: Call resend invitation action
    await new Promise(resolve => setTimeout(resolve, 1000));
    setResending(null);
  };
  
  const getStatusBadge = (status: string) => {
    const config: Record<string, { variant: "default" | "secondary" | "destructive" | "outline"; icon: React.ReactNode; label: string }> = {
      pending: { variant: "secondary", icon: <Clock className="w-3 h-3" />, label: "Pending" },
      notified: { variant: "outline", icon: <Mail className="w-3 h-3" />, label: "Notified" },
      viewed: { variant: "outline", icon: <Eye className="w-3 h-3" />, label: "Viewed" },
      signed: { variant: "default", icon: <CheckCircle className="w-3 h-3" />, label: "Signed" },
      declined: { variant: "destructive", icon: <AlertCircle className="w-3 h-3" />, label: "Declined" },
    };
    
    const { variant, icon, label } = config[status] || config.pending;
    
    return (
      <Badge variant={variant} className="gap-1">
        {icon}
        {label}
      </Badge>
    );
  };
  
  const isTokenExpired = (expiresAt: Date | null) => {
    if (!expiresAt) return false;
    return new Date(expiresAt) < new Date();
  };
  
  const canResend = (signer: Signer) => {
    return (
      agreementStatus === "pending_signatures" &&
      (signer.status === "notified" || signer.status === "viewed") &&
      isTokenExpired(signer.tokenExpiresAt)
    );
  };
  
  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead className="w-12">#</TableHead>
          <TableHead>Name</TableHead>
          <TableHead>Email</TableHead>
          <TableHead>Role</TableHead>
          <TableHead>Status</TableHead>
          <TableHead className="text-right">Actions</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {signers.map(signer => (
          <TableRow key={signer.id}>
            <TableCell className="font-medium">{signer.signingOrder}</TableCell>
            <TableCell>{signer.name}</TableCell>
            <TableCell className="text-gray-500">{signer.email}</TableCell>
            <TableCell className="text-gray-500">{signer.role}</TableCell>
            <TableCell>
              <div className="flex flex-col gap-1">
                {getStatusBadge(signer.status)}
                {signer.signedAt && (
                  <span className="text-xs text-gray-500">
                    {new Date(signer.signedAt).toLocaleDateString()}
                  </span>
                )}
                {isTokenExpired(signer.tokenExpiresAt) && signer.status !== "signed" && (
                  <span className="text-xs text-red-500">Link expired</span>
                )}
              </div>
            </TableCell>
            <TableCell className="text-right">
              {canResend(signer) && (
                <Button
                  size="sm"
                  variant="outline"
                  onClick={() => handleResend(signer.id)}
                  disabled={resending === signer.id}
                >
                  <RotateCcw className={"w-3 h-3 mr-1 " + (resending === signer.id ? "animate-spin" : "")} />
                  Resend
                </Button>
              )}
            </TableCell>
          </TableRow>
        ))}
      </TableBody>
    </Table>
  );
}
```
