```typescript
export async function getTemplate(templateId: number): Promise<TemplateData | null> {
  const template = await db.query.agreementTemplates.findFirst({
    where: eq(agreementTemplates.id, templateId),

  return {
    id: template.id,
    name: template.name,
    description: template.description,
    clauses: (template.content as { clauses: TemplateClause[] }).clauses || [],
    clauseOrder: (template.clauseOrder as string[]) || [],

    isActive: template.isActive ?? true,

}
```

Create TemplateEditor.tsx container:

```typescript
"use client";

import { useState, useCallback } from "react";
import { DndContext, closestCenter, DragEndEvent } from "@dnd-kit/core";
import { SortableContext, verticalListSortingStrategy, arrayMove } from "@dnd-kit/sortable";
import { ClauseList } from "./ClauseList";
import { VariablePalette } from "./VariablePalette";
import { PreviewPane } from "./PreviewPane";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { saveTemplate, reorderClauses, TemplateData, TemplateClause } from "@/app/[locale]/(dashboard)/templates/[templateId]/edit/actions";

interface TemplateEditorProps {
  initialData: TemplateData;
}

export function TemplateEditor({ initialData }: TemplateEditorProps) {
  const [template, setTemplate] = useState(initialData);
  const [isSaving, setIsSaving] = useState(false);
  const [activeTab, setActiveTab] = useState<"edit" | "preview">("edit");
  
  const handleDragEnd = useCallback((event: DragEndEvent) => {
    const { active, over } = event;
    if (over && active.id !== over.id) {
      const oldIndex = template.clauseOrder.indexOf(active.id as string);
      const newIndex = template.clauseOrder.indexOf(over.id as string);
      const newOrder = arrayMove(template.clauseOrder, oldIndex, newIndex);
      
      setTemplate(prev => ({ ...prev, clauseOrder: newOrder }));
      reorderClauses(template.id, newOrder);
    }
  }, [template.id, template.clauseOrder]);
  
  const handleSave = async () => {
    setIsSaving(true);
    await saveTemplate(template.id, {
      name: template.name,
      description: template.description,
      clauses: template.clauses,
      clauseOrder: template.clauseOrder,
      variables: template.variables,
    });
    setIsSaving(false);
  };
  
  const handleInsertVariable = (clauseId: string, variable: string, position: number) => {
    setTemplate(prev => ({
      ...prev,
      clauses: prev.clauses.map(c => 
        c.id === clauseId
          ? { ...c, content: c.content.slice(0, position) + `{{${variable}}}` + c.content.slice(position) }
          : c
      ),
    }));
  };
  
  return (
    <div className="flex h-[calc(100vh-200px)]">
      {/* Left: Variable Palette */}
      <div className="w-64 border-r p-4 overflow-y-auto">
        <VariablePalette onInsert={handleInsertVariable} />
      </div>
      
      {/* Center: Editor/Preview */}
      <div className="flex-1 p-4 overflow-y-auto">
        <div className="mb-4 flex items-center gap-4">
          <Input
            value={template.name}
            onChange={e => setTemplate(prev => ({ ...prev, name: e.target.value }))}
            className="text-xl font-semibold"
          />
          <Button onClick={handleSave} disabled={isSaving}>
            {isSaving ? "Saving..." : "Save"}
          </Button>
        </div>
        
        <Tabs value={activeTab} onValueChange={v => setActiveTab(v as "edit" | "preview")}>
          <TabsList>
            <TabsTrigger value="edit">Edit</TabsTrigger>
            <TabsTrigger value="preview">Preview</TabsTrigger>
          </TabsList>
          
          <TabsContent value="edit">
            <DndContext collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
              <SortableContext items={template.clauseOrder} strategy={verticalListSortingStrategy}>
                <ClauseList
                  clauses={template.clauses}
                  order={template.clauseOrder}
                  onUpdate={(clauseId, content) => 
                    setTemplate(prev => ({
                      ...prev,
                      clauses: prev.clauses.map(c => c.id === clauseId ? { ...c, content } : c),
                    }))
                  }
                />
              </SortableContext>
            </DndContext>
          </TabsContent>
          
          <TabsContent value="preview">
            <PreviewPane clauses={template.clauses} order={template.clauseOrder} />
          </TabsContent>
        </Tabs>
      </div>
    </div>
  );
}
```

```typescript
import { useSortable } from "@dnd-kit/sortable";
import { CSS } from "@dnd-kit/utilities";
import { ClauseEditor } from "./ClauseEditor";
import { GripVertical } from "lucide-react";

interface TemplateClause {
  id: string;
  title: string;
  content: string;
  order: number;
}

interface ClauseListProps {
  clauses: TemplateClause[];
  order: string[];
  onUpdate: (clauseId: string, content: string) => void;
}

export function ClauseList({ clauses, order, onUpdate }: ClauseListProps) {
  const orderedClauses = order
    .map(id => clauses.find(c => c.id === id))
    .filter((c): c is TemplateClause => c !== undefined);
  
  return (
    <div className="space-y-4">
      {orderedClauses.map(clause => (
        <SortableClauseItem
          key={clause.id}
          clause={clause}
          onUpdate={content => onUpdate(clause.id, content)}
        />
      ))}
    </div>
  );
}

function SortableClauseItem({
  clause,
  onUpdate,
}: {
  clause: TemplateClause;
  onUpdate: (content: string) => void;
}) {
  const {
    attributes,
    listeners,
    setNodeRef,
    transform,
    transition,
    isDragging,
  } = useSortable({ id: clause.id });
  
  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
    opacity: isDragging ? 0.5 : 1,
  };
  
  return (
    <div ref={setNodeRef} style={style} className="flex gap-2">
      <button
        {...attributes}
        {...listeners}
        className="cursor-grab p-2 text-gray-400 hover:text-gray-600"
      >
        <GripVertical className="w-5 h-5" />
      </button>
      <div className="flex-1">
        <ClauseEditor clause={clause} onUpdate={onUpdate} />
      </div>
    </div>
  );
}
```

```typescript
"use client";

import { useRef, useCallback } from "react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Textarea } from "@/components/ui/textarea";
import { Input } from "@/components/ui/input";

interface TemplateClause {
  id: string;
  title: string;
  content: string;
  order: number;
}

interface ClauseEditorProps {
  clause: TemplateClause;
  onUpdate: (content: string) => void;
}

export function ClauseEditor({ clause, onUpdate }: ClauseEditorProps) {
  const textareaRef = useRef<HTMLTextAreaElement>(null);
  
  const handleDrop = useCallback((e: React.DragEvent) => {
    e.preventDefault();
    const variable = e.dataTransfer.getData("text/plain");
    if (!variable || !textareaRef.current) return;
    
    const textarea = textareaRef.current;
    const cursorPos = textarea.selectionStart;
    const newContent = 
      clause.content.slice(0, cursorPos) + 
      `{{${variable}}}` + 
      clause.content.slice(cursorPos);
    
    onUpdate(newContent);
  }, [clause.content, onUpdate]);
  
  const handleDragOver = useCallback((e: React.DragEvent) => {
    e.preventDefault();
    e.dataTransfer.dropEffect = "copy";
  }, []);
  
  // Highlight variables in content for visual feedback
  const highlightVariables = (content: string) => {
    return content.replace(/\{\{([^}]+)\}\}/g, '<span class="bg-blue-100 text-blue-800 px-1 rounded">{{$1}}</span>');
  };
  
  return (
    <Card>
      <CardHeader className="py-3">
        <CardTitle className="text-base">{clause.title}</CardTitle>
      </CardHeader>
      <CardContent>
        <Textarea
          ref={textareaRef}
          value={clause.content}
          onChange={e => onUpdate(e.target.value)}
          onDrop={handleDrop}
          onDragOver={handleDragOver}
          rows={6}
          className="font-mono text-sm"
          placeholder="Enter clause content. Drag variables here to insert."
        />
        <p className="text-xs text-gray-500 mt-2">
          Tip: Drag variables from the left panel or type {"{{variable.name}}"} manually
        </p>
      </CardContent>
    </Card>
  );
}
```

```typescript
"use client";

import { useMemo } from "react";
import DOMPurify from "dompurify";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

interface TemplateClause {
  id: string;
  title: string;
  content: string;
  order: number;
}

interface PreviewPaneProps {
  clauses: TemplateClause[];
  order: string[];
}

// Sample data for preview (matches 59-03 variable structure)
const SAMPLE_DATA: Record<string, string> = {
  "client.name": "John Smith",
  "client.company": "Smith & Co.",
  "client.email": "john@smithco.com",
  "client.phone": "+1 555-0123",
  "client.address": "123 Main St, New York, NY 10001",
  "provider.name": "Acme Services",
  "provider.company": "Acme Services LLC",
  "provider.email": "hello@acme.com",

  "services.total": "$5,000.00",
  "services.breakdown": "SEO Audit: $2,000 | Content Strategy: $1,500 | Link Building: $1,500",
  "agreement.date": "May 2, 2026",
  "agreement.validUntil": "May 2, 2027",
  "agreement.number": "AGR-2026-0042",
  "signer.name": "John Smith",
  "signer.email": "john@smithco.com",
  "signer.role": "CEO",
  "payment.amount": "$5,000.00",
  "payment.currency": "USD",
  "payment.dueDate": "May 15, 2026",
  "payment.method": "Bank Transfer",
};

export function PreviewPane({ clauses, order }: PreviewPaneProps) {
  const orderedClauses = useMemo(() => 
    order
      .map(id => clauses.find(c => c.id === id))
      .filter((c): c is TemplateClause => c !== undefined),
    [clauses, order]
  );
  
  const resolveVariables = (content: string): string => {
    return content.replace(/\{\{([^}]+)\}\}/g, (match, variable) => {
      const value = SAMPLE_DATA[variable.trim()];
      if (value) {
        return `<span class="bg-green-100 text-green-800 px-1 rounded">${value}</span>`;
      }
      return `<span class="bg-red-100 text-red-800 px-1 rounded">${match}</span>`;
    });
  };
  
  // Sanitize with DOMPurify before rendering
  const sanitizeHtml = (html: string): string => {
    return DOMPurify.sanitize(html, {
      ALLOWED_TAGS: ["p", "br", "strong", "em", "span", "ul", "ol", "li"],
      ALLOWED_ATTR: ["class"],
    });
  };
  
  return (
    <div className="space-y-4">
      <div className="p-3 bg-blue-50 rounded-lg text-sm text-blue-800">
        Preview with sample data. Green = resolved, Red = unknown variable.
      </div>
```
